10/// Parse Rust source code into a RestrictedAst
11pub fn parse(input: &str) -> Result<RestrictedAst> {
12    let file: File = syn::parse_str(input)?;
13
14    let mut functions = Vec::new();
15    let mut entry_point = None;
16
17    for item in file.items {
18        match item {
19            Item::Fn(item_fn) => {
20                // Check if this is the main function marked with #[rash::main]
21                let is_main = item_fn.attrs.iter().any(|_attr| {
22                    // For now, accept any attribute as #[rash::main]
23                    // TODO: Proper attribute parsing
24                    true
25                }) || item_fn.sig.ident == "main";
26
27                let function = convert_function(item_fn)?;
28
29                if is_main {
30                    if entry_point.is_some() {
31                        return Err(Error::Validation(
32                            "Multiple #[rash::main] functions found".to_string(),
33                        ));
34                    }
35                    entry_point = Some(function.name.clone());
36                }
37
38                functions.push(function);
39            }
40            _ => {
41                return Err(Error::Validation(
42                    "Only functions are allowed in Rash code".to_string(),
43                ));
44            }
45        }
46    }
47
48    let entry_point = entry_point
49        .ok_or_else(|| Error::Validation("No #[rash::main] function found".to_string()))?;
50
51    Ok(RestrictedAst {
52        functions,
53        entry_point,
54    })
55}

57fn convert_function(item_fn: ItemFn) -> Result<Function> {
58    let name = item_fn.sig.ident.to_string();
59
60    // Convert parameters
61    let mut params = Vec::new();
62    for input in item_fn.sig.inputs {
63        match input {
64            FnArg::Typed(pat_type) => {
65                if let Pat::Ident(pat_ident) = &*pat_type.pat {
66                    let param_name = pat_ident.ident.to_string();
67                    let param_type = convert_type(&pat_type.ty)?;
68                    params.push(Parameter {
69                        name: param_name,
70                        param_type,
71                    });
72                } else {
73                    return Err(Error::Validation(
74                        "Complex parameter patterns not supported".to_string(),
75                    ));
76                }
77            }
78            FnArg::Receiver(_) => {
79                return Err(Error::Validation("Self parameters not allowed".to_string()));
80            }
81        }
82    }
83
84    // Convert return type
85    let return_type = match &item_fn.sig.output {
86        ReturnType::Default => Type::Str, // Default to unit type represented as string
87        ReturnType::Type(_, ty) => convert_type(ty)?,
88    };
89
90    // Convert function body
91    let body = convert_block(&item_fn.block)?;
92
93    Ok(Function {
94        name,
95        params,
96        return_type,
97        body,
98    })
99}

101fn convert_type(ty: &SynType) -> Result<Type> {
102    match ty {
103        SynType::Path(type_path) => {
104            let path_str = type_path
105                .path
106                .segments
107                .iter()
108                .map(|seg| seg.ident.to_string())
109                .collect::<Vec<_>>()
110                .join("::");
111
112            match path_str.as_str() {
113                "bool" => Ok(Type::Bool),
114                "u32" | "i32" => Ok(Type::U32), // Treat i32 as u32 for now
115                "str" | "String" => Ok(Type::Str),
116                path if path.starts_with("Result") => {
117                    // Simple Result type parsing - assumes Result<T, E>
118                    Ok(Type::Result {
119                        ok_type: Box::new(Type::Str), // Simplified
120                        err_type: Box::new(Type::Str),
121                    })
122                }
123                path if path.starts_with("Option") => {
124                    // Simple Option type parsing - assumes Option<T>
125                    Ok(Type::Option {
126                        inner_type: Box::new(Type::Str), // Simplified
127                    })
128                }
129                _ => Err(Error::Validation(format!("Unsupported type: {}", path_str))),
130            }
131        }
132        SynType::Reference(type_ref) => {
133            // Handle &str and other reference types
134            if let SynType::Path(path) = &*type_ref.elem {
135                let path_str = path
136                    .path
137                    .segments
138                    .iter()
139                    .map(|seg| seg.ident.to_string())
140                    .collect::<Vec<_>>()
141                    .join("::");
142
143                match path_str.as_str() {
144                    "str" => Ok(Type::Str),
145                    _ => Err(Error::Validation(format!(
146                        "Unsupported reference type: &{}",
147                        path_str
148                    ))),
149                }
150            } else {
151                Err(Error::Validation(
152                    "Complex reference types not supported".to_string(),
153                ))
154            }
155        }
156        _ => Err(Error::Validation("Complex types not supported".to_string())),
157    }
158}

160fn convert_block(block: &Block) -> Result<Vec<Stmt>> {
161    let mut statements = Vec::new();
162
163    for stmt in &block.stmts {
164        statements.push(convert_stmt(stmt)?);
165    }
166
167    Ok(statements)
168}

170fn convert_stmt(stmt: &SynStmt) -> Result<Stmt> {
171    match stmt {
172        SynStmt::Local(local) => {
173            if let Pat::Ident(pat_ident) = &local.pat {
174                let name = pat_ident.ident.to_string();
175                if let Some(init) = &local.init {
176                    let value = convert_expr(&init.expr)?;
177                    Ok(Stmt::Let { name, value })
178                } else {
179                    Err(Error::Validation(
180                        "Let bindings must have initializers".to_string(),
181                    ))
182                }
183            } else {
184                Err(Error::Validation(
185                    "Complex patterns not supported".to_string(),
186                ))
187            }
188        }
189        SynStmt::Expr(expr, _) => {
190            // Check if this is an if expression used as a statement
191            if let SynExpr::If(expr_if) = expr {
192                let condition = convert_expr(&expr_if.cond)?;
193                let then_block = convert_block(&expr_if.then_branch)?;
194                let else_block = if let Some((_, else_expr)) = &expr_if.else_branch {
195                    match &**else_expr {
196                        SynExpr::Block(block) => Some(convert_block(&block.block)?),
197                        SynExpr::If(_) => {
198                            // Handle else-if by converting to nested if statement
199                            Some(vec![Stmt::Expr(convert_expr(else_expr)?)])
200                        }
201                        _ => None,
202                    }
203                } else {
204                    None
205                };
206                Ok(Stmt::If {
207                    condition,
208                    then_block,
209                    else_block,
210                })
211            } else {
212                Ok(Stmt::Expr(convert_expr(expr)?))
213            }
214        }
215        _ => Err(Error::Validation("Unsupported statement type".to_string())),
216    }
217}

219fn convert_expr(expr: &SynExpr) -> Result<Expr> {
220    match expr {
221        SynExpr::Lit(expr_lit) => {
222            let literal = convert_literal(&expr_lit.lit)?;
223            Ok(Expr::Literal(literal))
224        }
225        SynExpr::Path(expr_path) => {
226            let name = expr_path
227                .path
228                .segments
229                .iter()
230                .map(|seg| seg.ident.to_string())
231                .collect::<Vec<_>>()
232                .join("::");
233            Ok(Expr::Variable(name))
234        }
235        SynExpr::Call(expr_call) => {
236            if let SynExpr::Path(path) = &*expr_call.func {
237                let name = path
238                    .path
239                    .segments
240                    .iter()
241                    .map(|seg| seg.ident.to_string())
242                    .collect::<Vec<_>>()
243                    .join("::");
244
245                let mut args = Vec::new();
246                for arg in &expr_call.args {
247                    args.push(convert_expr(arg)?);
248                }
249
250                Ok(Expr::FunctionCall { name, args })
251            } else {
252                Err(Error::Validation(
253                    "Complex function calls not supported".to_string(),
254                ))
255            }
256        }
257        SynExpr::Binary(expr_binary) => {
258            let left = Box::new(convert_expr(&expr_binary.left)?);
259            let right = Box::new(convert_expr(&expr_binary.right)?);
260            let op = convert_binary_op(&expr_binary.op)?;
261            Ok(Expr::Binary { op, left, right })
262        }
263        SynExpr::Unary(expr_unary) => {
264            let operand = Box::new(convert_expr(&expr_unary.expr)?);
265            let op = convert_unary_op(&expr_unary.op)?;
266            Ok(Expr::Unary { op, operand })
267        }
268        SynExpr::MethodCall(method_call) => {
269            let receiver = Box::new(convert_expr(&method_call.receiver)?);
270            let method = method_call.method.to_string();
271            let mut args = Vec::new();
272            for arg in &method_call.args {
273                args.push(convert_expr(arg)?);
274            }
275            Ok(Expr::MethodCall {
276                receiver,
277                method,
278                args,
279            })
280        }
281        SynExpr::Return(ret_expr) => {
282            if let Some(expr) = &ret_expr.expr {
283                convert_expr(expr)
284            } else {
285                Ok(Expr::Literal(crate::ast::restricted::Literal::Str(
286                    "()".to_string(),
287                )))
288            }
289        }
290        SynExpr::Paren(expr_paren) => {
291            // Handle parenthesized expressions by unwrapping them
292            convert_expr(&expr_paren.expr)
293        }
294        SynExpr::If(_) => {
295            // For now, reject if expressions in expression position
296            // They should be used as statements instead
297            Err(Error::Validation(
298                "If expressions not supported in expression position".to_string(),
299            ))
300        }
301        _ => Err(Error::Validation("Unsupported expression type".to_string())),
302    }
303}

305fn convert_literal(lit: &Lit) -> Result<Literal> {
306    match lit {
307        Lit::Bool(lit_bool) => Ok(Literal::Bool(lit_bool.value)),
308        Lit::Int(lit_int) => {
309            let value: u32 = lit_int
310                .base10_parse()
311                .map_err(|_| Error::Validation("Invalid integer literal".to_string()))?;
312            Ok(Literal::U32(value))
313        }
314        Lit::Str(lit_str) => Ok(Literal::Str(lit_str.value())),
315        _ => Err(Error::Validation("Unsupported literal type".to_string())),
316    }
317}

319fn convert_binary_op(op: &BinOp) -> Result<BinaryOp> {
320    match op {
321        BinOp::Add(_) => Ok(BinaryOp::Add),
322        BinOp::Sub(_) => Ok(BinaryOp::Sub),
323        BinOp::Mul(_) => Ok(BinaryOp::Mul),
324        BinOp::Div(_) => Ok(BinaryOp::Div),
325        BinOp::Eq(_) => Ok(BinaryOp::Eq),
326        BinOp::Ne(_) => Ok(BinaryOp::Ne),
327        BinOp::Lt(_) => Ok(BinaryOp::Lt),
328        BinOp::Le(_) => Ok(BinaryOp::Le),
329        BinOp::Gt(_) => Ok(BinaryOp::Gt),
330        BinOp::Ge(_) => Ok(BinaryOp::Ge),
331        BinOp::And(_) => Ok(BinaryOp::And),
332        BinOp::Or(_) => Ok(BinaryOp::Or),
333        _ => Err(Error::Validation("Unsupported binary operator".to_string())),
334    }
335}

337fn convert_unary_op(op: &UnOp) -> Result<UnaryOp> {
338    match op {
339        UnOp::Not(_) => Ok(UnaryOp::Not),
340        UnOp::Neg(_) => Ok(UnaryOp::Neg),
341        _ => Err(Error::Validation("Unsupported unary operator".to_string())),
342    }
