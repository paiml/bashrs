10/// Parse Rust source code into a RestrictedAst
11pub fn parse(input: &str) -> Result<RestrictedAst> {
12    let file: File = syn::parse_str(input)?;
13
14    let mut functions = Vec::new();
15    let mut entry_point = None;
16
17    for item in file.items {
18        match item {
19            Item::Fn(item_fn) => {
20                // Check if this is the main function marked with #[rash::main]
21                let is_main = item_fn.attrs.iter().any(|attr| {
22                    // Check if the attribute path matches "rash::main"
23                    let path = attr.path();
24                    path.segments.len() == 2
25                        && path.segments[0].ident == "rash"
26                        && path.segments[1].ident == "main"
27                }) || item_fn.sig.ident == "main";
28
29                let function = convert_function(item_fn)?;
30
31                if is_main {
32                    if entry_point.is_some() {
33                        return Err(Error::Validation(
34                            "Multiple #[rash::main] functions found".to_string(),
35                        ));
36                    }
37                    entry_point = Some(function.name.clone());
38                }
39
40                functions.push(function);
41            }
42            _ => {
43                return Err(Error::Validation(
44                    "Only functions are allowed in Rash code".to_string(),
45                ));
46            }
47        }
48    }
49
50    let entry_point = entry_point
51        .ok_or_else(|| Error::Validation("No #[rash::main] function found".to_string()))?;
52
53    Ok(RestrictedAst {
54        functions,
55        entry_point,
56    })
57}

59fn convert_function(item_fn: ItemFn) -> Result<Function> {
60    let name = item_fn.sig.ident.to_string();
61
62    // Convert parameters
63    let mut params = Vec::new();
64    for input in item_fn.sig.inputs {
65        match input {
66            FnArg::Typed(pat_type) => {
67                if let Pat::Ident(pat_ident) = &*pat_type.pat {
68                    let param_name = pat_ident.ident.to_string();
69                    let param_type = convert_type(&pat_type.ty)?;
70                    params.push(Parameter {
71                        name: param_name,
72                        param_type,
73                    });
74                } else {
75                    return Err(Error::Validation(
76                        "Complex parameter patterns not supported".to_string(),
77                    ));
78                }
79            }
80            FnArg::Receiver(_) => {
81                return Err(Error::Validation("Self parameters not allowed".to_string()));
82            }
83        }
84    }
85
86    // Convert return type
87    let return_type = match &item_fn.sig.output {
88        ReturnType::Default => Type::Str, // Default to unit type represented as string
89        ReturnType::Type(_, ty) => convert_type(ty)?,
90    };
91
92    // Convert function body
93    let body = convert_block(&item_fn.block)?;
94
95    Ok(Function {
96        name,
97        params,
98        return_type,
99        body,
100    })
101}

103fn convert_type(ty: &SynType) -> Result<Type> {
104    match ty {
105        SynType::Path(type_path) => {
106            let path_str = type_path
107                .path
108                .segments
109                .iter()
110                .map(|seg| seg.ident.to_string())
111                .collect::<Vec<_>>()
112                .join("::");
113
114            match path_str.as_str() {
115                "bool" => Ok(Type::Bool),
116                "u32" | "i32" => Ok(Type::U32), // Treat i32 as u32 for now
117                "str" | "String" => Ok(Type::Str),
118                path if path.starts_with("Result") => {
119                    // Simple Result type parsing - assumes Result<T, E>
120                    Ok(Type::Result {
121                        ok_type: Box::new(Type::Str), // Simplified
122                        err_type: Box::new(Type::Str),
123                    })
124                }
125                path if path.starts_with("Option") => {
126                    // Simple Option type parsing - assumes Option<T>
127                    Ok(Type::Option {
128                        inner_type: Box::new(Type::Str), // Simplified
129                    })
130                }
131                _ => Err(Error::Validation(format!("Unsupported type: {path_str}"))),
132            }
133        }
134        SynType::Reference(type_ref) => {
135            // Handle &str and other reference types
136            if let SynType::Path(path) = &*type_ref.elem {
137                let path_str = path
138                    .path
139                    .segments
140                    .iter()
141                    .map(|seg| seg.ident.to_string())
142                    .collect::<Vec<_>>()
143                    .join("::");
144
145                match path_str.as_str() {
146                    "str" => Ok(Type::Str),
147                    _ => Err(Error::Validation(format!(
148                        "Unsupported reference type: &{path_str}"
149                    ))),
150                }
151            } else {
152                Err(Error::Validation(
153                    "Complex reference types not supported".to_string(),
154                ))
155            }
156        }
157        _ => Err(Error::Validation("Complex types not supported".to_string())),
158    }
159}

161fn convert_block(block: &Block) -> Result<Vec<Stmt>> {
162    let mut statements = Vec::new();
163
164    for stmt in &block.stmts {
165        statements.push(convert_stmt(stmt)?);
166    }
167
168    Ok(statements)
169}

171fn convert_stmt(stmt: &SynStmt) -> Result<Stmt> {
172    match stmt {
173        SynStmt::Local(local) => {
174            if let Pat::Ident(pat_ident) = &local.pat {
175                let name = pat_ident.ident.to_string();
176                if let Some(init) = &local.init {
177                    let value = convert_expr(&init.expr)?;
178                    Ok(Stmt::Let { name, value })
179                } else {
180                    Err(Error::Validation(
181                        "Let bindings must have initializers".to_string(),
182                    ))
183                }
184            } else {
185                Err(Error::Validation(
186                    "Complex patterns not supported".to_string(),
187                ))
188            }
189        }
190        SynStmt::Expr(expr, _) => {
191            // Check if this is an if expression used as a statement
192            if let SynExpr::If(expr_if) = expr {
193                let condition = convert_expr(&expr_if.cond)?;
194                let then_block = convert_block(&expr_if.then_branch)?;
195                let else_block = if let Some((_, else_expr)) = &expr_if.else_branch {
196                    match &**else_expr {
197                        SynExpr::Block(block) => Some(convert_block(&block.block)?),
198                        SynExpr::If(_) => {
199                            // Handle else-if by converting to nested if statement
200                            Some(vec![Stmt::Expr(convert_expr(else_expr)?)])
201                        }
202                        _ => None,
203                    }
204                } else {
205                    None
206                };
207                Ok(Stmt::If {
208                    condition,
209                    then_block,
210                    else_block,
211                })
212            } else {
213                Ok(Stmt::Expr(convert_expr(expr)?))
214            }
215        }
216        _ => Err(Error::Validation("Unsupported statement type".to_string())),
217    }
218}

220fn convert_expr(expr: &SynExpr) -> Result<Expr> {
221    match expr {
222        SynExpr::Lit(expr_lit) => {
223            let literal = convert_literal(&expr_lit.lit)?;
224            Ok(Expr::Literal(literal))
225        }
226        SynExpr::Path(expr_path) => {
227            let name = expr_path
228                .path
229                .segments
230                .iter()
231                .map(|seg| seg.ident.to_string())
232                .collect::<Vec<_>>()
233                .join("::");
234            Ok(Expr::Variable(name))
235        }
236        SynExpr::Call(expr_call) => {
237            if let SynExpr::Path(path) = &*expr_call.func {
238                let name = path
239                    .path
240                    .segments
241                    .iter()
242                    .map(|seg| seg.ident.to_string())
243                    .collect::<Vec<_>>()
244                    .join("::");
245
246                let mut args = Vec::new();
247                for arg in &expr_call.args {
248                    args.push(convert_expr(arg)?);
249                }
250
251                Ok(Expr::FunctionCall { name, args })
252            } else {
253                Err(Error::Validation(
254                    "Complex function calls not supported".to_string(),
255                ))
256            }
257        }
258        SynExpr::Binary(expr_binary) => {
259            let left = Box::new(convert_expr(&expr_binary.left)?);
260            let right = Box::new(convert_expr(&expr_binary.right)?);
261            let op = convert_binary_op(&expr_binary.op)?;
262            Ok(Expr::Binary { op, left, right })
263        }
264        SynExpr::Unary(expr_unary) => {
265            let operand = Box::new(convert_expr(&expr_unary.expr)?);
266            let op = convert_unary_op(&expr_unary.op)?;
267            Ok(Expr::Unary { op, operand })
268        }
269        SynExpr::MethodCall(method_call) => {
270            let receiver = Box::new(convert_expr(&method_call.receiver)?);
271            let method = method_call.method.to_string();
272            let mut args = Vec::new();
273            for arg in &method_call.args {
274                args.push(convert_expr(arg)?);
275            }
276            Ok(Expr::MethodCall {
277                receiver,
278                method,
279                args,
280            })
281        }
282        SynExpr::Return(ret_expr) => {
283            if let Some(expr) = &ret_expr.expr {
284                convert_expr(expr)
285            } else {
286                Ok(Expr::Literal(crate::ast::restricted::Literal::Str(
287                    "()".to_string(),
288                )))
289            }
290        }
291        SynExpr::Paren(expr_paren) => {
292            // Handle parenthesized expressions by unwrapping them
293            convert_expr(&expr_paren.expr)
294        }
295        SynExpr::If(_) => {
296            // For now, reject if expressions in expression position
297            // They should be used as statements instead
298            Err(Error::Validation(
299                "If expressions not supported in expression position".to_string(),
300            ))
301        }
302        _ => Err(Error::Validation("Unsupported expression type".to_string())),
303    }
304}

306fn convert_literal(lit: &Lit) -> Result<Literal> {
307    match lit {
308        Lit::Bool(lit_bool) => Ok(Literal::Bool(lit_bool.value)),
309        Lit::Int(lit_int) => {
310            let value: u32 = lit_int
311                .base10_parse()
312                .map_err(|_| Error::Validation("Invalid integer literal".to_string()))?;
313            Ok(Literal::U32(value))
314        }
315        Lit::Str(lit_str) => Ok(Literal::Str(lit_str.value())),
316        _ => Err(Error::Validation("Unsupported literal type".to_string())),
317    }
318}

320fn convert_binary_op(op: &BinOp) -> Result<BinaryOp> {
321    match op {
322        BinOp::Add(_) => Ok(BinaryOp::Add),
323        BinOp::Sub(_) => Ok(BinaryOp::Sub),
324        BinOp::Mul(_) => Ok(BinaryOp::Mul),
325        BinOp::Div(_) => Ok(BinaryOp::Div),
326        BinOp::Eq(_) => Ok(BinaryOp::Eq),
327        BinOp::Ne(_) => Ok(BinaryOp::Ne),
328        BinOp::Lt(_) => Ok(BinaryOp::Lt),
329        BinOp::Le(_) => Ok(BinaryOp::Le),
330        BinOp::Gt(_) => Ok(BinaryOp::Gt),
331        BinOp::Ge(_) => Ok(BinaryOp::Ge),
332        BinOp::And(_) => Ok(BinaryOp::And),
333        BinOp::Or(_) => Ok(BinaryOp::Or),
334        _ => Err(Error::Validation("Unsupported binary operator".to_string())),
335    }
336}
337
338fn convert_unary_op(op: &UnOp) -> Result<UnaryOp> {
339    match op {
340        UnOp::Not(_) => Ok(UnaryOp::Not),
341        UnOp::Neg(_) => Ok(UnaryOp::Neg),
342        _ => Err(Error::Validation("Unsupported unary operator".to_string())),
343    }
