6/// Parse Rust source code into a RestrictedAst
7pub fn parse(input: &str) -> Result<RestrictedAst> {
8    let file: File = syn::parse_str(input)?;
9    
10    let mut functions = Vec::new();
11    let mut entry_point = None;
12    
13    for item in file.items {
14        match item {
15            Item::Fn(item_fn) => {
16                // Check if this is the main function marked with #[rash::main]
17                let is_main = item_fn.attrs.iter().any(|attr| {
18                    // For now, just check if function name is "main" 
19                    item_fn.sig.ident == "main"
20                }) || item_fn.sig.ident == "main";
21                
22                let function = convert_function(item_fn)?;
23                
24                if is_main {
25                    if entry_point.is_some() {
26                        return Err(Error::Validation("Multiple #[rash::main] functions found".to_string()));
27                    }
28                    entry_point = Some(function.name.clone());
29                }
30                
31                functions.push(function);
32            }
33            _ => {
34                return Err(Error::Validation("Only functions are allowed in Rash code".to_string()));
35            }
36        }
37    }
38    
39    let entry_point = entry_point.ok_or_else(|| {
40        Error::Validation("No #[rash::main] function found".to_string())
41    })?;
42    
43    Ok(RestrictedAst {
44        functions,
45        entry_point,
46    })
47}

49fn convert_function(item_fn: ItemFn) -> Result<Function> {
50    let name = item_fn.sig.ident.to_string();
51    
52    // Convert parameters
53    let mut params = Vec::new();
54    for input in item_fn.sig.inputs {
55        match input {
56            FnArg::Typed(pat_type) => {
57                if let Pat::Ident(pat_ident) = &*pat_type.pat {
58                    let param_name = pat_ident.ident.to_string();
59                    let param_type = convert_type(&*pat_type.ty)?;
60                    params.push(Parameter {
61                        name: param_name,
62                        param_type,
63                    });
64                } else {
65                    return Err(Error::Validation("Complex parameter patterns not supported".to_string()));
66                }
67            }
68            FnArg::Receiver(_) => {
69                return Err(Error::Validation("Self parameters not allowed".to_string()));
70            }
71        }
72    }
73    
74    // Convert return type
75    let return_type = match &item_fn.sig.output {
76        ReturnType::Default => Type::Str, // Default to unit type represented as string
77        ReturnType::Type(_, ty) => convert_type(ty)?,
78    };
79    
80    // Convert function body
81    let body = convert_block(&item_fn.block)?;
82    
83    Ok(Function {
84        name,
85        params,
86        return_type,
87        body,
88    })
89}

91fn convert_type(ty: &SynType) -> Result<Type> {
92    match ty {
93        SynType::Path(type_path) => {
94            let path_str = type_path.path.segments.iter()
95                .map(|seg| seg.ident.to_string())
96                .collect::<Vec<_>>()
97                .join("::");
98            
99            match path_str.as_str() {
100                "bool" => Ok(Type::Bool),
101                "u32" => Ok(Type::U32),
102                "str" | "&str" | "String" => Ok(Type::Str),
103                path if path.starts_with("Result") => {
104                    // Simple Result type parsing - assumes Result<T, E>
105                    Ok(Type::Result {
106                        ok_type: Box::new(Type::Str), // Simplified
107                        err_type: Box::new(Type::Str),
108                    })
109                }
110                path if path.starts_with("Option") => {
111                    // Simple Option type parsing - assumes Option<T>
112                    Ok(Type::Option {
113                        inner_type: Box::new(Type::Str), // Simplified
114                    })
115                }
116                _ => Err(Error::Validation(format!("Unsupported type: {}", path_str))),
117            }
118        }
119        _ => Err(Error::Validation("Complex types not supported".to_string())),
120    }
121}

123fn convert_block(block: &Block) -> Result<Vec<Stmt>> {
124    let mut statements = Vec::new();
125    
126    for stmt in &block.stmts {
127        statements.push(convert_stmt(stmt)?);
128    }
129    
130    Ok(statements)
131}

133fn convert_stmt(stmt: &SynStmt) -> Result<Stmt> {
134    match stmt {
135        SynStmt::Local(local) => {
136            if let Pat::Ident(pat_ident) = &local.pat {
137                let name = pat_ident.ident.to_string();
138                if let Some(init) = &local.init {
139                    let value = convert_expr(&init.expr)?;
140                    Ok(Stmt::Let { name, value })
141                } else {
142                    Err(Error::Validation("Let bindings must have initializers".to_string()))
143                }
144            } else {
145                Err(Error::Validation("Complex patterns not supported".to_string()))
146            }
147        }
148        SynStmt::Expr(expr, _) => {
149            Ok(Stmt::Expr(convert_expr(expr)?))
150        }
151        _ => Err(Error::Validation("Unsupported statement type".to_string())),
152    }
153}

155fn convert_expr(expr: &SynExpr) -> Result<Expr> {
156    match expr {
157        SynExpr::Lit(expr_lit) => {
158            let literal = convert_literal(&expr_lit.lit)?;
159            Ok(Expr::Literal(literal))
160        }
161        SynExpr::Path(expr_path) => {
162            let name = expr_path.path.segments.iter()
163                .map(|seg| seg.ident.to_string())
164                .collect::<Vec<_>>()
165                .join("::");
166            Ok(Expr::Variable(name))
167        }
168        SynExpr::Call(expr_call) => {
169            if let SynExpr::Path(path) = &*expr_call.func {
170                let name = path.path.segments.iter()
171                    .map(|seg| seg.ident.to_string())
172                    .collect::<Vec<_>>()
173                    .join("::");
174                
175                let mut args = Vec::new();
176                for arg in &expr_call.args {
177                    args.push(convert_expr(arg)?);
178                }
179                
180                Ok(Expr::FunctionCall { name, args })
181            } else {
182                Err(Error::Validation("Complex function calls not supported".to_string()))
183            }
184        }
185        SynExpr::Binary(expr_binary) => {
186            let left = Box::new(convert_expr(&expr_binary.left)?);
187            let right = Box::new(convert_expr(&expr_binary.right)?);
188            let op = convert_binary_op(&expr_binary.op)?;
189            Ok(Expr::Binary { op, left, right })
190        }
191        SynExpr::Unary(expr_unary) => {
192            let operand = Box::new(convert_expr(&expr_unary.expr)?);
193            let op = convert_unary_op(&expr_unary.op)?;
194            Ok(Expr::Unary { op, operand })
195        }
196        SynExpr::MethodCall(method_call) => {
197            let receiver = Box::new(convert_expr(&method_call.receiver)?);
198            let method = method_call.method.to_string();
199            let mut args = Vec::new();
200            for arg in &method_call.args {
201                args.push(convert_expr(arg)?);
202            }
203            Ok(Expr::MethodCall { receiver, method, args })
204        }
205        SynExpr::Return(ret_expr) => {
206            if let Some(expr) = &ret_expr.expr {
207                convert_expr(expr)
208            } else {
209                Ok(Expr::Literal(crate::ast::restricted::Literal::Str("()".to_string())))
210            }
211        }
212        _ => Err(Error::Validation("Unsupported expression type".to_string())),
213    }
214}

216fn convert_literal(lit: &Lit) -> Result<Literal> {
217    match lit {
218        Lit::Bool(lit_bool) => Ok(Literal::Bool(lit_bool.value)),
219        Lit::Int(lit_int) => {
220            let value: u32 = lit_int.base10_parse()
221                .map_err(|_| Error::Validation("Invalid integer literal".to_string()))?;
222            Ok(Literal::U32(value))
223        }
224        Lit::Str(lit_str) => Ok(Literal::Str(lit_str.value())),
225        _ => Err(Error::Validation("Unsupported literal type".to_string())),
226    }
227}

229fn convert_binary_op(op: &BinOp) -> Result<BinaryOp> {
230    match op {
231        BinOp::Add(_) => Ok(BinaryOp::Add),
232        BinOp::Sub(_) => Ok(BinaryOp::Sub),
233        BinOp::Mul(_) => Ok(BinaryOp::Mul),
234        BinOp::Div(_) => Ok(BinaryOp::Div),
235        BinOp::Eq(_) => Ok(BinaryOp::Eq),
236        BinOp::Ne(_) => Ok(BinaryOp::Ne),
237        BinOp::Lt(_) => Ok(BinaryOp::Lt),
238        BinOp::Le(_) => Ok(BinaryOp::Le),
239        BinOp::Gt(_) => Ok(BinaryOp::Gt),
240        BinOp::Ge(_) => Ok(BinaryOp::Ge),
241        BinOp::And(_) => Ok(BinaryOp::And),
242        BinOp::Or(_) => Ok(BinaryOp::Or),
243        _ => Err(Error::Validation("Unsupported binary operator".to_string())),
244    }
245}

247fn convert_unary_op(op: &UnOp) -> Result<UnaryOp> {
248    match op {
249        UnOp::Not(_) => Ok(UnaryOp::Not),
250        UnOp::Neg(_) => Ok(UnaryOp::Neg),
251        _ => Err(Error::Validation("Unsupported unary operator".to_string())),
252    }
