10pub struct Cli {

13
14    /// Verification stringency level

18    /// Target shell dialect

21
22    /// ShellCheck-compatible validation level

35#[derive(Subcommand)]
36pub enum Commands {

47        /// Emit verification proof
48        #[arg(long)]
49        emit_proof: bool,

51        /// Disable optimizations

55
56    /// Check Rust source for Rash compatibility
57    Check {
58        /// Input Rust file
59        #[arg(value_name = "FILE")]
60        input: PathBuf,
61    },
62
63    /// Initialize new Rash project
64    Init {
65        /// Project directory
66        #[arg(default_value = ".")]
67        path: PathBuf,
68
69        /// Project name
70        #[arg(long)]
71        name: Option<String>,
72    },
73
74    /// Verify shell script matches Rust source
75    Verify {
76        /// Rust source file
77        rust_source: PathBuf,
78
79        /// Shell script file
80        shell_script: PathBuf,
81    },
82
83    /// Generate formal verification inspection report
84    Inspect {
85        /// Input AST file (JSON) or inline AST specification
86        #[arg(value_name = "AST")]
87        input: String,
88
89        /// Output format
90        #[arg(long, default_value = "markdown")]
91        format: InspectionFormat,
92
93        /// Output file (defaults to stdout)
94        #[arg(short, long)]
95        output: Option<PathBuf>,
96
97        /// Include detailed traces
98        #[arg(long)]
99        detailed: bool,
100    },
