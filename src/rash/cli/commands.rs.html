13pub fn execute_command(cli: Cli) -> Result<()> {
14    // Initialize logging
15    let subscriber = tracing_subscriber::fmt()
16        .with_max_level(if cli.verbose {
17            tracing::Level::DEBUG
18        } else {
19            tracing::Level::INFO
20        })
21        .finish();
22    tracing::subscriber::set_global_default(subscriber)
23        .map_err(|e| Error::Internal(format!("Failed to initialize logging: {}", e)))?;
24
25    match cli.command {
26        Commands::Build {
27            input,
28            output,
29            emit_proof,
30            no_optimize,
31        } => {
32            info!("Building {} -> {}", input.display(), output.display());
33
34            let config = Config {
35                target: cli.target,
36                verify: cli.verify,
37                emit_proof,
38                optimize: !no_optimize,
39                validation_level: Some(cli.validation),
40                strict_mode: cli.strict,
41            };
42
43            build_command(&input, &output, config)
44        }
45
46        Commands::Check { input } => {
47            info!("Checking {}", input.display());
48            check_command(&input)
49        }
50
51        Commands::Init { path, name } => {
52            info!("Initializing project in {}", path.display());
53            init_command(&path, name.as_deref())
54        }
55
56        Commands::Verify {
57            rust_source,
58            shell_script,
59        } => {
60            info!(
61                "Verifying {} against {}",
62                shell_script.display(),
63                rust_source.display()
64            );
65            verify_command(&rust_source, &shell_script, cli.target, cli.verify)
66        }
67
68        Commands::Inspect {
69            input,
70            format,
71            output,
72            detailed,
73        } => {
74            info!("Generating inspection report for: {}", input);
75            inspect_command(&input, format, output.as_deref(), detailed)
76        }
77    }
78}

80fn build_command(input: &Path, output: &Path, config: Config) -> Result<()> {
81    // Read input file
82    let source = fs::read_to_string(input).map_err(Error::Io)?;
83
84    // Transpile
85    let shell_code = transpile(&source, config.clone())?;
86
87    // Write output
88    fs::write(output, shell_code).map_err(Error::Io)?;
89
90    info!("Successfully transpiled to {}", output.display());
91
92    // Generate proof if requested
93    if config.emit_proof {
94        let proof_path = output.with_extension("proof");
95        generate_proof(&source, &proof_path, &config)?;
96        info!("Proof generated at {}", proof_path.display());
97    }
98
99    Ok(())
100}

113fn init_command(path: &Path, name: Option<&str>) -> Result<()> {

120        path.file_name()
121            .and_then(|n| n.to_str())
122            .unwrap_or("my-installer"),
123    );
124
125    // Create Cargo.toml
126    let cargo_toml = format!(
127        r#"[package]
128name = "{}"
129version = "0.1.0"
130edition = "2021"
131
132[dependencies]
133# No dependencies needed - Rash transpiles to pure shell
134
135[[bin]]
136name = "install"
137path = "src/main.rs"
138"#,
139        project_name
140    );
141
142    fs::write(path.join("Cargo.toml"), cargo_toml).map_err(Error::Io)?;
143
144    // Create src directory
145    let src_dir = path.join("src");
146    fs::create_dir_all(&src_dir).map_err(Error::Io)?;
147
148    // Create main.rs with example installer
149    let main_rs = r#"/// Example installer script for Rash
150/// This will be transpiled to POSIX-compliant shell script
151use std::env;
152use std::fs;
153use std::path::Path;
154use std::process::{Command, exit};
155
156const VERSION: &str = "0.1.0";
157const BINARY_NAME: &str = "myapp";

162    
163    // Parse arguments

167        return;
168    }
169

208fn detect_platform() -> &'static str {
209    // Simplified platform detection
210    if cfg!(target_os = "linux") {
211        if cfg!(target_arch = "x86_64") {
212            "linux-amd64"
213        } else {
214            "linux-arm64"
215        }
216    } else if cfg!(target_os = "macos") {
217        "darwin-amd64"
218    } else {
219        panic!("Unsupported platform");
220    }
221}"#;
222
223    fs::write(src_dir.join("main.rs"), main_rs).map_err(Error::Io)?;
224
225    // Create .rash.toml
226    let rash_toml = r##"# Rash configuration file
227[transpiler]
228target = "posix"          # Target shell dialect
229strict_mode = true        # Fail on warnings
230preserve_comments = false # Strip comments for smaller output
231
232[validation]
233level = "strict"          # ShellCheck compliance level
234rules = ["all"]           # Can disable specific: ["-SC2034"]
235external_check = false    # Run actual shellcheck binary
236
237[output]
238shebang = "#!/bin/sh"     # POSIX shebang
239set_flags = "euf"         # set -euf (no pipefail in POSIX)
240optimize_size = true      # Minimize output script size
241
242[style]
243indent = "    "           # 4 spaces
244max_line_length = 100     # Wrap long commands
245"##;
246
247    fs::write(path.join(".rash.toml"), rash_toml).map_err(Error::Io)?;
248
249    info!("✓ Initialized Rash project '{}'", project_name);
250    info!("  Run 'cd {}' to enter the project", path.display());
251    info!("  Run 'rash build src/main.rs' to build");
252
253    Ok(())
254}

256fn verify_command(
257    rust_source: &Path,
258    shell_script: &Path,
259    target: crate::models::ShellDialect,
260    verify_level: crate::models::VerificationLevel,
261) -> Result<()> {
262    // Read both files
263    let rust_code = fs::read_to_string(rust_source).map_err(Error::Io)?;
264    let shell_code = fs::read_to_string(shell_script).map_err(Error::Io)?;
265
266    // Transpile Rust to shell
267    let config = Config {
268        target,
269        verify: verify_level,
270        emit_proof: false,
271        optimize: true,
272        strict_mode: true,
273        validation_level: Some(crate::validation::ValidationLevel::Strict),
274    };
275
276    let generated_shell = transpile(&rust_code, config)?;
277
278    // Compare generated vs actual
279    if normalize_shell_script(&generated_shell) == normalize_shell_script(&shell_code) {
280        info!("✓ Shell script matches Rust source");
281        Ok(())
282    } else {
283        warn!("Shell script does not match Rust source");
284        // TODO: Show diff
285        Err(Error::Verification("Script mismatch".to_string()))
286    }
287}

289fn generate_proof(source: &str, proof_path: &Path, config: &Config) -> Result<()> {
290    // For now, just create a simple proof file
291    let proof = format!(
292        r#"{{
293    "version": "1.0",
294    "source_hash": "{}",
295    "verification_level": "{:?}",
296    "target": "{:?}",
297    "timestamp": "{}",
298    "properties": ["no-injection", "deterministic"]
299}}"#,
300        blake3::hash(source.as_bytes()),
301        config.verify,
302        config.target,
303        chrono::Utc::now().to_rfc3339()
304    );
305
306    fs::write(proof_path, proof).map_err(Error::Io)?;
307
308    Ok(())
309}

311fn normalize_shell_script(script: &str) -> String {
312    // Remove comments and normalize whitespace for comparison
313    script
314        .lines()
315        .filter(|line| !line.trim().starts_with('#'))
316        .map(|line| line.trim())
317        .filter(|line| !line.is_empty())
318        .collect::<Vec<_>>()
319        .join("\n")
320}

322fn inspect_command(
323    input: &str,
324    format: InspectionFormat,
325    output: Option<&Path>,
326    _detailed: bool,
327) -> Result<()> {
328    use crate::formal::{AbstractState, ProofInspector, TinyAst};
329
330    // Parse input - for now, we'll support JSON AST or a few predefined examples
331    let ast = if input.starts_with('{') {
332        // JSON input
333        serde_json::from_str::<TinyAst>(input)
334            .map_err(|e| Error::Internal(format!("Invalid AST JSON: {}", e)))?
335    } else {
336        // Predefined examples or simple DSL
337        match input {
338            "echo-example" => TinyAst::ExecuteCommand {
339                command_name: "echo".to_string(),
340                args: vec!["Hello, World!".to_string()],
341            },
342            "bootstrap-example" => TinyAst::Sequence {
343                commands: vec![
344                    TinyAst::SetEnvironmentVariable {
345                        name: "INSTALL_DIR".to_string(),
346                        value: "/opt/rash".to_string(),
347                    },
348                    TinyAst::ExecuteCommand {
349                        command_name: "mkdir".to_string(),
350                        args: vec!["-p".to_string(), "/opt/rash/bin".to_string()],
351                    },
352                    TinyAst::ChangeDirectory {
353                        path: "/opt/rash".to_string(),
354                    },
355                    TinyAst::ExecuteCommand {
356                        command_name: "echo".to_string(),
357                        args: vec!["Installation ready".to_string()],
358                    },
359                ],
360            },
361            _ => {
362                return Err(Error::Internal(format!("Unknown example: {}. Try 'echo-example' or 'bootstrap-example', or provide JSON AST", input)));
363            }
364        }
365    };
366
367    // Validate the AST
368    if !ast.is_valid() {
369        return Err(Error::Validation("Invalid AST".to_string()));
370    }
371
372    // Generate inspection report
373    let mut initial_state = AbstractState::new();
374    // Add common directories for testing
375    initial_state.filesystem.insert(
376        std::path::PathBuf::from("/opt"),
377        crate::formal::FileSystemEntry::Directory,
378    );
379
380    let report = ProofInspector::inspect(&ast, initial_state);
381
382    // Format output
383    let output_content = match format {
384        InspectionFormat::Markdown => ProofInspector::generate_report(&report),
385        InspectionFormat::Json => serde_json::to_string_pretty(&report)
386            .map_err(|e| Error::Internal(format!("JSON serialization failed: {}", e)))?,
387        InspectionFormat::Html => {
388            // Convert markdown to HTML (simplified)
389            let markdown = ProofInspector::generate_report(&report);
390            format!(
391                r#"<!DOCTYPE html>
392<html>
393<head>
394    <title>Formal Verification Report</title>
395    <style>
396        body {{ font-family: Arial, sans-serif; margin: 2em; }}
397        pre {{ background: #f5f5f5; padding: 1em; border-radius: 4px; }}
398        .success {{ color: green; }}
399        .failure {{ color: red; }}
400        .warning {{ color: orange; }}
401    </style>
402</head>
403<body>
404<pre>{}</pre>
405</body>
406</html>"#,
407                markdown
408                    .replace('&', "&amp;")
409                    .replace('<', "&lt;")
410                    .replace('>', "&gt;")
411            )
412        }
413    };
414
415    // Write output
416    match output {
417        Some(path) => {
418            fs::write(path, &output_content).map_err(Error::Io)?;
419            info!("Inspection report written to {}", path.display());
420        }
421        None => {
422            println!("{}", output_content);
423        }
424    }
425
426    Ok(())
