79    /// Run the playground main loop
80    pub fn run(&mut self) -> Result<()> {
81        #[cfg(feature = "playground")]
82        {
83            use ratatui::{backend::CrosstermBackend, Terminal};
84            use std::io;
85
86            // Setup terminal
87            ratatui::crossterm::terminal::enable_raw_mode()
88                .map_err(|e| Error::Internal(format!("Failed to enable raw mode: {e}")))?;
89
90            let mut stdout = io::stdout();
91            ratatui::crossterm::execute!(
92                stdout,
93                ratatui::crossterm::terminal::EnterAlternateScreen,
94                ratatui::crossterm::event::EnableMouseCapture
95            )
96            .map_err(|e| Error::Internal(format!("Failed to setup terminal: {e}")))?;
97
98            let backend = CrosstermBackend::new(stdout);
99            let mut terminal = Terminal::new(backend)
100                .map_err(|e| Error::Internal(format!("Failed to create terminal: {e}")))?;
101
102            // Run main loop
103            let result = self.main_loop(&mut terminal);
104
105            // Cleanup terminal
106            ratatui::crossterm::terminal::disable_raw_mode()
107                .map_err(|e| Error::Internal(format!("Failed to disable raw mode: {e}")))?;
108
109            ratatui::crossterm::execute!(
110                terminal.backend_mut(),
111                ratatui::crossterm::terminal::LeaveAlternateScreen,
112                ratatui::crossterm::event::DisableMouseCapture
113            )
114            .map_err(|e| Error::Internal(format!("Failed to cleanup terminal: {e}")))?;
115
116            terminal
117                .show_cursor()
118                .map_err(|e| Error::Internal(format!("Failed to show cursor: {e}")))?;
119
120            result
121        }
122
123        #[cfg(not(feature = "playground"))]
124        Err(Error::Internal(
125            "Playground feature not enabled".to_string(),
126        ))
127    }

129    #[cfg(feature = "playground")]
130    fn main_loop(
131        &mut self,
132        terminal: &mut ratatui::Terminal<ratatui::backend::CrosstermBackend<std::io::Stdout>>,
133    ) -> Result<()> {
134        use ratatui::crossterm::event::{self, Event};
135        use std::time::Duration;
136
137        loop {
138            // Render frame
139            let frame_start = Instant::now();
140            terminal
141                .draw(|f| self.render_frame(f))
142                .map_err(|e| Error::Internal(format!("Render failed: {e}")))?;
143
144            self.frame_count += 1;
145            let render_time = frame_start.elapsed();
146
147            // Performance monitoring - warn if frame took too long
148            if render_time > Duration::from_millis(16) {
149                tracing::warn!("Slow frame: {:?} (target: 16ms)", render_time);
150            }
151
152            // Handle input with timeout for smooth animation
153            if event::poll(Duration::from_millis(16))
154                .map_err(|e| Error::Internal(format!("Event poll failed: {e}")))?
155            {
156                match event::read()
157                    .map_err(|e| Error::Internal(format!("Event read failed: {e}")))?
158                {
159                    Event::Key(key_event) => {
160                        if self.handle_key_event(key_event)? {
161                            break; // Exit requested
162                        }
163                    }
164                    Event::Resize(width, height) => {
165                        self.handle_resize(width, height)?;
166                    }
167                    _ => {}
168                }
169            }
170
171            // Process any pending computations
172            self.computation_dag.process_pending()?;
173        }
174
175        Ok(())
176    }

178    #[cfg(feature = "playground")]
179    fn render_frame(&mut self, f: &mut ratatui::Frame) {
180        use ratatui::{
181            layout::{Constraint, Direction, Layout},
182            style::{Color, Style},
183            widgets::{Block, Borders, Paragraph},
184        };
185
186        let chunks = Layout::default()
187            .direction(Direction::Vertical)
188            .constraints([
189                Constraint::Ratio(1, 2), // Source editor
190                Constraint::Ratio(1, 2), // Output panel
191            ])
192            .split(f.area());
193
194        // Render source editor
195        let source_content = self.document_store.get_content();
196        let source_widget = Paragraph::new(source_content)
197            .block(Block::default().title("Rust Source").borders(Borders::ALL))
198            .style(Style::default().fg(Color::White));
199
200        f.render_widget(source_widget, chunks[0]);
201
202        // Render transpiled output
203        let output_content = self.get_transpiled_output();
204        let output_widget = Paragraph::new(output_content)
205            .block(Block::default().title("Shell Output").borders(Borders::ALL))
206            .style(Style::default().fg(Color::Green));
207
208        f.render_widget(output_widget, chunks[1]);
209
210        // Render status line
211        self.render_status_line(f, f.area());
212    }

214    #[cfg(feature = "playground")]
215    fn render_status_line(&self, f: &mut ratatui::Frame, area: ratatui::layout::Rect) {
216        use ratatui::{
217            layout::Rect,
218            style::{Color, Style},
219            text::Line,
220            widgets::Paragraph,
221        };
222
223        let status_area = Rect {
224            x: area.x,
225            y: area.height.saturating_sub(1),
226            width: area.width,
227            height: 1,
228        };
229
230        let mode_str = match &self.editor_mode {
231            EditorMode::Normal { .. } => "NORMAL",
232            EditorMode::Insert { .. } => "INSERT",
233            EditorMode::Visual { .. } => "VISUAL",
234            EditorMode::Command { .. } => "COMMAND",
235        };
236
237        let fps = if self.frame_count > 0 {
238            self.frame_count as f64 / self.session_start.elapsed().as_secs_f64()
239        } else {
240            0.0
241        };
242
243        let status_text = format!(
244            " {} | FPS: {:.1} | Frame: {} ",
245            mode_str, fps, self.frame_count
246        );
247
248        let status_widget = Paragraph::new(Line::from(status_text))
249            .style(Style::default().bg(Color::Blue).fg(Color::White));
250
251        f.render_widget(status_widget, status_area);
252    }

254    fn handle_key_event(&mut self, key_event: ratatui::crossterm::event::KeyEvent) -> Result<bool> {
255        use ratatui::crossterm::event::KeyCode;
256
257        // Global shortcuts
258        match key_event.code {
259            KeyCode::Esc if matches!(self.editor_mode, EditorMode::Insert { .. }) => {
260                if self.vi_mode {
261                    self.editor_mode = EditorMode::Normal {
262                        pending_operator: None,
263                        count: None,
264                    };
265                }
266                return Ok(false);
267            }
268            KeyCode::Char('q') if matches!(self.editor_mode, EditorMode::Normal { .. }) => {
269                return Ok(true); // Exit
270            }
271            _ => {}
272        }
273
274        // Process through keymap engine
275        if let Some(action) = self.keymap_engine.process_key(key_event, &self.editor_mode) {
276            self.execute_action(action)?;
277        }
278
279        Ok(false)
280    }
