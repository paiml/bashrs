120    /// Highlight a line using SIMD-accelerated token classification
121    pub fn highlight_line(&mut self, line: &str, line_id: LineId) -> Vec<StyledToken> {
122        // Check cache first
123        if let Some(cached) = self.token_cache.get(&line_id) {
124            return cached.clone();
125        }
126        
127        // SIMD-accelerated highlighting
128        let tokens = self.highlight_line_simd(line);
129        
130        // Cache the result
131        self.token_cache.put(line_id, tokens.clone());
132        
133        tokens
134    }

136    #[cfg(feature = "playground")]
137    fn highlight_line_simd(&self, line: &str) -> Vec<StyledToken> {
138        use simdutf8::basic::from_utf8;
139        
140        // Validate UTF-8 using SIMD
141        if from_utf8(line.as_bytes()).is_err() {
142            return vec![StyledToken {
143                text: line.to_string(),
144                style: self.theme.default,
145            }];
146        }
147        
148        // Tokenize and classify
149        let mut tokens = Vec::new();
150        let mut chars = line.chars().peekable();
151        let mut current_token = String::new();
152        
153        while let Some(ch) = chars.next() {
154            match ch {
155                // String literals
156                '"' => {
157                    current_token.push(ch);
158                    while let Some(ch) = chars.next() {
159                        current_token.push(ch);
160                        if ch == '"' && !current_token.ends_with("\\\"") {
161                            break;
162                        }
163                    }
164                    tokens.push(StyledToken {
165                        text: std::mem::take(&mut current_token),
166                        style: self.theme.string,
167                    });
168                }
169                
170                // Comments
171                '/' if chars.peek() == Some(&'/') => {
172                    current_token.push(ch);
173                    for ch in chars.by_ref() {
174                        current_token.push(ch);
175                    }
176                    tokens.push(StyledToken {
177                        text: std::mem::take(&mut current_token),
178                        style: self.theme.comment,
179                    });
180                }
181                
182                // Numbers
183                '0'..='9' => {
184                    current_token.push(ch);
185                    while let Some(&next_ch) = chars.peek() {
186                        if next_ch.is_alphanumeric() || next_ch == '.' || next_ch == '_' {
187                            current_token.push(chars.next().unwrap());
188                        } else {
189                            break;
190                        }
191                    }
192                    tokens.push(StyledToken {
193                        text: std::mem::take(&mut current_token),
194                        style: self.theme.number,
195                    });
196                }
197                
198                // Identifiers and keywords
199                'a'..='z' | 'A'..='Z' | '_' => {
200                    current_token.push(ch);
201                    while let Some(&next_ch) = chars.peek() {
202                        if next_ch.is_alphanumeric() || next_ch == '_' {
203                            current_token.push(chars.next().unwrap());
204                        } else {
205                            break;
206                        }
207                    }
208                    
209                    let style = if self.is_keyword(&current_token) {
210                        self.theme.keyword
211                    } else if self.is_function_call(&current_token, &mut chars) {
212                        self.theme.function
213                    } else {
214                        self.theme.variable
215                    };
216                    
217                    tokens.push(StyledToken {
218                        text: std::mem::take(&mut current_token),
219                        style,
220                    });
221                }
222                
223                // Operators
224                '+' | '-' | '*' | '/' | '=' | '<' | '>' | '!' | '&' | '|' | '^' | '%' => {
225                    current_token.push(ch);
226                    // Check for multi-char operators
227                    if let Some(&next_ch) = chars.peek() {
228                        if matches!((ch, next_ch), 
229                            ('=', '=') | ('!', '=') | ('<', '=') | ('>', '=') |
230                            ('&', '&') | ('|', '|') | ('<', '<') | ('>', '>') |
231                            ('+', '=') | ('-', '=') | ('*', '=') | ('/', '='))
232                        {
233                            current_token.push(chars.next().unwrap());
234                        }
235                    }
236                    tokens.push(StyledToken {
237                        text: std::mem::take(&mut current_token),
238                        style: self.theme.operator,
239                    });
240                }
241                
242                // Whitespace and other characters
243                _ => {
244                    if !current_token.is_empty() {
245                        tokens.push(StyledToken {
246                            text: std::mem::take(&mut current_token),
247                            style: self.theme.default,
248                        });
249                    }
250                    tokens.push(StyledToken {
251                        text: ch.to_string(),
252                        style: self.theme.default,
253                    });
254                }
255            }
256        }
257        
258        // Handle any remaining token
259        if !current_token.is_empty() {
260            tokens.push(StyledToken {
261                text: current_token,
262                style: self.theme.default,
263            });
264        }
265        
266        tokens
267    }

277    fn is_keyword(&self, word: &str) -> bool {
278        matches!(word,
279            "fn" | "let" | "mut" | "if" | "else" | "while" | "for" | "loop" |
280            "match" | "return" | "break" | "continue" | "struct" | "enum" |
281            "impl" | "trait" | "pub" | "mod" | "use" | "self" | "super" |
282            "crate" | "const" | "static" | "type" | "where" | "async" |
283            "await" | "move" | "ref" | "in" | "as" | "true" | "false"
284        )
285    }
286    
287    fn is_function_call(&self, _word: &str, chars: &mut std::iter::Peekable<std::str::Chars>) -> bool {
288        // Check if followed by '('
289        let mut _whitespace_count = 0;
290        while let Some(&ch) = chars.peek() {
291            if ch == '(' {
292                return true;
293            } else if ch.is_whitespace() {
294                _whitespace_count += 1;
295                chars.next();
296            } else {
297                break;
298            }
299        }
300        false
301    }

319    /// Check if a byte slice contains only ASCII alphanumeric characters
320    #[cfg(target_arch = "x86_64")]
321    pub fn is_ascii_alphanumeric_simd(bytes: &[u8]) -> bool {
322        use std::arch::x86_64::*;
323        
324        unsafe {
325            let chunks = bytes.chunks_exact(16);
326            let remainder = chunks.remainder();
327            
328            for chunk in chunks {
329                let data = _mm_loadu_si128(chunk.as_ptr() as *const __m128i);
330                
331                // Check for ASCII range
332                let ascii_mask = _mm_cmplt_epi8(data, _mm_set1_epi8(-128i8));
333                if _mm_movemask_epi8(ascii_mask) != 0xFFFF {
334                    return false;
335                }
336                
337                // Check for alphanumeric
338                // Implement comparisons using available intrinsics
339                // For >= we use NOT(a < b), for <= we use NOT(a > b)
340                let zero = _mm_set1_epi8(b'0' as i8);
341                let nine = _mm_set1_epi8(b'9' as i8);
342                let is_digit = _mm_and_si128(
343                    _mm_xor_si128(_mm_cmplt_epi8(data, zero), _mm_set1_epi8(-1i8)),
344                    _mm_xor_si128(_mm_cmpgt_epi8(data, nine), _mm_set1_epi8(-1i8))
345                );
346                
347                let upper_a = _mm_set1_epi8(b'A' as i8);
348                let upper_z = _mm_set1_epi8(b'Z' as i8);
349                let is_upper = _mm_and_si128(
350                    _mm_xor_si128(_mm_cmplt_epi8(data, upper_a), _mm_set1_epi8(-1i8)),
351                    _mm_xor_si128(_mm_cmpgt_epi8(data, upper_z), _mm_set1_epi8(-1i8))
352                );
353                
354                let lower_a = _mm_set1_epi8(b'a' as i8);
355                let lower_z = _mm_set1_epi8(b'z' as i8);
356                let is_lower = _mm_and_si128(
357                    _mm_xor_si128(_mm_cmplt_epi8(data, lower_a), _mm_set1_epi8(-1i8)),
358                    _mm_xor_si128(_mm_cmpgt_epi8(data, lower_z), _mm_set1_epi8(-1i8))
359                );
360                
361                let is_underscore = _mm_cmpeq_epi8(data, _mm_set1_epi8(b'_' as i8));
362                
363                let is_valid = _mm_or_si128(
364                    _mm_or_si128(is_digit, is_upper),
365                    _mm_or_si128(is_lower, is_underscore)
366                );
367                
368                if _mm_movemask_epi8(is_valid) != 0xFFFF {
369                    return false;
370                }
371            }
372            
373            // Check remainder
374            remainder.iter().all(|&b| b.is_ascii_alphanumeric() || b == b'_')
375        }
376    }
