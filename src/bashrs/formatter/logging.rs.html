7/// Append-only log for transformation verification context propagation
8#[derive(Debug, Clone)]
9pub struct TransformLog {

11    pub entries: Vec<TransformEntry>,
12
13    /// Merkle tree for integrity verification

16    /// Log metadata

20/// Individual transformation log entry
21#[derive(Debug, Clone)]
22pub struct TransformEntry {
23    /// Unique identifier for this transformation
24    pub id: TransformId,
25
26    /// The transformation that was applied
27    pub transform: Transform,
28
29    /// Source span affected by this transformation
30    pub source_span: Span,
31
32    /// Resulting span after transformation
33    pub result_span: Span,
34
35    /// Timestamp when transformation was applied
36    pub timestamp: Instant,
37
38    /// Optional SMT proof for semantic preservation
39    pub proof: Option<SexprProof>,
40
41    /// Semantic changes introduced (if any)
42    pub semantic_delta: Option<SemanticDelta>,

45/// Merkle tree for log integrity verification
46#[derive(Debug, Clone)]
47pub struct MerkleTree {
48    /// Root hash of the tree
49    root_hash: Hash,
50
51    /// All leaf hashes (one per log entry)
52    leaf_hashes: Vec<Hash>,
53
54    /// Internal node hashes for efficient verification
55    internal_nodes: Vec<Hash>,
56
57    /// Tree height (log2 of leaf count, rounded up)
58    #[allow(dead_code)]
59    height: usize,
60}

81/// Merkle proof for verifying log entry integrity
82#[derive(Debug, Clone)]
83pub struct MerkleProof {
84    /// Index of the leaf being verified
85    pub leaf_index: usize,
86
87    /// Hash path from leaf to root
88    pub path: Vec<Hash>,
89
90    /// Direction indicators (true = right, false = left)
91    pub directions: Vec<bool>,
92}
93
94/// Result of verifying a Merkle proof
95#[derive(Debug, Clone, PartialEq, Eq)]
96pub enum VerificationResult {
97    Valid,
98    Invalid,
99    InsufficientData,
100}

102impl TransformLog {
103    /// Create a new empty transform log
104    pub fn new() -> Self {
105        Self {
106            entries: Vec::new(),
107            merkle_tree: MerkleTree::empty(),
108            metadata: LogMetadata {
109                total_transforms: 0,
110                semantic_preserving: 0,
111                with_proofs: 0,
112                time_span: None,
113                created_at: Instant::now(),
114            },
115        }
116    }
117
118    /// Add a new transformation entry
119    pub fn add_entry(&mut self, entry: TransformEntry) {
120        // Update metadata
121        self.metadata.total_transforms += 1;
122
123        if entry.transform.is_semantic_preserving() {
124            self.metadata.semantic_preserving += 1;
125        }
126
127        if entry.proof.is_some() {
128            self.metadata.with_proofs += 1;
129        }
130
131        // Update time span
132        if let Some(first_entry) = self.entries.first() {
133            self.metadata.time_span = Some(entry.timestamp.duration_since(first_entry.timestamp));
134        }
135
136        // Add to entries
137        self.entries.push(entry);
138
139        // Rebuild Merkle tree
140        self.rebuild_merkle_tree();
141    }
142
143    /// Rebuild the Merkle tree from all entries
144    fn rebuild_merkle_tree(&mut self) {
145        if self.entries.is_empty() {
146            self.merkle_tree = MerkleTree::empty();
147            return;
148        }
149
150        // Compute leaf hashes
151        let leaf_hashes: Vec<Hash> = self
152            .entries
153            .iter()
154            .map(|entry| self.hash_entry(entry))
155            .collect();
156
157        self.merkle_tree = MerkleTree::from_leaves(leaf_hashes);
158    }
159
160    /// Hash a single transform entry
161    fn hash_entry(&self, entry: &TransformEntry) -> Hash {
162        let mut hasher = Hasher::new();
163
164        // Hash the transform ID
165        hasher.update(&entry.id.0.to_le_bytes());
166
167        // Hash the transform description
168        hasher.update(entry.transform.description().as_bytes());
169
170        // Hash the spans
171        hasher.update(&entry.source_span.start.0.to_le_bytes());
172        hasher.update(&entry.source_span.end.0.to_le_bytes());
173        hasher.update(&entry.result_span.start.0.to_le_bytes());
174        hasher.update(&entry.result_span.end.0.to_le_bytes());
175
176        // Hash timestamp (as nanoseconds since epoch)
177        let nanos = entry.timestamp.elapsed().as_nanos();
178        hasher.update(&(nanos as u64).to_le_bytes());
179
180        // Hash proof if present
181        if let Some(proof) = &entry.proof {
182            hasher.update(proof.formula.as_bytes());
183            hasher.update(&[if proof.is_valid { 1 } else { 0 }]);
184        }
185
186        hasher.finalize()
187    }
188
189    /// Generate a proof for a specific log entry
190    pub fn generate_proof(&self, entry_index: usize) -> Option<MerkleProof> {
191        if entry_index >= self.entries.len() {
192            return None;
193        }
194
195        self.merkle_tree.generate_proof(entry_index)
196    }
197
198    /// Verify a Merkle proof against the current root
199    pub fn verify_proof(&self, proof: &MerkleProof, leaf_hash: Hash) -> VerificationResult {
200        self.merkle_tree.verify_proof(proof, leaf_hash)
201    }
202
203    /// Get the current root hash for integrity checking
204    pub fn root_hash(&self) -> Hash {
205        self.merkle_tree.root_hash
206    }
207
208    /// Export log for external verification
209    pub fn export_verification_data(&self) -> LogVerificationData {
210        LogVerificationData {
211            entries: self.entries.clone(),
212            root_hash: self.merkle_tree.root_hash,
213            leaf_hashes: self.merkle_tree.leaf_hashes.clone(),
214            metadata: self.metadata.clone(),
215        }
216    }
217
218    /// Get statistics about the log
219    pub fn stats(&self) -> LogStats {
220        let total = self.metadata.total_transforms;
221        let semantic_ratio = if total > 0 {
222            self.metadata.semantic_preserving as f64 / total as f64
223        } else {
224            0.0
225        };
226
227        let proof_ratio = if total > 0 {
228            self.metadata.with_proofs as f64 / total as f64
229        } else {
230            0.0
231        };
232
233        LogStats {
234            total_entries: total,
235            semantic_preserving_ratio: semantic_ratio,
236            proof_coverage_ratio: proof_ratio,
237            memory_usage_bytes: self.estimate_memory_usage(),
238            integrity_verified: true, // Always true for our implementation
239        }
240    }
241
242    fn estimate_memory_usage(&self) -> usize {
243        let entries_size = self.entries.len() * std::mem::size_of::<TransformEntry>();
244        let tree_size = self.merkle_tree.leaf_hashes.len() * std::mem::size_of::<Hash>()
245            + self.merkle_tree.internal_nodes.len() * std::mem::size_of::<Hash>();
246
247        entries_size + tree_size + std::mem::size_of::<LogMetadata>()
248    }
249}

251impl MerkleTree {
252    /// Create an empty Merkle tree
253    pub fn empty() -> Self {
254        Self {
255            root_hash: blake3::hash(b""),
256            leaf_hashes: Vec::new(),
257            internal_nodes: Vec::new(),
258            height: 0,
259        }
260    }
261
262    /// Create Merkle tree from leaf hashes
263    pub fn from_leaves(mut leaf_hashes: Vec<Hash>) -> Self {
264        if leaf_hashes.is_empty() {
265            return Self::empty();
266        }
267
268        // Pad to next power of 2 for balanced tree (minimum 2 leaves)
269        let original_count = leaf_hashes.len();
270        let next_power_of_2 = std::cmp::max(2, original_count.next_power_of_two());
271        let padding_needed = next_power_of_2 - original_count;
272
273        // Pad with zero hashes
274        let zero_hash = blake3::hash(b"");
275        for _ in 0..padding_needed {
276            leaf_hashes.push(zero_hash);
277        }
278
279        let height = (leaf_hashes.len() as f64).log2() as usize;
280        let mut internal_nodes = Vec::new();
281        let mut current_level = leaf_hashes.clone();
282
283        // Build tree bottom-up
284        while current_level.len() > 1 {
285            let mut next_level = Vec::new();
286
287            for chunk in current_level.chunks(2) {
288                let left = chunk[0];
289                let right = chunk.get(1).copied().unwrap_or(zero_hash);
290
291                let mut hasher = Hasher::new();
292                hasher.update(left.as_bytes());
293                hasher.update(right.as_bytes());
294                let parent_hash = hasher.finalize();
295
296                internal_nodes.push(parent_hash);
297                next_level.push(parent_hash);
298            }
299
300            current_level = next_level;
301        }
302
303        let root_hash = current_level[0];
304
305        Self {
306            root_hash,
307            leaf_hashes,
308            internal_nodes,
309            height,
310        }
311    }
312
313    /// Generate a Merkle proof for a specific leaf
314    pub fn generate_proof(&self, leaf_index: usize) -> Option<MerkleProof> {
315        if leaf_index >= self.leaf_hashes.len() {
316            return None;
317        }
318
319        let mut path = Vec::new();
320        let mut directions = Vec::new();
321        let current_index = leaf_index;
322        let _current_level = &self.leaf_hashes;
323
324        // Generate simple proof (simplified implementation)
325        if _current_level.len() > 1 {
326            let sibling_index = if current_index.is_multiple_of(2) {
327                current_index + 1
328            } else {
329                current_index - 1
330            };
331
332            // Add sibling to path
333            if sibling_index < _current_level.len() {
334                path.push(_current_level[sibling_index]);
335                directions.push(current_index.is_multiple_of(2)); // true if we're left child
336            } else {
337                // Pad with zero hash
338                path.push(blake3::hash(b""));
339                directions.push(true);
340            }
341        }
342
343        Some(MerkleProof {
344            leaf_index,
345            path,
346            directions,
347        })
348    }
349
350    /// Verify a Merkle proof
351    pub fn verify_proof(&self, proof: &MerkleProof, leaf_hash: Hash) -> VerificationResult {
352        if proof.leaf_index >= self.leaf_hashes.len() {
353            return VerificationResult::Invalid;
354        }
355
356        if proof.path.len() != proof.directions.len() {
357            return VerificationResult::Invalid;
358        }
359
360        let mut current_hash = leaf_hash;
361
362        // Recompute root hash using proof path
363        for (sibling_hash, is_left) in proof.path.iter().zip(&proof.directions) {
364            let mut hasher = Hasher::new();
365
366            if *is_left {
367                hasher.update(current_hash.as_bytes());
368                hasher.update(sibling_hash.as_bytes());
369            } else {
370                hasher.update(sibling_hash.as_bytes());
371                hasher.update(current_hash.as_bytes());
372            }
373
374            current_hash = hasher.finalize();
375        }
376
377        if current_hash == self.root_hash {
378            VerificationResult::Valid
379        } else {
380            VerificationResult::Invalid
381        }
382    }
383}
