1//! Shell dialect detection and compatibility system
2
3use std::collections::HashMap;
4
5/// Shell dialect variants with version information
6#[derive(Debug, Clone, PartialEq, Default)]
7pub enum ShellDialect {
8    #[default]
9    Posix, // IEEE Std 1003.1-2017
10    Bash5_2,    // GNU Bash 5.2.21
11    Ksh93uPlus, // AT&T KornShell 93u+ 2012-08-01
12    Zsh5_9,     // Z shell 5.9
13    Dash0_5_12, // Debian Almquist Shell
14    Inferred(Box<InferenceConfidence>),
15}
16
17/// Confidence scoring for dialect inference
18#[derive(Debug, Clone, PartialEq)]
19pub struct InferenceConfidence {
20    pub dialect: Box<ShellDialect>,
21    pub confidence: f32, // 0.0-1.0
22    pub evidence: InferenceEvidence,
23}
24
25#[derive(Debug, Clone, Copy, PartialEq, Eq)]
26pub enum InferenceEvidence {
27    Shebang(&'static str),
28    Syntax(SyntaxFeature),
29    Builtins(BuiltinProfile),
30}
31
32#[derive(Debug, Clone, Copy, PartialEq, Eq)]
33pub enum SyntaxFeature {
34    BashArrays,       // array=( ... )
35    BashProcessSubst, // <( ... )
36    ZshGlobs,         // **/ recursive globs
37    KshFunctions,     // function name { ... }
38    PosixFunctions,   // name() { ... }
39    BashConditionals, // [[ ... ]]
40    BashArithmetic,   // (( ... ))
41}
42
43#[derive(Debug, Clone, Copy, PartialEq, Eq)]
44pub enum BuiltinProfile {
45    BashReadarray,
46    ZshZparseopts,
47    KshTypeset,
48    DashLocal,
49}
50
51/// Core shell dialects for scoring (excludes Inferred variant)
52#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
53pub enum CoreDialect {
54    Posix,
55    Bash5_2,
56    Ksh93uPlus,
57    Zsh5_9,
58    Dash0_5_12,
59}
60
61impl CoreDialect {
62    pub fn to_shell_dialect(self) -> ShellDialect {
63        match self {
64            CoreDialect::Posix => ShellDialect::Posix,
65            CoreDialect::Bash5_2 => ShellDialect::Bash5_2,
66            CoreDialect::Ksh93uPlus => ShellDialect::Ksh93uPlus,
67            CoreDialect::Zsh5_9 => ShellDialect::Zsh5_9,
68            CoreDialect::Dash0_5_12 => ShellDialect::Dash0_5_12,
69        }
70    }
71}
72
73/// Dialect scoring system for inference
74pub struct DialectScorer {
75    scores: HashMap<CoreDialect, f32>,
76    evidence: Vec<(InferenceEvidence, f32)>,
77}
78
79impl DialectScorer {
80    pub fn new() -> Self {
81        let mut scores = HashMap::new();
82        scores.insert(CoreDialect::Posix, 0.1); // Base score for POSIX compatibility
83        scores.insert(CoreDialect::Bash5_2, 0.0);
84        scores.insert(CoreDialect::Ksh93uPlus, 0.0);
85        scores.insert(CoreDialect::Zsh5_9, 0.0);
86        scores.insert(CoreDialect::Dash0_5_12, 0.0);
87
88        Self {
89            scores,
90            evidence: Vec::new(),
91        }
92    }
93
94    pub fn add_evidence(&mut self, evidence: InferenceEvidence, weight: f32) {
95        self.evidence.push((evidence, weight));
96
97        match evidence {
98            InferenceEvidence::Shebang(shebang) => {
99                self.score_shebang(shebang, weight);
100            }
101            InferenceEvidence::Syntax(feature) => {
102                self.score_syntax_feature(feature, weight);
103            }
104            InferenceEvidence::Builtins(profile) => {
105                self.score_builtin_profile(profile, weight);
106            }
107        }
108    }
109
110    pub fn compute_confidence(&self) -> InferenceConfidence {
111        let (best_core_dialect, best_score) = self
112            .scores
113            .iter()
114            .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap())
115            .map(|(d, s)| (*d, *s))
116            .unwrap_or((CoreDialect::Posix, 0.1));
117
118        let best_dialect = best_core_dialect.to_shell_dialect();
119
120        let total_evidence_weight: f32 = self.evidence.iter().map(|(_, w)| w).sum();
121        let confidence = if total_evidence_weight > 0.0 {
122            (best_score / total_evidence_weight).min(1.0)
123        } else {
124            0.1
125        };
126
127        // Find the strongest evidence
128        let strongest_evidence = self
129            .evidence
130            .iter()
131            .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap())
132            .map(|(e, _)| *e)
133            .unwrap_or(InferenceEvidence::Syntax(SyntaxFeature::PosixFunctions));
134
135        InferenceConfidence {
136            dialect: Box::new(best_dialect),
137            confidence,
138            evidence: strongest_evidence,
139        }
140    }
141
142    fn score_shebang(&mut self, shebang: &str, weight: f32) {
143        if shebang.contains("bash") {
144            *self.scores.get_mut(&CoreDialect::Bash5_2).unwrap() += weight;
145        } else if shebang.contains("zsh") {
146            *self.scores.get_mut(&CoreDialect::Zsh5_9).unwrap() += weight;
147        } else if shebang.contains("ksh") {
148            *self.scores.get_mut(&CoreDialect::Ksh93uPlus).unwrap() += weight;
149        } else if shebang.contains("dash") {
150            *self.scores.get_mut(&CoreDialect::Dash0_5_12).unwrap() += weight;
151        } else if shebang.contains("sh") {
152            *self.scores.get_mut(&CoreDialect::Posix).unwrap() += weight;
153        }
154    }
155
156    fn score_syntax_feature(&mut self, feature: SyntaxFeature, weight: f32) {
157        match feature {
158            SyntaxFeature::BashArrays
159            | SyntaxFeature::BashProcessSubst
160            | SyntaxFeature::BashConditionals
161            | SyntaxFeature::BashArithmetic => {
162                *self.scores.get_mut(&CoreDialect::Bash5_2).unwrap() += weight;
163            }
164            SyntaxFeature::ZshGlobs => {
165                *self.scores.get_mut(&CoreDialect::Zsh5_9).unwrap() += weight;
166            }
167            SyntaxFeature::KshFunctions => {
168                *self.scores.get_mut(&CoreDialect::Ksh93uPlus).unwrap() += weight;
169            }
170            SyntaxFeature::PosixFunctions => {
171                *self.scores.get_mut(&CoreDialect::Posix).unwrap() += weight;
172            }
173        }
174    }
175
176    fn score_builtin_profile(&mut self, profile: BuiltinProfile, weight: f32) {
177        match profile {
178            BuiltinProfile::BashReadarray => {
179                *self.scores.get_mut(&CoreDialect::Bash5_2).unwrap() += weight;
180            }
181            BuiltinProfile::ZshZparseopts => {
182                *self.scores.get_mut(&CoreDialect::Zsh5_9).unwrap() += weight;
183            }
184            BuiltinProfile::KshTypeset => {
185                *self.scores.get_mut(&CoreDialect::Ksh93uPlus).unwrap() += weight;
186            }
187            BuiltinProfile::DashLocal => {
188                *self.scores.get_mut(&CoreDialect::Dash0_5_12).unwrap() += weight;
189            }
190        }
191    }
192}

200impl ShellDialect {
201    /// Infer dialect with confidence scoring
202    pub fn infer(source: &[u8]) -> InferenceConfidence {
203        let mut scorer = DialectScorer::new();
204
205        // Convert to string for analysis
206        let source_str = String::from_utf8_lossy(source);
207
208        // Shebang provides strongest signal (weight: 0.7)
209        if let Some(shebang) = Self::parse_shebang(&source_str) {
210            scorer.add_evidence(InferenceEvidence::Shebang(shebang), 0.7);
211        }
212
213        // Syntactic constructs (weight: 0.2)
214        let syntax_features = Self::extract_syntax_features(&source_str);
215        for feature in syntax_features {
216            scorer.add_evidence(InferenceEvidence::Syntax(feature), 0.2);
217        }
218
219        // Builtin usage patterns (weight: 0.1)
220        let builtin_profile = Self::profile_builtins(&source_str);
221        if let Some(profile) = builtin_profile {
222            scorer.add_evidence(InferenceEvidence::Builtins(profile), 0.1);
223        }
224
225        scorer.compute_confidence()
226    }
227
228    fn parse_shebang(source: &str) -> Option<&'static str> {
229        let first_line = source.lines().next()?;
230        if first_line.starts_with("#!") {
231            if first_line.contains("bash") {
232                Some("bash")
233            } else if first_line.contains("zsh") {
234                Some("zsh")
235            } else if first_line.contains("ksh") {
236                Some("ksh")
237            } else if first_line.contains("dash") {
238                Some("dash")
239            } else if first_line.contains("/sh") || first_line.ends_with("sh") {
240                Some("sh")
241            } else {
242                None
243            }
244        } else {
245            None
246        }
247    }
248
249    fn extract_syntax_features(source: &str) -> Vec<SyntaxFeature> {
250        let mut features = Vec::new();
251
252        // Check for bash arrays
253        if source.contains("=(") && source.contains(")") {
254            features.push(SyntaxFeature::BashArrays);
255        }
256
257        // Check for process substitution
258        if source.contains("<(") || source.contains(">(") {
259            features.push(SyntaxFeature::BashProcessSubst);
260        }
261
262        // Check for bash conditionals
263        if source.contains("[[") && source.contains("]]") {
264            features.push(SyntaxFeature::BashConditionals);
265        }
266
267        // Check for bash arithmetic
268        if source.contains("((") && source.contains("))") {
269            features.push(SyntaxFeature::BashArithmetic);
270        }
271
272        // Check for function definitions
273        if source.contains("function ") {
274            features.push(SyntaxFeature::KshFunctions);
275        }
276        if source.contains("()") && source.contains("{") {
277            features.push(SyntaxFeature::PosixFunctions);
278        }
279
280        // Check for zsh globs
281        if source.contains("**/") {
282            features.push(SyntaxFeature::ZshGlobs);
283        }
284
285        features
286    }
287
288    fn profile_builtins(source: &str) -> Option<BuiltinProfile> {
289        if source.contains("readarray") || source.contains("mapfile") {
290            Some(BuiltinProfile::BashReadarray)
291        } else if source.contains("zparseopts") {
292            Some(BuiltinProfile::ZshZparseopts)
293        } else if source.contains("typeset") {
294            Some(BuiltinProfile::KshTypeset)
295        } else if source.contains("local ") {
296            Some(BuiltinProfile::DashLocal)
297        } else {
298            None
299        }
300    }
301
302    /// Get display name for the dialect
303    pub fn display_name(&self) -> &'static str {
304        match self {
305            ShellDialect::Posix => "POSIX",
306            ShellDialect::Bash5_2 => "Bash 5.2",
307            ShellDialect::Ksh93uPlus => "KornShell 93u+",
308            ShellDialect::Zsh5_9 => "Z shell 5.9",
309            ShellDialect::Dash0_5_12 => "Dash 0.5.12",
310            ShellDialect::Inferred(conf) => conf.dialect.display_name(),
311        }
312    }
313
314    /// Check if this dialect supports a given feature
315    pub fn supports_feature(&self, feature: SyntaxFeature) -> bool {
316        match (self, feature) {
317            (ShellDialect::Bash5_2, SyntaxFeature::BashArrays) => true,
318            (ShellDialect::Bash5_2, SyntaxFeature::BashProcessSubst) => true,
319            (ShellDialect::Bash5_2, SyntaxFeature::BashConditionals) => true,
320            (ShellDialect::Bash5_2, SyntaxFeature::BashArithmetic) => true,
321            (ShellDialect::Zsh5_9, SyntaxFeature::ZshGlobs) => true,
322            (ShellDialect::Ksh93uPlus, SyntaxFeature::KshFunctions) => true,
323            (_, SyntaxFeature::PosixFunctions) => true, // All shells support POSIX functions
324            _ => false,
325        }
326    }
327}

329/// Feature compatibility between dialects
330#[derive(Debug, Clone, Copy, PartialEq, Eq)]
331pub enum Compatibility {
332    /// Feature is directly compatible
333    Direct,
334    /// Feature is incompatible
335    Incompatible,
336}
337
338/// Check compatibility between source and target dialects for a feature
339pub fn check_compatibility(
340    _source: ShellDialect,
341    target: ShellDialect,
342    feature: SyntaxFeature,
343) -> Compatibility {
344    // If target supports the feature, it's compatible
345    if target.supports_feature(feature) {
346        Compatibility::Direct
347    } else {
348        Compatibility::Incompatible
349    }
350}
