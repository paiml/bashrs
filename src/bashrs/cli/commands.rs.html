13pub fn execute_command(cli: Cli) -> Result<()> {
14    // Initialize logging
15    let subscriber = tracing_subscriber::fmt()
16        .with_max_level(if cli.verbose {
17            tracing::Level::DEBUG
18        } else {
19            tracing::Level::INFO
20        })
21        .finish();
22    tracing::subscriber::set_global_default(subscriber)
23        .map_err(|e| Error::Internal(format!("Failed to initialize logging: {e}")))?;
24
25    match cli.command {
26        Commands::Build {
27            input,
28            output,
29            emit_proof,
30            no_optimize,
31        } => {
32            info!("Building {} -> {}", input.display(), output.display());
33
34            let config = Config {
35                target: cli.target,
36                verify: cli.verify,
37                emit_proof,
38                optimize: !no_optimize,
39                validation_level: Some(cli.validation),
40                strict_mode: cli.strict,
41            };
42
43            build_command(&input, &output, config)
44        }
45
46        Commands::Check { input } => {
47            info!("Checking {}", input.display());
48            check_command(&input)
49        }
50
51        Commands::Init { path, name } => {
52            info!("Initializing project in {}", path.display());
53            init_command(&path, name.as_deref())
54        }
55
56        Commands::Verify {
57            rust_source,
58            shell_script,
59        } => {
60            info!(
61                "Verifying {} against {}",
62                shell_script.display(),
63                rust_source.display()
64            );
65            verify_command(&rust_source, &shell_script, cli.target, cli.verify)
66        }
67
68        Commands::Inspect {
69            input,
70            format,
71            output,
72            detailed,
73        } => {
74            info!("Generating inspection report for: {}", input);
75            inspect_command(&input, format, output.as_deref(), detailed)
76        }
77
78        Commands::Compile {
79            rust_source,
80            output,
81            runtime,
82            self_extracting,
83            container,
84            container_format,
85        } => {
86            let config = Config {
87                target: cli.target,
88                verify: cli.verify,
89                emit_proof: false,
90                optimize: true,
91                validation_level: Some(cli.validation),
92                strict_mode: cli.strict,
93            };
94
95            handle_compile(
96                &rust_source,
97                &output,
98                runtime,
99                self_extracting,
100                container,
101                container_format,
102                &config,
103            )
104        }
105
106        Commands::Lint { input, format, fix } => {
107            info!("Linting {}", input.display());
108            lint_command(&input, format, fix)
109        } // Playground feature removed in v1.0 - will be moved to separate rash-playground crate in v1.1
110    }
111}

113fn build_command(input: &Path, output: &Path, config: Config) -> Result<()> {
114    // Read input file
115    let source = fs::read_to_string(input).map_err(Error::Io)?;
116
117    // Transpile
118    let shell_code = transpile(&source, config.clone())?;
119
120    // Write output
121    fs::write(output, shell_code).map_err(Error::Io)?;
122
123    info!("Successfully transpiled to {}", output.display());
124
125    // Generate proof if requested
126    if config.emit_proof {
127        let proof_path = output.with_extension("proof");
128        generate_proof(&source, &proof_path, &config)?;
129        info!("Proof generated at {}", proof_path.display());
130    }
131
132    Ok(())
133}

146fn init_command(path: &Path, name: Option<&str>) -> Result<()> {
147    // Create directory if it doesn't exist
148    if !path.exists() {
149        fs::create_dir_all(path).map_err(Error::Io)?;
150    }
151
152    let project_name = name.unwrap_or(
153        path.file_name()
154            .and_then(|n| n.to_str())
155            .unwrap_or("my-installer"),
156    );
157
158    // Create Cargo.toml
159    let cargo_toml = format!(
160        r#"[package]
161name = "{project_name}"
162version = "0.1.0"
163edition = "2021"
164
165[dependencies]
166# No dependencies needed - Rash transpiles to pure shell
167
168[[bin]]
169name = "install"
170path = "src/main.rs"
171"#
172    );
173
174    fs::write(path.join("Cargo.toml"), cargo_toml).map_err(Error::Io)?;
175
176    // Create src directory
177    let src_dir = path.join("src");
178    fs::create_dir_all(&src_dir).map_err(Error::Io)?;
179
180    // Create main.rs with example installer
181    let main_rs = r#"/// Example installer script for Rash
182/// This will be transpiled to POSIX-compliant shell script
183use std::env;
184use std::fs;
185use std::path::Path;
186use std::process::{Command, exit};
187
188const VERSION: &str = "0.1.0";
189const BINARY_NAME: &str = "myapp";
190
191fn main() {
192    println!("{} installer v{}", BINARY_NAME, VERSION);
193    println!("=======================");
194    
195    // Parse arguments
196    let args: Vec<String> = env::args().collect();
197    if args.contains(&"--help".to_string()) {
198        print_help();
199        return;
200    }
201    
202    // Determine installation directory
203    let prefix = env::var("PREFIX").unwrap_or_else(|_| "/usr/local".to_string());
204    let bin_dir = format!("{}/bin", prefix);
205    
206    println!("Installing to: {}", bin_dir);
207    
208    // Create directory
209    if let Err(e) = fs::create_dir_all(&bin_dir) {
210        eprintln!("Failed to create directory: {}", e);
211        exit(1);
212    }
213    
214    // Download binary (example URL)
215    let url = format!(
216        "https://github.com/example/{}/releases/download/v{}/{}-{}.tar.gz",
217        BINARY_NAME, VERSION, BINARY_NAME, detect_platform()
218    );
219    
220    println!("Downloading from: {}", url);
221    
222    // TODO: Add your actual installation logic here
223    // - Download binary
224    // - Verify checksum
225    // - Extract and install
226    // - Set permissions
227    
228    println!("\n✓ {} installed successfully!", BINARY_NAME);
229    println!("\nTo get started, run:");
230    println!("  {} --help", BINARY_NAME);

240fn detect_platform() -> &'static str {
241    // Simplified platform detection
242    if cfg!(target_os = "linux") {
243        if cfg!(target_arch = "x86_64") {
244            "linux-amd64"
245        } else {
246            "linux-arm64"
247        }
248    } else if cfg!(target_os = "macos") {
249        "darwin-amd64"
250    } else {
251        panic!("Unsupported platform");
252    }
253}"#;
254
255    fs::write(src_dir.join("main.rs"), main_rs).map_err(Error::Io)?;
256
257    // Create .rash.toml
258    let rash_toml = r##"# Rash configuration file
259[transpiler]
260target = "posix"          # Target shell dialect
261strict_mode = true        # Fail on warnings
262preserve_comments = false # Strip comments for smaller output
263
264[validation]
265level = "strict"          # ShellCheck compliance level
266rules = ["all"]           # Can disable specific: ["-SC2034"]
267external_check = false    # Run actual shellcheck binary
268
269[output]
270shebang = "#!/bin/sh"     # POSIX shebang
271set_flags = "euf"         # set -euf (no pipefail in POSIX)
272optimize_size = true      # Minimize output script size
273
274[style]
275indent = "    "           # 4 spaces
276max_line_length = 100     # Wrap long commands
277"##;
278
279    fs::write(path.join(".rash.toml"), rash_toml).map_err(Error::Io)?;
280
281    info!("✓ Initialized Rash project '{}'", project_name);
282    info!("  Run 'cd {}' to enter the project", path.display());
283    info!("  Run 'rash build src/main.rs' to build");
284
285    Ok(())
286}

288fn verify_command(
289    rust_source: &Path,
290    shell_script: &Path,
291    target: crate::models::ShellDialect,
292    verify_level: crate::models::VerificationLevel,
293) -> Result<()> {
294    // Read both files
295    let rust_code = fs::read_to_string(rust_source).map_err(Error::Io)?;
296    let shell_code = fs::read_to_string(shell_script).map_err(Error::Io)?;
297
298    // Transpile Rust to shell
299    let config = Config {
300        target,
301        verify: verify_level,
302        emit_proof: false,
303        optimize: true,
304        strict_mode: true,
305        validation_level: Some(crate::validation::ValidationLevel::Strict),
306    };
307
308    let generated_shell = transpile(&rust_code, config)?;
309
310    // Compare generated vs actual
311    if normalize_shell_script(&generated_shell) == normalize_shell_script(&shell_code) {
312        info!("✓ Shell script matches Rust source");
313        Ok(())
314    } else {
315        warn!("Shell script does not match Rust source");
316        // TODO: Show diff
317        Err(Error::Verification("Script mismatch".to_string()))
318    }
319}

321fn generate_proof(source: &str, proof_path: &Path, config: &Config) -> Result<()> {
322    // For now, just create a simple proof file
323    let proof = format!(
324        r#"{{
325    "version": "1.0",
326    "source_hash": "{}",
327    "verification_level": "{:?}",
328    "target": "{:?}",
329    "timestamp": "{}",
330    "properties": ["no-injection", "deterministic"]
331}}"#,
332        blake3::hash(source.as_bytes()),
333        config.verify,
334        config.target,
335        chrono::Utc::now().to_rfc3339()
336    );
337
338    fs::write(proof_path, proof).map_err(Error::Io)?;
339
340    Ok(())
341}

343fn normalize_shell_script(script: &str) -> String {
344    // Remove comments and normalize whitespace for comparison
345    script
346        .lines()
347        .filter(|line| !line.trim().starts_with('#'))
348        .map(|line| line.trim())
349        .filter(|line| !line.is_empty())
350        .collect::<Vec<_>>()
351        .join("\n")
352}

354fn inspect_command(
355    input: &str,
356    format: InspectionFormat,
357    output: Option<&Path>,
358    _detailed: bool,
359) -> Result<()> {
360    use crate::formal::{AbstractState, ProofInspector, TinyAst};
361
362    // Parse input - for now, we'll support JSON AST or a few predefined examples
363    let ast = if input.starts_with('{') {
364        // JSON input
365        serde_json::from_str::<TinyAst>(input)
366            .map_err(|e| Error::Internal(format!("Invalid AST JSON: {e}")))?
367    } else {
368        // Predefined examples or simple DSL
369        match input {
370            "echo-example" => TinyAst::ExecuteCommand {
371                command_name: "echo".to_string(),
372                args: vec!["Hello, World!".to_string()],
373            },
374            "bootstrap-example" => TinyAst::Sequence {
375                commands: vec![
376                    TinyAst::SetEnvironmentVariable {
377                        name: "INSTALL_DIR".to_string(),
378                        value: "/opt/rash".to_string(),
379                    },
380                    TinyAst::ExecuteCommand {
381                        command_name: "mkdir".to_string(),
382                        args: vec!["-p".to_string(), "/opt/rash/bin".to_string()],
383                    },
384                    TinyAst::ChangeDirectory {
385                        path: "/opt/rash".to_string(),
386                    },
387                    TinyAst::ExecuteCommand {
388                        command_name: "echo".to_string(),
389                        args: vec!["Installation ready".to_string()],
390                    },
391                ],
392            },
393            _ => {
394                return Err(Error::Internal(format!("Unknown example: {input}. Try 'echo-example' or 'bootstrap-example', or provide JSON AST")));
395            }
396        }
397    };
398
399    // Validate the AST
400    if !ast.is_valid() {
401        return Err(Error::Validation("Invalid AST".to_string()));
402    }
403
404    // Generate inspection report
405    let mut initial_state = AbstractState::new();
406    // Add common directories for testing
407    initial_state.filesystem.insert(
408        std::path::PathBuf::from("/opt"),
409        crate::formal::FileSystemEntry::Directory,
410    );
411
412    let report = ProofInspector::inspect(&ast, initial_state);
413
414    // Format output
415    let output_content = match format {
416        InspectionFormat::Markdown => ProofInspector::generate_report(&report),
417        InspectionFormat::Json => serde_json::to_string_pretty(&report)
418            .map_err(|e| Error::Internal(format!("JSON serialization failed: {e}")))?,
419        InspectionFormat::Html => {
420            // Convert markdown to HTML (simplified)
421            let markdown = ProofInspector::generate_report(&report);
422            format!(
423                r#"<!DOCTYPE html>
424<html>
425<head>
426    <title>Formal Verification Report</title>
427    <style>
428        body {{ font-family: Arial, sans-serif; margin: 2em; }}
429        pre {{ background: #f5f5f5; padding: 1em; border-radius: 4px; }}
430        .success {{ color: green; }}
431        .failure {{ color: red; }}
432        .warning {{ color: orange; }}
433    </style>
434</head>
435<body>
436<pre>{}</pre>
437</body>
438</html>"#,
439                markdown
440                    .replace('&', "&amp;")
441                    .replace('<', "&lt;")
442                    .replace('>', "&gt;")
443            )
444        }
445    };
446
447    // Write output
448    match output {
449        Some(path) => {
450            fs::write(path, &output_content).map_err(Error::Io)?;
451            info!("Inspection report written to {}", path.display());
452        }
453        None => {
454            println!("{output_content}");
455        }
456    }
457
458    Ok(())
459}

461fn handle_compile(
462    rust_source: &Path,
463    output: &Path,
464    runtime: CompileRuntime,
465    self_extracting: bool,
466    container: bool,
467    container_format: ContainerFormatArg,
468    config: &Config,
469) -> Result<()> {
470    use crate::compiler::{create_self_extracting_script, BinaryCompiler, RuntimeType};
471    use crate::container::{ContainerFormat, DistrolessBuilder};
472
473    info!(
474        "Compiling {} to {}",
475        rust_source.display(),
476        output.display()
477    );
478
479    // Read and transpile the source
480    let source = fs::read_to_string(rust_source).map_err(Error::Io)?;
481    let shell_code = transpile(&source, config.clone())?;
482
483    if self_extracting {
484        // Create self-extracting script
485        let output_str = output
486            .to_str()
487            .ok_or_else(|| Error::Validation("Output path contains invalid UTF-8".to_string()))?;
488        create_self_extracting_script(&shell_code, output_str)?;
489        info!("Created self-extracting script at {}", output.display());
490    } else if container {
491        // Create container image
492        let runtime_type = match runtime {
493            CompileRuntime::Dash => RuntimeType::Dash,
494            CompileRuntime::Busybox => RuntimeType::Busybox,
495            CompileRuntime::Minimal => RuntimeType::Minimal,
496        };
497
498        let compiler = BinaryCompiler::new(runtime_type);
499        let binary = compiler.compile(&shell_code)?;
500
501        let format = match container_format {
502            ContainerFormatArg::Oci => ContainerFormat::OCI,
503            ContainerFormatArg::Docker => ContainerFormat::Docker,
504        };
505
506        let builder = DistrolessBuilder::new(binary).with_format(format);
507        let container_data = builder.build()?;
508
509        fs::write(output, container_data).map_err(Error::Io)?;
510        info!("Created container image at {}", output.display());
511    } else {
512        // Create standalone binary (not fully implemented)
513        warn!(
514            "Binary compilation not yet fully implemented, creating self-extracting script instead"
515        );
516        let output_str = output
517            .to_str()
518            .ok_or_else(|| Error::Validation("Output path contains invalid UTF-8".to_string()))?;
519        create_self_extracting_script(&shell_code, output_str)?;
520    }
521
522    Ok(())
523}

525fn lint_command(input: &Path, format: LintFormat, _fix: bool) -> Result<()> {
526    use crate::linter::{rules::lint_shell, output::{write_results, OutputFormat}};
527
528    // Read input file
529    let source = fs::read_to_string(input).map_err(Error::Io)?;
530
531    // Run linter
532    let result = lint_shell(&source);
533
534    // Convert format
535    let output_format = match format {
536        LintFormat::Human => OutputFormat::Human,
537        LintFormat::Json => OutputFormat::Json,
538        LintFormat::Sarif => OutputFormat::Sarif,
539    };
540
541    // Write results to stdout
542    let file_path = input.to_str().unwrap_or("unknown");
543    write_results(&mut std::io::stdout(), &result, output_format, file_path)
544        .map_err(|e| Error::Internal(format!("Failed to write lint results: {e}")))?;
545
546    // Exit with appropriate code
547    if result.has_errors() {
548        std::process::exit(2);
549    } else if result.has_warnings() {
550        std::process::exit(1);
551    }
552
553    Ok(())
554}
