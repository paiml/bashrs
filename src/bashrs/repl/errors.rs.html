14/// Structured error message
15#[derive(Debug, Clone)]
16pub struct ErrorMessage {
17    /// Error type (Parse, Lint, Command, Runtime)
18    pub error_type: ErrorType,
19
20    /// Error code (optional, e.g., "DET001", "E001")
21    pub code: Option<String>,
22
23    /// Severity level
24    pub severity: Severity,
25
26    /// Main error message
27    pub message: String,
28
29    /// Source code context (line with error)
30    pub context: Option<SourceContext>,
31
32    /// Detailed explanation (optional)
33    pub explanation: Option<String>,
34
35    /// Suggested fix (optional)
36    pub suggestion: Option<Suggestion>,
37
38    /// Related help topics (optional)
39    pub help_topics: Vec<String>,
40}

84/// Format error message for display
85pub fn format_error(error: &ErrorMessage) -> String {
86    let mut output = String::new();
87
88    // Header with icon, severity, and type
89    let icon = match error.severity {
90        Severity::Error => "✗",
91        Severity::Warning => "⚠",
92        Severity::Info => "ℹ",
93    };
94
95    let severity_str = match error.severity {
96        Severity::Error => "Error",
97        Severity::Warning => "Warning",
98        Severity::Info => "Info",
99    };
100
101    let type_str = match error.error_type {
102        ErrorType::Parse => "Parse",
103        ErrorType::Lint => "Lint",
104        ErrorType::Command => "Command",
105        ErrorType::Runtime => "Runtime",
106    };
107
108    // Format header
109    if let Some(code) = &error.code {
110        output.push_str(&format!("{} {} {} [{}]\n\n", icon, type_str, severity_str, code));
111    } else {
112        output.push_str(&format!("{} {} {}\n\n", icon, type_str, severity_str));
113    }
114
115    // Add source context if available
116    if let Some(context) = &error.context {
117        output.push_str(&format_source_context(context));
118        output.push('\n');
119    }
120
121    // Add main message
122    output.push_str(&format!("  {}\n", error.message));
123
124    // Add explanation if available
125    if let Some(explanation) = &error.explanation {
126        output.push_str(&format!("\n  Problem: {}\n", explanation));
127    }
128
129    // Add suggestion if available
130    if let Some(suggestion) = &error.suggestion {
131        output.push_str(&format!("\n  Suggestion: {}\n", suggestion.description));
132
133        if let Some(fixed) = &suggestion.fixed_code {
134            output.push_str(&format!("  Try: {}\n", fixed));
135        }
136
137        if suggestion.auto_fixable {
138            output.push_str("\n  Auto-fix available: Run ':purify' to see safe version\n");
139        }
140    }
141
142    // Add help topics if available
143    if !error.help_topics.is_empty() {
144        output.push_str("\n  For more help, try:\n");
145        for topic in &error.help_topics {
146            output.push_str(&format!("    :help {}\n", topic));
147        }
148    }
149
150    output
151}

153/// Create error message for parse errors
154pub fn format_parse_error(error_msg: &str, line: usize, column: usize, source: &str) -> ErrorMessage {
155    let lines: Vec<&str> = source.lines().collect();
156    let source_line = if line > 0 && line <= lines.len() {
157        lines[line - 1].to_string()
158    } else {
159        String::new()
160    };
161
162    let context = if !source_line.is_empty() {
163        Some(SourceContext {
164            line,
165            column,
166            source_line: source_line.clone(),
167            length: 1,
168        })
169    } else {
170        None
171    };
172
173    // Try to suggest a fix based on common parse errors
174    let suggestion = if error_msg.contains("then") {
175        Some(Suggestion {
176            description: "Did you forget 'then'?".to_string(),
177            fixed_code: Some(format!("{}; then", source_line)),
178            auto_fixable: false,
179        })
180    } else {
181        None
182    };
183
184    ErrorMessage {
185        error_type: ErrorType::Parse,
186        code: Some("P001".to_string()),
187        severity: Severity::Error,
188        message: error_msg.to_string(),
189        context,
190        explanation: None,
191        suggestion,
192        help_topics: vec![],
193    }
194}

196/// Create error message for lint violations
197pub fn format_lint_error(diagnostic: &Diagnostic, source: &str) -> ErrorMessage {
198    let lines: Vec<&str> = source.lines().collect();
199    let line_idx = diagnostic.span.start_line.saturating_sub(1);
200    let source_line = if line_idx < lines.len() {
201        lines[line_idx].to_string()
202    } else {
203        String::new()
204    };
205
206    let context = if !source_line.is_empty() {
207        Some(SourceContext {
208            line: diagnostic.span.start_line,
209            column: diagnostic.span.start_col,
210            source_line,
211            length: diagnostic
212                .span
213                .end_col
214                .saturating_sub(diagnostic.span.start_col)
215                .max(1),
216        })
217    } else {
218        None
219    };
220
221    // Convert LintSeverity to our Severity
222    let severity = match diagnostic.severity {
223        LintSeverity::Error => Severity::Error,
224        LintSeverity::Warning => Severity::Warning,
225        LintSeverity::Info => Severity::Info,
226        LintSeverity::Note => Severity::Info,
227        LintSeverity::Perf => Severity::Warning,
228        LintSeverity::Risk => Severity::Warning,
229    };
230
231    // Determine explanation based on violation code
232    let explanation = match diagnostic.code.as_str() {
233        code if code.starts_with("IDEM") => {
234            Some("Command will fail if run multiple times".to_string())
235        }
236        code if code.starts_with("DET") => {
237            Some("Command produces different output on each run".to_string())
238        }
239        code if code.starts_with("SEC") => Some("Security vulnerability detected".to_string()),
240        _ => None,
241    };
242
243    // Create suggestion from fix if available
244    let suggestion = diagnostic.fix.as_ref().map(|fix| Suggestion {
245        description: "Use idempotent version".to_string(),
246        fixed_code: Some(fix.replacement.clone()),
247        auto_fixable: true, // bashrs can auto-fix lint issues via purify
248    });
249
250    ErrorMessage {
251        error_type: ErrorType::Lint,
252        code: Some(diagnostic.code.clone()),
253        severity,
254        message: diagnostic.message.clone(),
255        context,
256        explanation,
257        suggestion,
258        help_topics: vec!["purify".to_string()],
259    }
260}

262/// Create error message for unknown commands
263pub fn format_command_error(command: &str, available_commands: &[&str]) -> ErrorMessage {
264    let message = format!("Unknown command: '{}'", command);
265
266    // Try to suggest similar command
267    let suggestion = suggest_command(command, available_commands).map(|suggested| Suggestion {
268        description: format!("Did you mean: '{}'?", suggested),
269        fixed_code: Some(suggested),
270        auto_fixable: false,
271    });
272
273    // Build list of available commands for explanation
274    let commands_list = available_commands.join(", ");
275    let explanation = Some(format!("Available commands: {}", commands_list));
276
277    ErrorMessage {
278        error_type: ErrorType::Command,
279        code: None,
280        severity: Severity::Error,
281        message,
282        context: None,
283        explanation,
284        suggestion,
285        help_topics: vec!["commands".to_string()],
286    }
287}

289/// Suggest similar command using edit distance
290pub fn suggest_command(input: &str, commands: &[&str]) -> Option<String> {
291    if commands.is_empty() {
292        return None;
293    }
294
295    // Calculate Levenshtein distance to each command
296    let mut best_match: Option<(String, usize)> = None;
297
298    for &command in commands {
299        let distance = levenshtein_distance(input, command);
300
301        // Only suggest if distance is small (< 3)
302        if distance < 3 {
303            if let Some((_, best_distance)) = &best_match {
304                if distance < *best_distance {
305                    best_match = Some((command.to_string(), distance));
306                }
307            } else {
308                best_match = Some((command.to_string(), distance));
309            }
310        }
311    }
312
313    best_match.map(|(cmd, _)| cmd)
314}

316/// Calculate Levenshtein distance between two strings
317fn levenshtein_distance(s1: &str, s2: &str) -> usize {
318    let len1 = s1.len();
319    let len2 = s2.len();
320
321    if len1 == 0 {
322        return len2;
323    }
324    if len2 == 0 {
325        return len1;
326    }
327
328    let mut matrix = vec![vec![0; len2 + 1]; len1 + 1];
329
330    // Initialize first column and row
331    for i in 0..=len1 {
332        matrix[i][0] = i;
333    }
334    for j in 0..=len2 {
335        matrix[0][j] = j;
336    }
337
338    // Calculate distances
339    let s1_chars: Vec<char> = s1.chars().collect();
340    let s2_chars: Vec<char> = s2.chars().collect();
341
342    for i in 1..=len1 {
343        for j in 1..=len2 {
344            let cost = if s1_chars[i - 1] == s2_chars[j - 1] {
345                0
346            } else {
347                1
348            };
349
350            matrix[i][j] = (matrix[i - 1][j] + 1) // deletion
351                .min(matrix[i][j - 1] + 1) // insertion
352                .min(matrix[i - 1][j - 1] + cost); // substitution
353        }
354    }
355
356    matrix[len1][len2]
357}

359/// Format source context with caret indicator
360pub fn format_source_context(context: &SourceContext) -> String {
361    let mut output = String::new();
362
363    // Show line number and source
364    output.push_str(&format!("    {} | {}\n", context.line, context.source_line));
365
366    // Show caret indicator pointing to the error
367    let col = context.column.saturating_sub(1);
368    let line_num_width = format!("    {} | ", context.line).len();
369
370    output.push_str(&" ".repeat(line_num_width));
371    output.push_str(&" ".repeat(col));
372    output.push_str(&"^".repeat(context.length));
373
374    output
375}
