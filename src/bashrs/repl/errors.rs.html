14/// Structured error message with source context and actionable suggestions.
15///
16/// `ErrorMessage` provides rich, user-friendly error messages for REPL interactions.
17/// Each error includes:
18/// - **Type and severity**: Categorize the error for appropriate handling
19/// - **Source context**: Show exactly where the error occurred
20/// - **Explanation**: Describe why it's a problem
21/// - **Suggestion**: Provide actionable fixes
22/// - **Help topics**: Link to relevant documentation
23///
24/// # Examples
25///
26/// ## Parse error with source context
27///
28/// ```
29/// use bashrs::repl::errors::{ErrorMessage, ErrorType, Severity, SourceContext};
30///
31/// let error = ErrorMessage {
32///     error_type: ErrorType::Parse,
33///     code: Some("P001".to_string()),
34///     severity: Severity::Error,
35///     message: "Expected 'then' after condition".to_string(),
36///     context: Some(SourceContext {
37///         line: 1,
38///         column: 13,
39///         source_line: "if [ -f file".to_string(),
40///         length: 1,
41///     }),
42///     explanation: None,
43///     suggestion: None,
44///     help_topics: vec![],
45/// };
46///
47/// assert_eq!(error.error_type, ErrorType::Parse);
48/// ```
49///
50/// ## Format error for display
51///
52/// ```
53/// use bashrs::repl::errors::{format_parse_error, format_error};
54///
55/// let error = format_parse_error("Expected 'then'", 1, 13, "if [ -f file");
56/// let formatted = format_error(&error);
57///
58/// assert!(formatted.contains("Parse Error"));
59/// assert!(formatted.contains("if [ -f file"));
60/// ```
61#[derive(Debug, Clone)]
62pub struct ErrorMessage {
63    /// Error type (Parse, Lint, Command, Runtime).
64    pub error_type: ErrorType,
65
66    /// Error code (e.g., "DET001", "P001").
67    ///
68    /// Codes enable users to search for specific errors in documentation.
69    pub code: Option<String>,
70
71    /// Severity level (Error, Warning, Info).
72    pub severity: Severity,
73
74    /// Main error message.
75    ///
76    /// Should be concise and describe what went wrong.
77    pub message: String,
78
79    /// Source code context showing where the error occurred.
80    ///
81    /// Includes line number, column, and the problematic source line.
82    pub context: Option<SourceContext>,
83
84    /// Detailed explanation of why this is a problem.
85    ///
86    /// Helps users understand the underlying issue.
87    pub explanation: Option<String>,
88
89    /// Suggested fix with optional corrected code.
90    pub suggestion: Option<Suggestion>,
91
92    /// Related help topics.
93    ///
94    /// Users can type `:help <topic>` for more information.
95    pub help_topics: Vec<String>,
96}

225/// Formats an error message for display in the terminal.
226///
227/// Produces a user-friendly, color-coded error message with:
228/// - Icon and severity (✗, ⚠, ℹ)
229/// - Error type and code
230/// - Source context with caret indicators
231/// - Explanation and suggestions
232/// - Related help topics
233///
234/// # Arguments
235///
236/// * `error` - The error message to format
237///
238/// # Returns
239///
240/// Formatted string ready for terminal display
241///
242/// # Examples
243///
244/// ```
245/// use bashrs::repl::errors::{ErrorMessage, ErrorType, Severity, format_error};
246///
247/// let error = ErrorMessage {
248///     error_type: ErrorType::Command,
249///     code: None,
250///     severity: Severity::Error,
251///     message: "Unknown command 'foo'".to_string(),
252///     context: None,
253///     explanation: Some("Command not recognized".to_string()),
254///     suggestion: None,
255///     help_topics: vec!["commands".to_string()],
256/// };
257///
258/// let formatted = format_error(&error);
259/// assert!(formatted.contains("Unknown command"));
260/// assert!(formatted.contains(":help commands"));
261/// ```
262pub fn format_error(error: &ErrorMessage) -> String {
263    let mut output = String::new();
264
265    // Header with icon, severity, and type
266    let icon = match error.severity {
267        Severity::Error => "✗",
268        Severity::Warning => "⚠",
269        Severity::Info => "ℹ",
270    };
271
272    let severity_str = match error.severity {
273        Severity::Error => "Error",
274        Severity::Warning => "Warning",
275        Severity::Info => "Info",
276    };
277
278    let type_str = match error.error_type {
279        ErrorType::Parse => "Parse",
280        ErrorType::Lint => "Lint",
281        ErrorType::Command => "Command",
282        ErrorType::Runtime => "Runtime",
283    };
284
285    // Format header
286    if let Some(code) = &error.code {
287        output.push_str(&format!(
288            "{} {} {} [{}]\n\n",
289            icon, type_str, severity_str, code
290        ));
291    } else {
292        output.push_str(&format!("{} {} {}\n\n", icon, type_str, severity_str));
293    }
294
295    // Add source context if available
296    if let Some(context) = &error.context {
297        output.push_str(&format_source_context(context));
298        output.push('\n');
299    }
300
301    // Add main message
302    output.push_str(&format!("  {}\n", error.message));
303
304    // Add explanation if available
305    if let Some(explanation) = &error.explanation {
306        output.push_str(&format!("\n  Problem: {}\n", explanation));
307    }
308
309    // Add suggestion if available
310    if let Some(suggestion) = &error.suggestion {
311        output.push_str(&format!("\n  Suggestion: {}\n", suggestion.description));
312
313        if let Some(fixed) = &suggestion.fixed_code {
314            output.push_str(&format!("  Try: {}\n", fixed));
315        }
316
317        if suggestion.auto_fixable {
318            output.push_str("\n  Auto-fix available: Run ':purify' to see safe version\n");
319        }
320    }
321
322    // Add help topics if available
323    if !error.help_topics.is_empty() {
324        output.push_str("\n  For more help, try:\n");
325        for topic in &error.help_topics {
326            output.push_str(&format!("    :help {}\n", topic));
327        }
328    }
329
330    output
331}

333/// Creates an error message for parse errors (syntax errors).
334///
335/// Analyzes the parse error and constructs a helpful error message with
336/// source context and common fix suggestions.
337///
338/// # Arguments
339///
340/// * `error_msg` - The parser's error message
341/// * `line` - Line number where error occurred (1-indexed)
342/// * `column` - Column number where error occurred (1-indexed)
343/// * `source` - The source code being parsed
344///
345/// # Returns
346///
347/// Structured `ErrorMessage` with code P001
348///
349/// # Examples
350///
351/// ```
352/// use bashrs::repl::errors::{format_parse_error, ErrorType};
353///
354/// let error = format_parse_error(
355///     "Expected 'then'",
356///     1,
357///     13,
358///     "if [ -f test"
359/// );
360///
361/// assert_eq!(error.error_type, ErrorType::Parse);
362/// assert_eq!(error.code, Some("P001".to_string()));
363/// assert!(error.context.is_some());
364/// ```
365pub fn format_parse_error(
366    error_msg: &str,
367    line: usize,
368    column: usize,
369    source: &str,
370) -> ErrorMessage {
371    let lines: Vec<&str> = source.lines().collect();
372    let source_line = if line > 0 && line <= lines.len() {
373        lines
374            .get(line - 1)
375            .map(|s| s.to_string())
376            .unwrap_or_default()
377    } else {
378        String::new()
379    };
380
381    let context = if !source_line.is_empty() {
382        Some(SourceContext {
383            line,
384            column,
385            source_line: source_line.clone(),
386            length: 1,
387        })
388    } else {
389        None
390    };
391
392    // Try to suggest a fix based on common parse errors
393    let suggestion = if error_msg.contains("then") {
394        Some(Suggestion {
395            description: "Did you forget 'then'?".to_string(),
396            fixed_code: Some(format!("{}; then", source_line)),
397            auto_fixable: false,
398        })
399    } else {
400        None
401    };
402
403    ErrorMessage {
404        error_type: ErrorType::Parse,
405        code: Some("P001".to_string()),
406        severity: Severity::Error,
407        message: error_msg.to_string(),
408        context,
409        explanation: None,
410        suggestion,
411        help_topics: vec![],
412    }
413}

415/// Creates an error message for lint violations (code quality issues).
416///
417/// Converts a linter diagnostic into a user-friendly error message with
418/// explanations and auto-fix suggestions.
419///
420/// # Arguments
421///
422/// * `diagnostic` - The linter diagnostic
423/// * `source` - The source code being linted
424///
425/// # Returns
426///
427/// Structured `ErrorMessage` with lint code (DET*, IDEM*, SEC*)
428///
429/// # Examples
430///
431/// ```
432/// use bashrs::repl::errors::format_lint_error;
433/// use bashrs::linter::{Diagnostic, Severity, Span, Fix};
434///
435/// let diagnostic = Diagnostic {
436///     code: "IDEM001".to_string(),
437///     severity: Severity::Error,
438///     message: "mkdir without -p".to_string(),
439///     span: Span::new(1, 1, 1, 11),
440///     fix: Some(Fix::new("mkdir -p /tmp")),
441/// };
442///
443/// let error = format_lint_error(&diagnostic, "mkdir /tmp");
444///
445/// assert_eq!(error.code, Some("IDEM001".to_string()));
446/// assert!(error.suggestion.is_some());
447/// ```
448pub fn format_lint_error(diagnostic: &Diagnostic, source: &str) -> ErrorMessage {
449    let lines: Vec<&str> = source.lines().collect();
450    let line_idx = diagnostic.span.start_line.saturating_sub(1);
451    let source_line = if line_idx < lines.len() {
452        lines
453            .get(line_idx)
454            .map(|s| s.to_string())
455            .unwrap_or_default()
456    } else {
457        String::new()
458    };
459
460    let context = if !source_line.is_empty() {
461        Some(SourceContext {
462            line: diagnostic.span.start_line,
463            column: diagnostic.span.start_col,
464            source_line,
465            length: diagnostic
466                .span
467                .end_col
468                .saturating_sub(diagnostic.span.start_col)
469                .max(1),
470        })
471    } else {
472        None
473    };
474
475    // Convert LintSeverity to our Severity
476    let severity = match diagnostic.severity {
477        LintSeverity::Error => Severity::Error,
478        LintSeverity::Warning => Severity::Warning,
479        LintSeverity::Info => Severity::Info,
480        LintSeverity::Note => Severity::Info,
481        LintSeverity::Perf => Severity::Warning,
482        LintSeverity::Risk => Severity::Warning,
483    };
484
485    // Determine explanation based on violation code
486    let explanation = match diagnostic.code.as_str() {
487        code if code.starts_with("IDEM") => {
488            Some("Command will fail if run multiple times".to_string())
489        }
490        code if code.starts_with("DET") => {
491            Some("Command produces different output on each run".to_string())
492        }
493        code if code.starts_with("SEC") => Some("Security vulnerability detected".to_string()),
494        _ => None,
495    };
496
497    // Create suggestion from fix if available
498    let suggestion = diagnostic.fix.as_ref().map(|fix| Suggestion {
499        description: "Use idempotent version".to_string(),
500        fixed_code: Some(fix.replacement.clone()),
501        auto_fixable: true, // bashrs can auto-fix lint issues via purify
502    });
503
504    ErrorMessage {
505        error_type: ErrorType::Lint,
506        code: Some(diagnostic.code.clone()),
507        severity,
508        message: diagnostic.message.clone(),
509        context,
510        explanation,
511        suggestion,
512        help_topics: vec!["purify".to_string()],
513    }
514}

516/// Creates an error message for unknown REPL commands.
517///
518/// Uses edit distance to suggest similar commands and lists all available commands.
519///
520/// # Arguments
521///
522/// * `command` - The unknown command entered by the user
523/// * `available_commands` - List of valid command names
524///
525/// # Returns
526///
527/// Structured `ErrorMessage` with command suggestion if similar command found
528///
529/// # Examples
530///
531/// ```
532/// use bashrs::repl::errors::format_command_error;
533///
534/// let error = format_command_error(
535///     "purfy",
536///     &["purify", "lint", "quit", "help"]
537/// );
538///
539/// assert!(error.message.contains("purfy"));
540/// assert!(error.suggestion.is_some());
541/// assert!(error.explanation.is_some());
542/// ```
543pub fn format_command_error(command: &str, available_commands: &[&str]) -> ErrorMessage {
544    let message = format!("Unknown command: '{}'", command);
545
546    // Try to suggest similar command
547    let suggestion = suggest_command(command, available_commands).map(|suggested| Suggestion {
548        description: format!("Did you mean: '{}'?", suggested),
549        fixed_code: Some(suggested),
550        auto_fixable: false,
551    });
552
553    // Build list of available commands for explanation
554    let commands_list = available_commands.join(", ");
555    let explanation = Some(format!("Available commands: {}", commands_list));
556
557    ErrorMessage {
558        error_type: ErrorType::Command,
559        code: None,
560        severity: Severity::Error,
561        message,
562        context: None,
563        explanation,
564        suggestion,
565        help_topics: vec!["commands".to_string()],
566    }
567}

569/// Suggest similar command using edit distance
570pub fn suggest_command(input: &str, commands: &[&str]) -> Option<String> {
571    if commands.is_empty() {
572        return None;
573    }
574
575    // Calculate Levenshtein distance to each command
576    let mut best_match: Option<(String, usize)> = None;
577
578    for &command in commands {
579        let distance = levenshtein_distance(input, command);
580
581        // Only suggest if distance is small (< 3)
582        if distance < 3 {
583            if let Some((_, best_distance)) = &best_match {
584                if distance < *best_distance {
585                    best_match = Some((command.to_string(), distance));
586                }
587            } else {
588                best_match = Some((command.to_string(), distance));
589            }
590        }
591    }
592
593    best_match.map(|(cmd, _)| cmd)
594}

596/// Calculate Levenshtein distance between two strings
597///
598/// This is a standard dynamic programming algorithm with guaranteed safe indexing
599/// because the matrix is pre-allocated to (len1+1) x (len2+1) dimensions.
600#[allow(clippy::indexing_slicing, clippy::needless_range_loop)]
601fn levenshtein_distance(s1: &str, s2: &str) -> usize {
602    let len1 = s1.len();
603    let len2 = s2.len();
604
605    if len1 == 0 {
606        return len2;
607    }
608    if len2 == 0 {
609        return len1;
610    }
611
612    let mut matrix = vec![vec![0; len2 + 1]; len1 + 1];
613
614    // Initialize first column and row
615    for i in 0..=len1 {
616        matrix[i][0] = i;
617    }
618    for j in 0..=len2 {
619        matrix[0][j] = j;
620    }
621
622    // Calculate distances
623    let s1_chars: Vec<char> = s1.chars().collect();
624    let s2_chars: Vec<char> = s2.chars().collect();
625
626    for i in 1..=len1 {
627        for j in 1..=len2 {
628            let cost = if s1_chars[i - 1] == s2_chars[j - 1] {
629                0
630            } else {
631                1
632            };
633
634            matrix[i][j] = (matrix[i - 1][j] + 1) // deletion
635                .min(matrix[i][j - 1] + 1) // insertion
636                .min(matrix[i - 1][j - 1] + cost); // substitution
637        }
638    }
639
640    matrix[len1][len2]
641}

643/// Format source context with caret indicator
644pub fn format_source_context(context: &SourceContext) -> String {
645    let mut output = String::new();
646
647    // Show line number and source
648    output.push_str(&format!("    {} | {}\n", context.line, context.source_line));
649
650    // Show caret indicator pointing to the error
651    let col = context.column.saturating_sub(1);
652    let line_num_width = format!("    {} | ", context.line).len();
653
654    output.push_str(&" ".repeat(line_num_width));
655    output.push_str(&" ".repeat(col));
656    output.push_str(&"^".repeat(context.length));
657
658    output
659}
