14/// Parse bash input and return AST
15///
16/// # Examples
17///
18/// ```
19/// use bashrs::repl::parser::parse_bash;
20///
21/// let result = parse_bash("echo hello");
22/// assert!(result.is_ok());
23/// ```
24pub fn parse_bash(input: &str) -> Result<BashAst, ParseError> {
25    let mut parser = BashParser::new(input)?;
26    parser.parse()
27}
28
29/// Format parse error for display in REPL with enhanced context
30///
31/// Provides helpful error messages with line numbers for better debugging.
32/// Handles all ParseError variants with appropriate formatting.
33///
34/// # Examples
35///
36/// ```
37/// use bashrs::repl::parser::{parse_bash, format_parse_error};
38///
39/// let result = parse_bash("if then fi");  // Missing condition
40/// assert!(result.is_err());
41///
42/// let error = result.unwrap_err();
43/// let formatted = format_parse_error(&error);
44/// assert!(formatted.contains("Syntax error"));
45/// ```
46pub fn format_parse_error(error: &ParseError) -> String {
47    match error {
48        ParseError::UnexpectedToken {
49            expected,
50            found,
51            line,
52        } => {
53            format!(
54                "Syntax error at line {}: expected {}, but found {}",
55                line, expected, found
56            )
57        }
58        ParseError::UnexpectedEof => String::from(
59            "Syntax error: unexpected end of file (did you forget to close a quote or bracket?)",
60        ),
61        ParseError::InvalidSyntax(msg) => {
62            // Try to add generic line info even if not provided in the variant
63            format!("Syntax error: {}", msg)
64        }
65        ParseError::LexerError(lexer_err) => {
66            format!("Lexer error: {}", lexer_err)
67        }
68    }
69}
