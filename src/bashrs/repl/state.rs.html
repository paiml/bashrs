16/// Mutable state for a REPL session
17///
18/// Inspired by Ruchy REPL state management:
19/// - Command history for navigation (up/down arrows)
20/// - Session variables for persistence across commands
21/// - Exit flag for clean shutdown
22/// - Error tracking for debugging
23/// - Mode switching for different behaviors (Normal, Purify, Lint, Debug, Explain)
24/// - Loaded scripts tracking for :reload command
25/// - Functions extracted from loaded scripts
26#[derive(Debug, Clone, PartialEq, Eq, Default)]
27pub struct ReplState {
28    /// Command history (for up/down arrow navigation)
29    history: Vec<String>,
30
31    /// Session variables (persist across commands)
32    variables: HashMap<String, String>,
33
34    /// Exit requested flag (for clean shutdown)
35    exit_requested: bool,
36
37    /// Error count (for debugging and statistics)
38    error_count: usize,
39
40    /// Current REPL mode
41    mode: ReplMode,
42
43    /// Last loaded script path (for :reload command)
44    last_loaded_script: Option<PathBuf>,
45
46    /// Functions extracted from loaded scripts
47    loaded_functions: Vec<String>,
48}
49
50impl ReplState {
51    /// Create a new REPL state
52    pub fn new() -> Self {
53        Self::default()
54    }
55
56    /// Add a command to history
57    pub fn add_history(&mut self, command: String) {
58        self.history.push(command);
59    }
60
61    /// Get command history
62    pub fn history(&self) -> &[String] {
63        &self.history
64    }
65
66    /// Get a specific history entry
67    pub fn get_history(&self, index: usize) -> Option<&String> {
68        self.history.get(index)
69    }
70
71    /// Clear command history
72    pub fn clear_history(&mut self) {
73        self.history.clear();
74    }
75
76    /// Set a session variable
77    pub fn set_variable(&mut self, name: String, value: String) {
78        self.variables.insert(name, value);
79    }
80
81    /// Get a session variable
82    pub fn get_variable(&self, name: &str) -> Option<&String> {
83        self.variables.get(name)
84    }
85
86    /// Remove a session variable
87    pub fn remove_variable(&mut self, name: &str) -> Option<String> {
88        self.variables.remove(name)
89    }
90
91    /// Get all variables
92    pub fn variables(&self) -> &HashMap<String, String> {
93        &self.variables
94    }
95
96    /// Clear all variables
97    pub fn clear_variables(&mut self) {
98        self.variables.clear();
99    }
100
101    /// Request exit
102    pub fn request_exit(&mut self) {
103        self.exit_requested = true;
104    }
105
106    /// Check if exit was requested
107    pub fn should_exit(&self) -> bool {
108        self.exit_requested
109    }
110
111    /// Increment error count
112    pub fn record_error(&mut self) {
113        self.error_count += 1;
114    }
115
116    /// Get error count
117    pub fn error_count(&self) -> usize {
118        self.error_count
119    }
120
121    /// Reset error count
122    pub fn reset_error_count(&mut self) {
123        self.error_count = 0;
124    }
125
126    /// Get history length
127    pub fn history_len(&self) -> usize {
128        self.history.len()
129    }
130
131    /// Get variable count
132    pub fn variable_count(&self) -> usize {
133        self.variables.len()
134    }
135
136    /// Get current REPL mode
137    pub fn mode(&self) -> ReplMode {
138        self.mode
139    }
140
141    /// Set REPL mode
142    pub fn set_mode(&mut self, mode: ReplMode) {
143        self.mode = mode;
144    }
145
146    /// Set last loaded script
147    pub fn set_last_loaded_script(&mut self, path: PathBuf) {
148        self.last_loaded_script = Some(path);
149    }
150
151    /// Get last loaded script
152    pub fn last_loaded_script(&self) -> Option<&PathBuf> {
153        self.last_loaded_script.as_ref()
154    }
155
156    /// Add a function to loaded functions
157    pub fn add_function(&mut self, name: String) {
158        if !self.loaded_functions.contains(&name) {
159            self.loaded_functions.push(name);
160        }
161    }
162
163    /// Get loaded functions
164    pub fn loaded_functions(&self) -> &[String] {
165        &self.loaded_functions
166    }
167
168    /// Clear all loaded functions
169    pub fn clear_functions(&mut self) {
170        self.loaded_functions.clear();
171    }
172
173    /// Get function count
174    pub fn function_count(&self) -> usize {
175        self.loaded_functions.len()
176    }
177}

434            let mut state = ReplState::new();

436            for cmd in &commands {

441
442            for (i, cmd) in commands.iter().enumerate() {
443                prop_assert_eq!(state.get_history(i), Some(cmd));
444            }
445        }

447        #[test]
448        fn prop_variables_preserve_last_write(
449            updates in prop::collection::vec(("[A-Z]+", "[a-z0-9]+"), 1..50)
450        ) {
451            let mut state = ReplState::new();
452            let mut expected = HashMap::new();
453
454            for (key, value) in updates {
455                state.set_variable(key.clone(), value.clone());
456                expected.insert(key, value);
457            }
458
459            prop_assert_eq!(state.variable_count(), expected.len());
460
461            for (key, value) in &expected {
462                prop_assert_eq!(state.get_variable(key), Some(value));
463            }
464        }
465
466        #[test]
467        fn prop_error_count_never_decreases_except_reset(
468            error_counts in prop::collection::vec(0usize..100, 1..50)
469        ) {
470            let mut state = ReplState::new();
471            let mut total_errors = 0;
472
473            for count in error_counts {
474                for _ in 0..count {
475                    state.record_error();
476                    total_errors += 1;
477                }
478
479                prop_assert_eq!(state.error_count(), total_errors);
480            }
481
482            state.reset_error_count();
483            prop_assert_eq!(state.error_count(), 0);
484        }
