12use std::collections::HashMap;
13
14/// Mutable state for a REPL session
15///
16/// Inspired by Ruchy REPL state management:
17/// - Command history for navigation (up/down arrows)
18/// - Session variables for persistence across commands
19/// - Exit flag for clean shutdown
20/// - Error tracking for debugging
21#[derive(Debug, Clone, PartialEq, Eq)]
22pub struct ReplState {
23    /// Command history (for up/down arrow navigation)
24    history: Vec<String>,
25
26    /// Session variables (persist across commands)
27    variables: HashMap<String, String>,
28
29    /// Exit requested flag (for clean shutdown)
30    exit_requested: bool,
31
32    /// Error count (for debugging and statistics)
33    error_count: usize,
34}
35
36impl Default for ReplState {
37    fn default() -> Self {
38        Self {
39            history: Vec::new(),
40            variables: HashMap::new(),
41            exit_requested: false,
42            error_count: 0,
43        }
44    }
45}
46
47impl ReplState {
48    /// Create a new REPL state
49    pub fn new() -> Self {
50        Self::default()
51    }
52
53    /// Add a command to history
54    pub fn add_history(&mut self, command: String) {
55        self.history.push(command);
56    }
57
58    /// Get command history
59    pub fn history(&self) -> &[String] {
60        &self.history
61    }
62
63    /// Get a specific history entry
64    pub fn get_history(&self, index: usize) -> Option<&String> {
65        self.history.get(index)
66    }
67
68    /// Clear command history
69    pub fn clear_history(&mut self) {
70        self.history.clear();
71    }
72
73    /// Set a session variable
74    pub fn set_variable(&mut self, name: String, value: String) {
75        self.variables.insert(name, value);
76    }
77
78    /// Get a session variable
79    pub fn get_variable(&self, name: &str) -> Option<&String> {
80        self.variables.get(name)
81    }
82
83    /// Remove a session variable
84    pub fn remove_variable(&mut self, name: &str) -> Option<String> {
85        self.variables.remove(name)
86    }
87
88    /// Get all variables
89    pub fn variables(&self) -> &HashMap<String, String> {
90        &self.variables
91    }
92
93    /// Clear all variables
94    pub fn clear_variables(&mut self) {
95        self.variables.clear();
96    }
97
98    /// Request exit
99    pub fn request_exit(&mut self) {
100        self.exit_requested = true;
101    }
102
103    /// Check if exit was requested
104    pub fn should_exit(&self) -> bool {
105        self.exit_requested
106    }
107
108    /// Increment error count
109    pub fn record_error(&mut self) {
110        self.error_count += 1;
111    }
112
113    /// Get error count
114    pub fn error_count(&self) -> usize {
115        self.error_count
116    }
117
118    /// Reset error count
119    pub fn reset_error_count(&mut self) {
120        self.error_count = 0;
121    }
122
123    /// Get history length
124    pub fn history_len(&self) -> usize {
125        self.history.len()
126    }
127
128    /// Get variable count
129    pub fn variable_count(&self) -> usize {
130        self.variables.len()
131    }
132}

289        fn prop_variables_preserve_last_write(
290            updates in prop::collection::vec(("[A-Z]+", "[a-z0-9]+"), 1..50)
291        ) {
292            let mut state = ReplState::new();
293            let mut expected = HashMap::new();
294
295            for (key, value) in updates {
296                state.set_variable(key.clone(), value.clone());
297                expected.insert(key, value);
298            }
299
300            prop_assert_eq!(state.variable_count(), expected.len());
301
302            for (key, value) in &expected {
303                prop_assert_eq!(state.get_variable(key), Some(value));
304            }
305        }

307        #[test]
308        fn prop_error_count_never_decreases_except_reset(
309            error_counts in prop::collection::vec(0usize..100, 1..50)
310        ) {
311            let mut state = ReplState::new();
312            let mut total_errors = 0;
313
314            for count in error_counts {
315                for _ in 0..count {
316                    state.record_error();
317                    total_errors += 1;
318                }
319
320                prop_assert_eq!(state.error_count(), total_errors);
321            }
322
323            state.reset_error_count();
324            prop_assert_eq!(state.error_count(), 0);
325        }
