15/// Mutable state for a REPL session
16///
17/// Inspired by Ruchy REPL state management:
18/// - Command history for navigation (up/down arrows)
19/// - Session variables for persistence across commands
20/// - Exit flag for clean shutdown
21/// - Error tracking for debugging
22/// - Mode switching for different behaviors (Normal, Purify, Lint, Debug, Explain)
23#[derive(Debug, Clone, PartialEq, Eq)]
24pub struct ReplState {
25    /// Command history (for up/down arrow navigation)
26    history: Vec<String>,
27
28    /// Session variables (persist across commands)
29    variables: HashMap<String, String>,
30
31    /// Exit requested flag (for clean shutdown)
32    exit_requested: bool,
33
34    /// Error count (for debugging and statistics)
35    error_count: usize,
36
37    /// Current REPL mode
38    mode: ReplMode,
39}
40
41impl Default for ReplState {
42    fn default() -> Self {
43        Self {
44            history: Vec::new(),
45            variables: HashMap::new(),
46            exit_requested: false,
47            error_count: 0,
48            mode: ReplMode::default(),
49        }
50    }
51}
52
53impl ReplState {
54    /// Create a new REPL state
55    pub fn new() -> Self {
56        Self::default()
57    }
58
59    /// Add a command to history
60    pub fn add_history(&mut self, command: String) {
61        self.history.push(command);
62    }
63
64    /// Get command history
65    pub fn history(&self) -> &[String] {
66        &self.history
67    }
68
69    /// Get a specific history entry
70    pub fn get_history(&self, index: usize) -> Option<&String> {
71        self.history.get(index)
72    }
73
74    /// Clear command history
75    pub fn clear_history(&mut self) {
76        self.history.clear();
77    }
78
79    /// Set a session variable
80    pub fn set_variable(&mut self, name: String, value: String) {
81        self.variables.insert(name, value);
82    }
83
84    /// Get a session variable
85    pub fn get_variable(&self, name: &str) -> Option<&String> {
86        self.variables.get(name)
87    }
88
89    /// Remove a session variable
90    pub fn remove_variable(&mut self, name: &str) -> Option<String> {
91        self.variables.remove(name)
92    }
93
94    /// Get all variables
95    pub fn variables(&self) -> &HashMap<String, String> {
96        &self.variables
97    }
98
99    /// Clear all variables
100    pub fn clear_variables(&mut self) {
101        self.variables.clear();
102    }
103
104    /// Request exit
105    pub fn request_exit(&mut self) {
106        self.exit_requested = true;
107    }
108
109    /// Check if exit was requested
110    pub fn should_exit(&self) -> bool {
111        self.exit_requested
112    }
113
114    /// Increment error count
115    pub fn record_error(&mut self) {
116        self.error_count += 1;
117    }
118
119    /// Get error count
120    pub fn error_count(&self) -> usize {
121        self.error_count
122    }
123
124    /// Reset error count
125    pub fn reset_error_count(&mut self) {
126        self.error_count = 0;
127    }
128
129    /// Get history length
130    pub fn history_len(&self) -> usize {
131        self.history.len()
132    }
133
134    /// Get variable count
135    pub fn variable_count(&self) -> usize {
136        self.variables.len()
137    }
138
139    /// Get current REPL mode
140    pub fn mode(&self) -> ReplMode {
141        self.mode
142    }
143
144    /// Set REPL mode
145    pub fn set_mode(&mut self, mode: ReplMode) {
146        self.mode = mode;
147    }
148}

346        #[test]
347        fn prop_variables_preserve_last_write(
348            updates in prop::collection::vec(("[A-Z]+", "[a-z0-9]+"), 1..50)
349        ) {
350            let mut state = ReplState::new();
351            let mut expected = HashMap::new();
352
353            for (key, value) in updates {
354                state.set_variable(key.clone(), value.clone());
355                expected.insert(key, value);
356            }
357
358            prop_assert_eq!(state.variable_count(), expected.len());
359
360            for (key, value) in &expected {
361                prop_assert_eq!(state.get_variable(key), Some(value));
362            }
363        }

365        #[test]
366        fn prop_error_count_never_decreases_except_reset(
367            error_counts in prop::collection::vec(0usize..100, 1..50)
368        ) {
369            let mut state = ReplState::new();
370            let mut total_errors = 0;
371
372            for count in error_counts {
373                for _ in 0..count {
374                    state.record_error();
375                    total_errors += 1;
376                }
377
378                prop_assert_eq!(state.error_count(), total_errors);
379            }
380
381            state.reset_error_count();
382            prop_assert_eq!(state.error_count(), 0);
383        }
