22/// Tab completion helper for bashrs REPL
23///
24/// Provides intelligent completion for:
25/// - REPL commands (:mode, :parse, :purify, :lint, :load, :source, :functions, :reload, :history, :vars, :clear)
26/// - Mode names (normal, purify, lint, debug, explain)
27/// - File paths (for :load and :source commands)
28/// - Common bash constructs in explain mode
29///
30/// # Examples
31///
32/// ```
33/// use bashrs::repl::completion::ReplCompleter;
34///
35/// let completer = ReplCompleter::new();
36/// // User types ":mo" + Tab → completes to ":mode"
37/// // User types ":mode p" + Tab → completes to ":mode purify"
38/// // User types ":load ex" + Tab → completes to ":load examples/"
39/// ```
40#[derive(Debug, Clone)]
41pub struct ReplCompleter {
42    /// REPL commands (without : prefix)
43    commands: Vec<String>,
44    /// Available mode names
45    modes: Vec<String>,
46    /// Common bash constructs for explain mode
47    bash_constructs: Vec<String>,
48}
49
50impl ReplCompleter {

56                "parse".to_string(),
57                "purify".to_string(),

62                "reload".to_string(),

64                "vars".to_string(),
65                "clear".to_string(),
66            ],
67            modes: vec![
68                "normal".to_string(),
69                "purify".to_string(),
70                "lint".to_string(),
71                "debug".to_string(),
72                "explain".to_string(),
73            ],
74            bash_constructs: vec![
75                "${var:-default}".to_string(),
76                "${var:=default}".to_string(),
77                "${var:?error}".to_string(),
78                "${var:+alternate}".to_string(),
79                "${#var}".to_string(),
80                "for i in".to_string(),
81                "if [".to_string(),
82                "while".to_string(),
83                "case".to_string(),
84            ],
85        }

88    /// Get command completions (for lines starting with :)
89    fn complete_command(&self, word: &str) -> Vec<Pair> {
90        let word_lower = word.to_lowercase();
91
92        self.commands
93            .iter()
94            .filter(|cmd| cmd.starts_with(&word_lower))
95            .map(|cmd| Pair {
96                display: cmd.clone(),
97                replacement: format!(":{}", cmd),
98            })
99            .collect()
100    }
101
102    /// Get mode name completions (for `:mode <tab>`)
103    fn complete_mode(&self, word: &str) -> Vec<Pair> {
104        let word_lower = word.to_lowercase();
105
106        self.modes
107            .iter()
108            .filter(|mode| mode.starts_with(&word_lower))
109            .map(|mode| Pair {
110                display: mode.clone(),
111                replacement: mode.clone(),
112            })
113            .collect()
114    }
115
116    /// Get bash construct completions (for explain mode)
117    fn complete_bash_construct(&self, word: &str) -> Vec<Pair> {
118        self.bash_constructs
119            .iter()
120            .filter(|construct| construct.starts_with(word))
121            .map(|construct| Pair {
122                display: construct.clone(),
123                replacement: construct.clone(),
124            })
125            .collect()
126    }

128    /// Get file path completions (for :load and :source commands)
129    fn complete_file_path(&self, partial_path: &str) -> Vec<Pair> {
130        // Split path into directory and filename prefix
131        let path = Path::new(partial_path);
132        let (dir_path, file_prefix) = if partial_path.ends_with('/') {
133            // User typed a directory ending with /
134            (path, "")
135        } else {
136            // Extract directory and filename
137            match path.parent() {
138                Some(parent) if !parent.as_os_str().is_empty() => {
139                    let prefix = path.file_name().and_then(|n| n.to_str()).unwrap_or("");
140                    (parent, prefix)
141                }
142                _ => {
143                    // No parent directory, search current directory
144                    let prefix = path.to_str().unwrap_or("");
145                    (Path::new("."), prefix)
146                }
147            }
148        };
149
150        // Read directory contents
151        let entries = match fs::read_dir(dir_path) {
152            Ok(entries) => entries,
153            Err(_) => return Vec::new(), // Directory doesn't exist or no permission
154        };
155
156        // Filter and map to completion pairs
157        let mut completions = Vec::new();
158        for entry in entries.flatten() {
159            let file_name_os = entry.file_name();
160            let file_name = match file_name_os.to_str() {
161                Some(name) => name,
162                None => continue, // Skip invalid UTF-8
163            };
164
165            // Filter by prefix
166            if !file_name.starts_with(file_prefix) {
167                continue;
168            }
169
170            // Skip hidden files (starting with .)
171            if file_name.starts_with('.') && !file_prefix.starts_with('.') {
172                continue;
173            }
174
175            // Build completion path
176            let is_dir = entry.file_type().map(|ft| ft.is_dir()).unwrap_or(false);
177            let display_name = if is_dir {
178                format!("{}/", file_name)
179            } else {
180                file_name.to_string()
181            };
182
183            // Build replacement path
184            let replacement = if dir_path.to_str() == Some(".") {
185                // Current directory - don't include "./"
186                if is_dir {
187                    format!("{}/", file_name)
188                } else {
189                    file_name.to_string()
190                }
191            } else {
192                // Other directory - include full path
193                let full_path = dir_path.join(file_name);
194                if is_dir {
195                    // Append / to directories
196                    full_path
197                        .to_str()
198                        .map(|s| format!("{}/", s))
199                        .unwrap_or_default()
200                } else {
201                    full_path.to_str().unwrap_or("").to_string()
202                }
203            };
204
205            completions.push(Pair {
206                display: display_name,
207                replacement,
208            });
209        }
210
211        // Sort: directories first, then files, alphabetically
212        completions.sort_by(|a, b| {
213            let a_is_dir = a.display.ends_with('/');
214            let b_is_dir = b.display.ends_with('/');
215
216            match (a_is_dir, b_is_dir) {
217                (true, false) => std::cmp::Ordering::Less,
218                (false, true) => std::cmp::Ordering::Greater,
219                _ => a.display.cmp(&b.display),
220            }
221        });
222
223        completions
224    }
225}

233impl Completer for ReplCompleter {
234    type Candidate = Pair;
235
236    fn complete(
237        &self,
238        line: &str,
239        pos: usize,
240        _ctx: &Context<'_>,
241    ) -> Result<(usize, Vec<Pair>), ReadlineError> {
242        let line_before_cursor = &line[..pos];
243
244        // Complete REPL commands (lines starting with :)
245        if line_before_cursor.starts_with(':') {
246            // Check if we're completing a file path after `:load ` or `:source `
247            if line_before_cursor.starts_with(":load ") {
248                let path_start = 6; // Position after ":load "
249                let partial_path = &line_before_cursor[path_start..];
250                let completions = self.complete_file_path(partial_path);
251                return Ok((path_start, completions));
252            }
253
254            if line_before_cursor.starts_with(":source ") {
255                let path_start = 8; // Position after ":source "
256                let partial_path = &line_before_cursor[path_start..];
257                let completions = self.complete_file_path(partial_path);
258                return Ok((path_start, completions));
259            }
260
261            // Check if we're completing a mode name after `:mode `
262            if line_before_cursor.starts_with(":mode ") {
263                let mode_start = 6; // Position after ":mode "
264                let word = &line_before_cursor[mode_start..];
265                let completions = self.complete_mode(word);
266                return Ok((mode_start, completions));
267            }
268
269            // Complete the command itself
270            let word = &line_before_cursor[1..]; // Skip the ':'
271            let completions = self.complete_command(word);
272            return Ok((0, completions));
273        }
274
275        // Complete bash constructs (for any other input)
276        // Find the last word boundary
277        let word_start = line_before_cursor
278            .rfind(char::is_whitespace)
279            .map(|i| i + 1)
280            .unwrap_or(0);
281
282        let word = &line_before_cursor[word_start..];
283        let completions = self.complete_bash_construct(word);
284
285        Ok((word_start, completions))
286    }
287}

512        use tempfile::TempDir;
513
514        let temp_dir = TempDir::new().unwrap();
515        let dir1 = temp_dir.path().join("dir_a");
516        let dir2 = temp_dir.path().join("dir_b");
517        let file1 = temp_dir.path().join("file_a.sh");
518        std::fs::create_dir(&dir1).unwrap();
519        std::fs::create_dir(&dir2).unwrap();
520        std::fs::File::create(&file1).unwrap();
521
522        let completer = ReplCompleter::new();
523        let path_str = format!("{}/", temp_dir.path().display());
524        let completions = completer.complete_file_path(&path_str);
525
526        // Directories should come before files
527        let dir_positions: Vec<usize> = completions
528            .iter()
529            .enumerate()
530            .filter(|(_, p)| p.display.ends_with('/'))
531            .map(|(i, _)| i)
532            .collect();
533
534        let file_positions: Vec<usize> = completions
535            .iter()
536            .enumerate()
537            .filter(|(_, p)| !p.display.ends_with('/'))
538            .map(|(i, _)| i)
539            .collect();
540
541        if !dir_positions.is_empty() && !file_positions.is_empty() {
542            assert!(dir_positions.iter().max().unwrap() < file_positions.iter().min().unwrap());
543        }
544    }
