15/// A stack frame in the call stack
16///
17/// Represents a function call or execution context with its name
18/// and the line number where it was called from.
19#[derive(Debug, Clone, PartialEq, Eq)]
20pub struct StackFrame {
21    /// Name of the function or context
22    pub name: String,
23    /// Line number where this frame was called (1-indexed)
24    pub line: usize,
25}
26
27impl StackFrame {
28    /// Create a new stack frame
29    pub fn new(name: impl Into<String>, line: usize) -> Self {
30        Self {
31            name: name.into(),
32            line,
33        }
34    }
35}

51/// Debug session for step-by-step execution of bash scripts
52///
53/// Tracks the current execution state including:
54/// - Script lines
55/// - Current line number
56/// - Breakpoints
57/// - Variables
58#[derive(Debug, Clone)]
59pub struct DebugSession {
60    /// Lines of the script being debugged
61    lines: Vec<String>,
62
63    /// Current line number (0-indexed)
64    current_line: usize,
65
66    /// Breakpoint manager
67    breakpoints: BreakpointManager,
68
69    /// Session variables
70    variables: HashMap<String, String>,
71
72    /// Call stack for tracking function calls
73    call_stack: Vec<StackFrame>,
74
75    /// Purified version of the script lines (if purification succeeded)
76    purified_lines: Option<Vec<String>>,
77
78    /// Whether execution is complete
79    finished: bool,
80}
81
82impl DebugSession {
83    /// Create a new debug session from script content
84    pub fn new(script: &str) -> Self {
85        let lines: Vec<String> = script.lines().map(|l| l.to_string()).collect();
86
87        // Initialize call stack with main frame
88        let mut call_stack = Vec::new();
89        call_stack.push(StackFrame::new("<main>", 0));
90
91        // Attempt to purify the script for comparison
92        let purified_lines = purify_bash(script)
93            .ok()
94            .map(|purified| purified.lines().map(|l| l.to_string()).collect());
95
96        Self {
97            lines,
98            current_line: 0,
99            breakpoints: BreakpointManager::new(),
100            variables: HashMap::new(),
101            call_stack,
102            purified_lines,
103            finished: false,
104        }
105    }
106
107    /// Get the current line number (1-indexed for user display)
108    pub fn current_line(&self) -> usize {
109        self.current_line + 1
110    }
111
112    /// Get the current line content
113    pub fn current_line_content(&self) -> Option<&str> {
114        self.lines.get(self.current_line).map(|s| s.as_str())
115    }

117    /// Check if execution is finished
118    pub fn is_finished(&self) -> bool {
119        self.finished
120    }
121
122    /// Get total number of lines
123    pub fn total_lines(&self) -> usize {
124        self.lines.len()
125    }

127    /// Execute one line (step)
128    ///
129    /// Executes the current line and advances to the next line.
130    /// Returns the output from executing the line.
131    ///
132    /// # Returns
133    /// - `Some(output)` if a line was executed
134    /// - `None` if execution is finished
135    pub fn step(&mut self) -> Option<String> {
136        // Check if finished
137        if self.finished || self.current_line >= self.lines.len() {
138            self.finished = true;
139            return None;
140        }
141
142        // Get current line
143        let line = self.lines[self.current_line].clone();
144
145        // Execute the line (simplified - just echo for now)
146        let output = format!("Executed: {}", line);
147
148        // Advance to next line
149        self.current_line += 1;
150
151        // Check if finished
152        if self.current_line >= self.lines.len() {
153            self.finished = true;
154        }
155
156        Some(output)
157    }
158
159    /// Set a breakpoint at the specified line (1-indexed)
160    pub fn set_breakpoint(&mut self, line: usize) -> bool {
161        if line == 0 || line > self.lines.len() {
162            return false;
163        }
164        self.breakpoints.set_breakpoint(line)
165    }
166
167    /// Check if current line has a breakpoint
168    pub fn at_breakpoint(&self) -> bool {
169        self.breakpoints.is_breakpoint_hit(self.current_line())
170    }
171
172    /// Continue execution until a breakpoint is hit or execution finishes
173    ///
174    /// Executes lines repeatedly using step() until:
175    /// - A breakpoint is encountered (returns BreakpointHit with line number)
176    /// - Execution completes (returns Finished)
177    ///
178    /// # Returns
179    /// - `ContinueResult::BreakpointHit(line)` if stopped at breakpoint
180    /// - `ContinueResult::Finished` if execution completed
181    pub fn continue_execution(&mut self) -> ContinueResult {
182        loop {
183            // Check if at breakpoint before executing
184            if self.at_breakpoint() {
185                return ContinueResult::BreakpointHit(self.current_line());
186            }
187
188            // Execute one step
189            match self.step() {
190                Some(_output) => {
191                    // Line executed, check if we hit a breakpoint
192                    if self.at_breakpoint() {
193                        return ContinueResult::BreakpointHit(self.current_line());
194                    }
195                }
196                None => {
197                    // Execution finished
198                    return ContinueResult::Finished;
199                }
200            }
201        }
202    }

279    /// List all environment variables
280    ///
281    /// Returns a vector of (name, value) tuples for all environment variables.
282    /// Variables are sorted by name for consistency.
283    ///
284    /// # Returns
285    /// Vector of (variable_name, variable_value) tuples, sorted by name
286    pub fn list_env(&self) -> Vec<(String, String)> {
287        let mut env_vars: Vec<(String, String)> = std::env::vars().collect();
288        env_vars.sort_by(|(k1, _), (k2, _)| k1.cmp(k2));
289        env_vars
290    }

292    /// Filter environment variables by prefix
293    ///
294    /// Returns environment variables whose names start with the given prefix.
295    /// Results are sorted by name for consistency.
296    ///
297    /// # Arguments
298    /// * `prefix` - Prefix to filter by (case-sensitive)
299    ///
300    /// # Returns
301    /// Vector of (variable_name, variable_value) tuples matching the prefix
302    pub fn filter_env(&self, prefix: &str) -> Vec<(String, String)> {
303        let mut filtered: Vec<(String, String)> = std::env::vars()
304            .filter(|(name, _)| name.starts_with(prefix))
305            .collect();
306        filtered.sort_by(|(k1, _), (k2, _)| k1.cmp(k2));
307        filtered
308    }

323    /// Pop the most recent frame from the call stack
324    ///
325    /// Removes the top frame from the call stack (when returning from a function).
326    /// Does nothing if only the main frame remains.
327    pub fn pop_frame(&mut self) {
328        // Keep at least the main frame
329        if self.call_stack.len() > 1 {
330            self.call_stack.pop();
331        }
332    }

347
348    /// Compare current line with its purified version
349    ///
350    /// Returns a comparison showing the original line and its purified version,
351    /// or None if purification failed or line is out of bounds.
352    ///
353    /// # Returns
354    /// LineComparison with original, purified, and whether they differ
355    pub fn compare_current_line(&self) -> Option<LineComparison> {
356        // Get original line
357        let original = self.lines.get(self.current_line)?.clone();
358
359        // Get purified version if available
360        let purified_lines = self.purified_lines.as_ref()?;
361        let purified = purified_lines.get(self.current_line)?.clone();
362
363        // Compare
364        let differs = original != purified;
365
366        Some(LineComparison {
367            original,
368            purified,
369            differs,
370        })
371    }
372
373    /// Format diff highlighting for a line comparison
374    ///
375    /// Creates a visual diff showing the differences between original
376    /// and purified versions.
377    ///
378    /// # Arguments
379    /// * `comparison` - The line comparison to format
380    ///
381    /// # Returns
382    /// Formatted string showing the diff with highlighting
383    pub fn format_diff_highlighting(&self, comparison: &LineComparison) -> String {
384        if !comparison.differs {
385            return format!("  {}\n(no changes)", comparison.original);
386        }
387
388        format!("- {}\n+ {}", comparison.original, comparison.purified)
389    }
