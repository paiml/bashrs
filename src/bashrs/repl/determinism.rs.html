1//! Determinism checking for bash scripts
2//!
3//! Detects non-deterministic patterns that violate deterministic execution:
4//! - $RANDOM: Random number generation
5//! - $$: Process ID
6//! - Timestamps: date commands
7//! - $BASHPID: Bash-specific process ID
8//! - $SRANDOM: Cryptographic random (Bash 5.1+)
9
10/// Detects non-deterministic patterns in bash scripts
11#[derive(Debug, Clone, PartialEq)]
12pub struct DeterminismChecker {
13    /// Patterns detected in the script
14    detections: Vec<DeterminismIssue>,
15}
16
17/// A single non-deterministic pattern detection
18#[derive(Debug, Clone, PartialEq)]
19pub struct DeterminismIssue {
20    /// Line number where pattern was found (1-indexed)
21    pub line: usize,
22    /// Type of non-deterministic pattern
23    pub pattern_type: NonDeterministicPattern,
24    /// The actual code that triggered the detection
25    pub code: String,
26    /// Human-readable explanation
27    pub explanation: String,
28}
29
30/// Types of non-deterministic patterns
31#[derive(Debug, Clone, PartialEq, Eq)]
32pub enum NonDeterministicPattern {
33    /// $RANDOM variable
34    Random,
35    /// $$ (process ID)
36    ProcessId,
37    /// date command or timestamp generation
38    Timestamp,
39    /// $BASHPID variable
40    BashPid,
41    /// $SRANDOM (cryptographic random, Bash 5.1+)
42    SecureRandom,
43}
44
45impl DeterminismChecker {
46    /// Create a new determinism checker
47    pub fn new() -> Self {
48        Self {
49            detections: Vec::new(),
50        }
51    }
52
53    /// Scan bash script for non-deterministic patterns
54    ///
55    /// Returns: List of detected issues
56    pub fn scan(&mut self, script: &str) -> Vec<DeterminismIssue> {
57        // Clear previous results
58        self.detections.clear();
59
60        for (line_num, line) in script.lines().enumerate() {
61            let line_num = line_num + 1; // 1-indexed
62
63            // Detect $RANDOM
64            if line.contains("$RANDOM") || line.contains("${RANDOM}") {
65                self.detections.push(DeterminismIssue {
66                    line: line_num,
67                    pattern_type: NonDeterministicPattern::Random,
68                    code: line.to_string(),
69                    explanation:
70                        "Uses $RANDOM which produces different values on each run".to_string(),
71                });
72            }
73
74            // Detect $$ (process ID)
75            if line.contains("$$") {
76                self.detections.push(DeterminismIssue {
77                    line: line_num,
78                    pattern_type: NonDeterministicPattern::ProcessId,
79                    code: line.to_string(),
80                    explanation: "Uses $$ (process ID) which changes on each execution"
81                        .to_string(),
82                });
83            }
84
85            // Detect timestamps: date, $(date), `date`, etc.
86            if line.contains("date")
87                && (line.contains("$(date")
88                    || line.contains("`date")
89                    || line.contains("date +")
90                    || line.contains("date -"))
91            {
92                self.detections.push(DeterminismIssue {
93                    line: line_num,
94                    pattern_type: NonDeterministicPattern::Timestamp,
95                    code: line.to_string(),
96                    explanation: "Uses date command which produces different values over time"
97                        .to_string(),
98                });
99            }
100
101            // Detect $BASHPID
102            if line.contains("$BASHPID") || line.contains("${BASHPID}") {
103                self.detections.push(DeterminismIssue {
104                    line: line_num,
105                    pattern_type: NonDeterministicPattern::BashPid,
106                    code: line.to_string(),
107                    explanation: "Uses $BASHPID which changes on each execution".to_string(),
108                });
109            }
110
111            // Detect $SRANDOM (Bash 5.1+)
112            if line.contains("$SRANDOM") || line.contains("${SRANDOM}") {
113                self.detections.push(DeterminismIssue {
114                    line: line_num,
115                    pattern_type: NonDeterministicPattern::SecureRandom,
116                    code: line.to_string(),
117                    explanation:
118                        "Uses $SRANDOM (cryptographic random) which produces different values on each run"
119                            .to_string(),
120                });
121            }
122        }
123
124        self.detections.clone()
125    }
126
127    /// Check if script is deterministic (no issues found)
128    pub fn is_deterministic(&self) -> bool {
129        self.detections.is_empty()
130    }
131
132    /// Get count of issues by pattern type
133    pub fn count_by_pattern(&self, pattern: NonDeterministicPattern) -> usize {
134        self.detections
135            .iter()
136            .filter(|issue| issue.pattern_type == pattern)
137            .count()
138    }
139}

144    }

392/// Verifies determinism by running scripts multiple times and comparing outputs
393#[derive(Debug, Clone)]
394pub struct ReplayVerifier {
395    /// Number of replay runs to perform (default: 2, min: 2)
396    replay_count: usize,
397}
398
399/// Result of replay verification
400#[derive(Debug, Clone, PartialEq)]
401pub struct ReplayResult {
402    /// Whether all runs produced identical output
403    pub is_deterministic: bool,
404    /// Outputs from each run
405    pub runs: Vec<RunOutput>,
406    /// Lines that differ between runs (if non-deterministic)
407    pub differences: Vec<OutputDifference>,
408}
409
410/// Output from a single script execution
411#[derive(Debug, Clone, PartialEq)]
412pub struct RunOutput {
413    /// Run number (1-indexed)
414    pub run_number: usize,
415    /// Standard output
416    pub stdout: String,
417    /// Standard error
418    pub stderr: String,
419    /// Exit code
420    pub exit_code: i32,
421}
422
423/// Difference between two runs
424#[derive(Debug, Clone, PartialEq)]
425pub struct OutputDifference {
426    /// Line number where difference occurs (1-indexed)
427    pub line: usize,
428    /// Output from first run
429    pub run1: String,
430    /// Output from second run
431    pub run2: String,
432}
433
434impl ReplayVerifier {
435    /// Create a new replay verifier with default settings
436    pub fn new() -> Self {
437        Self { replay_count: 2 }
438    }
439
440    /// Set number of replay runs (min: 2)
441    pub fn with_replay_count(mut self, count: usize) -> Self {
442        self.replay_count = count.max(2);
443        self
444    }
445
446    /// Verify determinism by running script multiple times
447    pub fn verify(&self, script: &str) -> ReplayResult {
448        let mut runs = Vec::new();
449
450        // Run script multiple times
451        for run_number in 1..=self.replay_count {
452            let mut output = Self::execute_script(script);
453            output.run_number = run_number;
454            runs.push(output);
455        }
456
457        // Compare outputs between first two runs
458        let differences = if runs.len() >= 2 {
459            if let (Some(run0), Some(run1)) = (runs.first(), runs.get(1)) {
460                Self::find_differences(run0, run1)
461            } else {
462                Vec::new()
463            }
464        } else {
465            Vec::new()
466        };
467
468        let is_deterministic = differences.is_empty();
469
470        ReplayResult {
471            is_deterministic,
472            runs,
473            differences,
474        }
475    }
476
477    /// Execute bash script and capture output
478    fn execute_script(script: &str) -> RunOutput {
479        use std::process::Command;
480
481        match Command::new("bash").arg("-c").arg(script).output() {
482            Ok(output) => RunOutput {
483                run_number: 0, // Will be set by caller
484                stdout: String::from_utf8_lossy(&output.stdout).to_string(),
485                stderr: String::from_utf8_lossy(&output.stderr).to_string(),
486                exit_code: output.status.code().unwrap_or(-1),
487            },
488            Err(e) => RunOutput {
489                run_number: 0,
490                stdout: String::new(),
491                stderr: format!("Failed to execute bash script: {}", e),
492                exit_code: -1,
493            },
494        }
495    }
496
497    /// Find differences between two runs
498    fn find_differences(run1: &RunOutput, run2: &RunOutput) -> Vec<OutputDifference> {
499        let mut differences = Vec::new();
500
501        // Compare stdout line by line
502        let lines1: Vec<&str> = run1.stdout.lines().collect();
503        let lines2: Vec<&str> = run2.stdout.lines().collect();
504
505        let max_lines = lines1.len().max(lines2.len());
506        for i in 0..max_lines {
507            let line1 = lines1.get(i).unwrap_or(&"");
508            let line2 = lines2.get(i).unwrap_or(&"");
509
510            if line1 != line2 {
511                differences.push(OutputDifference {
512                    line: i + 1, // 1-indexed
513                    run1: line1.to_string(),
514                    run2: line2.to_string(),
515                });
516            }
517        }
518
519        differences
520    }
521}
522
523impl Default for ReplayVerifier {
524    fn default() -> Self {
525        Self::new()
526    }

531/// Format replay verification differences for display
532///
533/// Takes a vector of OutputDifference and formats them in a human-friendly way.
534/// Shows line numbers and the actual output from each run.
535///
536/// # Examples
537///
538/// ```
539/// use bashrs::repl::determinism::OutputDifference;
540/// use bashrs::repl::determinism::format_replay_diff;
541///
542/// let differences = vec![
543///     OutputDifference {
544///         line: 1,
545///         run1: "Random: 12345".to_string(),
546///         run2: "Random: 67890".to_string(),
547///     },
548/// ];
549///
550/// let formatted = format_replay_diff(&differences);
551/// assert!(formatted.contains("Line 1:"));
552/// ```
553pub fn format_replay_diff(differences: &[OutputDifference]) -> String {
554    if differences.is_empty() {
555        return String::from(" No differences detected - script is deterministic");
556    }
557
558    let mut output = String::new();
559    output.push_str(" Non-deterministic output detected!\n\n");
560    output.push_str(&format!("Found {} difference(s):\n\n", differences.len()));
561
562    for diff in differences {
563        output.push_str(&format!("Line {}:\n", diff.line));
564        output.push_str(&format!("  Run 1: {}\n", diff.run1));
565        output.push_str(&format!("  Run 2: {}\n", diff.run2));
566        output.push('\n');
567    }
568
569    output
570}
571
572impl ReplayResult {
573    /// Format replay result for display
574    ///
575    /// Shows determinism status, run count, exit codes, and any differences.
576    pub fn format_result(&self) -> String {
577        let mut output = String::new();
578
579        // Show determinism status
580        if self.is_deterministic {
581            output.push_str(" Script is deterministic\n");
582        } else {
583            output.push_str(" Script is non-deterministic\n");
584        }
585
586        // Show run count
587        output.push_str(&format!("\nRuns: {}\n", self.runs.len()));
588
589        // Show exit codes
590        output.push_str("Exit codes: ");
591        for run in &self.runs {
592            output.push_str(&format!("{} ", run.exit_code));
593        }
594        output.push('\n');
595
596        // Show differences if any
597        if !self.differences.is_empty() {
598            output.push('\n');
599            output.push_str(&format_replay_diff(&self.differences));
600        }
601
602        output
603    }
604}

1034// ===== IDEMPOTENCY CHECKER (REPL-012-001) =====
1035
1036// Idempotency checking for bash scripts
1037//
1038// Detects non-idempotent operations that may fail on re-run:
1039// - mkdir without -p: Fails if directory exists
1040// - rm without -f: Fails if file doesn't exist
1041// - ln -s without -f: Fails if symlink exists
1042
1043/// Detects non-idempotent operations in bash scripts
1044#[derive(Debug, Clone, PartialEq)]
1045pub struct IdempotencyChecker {
1046    /// Issues detected in the script
1047    detections: Vec<IdempotencyIssue>,
1048}
1049
1050/// A single non-idempotent operation detection
1051#[derive(Debug, Clone, PartialEq)]
1052pub struct IdempotencyIssue {
1053    /// Line number where operation was found (1-indexed)
1054    pub line: usize,
1055    /// Type of non-idempotent operation
1056    pub operation_type: NonIdempotentOperation,
1057    /// The actual code that triggered the detection
1058    pub code: String,
1059    /// Human-readable explanation
1060    pub explanation: String,
1061    /// Suggested fix
1062    pub suggestion: String,
1063}
1064
1065/// Types of non-idempotent operations
1066#[derive(Debug, Clone, Copy, PartialEq, Eq)]
1067pub enum NonIdempotentOperation {
1068    /// mkdir without -p flag
1069    MkdirWithoutP,
1070    /// rm without -f flag
1071    RmWithoutF,
1072    /// ln -s without -f flag
1073    LnWithoutF,
1074}
1075
1076impl IdempotencyChecker {
1077    /// Create a new idempotency checker
1078    pub fn new() -> Self {
1079        Self {
1080            detections: Vec::new(),
1081        }
1082    }
1083
1084    /// Scan bash script for non-idempotent operations
1085    ///
1086    /// Returns: List of detected issues
1087    pub fn scan(&mut self, script: &str) -> Vec<IdempotencyIssue> {
1088        // Clear previous results
1089        self.detections.clear();
1090
1091        for (line_num, line) in script.lines().enumerate() {
1092            let line_num = line_num + 1; // 1-indexed
1093
1094            // Detect mkdir without -p
1095            if line.contains("mkdir ") && !line.contains("mkdir -p") {
1096                self.detections.push(IdempotencyIssue {
1097                    line: line_num,
1098                    operation_type: NonIdempotentOperation::MkdirWithoutP,
1099                    code: line.to_string(),
1100                    explanation: "mkdir without -p fails if directory already exists".to_string(),
1101                    suggestion: "Add -p flag: mkdir -p".to_string(),
1102                });
1103            }
1104
1105            // Detect rm without -f (but not rm -rf)
1106            if line.contains("rm ") && !line.contains(" -f") && !line.contains(" -rf") {
1107                self.detections.push(IdempotencyIssue {
1108                    line: line_num,
1109                    operation_type: NonIdempotentOperation::RmWithoutF,
1110                    code: line.to_string(),
1111                    explanation: "rm without -f fails if file doesn't exist".to_string(),
1112                    suggestion: "Add -f flag: rm -f".to_string(),
1113                });
1114            }
1115
1116            // Detect ln -s without -f
1117            if line.contains("ln -s") && !line.contains("ln -sf") {
1118                self.detections.push(IdempotencyIssue {
1119                    line: line_num,
1120                    operation_type: NonIdempotentOperation::LnWithoutF,
1121                    code: line.to_string(),
1122                    explanation: "ln -s without -f fails if symlink already exists".to_string(),
1123                    suggestion: "Add -f flag: ln -sf".to_string(),
1124                });
1125            }
1126        }
1127
1128        self.detections.clone()
1129    }
1130
1131    /// Check if script is idempotent (no issues found)
1132    pub fn is_idempotent(&self) -> bool {
1133        self.detections.is_empty()
1134    }
1135
1136    /// Get count of issues by operation type
1137    pub fn count_by_operation(&self, operation: NonIdempotentOperation) -> usize {
1138        self.detections
1139            .iter()
1140            .filter(|issue| issue.operation_type == operation)
1141            .count()
1142    }
1143}
1144
1145impl Default for IdempotencyChecker {
1146    fn default() -> Self {
1147        Self::new()
1148    }
1149}

1434    ///
1435    /// Shows idempotency status, issue counts, operation breakdown,
1436    /// and detailed issue reports.
1437    ///
1438    /// # Examples
1439    ///
1440    /// ```
1441    /// use bashrs::repl::IdempotencyChecker;

1453        // Show idempotency status
1454        if self.is_idempotent() {
1455            output.push_str(" Script is idempotent\n");
1456            output.push_str("  Safe to run multiple times\n");
1457        } else {
1458            output.push_str("  Script is non-idempotent\n");
1459            output.push_str(&format!("  {} issue(s) found\n", self.detections.len()));
1460        }
1461
1462        // Show issue breakdown by type
1463        let mkdir_count = self.count_by_operation(NonIdempotentOperation::MkdirWithoutP);
1464        let rm_count = self.count_by_operation(NonIdempotentOperation::RmWithoutF);
1465        let ln_count = self.count_by_operation(NonIdempotentOperation::LnWithoutF);
1466
1467        if mkdir_count > 0 || rm_count > 0 || ln_count > 0 {
1468            output.push_str("\nOperation breakdown:\n");
1469            if mkdir_count > 0 {
1470                output.push_str(&format!("  mkdir (missing -p): {}\n", mkdir_count));
1471            }
1472            if rm_count > 0 {
1473                output.push_str(&format!("  rm (missing -f):    {}\n", rm_count));
1474            }
1475            if ln_count > 0 {
1476                output.push_str(&format!("  ln (missing -f):    {}\n", ln_count));
1477            }
1478        }
1479
1480        // Show detailed issues
1481        if !self.detections.is_empty() {
1482            output.push('\n');
1483            output.push_str(&format_idempotency_report(&self.detections));
1484        }
1485
1486        output
1487    }

1711/// Verifies script idempotency by running multiple times and comparing results
1712///
1713/// Runs a script N times (default: 3) and checks if all runs produce
1714/// identical output and exit codes.
1715#[derive(Debug, Clone)]
1716pub struct IdempotencyVerifier {
1717    /// Number of verification runs to perform (default: 3, min: 2)
1718    run_count: usize,
1719}
1720
1721impl IdempotencyVerifier {
1722    /// Create a new idempotency verifier with default settings (3 runs)
1723    pub fn new() -> Self {
1724        Self { run_count: 3 }
1725    }
1726
1727    /// Create verifier with custom run count (minimum 2)
1728    pub fn with_run_count(count: usize) -> Self {
1729        Self {
1730            run_count: count.max(2),
1731        }
1732    }
1733
1734    /// Verify script idempotency by running multiple times
1735    pub fn verify(&self, script: &str) -> IdempotencyResult {
1736        let mut runs = Vec::new();
1737
1738        // Run script multiple times
1739        for run_number in 1..=self.run_count {
1740            let mut output = ReplayVerifier::execute_script(script);
1741            output.run_number = run_number;
1742            runs.push(output);
1743        }
1744
1745        // Compare all runs against the first run
1746        let mut differences = Vec::new();
1747        if let Some(first_run) = runs.first() {
1748            // Compare each subsequent run with the first run
1749            for run in runs.iter().skip(1) {
1750                let run_diffs = ReplayVerifier::find_differences(first_run, run);
1751                // Merge differences (avoid duplicates)
1752                for diff in run_diffs {
1753                    if !differences.iter().any(|d: &OutputDifference| {
1754                        d.line == diff.line && d.run1 == diff.run1 && d.run2 == diff.run2
1755                    }) {
1756                        differences.push(diff);
1757                    }
1758                }
1759            }
1760        }
1761
1762        let is_idempotent = differences.is_empty();
1763
1764        IdempotencyResult {
1765            is_idempotent,
1766            run_count: self.run_count,
1767            runs,
1768            differences,
1769        }
1770    }
1771}
1772
1773impl Default for IdempotencyVerifier {
1774    fn default() -> Self {
1775        Self::new()
1776    }
1777}

1779/// Result of idempotency verification
1780#[derive(Debug, Clone, PartialEq)]
1781pub struct IdempotencyResult {
1782    /// Whether all runs produced identical output
1783    pub is_idempotent: bool,
1784    /// Number of runs performed
1785    pub run_count: usize,
1786    /// Outputs from each run
1787    pub runs: Vec<RunOutput>,
1788    /// Lines that differ between runs (if non-idempotent)
1789    pub differences: Vec<OutputDifference>,
1790}
1791
1792impl IdempotencyResult {
1793    /// Format verification result for display
1794    pub fn format_result(&self) -> String {
1795        let mut output = String::new();
1796
1797        // Show idempotency status
1798        if self.is_idempotent {
1799            output.push_str(" Script is idempotent\n");
1800            output.push_str(&format!("  Verified across {} runs\n", self.run_count));
1801        } else {
1802            output.push_str(" Script is non-idempotent\n");
1803            output.push_str(&format!("  Tested {} runs\n", self.run_count));
1804        }
1805
1806        // Show run count
1807        output.push_str(&format!("\nRuns: {}\n", self.runs.len()));
1808
1809        // Show exit codes
1810        output.push_str("Exit codes: ");
1811        for run in &self.runs {
1812            output.push_str(&format!("{} ", run.exit_code));
1813        }
1814        output.push('\n');
1815
1816        // Show differences if any
1817        if !self.differences.is_empty() {
1818            output.push('\n');
1819            output.push_str(&format_replay_diff(&self.differences));
1820        }
1821
1822        output
1823    }
