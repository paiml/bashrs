11use crate::linter::{lint_shell, LintResult, Severity};

13/// Lint bash input and return diagnostics
14///
15/// # Examples
16///
17/// ```
18/// use bashrs::repl::linter::lint_bash;
19///
20/// let result = lint_bash("cat file.txt | grep pattern");
21/// assert!(result.is_ok());
22/// ```
23pub fn lint_bash(input: &str) -> anyhow::Result<LintResult> {
24    let result = lint_shell(input);
25    Ok(result)
26}
27
28/// Format lint results for display in REPL
29pub fn format_lint_results(result: &LintResult) -> String {
30    let mut output = String::new();
31
32    if result.diagnostics.is_empty() {
33        output.push_str("‚úì No issues found!\n");
34        return output;
35    }
36
37    // Count by severity
38    let errors = result
39        .diagnostics
40        .iter()
41        .filter(|d| d.severity == Severity::Error)
42        .count();
43    let warnings = result
44        .diagnostics
45        .iter()
46        .filter(|d| d.severity == Severity::Warning)
47        .count();
48    let info = result
49        .diagnostics
50        .iter()
51        .filter(|d| d.severity == Severity::Info)
52        .count();
53
54    output.push_str(&format!("Found {} issue(s):\n", result.diagnostics.len()));
55    if errors > 0 {
56        output.push_str(&format!("  ‚úó {} error(s)\n", errors));
57    }
58    if warnings > 0 {
59        output.push_str(&format!("  ‚ö† {} warning(s)\n", warnings));
60    }
61    if info > 0 {
62        output.push_str(&format!("  ‚Ñπ {} info\n", info));
63    }
64
65    output.push('\n');
66
67    // Show diagnostics
68    for (i, diag) in result.diagnostics.iter().enumerate() {
69        let severity_icon = match diag.severity {
70            Severity::Error => "‚úó",
71            Severity::Warning => "‚ö†",
72            Severity::Info => "‚Ñπ",
73            Severity::Note => "üìù",
74            Severity::Perf => "‚ö°",
75            Severity::Risk => "‚ö†",
76        };
77
78        output.push_str(&format!(
79            "[{}] {} {} - {}\n",
80            i + 1,
81            severity_icon,
82            diag.code,
83            diag.message
84        ));
85
86        if diag.span.start_line > 0 {
87            output.push_str(&format!("    Line {}\n", diag.span.start_line));
88        }
89    }
90
91    output
92}
