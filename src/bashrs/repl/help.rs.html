11/// Show help for the specified topic
12///
13/// # Arguments
14/// * `topic` - Optional help topic. If None, shows general help.
15///
16/// # Examples
17///
18/// ```
19/// use bashrs::repl::help::show_help;
20///
21/// // General help
22/// let help = show_help(None);
23/// assert!(help.contains("bashrs REPL"));
24///
25/// // Specific topic
26/// let help = show_help(Some("purify"));
27/// assert!(help.contains("purification"));
28/// ```
29pub fn show_help(topic: Option<&str>) -> String {
30    match topic {
31        None => show_general_help(),
32        Some("commands") | Some("command") => show_commands_help(),
33        Some("modes") | Some("mode") => show_modes_help(),
34        Some("purify") => show_purify_help(),
35        Some("lint") => show_lint_help(),
36        Some("parse") => show_parse_help(),
37        Some("explain") => show_explain_help(),
38        Some("debug") => show_debug_help(),
39        Some("history") => show_history_help(),
40        Some("variables") | Some("vars") => show_variables_help(),
41        Some("shortcuts") | Some("keys") => show_shortcuts_help(),
42        Some(unknown) => format!(
43            "Unknown help topic: '{}'\n\nAvailable topics:\n\
44             - commands   - List all REPL commands\n\
45             - modes      - Explain REPL modes\n\
46             - purify     - Learn about bash purification\n\
47             - lint       - Learn about linting\n\
48             - parse      - Learn about parsing\n\
49             - explain    - Learn about explanations\n\
50             - history    - Learn about history features\n\
51             - variables  - Learn about session variables\n\
52             - shortcuts  - Learn about keyboard shortcuts\n\n\
53             Try: :help <topic>",
54            unknown
55        ),
56    }
57}
