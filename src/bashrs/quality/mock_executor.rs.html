1//! Mock Command Executor for Testing Quality Gates
2//!
3//! This module provides infrastructure for mocking external command execution
4//! to enable unit testing of quality gate logic without running actual commands.
5//!
6//! # Design
7//!
8//! Uses trait-based dependency injection to allow:
9//! - Real execution in production (`RealCommandExecutor`)
10//! - Mock execution in tests (`MockCommandExecutor`)
11//!
12//! # Example
13//!
14//! ```ignore
15//! let mut mock = MockCommandExecutor::new();
16//! mock.register("cargo", &["clippy"], CommandResult::success("Clippy passed"));
17//! let result = mock.execute("cargo", &["clippy"]);
18//! assert!(result.success);
19//! ```
20
21use std::collections::HashMap;
22use std::process::{Command, Output};
23use std::sync::{Arc, Mutex};
24
25/// Result of command execution
26#[derive(Debug, Clone)]
27pub struct CommandResult {
28    /// Whether the command succeeded (exit code 0)
29    pub success: bool,
30    /// Exit code (0 for success)
31    pub exit_code: i32,
32    /// Standard output
33    pub stdout: String,
34    /// Standard error
35    pub stderr: String,
36}
37
38impl CommandResult {
39    /// Create a successful result
40    pub fn success(stdout: &str) -> Self {
41        Self {
42            success: true,
43            exit_code: 0,
44            stdout: stdout.to_string(),
45            stderr: String::new(),
46        }
47    }
48
49    /// Create a failed result
50    pub fn failure(stderr: &str, exit_code: i32) -> Self {
51        Self {
52            success: false,
53            exit_code,
54            stdout: String::new(),
55            stderr: stderr.to_string(),
56        }
57    }
58
59    /// Create a result with both stdout and stderr
60    pub fn with_output(success: bool, stdout: &str, stderr: &str) -> Self {
61        Self {
62            success,
63            exit_code: if success { 0 } else { 1 },
64            stdout: stdout.to_string(),
65            stderr: stderr.to_string(),
66        }
67    }
68}
69
70impl From<Output> for CommandResult {
71    fn from(output: Output) -> Self {
72        Self {
73            success: output.status.success(),
74            exit_code: output.status.code().unwrap_or(1),
75            stdout: String::from_utf8_lossy(&output.stdout).to_string(),
76            stderr: String::from_utf8_lossy(&output.stderr).to_string(),
77        }
78    }
79}

81/// Trait for executing external commands
82///
83/// Implementations can execute real commands or return mock results.
84pub trait CommandExecutor: Send + Sync {
85    /// Execute a command with arguments
86    fn execute(&self, program: &str, args: &[&str]) -> Result<CommandResult, String>;
87}
88
89/// Real command executor using std::process::Command
90#[derive(Debug, Default)]
91pub struct RealCommandExecutor;
92
93impl CommandExecutor for RealCommandExecutor {
94    fn execute(&self, program: &str, args: &[&str]) -> Result<CommandResult, String> {
95        Command::new(program)
96            .args(args)
97            .output()
98            .map(CommandResult::from)
99            .map_err(|e| format!("Failed to execute {}: {}", program, e))
100    }
101}

103/// Key for identifying a command (program + args combination)
104#[derive(Debug, Clone, PartialEq, Eq, Hash)]
105pub struct CommandKey {
106    pub program: String,
107    pub args: Vec<String>,
108}
109
110impl CommandKey {
111    pub fn new(program: &str, args: &[&str]) -> Self {
112        Self {
113            program: program.to_string(),
114            args: args.iter().map(|s| s.to_string()).collect(),
115        }
116    }
117}

119/// Mock command executor for testing
120///
121/// Allows registering expected commands and their results.
122#[derive(Debug, Default, Clone)]
123pub struct MockCommandExecutor {
124    /// Registered command results
125    results: Arc<Mutex<HashMap<CommandKey, CommandResult>>>,
126    /// Fallback result for unregistered commands
127    fallback: Option<CommandResult>,
128    /// Track executed commands for verification
129    executions: Arc<Mutex<Vec<CommandKey>>>,
130}
131
132impl MockCommandExecutor {
133    /// Create a new mock executor
134    pub fn new() -> Self {
135        Self::default()
136    }
137
138    /// Create a mock executor with a fallback result
139    pub fn with_fallback(result: CommandResult) -> Self {
140        Self {
141            fallback: Some(result),
142            ..Default::default()
143        }
144    }
145
146    /// Register an expected command and its result
147    pub fn register(&mut self, program: &str, args: &[&str], result: CommandResult) {
148        let key = CommandKey::new(program, args);
149        if let Ok(mut guard) = self.results.lock() {
150            guard.insert(key, result);
151        }
152    }
153
154    /// Register a command that matches any args for the program
155    pub fn register_program(&mut self, program: &str, result: CommandResult) {
156        // Register with empty args as a wildcard
157        let key = CommandKey::new(program, &[]);
158        if let Ok(mut guard) = self.results.lock() {
159            guard.insert(key, result);
160        }
161    }
162
163    /// Get all executed commands (for test verification)
164    pub fn get_executions(&self) -> Vec<CommandKey> {
165        self.executions
166            .lock()
167            .map(|guard| guard.clone())
168            .unwrap_or_default()
169    }
170
171    /// Verify a command was executed
172    pub fn was_executed(&self, program: &str, args: &[&str]) -> bool {
173        let key = CommandKey::new(program, args);
174        self.executions
175            .lock()
176            .map(|guard| guard.contains(&key))
177            .unwrap_or(false)
178    }
179}
180
181impl CommandExecutor for MockCommandExecutor {
182    fn execute(&self, program: &str, args: &[&str]) -> Result<CommandResult, String> {
183        let key = CommandKey::new(program, args);
184
185        // Record execution
186        if let Ok(mut guard) = self.executions.lock() {
187            guard.push(key.clone());
188        }
189
190        // Try exact match first
191        let results = match self.results.lock() {
192            Ok(guard) => guard,
193            Err(_) => return Err("Failed to acquire lock".to_string()),
194        };
195        if let Some(result) = results.get(&key) {
196            return Ok(result.clone());
197        }
198
199        // Try program-only match (wildcard)
200        let program_key = CommandKey::new(program, &[]);
201        if let Some(result) = results.get(&program_key) {
202            return Ok(result.clone());
203        }
204
205        // Use fallback if available
206        if let Some(ref fallback) = self.fallback {
207            return Ok(fallback.clone());
208        }
209
210        Err(format!("No mock registered for: {} {:?}", program, args))
211    }
212}

218    #[test]
219    fn test_command_result_success() {
220        let result = CommandResult::success("output");
221        assert!(result.success);
222        assert_eq!(result.exit_code, 0);
223        assert_eq!(result.stdout, "output");
224        assert!(result.stderr.is_empty());
225    }
226
227    #[test]
228    fn test_command_result_failure() {
229        let result = CommandResult::failure("error message", 1);
230        assert!(!result.success);
231        assert_eq!(result.exit_code, 1);
232        assert!(result.stdout.is_empty());
233        assert_eq!(result.stderr, "error message");
234    }
235
236    #[test]
237    fn test_command_result_with_output() {
238        let result = CommandResult::with_output(true, "stdout", "stderr");
239        assert!(result.success);
240        assert_eq!(result.stdout, "stdout");
241        assert_eq!(result.stderr, "stderr");
242    }

244    #[test]
245    fn test_mock_executor_register_and_execute() {
246        let mut mock = MockCommandExecutor::new();
247        mock.register("echo", &["hello"], CommandResult::success("hello\n"));
248
249        let result = mock.execute("echo", &["hello"]).expect("should succeed");
250        assert!(result.success);

270    #[test]
271    fn test_mock_executor_execution_tracking() {
272        let mock = MockCommandExecutor::with_fallback(CommandResult::success("ok"));
273
274        mock.execute("cmd1", &["a", "b"]).expect("should succeed");
275        mock.execute("cmd2", &["c"]).expect("should succeed");
276
277        assert!(mock.was_executed("cmd1", &["a", "b"]));
278        assert!(mock.was_executed("cmd2", &["c"]));
279        assert!(!mock.was_executed("cmd3", &[]));
280    }

284        let mut mock = MockCommandExecutor::new();

299        let key3 = CommandKey::new("cargo", &["build"]);
