20    pub fn validate_ast(&self, ast: &RestrictedAst) -> RashResult<()> {
21        if self.level == ValidationLevel::None {
22            return Ok(());
23        }
24
25        // Validate all functions
26        for function in &ast.functions {
27            // Validate function name is not a shell builtin
28            self.validate_function_name(&function.name)?;
29
30            // Validate function body
31            for stmt in &function.body {
32                self.validate_stmt(stmt)?;
33            }
34        }
35        Ok(())
36    }

38    pub fn validate_ir(&self, ir: &ShellIR) -> RashResult<()> {
39        if self.level == ValidationLevel::None {
40            return Ok(());
41        }
42
43        self.validate_ir_recursive(ir)
44    }

46    pub fn validate_output(&self, _shell_script: &str) -> RashResult<()> {
47        if self.level == ValidationLevel::None {
48            return Ok(());
49        }
50
51        #[cfg(debug_assertions)]
52        self.verify_with_embedded_rules(_shell_script)?;
53
54        Ok(())
55    }

57    fn validate_stmt(&self, stmt: &crate::ast::Stmt) -> RashResult<()> {
58        use crate::ast::Stmt;
59
60        match stmt {
61            Stmt::Let { value, .. } => {
62                self.validate_expr(value)?;
63            }
64            Stmt::Expr(expr) => {
65                self.validate_expr(expr)?;
66            }
67            Stmt::If {
68                condition,
69                then_block,
70                else_block,
71            } => {
72                self.validate_expr(condition)?;
73                for stmt in then_block {
74                    self.validate_stmt(stmt)?;
75                }
76                if let Some(else_stmts) = else_block {
77                    for stmt in else_stmts {
78                        self.validate_stmt(stmt)?;
79                    }
80                }
81            }
82            _ => {}
83        }
84        Ok(())
85    }

87    fn validate_expr(&self, expr: &crate::ast::Expr) -> RashResult<()> {
88        use crate::ast::Expr;
89
90        match expr {
91            Expr::Literal(lit) => self.validate_literal(lit),
92            Expr::Variable(name) => self.validate_variable_name(name),
93            Expr::Binary { left, right, .. } => self.validate_binary_expr(left, right),
94            Expr::Unary { operand, .. } => self.validate_expr(operand),
95            Expr::FunctionCall { name, args } => self.validate_function_call(name, args),
96            Expr::MethodCall {
97                receiver,
98                method,
99                args,
100            } => self.validate_method_call(receiver, method, args),
101            Expr::Array(items) => self.validate_array_items(items),
102            Expr::Index { object, index } => self.validate_index_expr(object, index),
103            Expr::Try { expr } => self.validate_expr(expr),
104            Expr::Block(stmts) => self.validate_block_statements(stmts),
105            Expr::Range { start, end, .. } => {
106                self.validate_expr(start)?;
107                self.validate_expr(end)
108            }
109        }
110    }

122    fn validate_string_literal(&self, s: &str) -> RashResult<()> {
123        if self.level == ValidationLevel::None {
124            return Ok(());
125        }
126
127        // Special handling for shellshock-style attacks
128        if s.contains("() { :; }") {
129            return Err(RashError::ValidationError(
130                "Shellshock-style function definition detected in string literal".to_string(),
131            ));
132        }
133
134        // Check for command injection patterns
135        let dangerous_patterns = [
136            (
137                "; ",
138                "Semicolon command separator detected in string literal",
139            ),
140            ("| ", "Pipe operator detected in string literal"),
141            ("$(", "Command substitution detected in string literal"),
142            (
143                "`",
144                "Backtick command substitution detected in string literal (SC2006)",
145            ),
146            ("&& ", "AND operator detected in string literal"),
147            ("|| ", "OR operator detected in string literal"),
148            (
149                "'; ",
150                "Quote escape with semicolon detected in string literal",
151            ),
152            (
153                "\"; ",
154                "Quote escape with semicolon detected in string literal",
155            ),
156            ("<<", "Here-doc syntax detected in string literal"),
157            ("eval ", "eval command detected in string literal"),
158            ("exec ", "exec command detected in string literal"),
159        ];
160
161        for (pattern, message) in &dangerous_patterns {
162            if s.contains(pattern) {
163                return Err(RashError::ValidationError(format!(
164                    "{}: '{}'",
165                    message,
166                    s.chars().take(50).collect::<String>()
167                )));
168            }
169        }
170
171        // Check for newlines/carriage returns followed by dangerous commands
172        if s.contains('\n') || s.contains('\r') {
173            let lines: Vec<&str> = s.split(&['\n', '\r'][..]).collect();
174            for line in lines {
175                let trimmed = line.trim();
176                // Check if any line starts with a shell command that could be dangerous
177                let dangerous_starts = ["rm ", "curl ", "wget ", "eval ", "exec ", "bash ", "sh "];
178                for start in &dangerous_starts {
179                    if trimmed.starts_with(start) {
180                        return Err(RashError::ValidationError(format!(
181                            "Newline followed by dangerous command detected: '{}'",
182                            trimmed.chars().take(50).collect::<String>()
183                        )));
184                    }
185                }
186            }
187        }
188
189        Ok(())
190    }

192    fn validate_function_name(&self, name: &str) -> RashResult<()> {
193        if name.is_empty() {
194            return Err(RashError::ValidationError(
195                "Empty function name".to_string(),
196            ));
197        }
198
199        // Check for reserved shell builtins that cannot be redefined
200        // These are POSIX special builtins and common builtins that cause errors
201        let reserved_builtins = [
202            "break", "continue", "exit", "return", "shift", "trap", "unset", "export", "readonly",
203            "set", "times", "exec", "eval", ".", ":", "true", "false", "test", "[",
204        ];
205
206        if reserved_builtins.contains(&name) {
207            return Err(RashError::ValidationError(format!(
208                "Function name '{}' is a reserved shell builtin and cannot be redefined",
209                name
210            )));
211        }
212
213        Ok(())
214    }

216    fn validate_variable_name(&self, name: &str) -> RashResult<()> {
217        if name.is_empty() {
218            return Err(RashError::ValidationError(
219                "Empty variable name".to_string(),
220            ));
221        }
222        if name.contains(char::is_whitespace) {
223            return Err(RashError::ValidationError(format!(
224                "Variable name '{name}' contains whitespace"
225            )));
226        }
227        Ok(())
228    }

292    fn validate_ir_recursive(&self, ir: &ShellIR) -> RashResult<()> {
293        match ir {
294            ShellIR::Let { value, .. } => {
295                self.validate_shell_value(value)?;
296            }
297            ShellIR::Exec { cmd, .. } => {
298                for arg in &cmd.args {
299                    self.validate_shell_value(arg)?;
300                }
301            }
302            ShellIR::If {
303                test,
304                then_branch,
305                else_branch,
306            } => {
307                self.validate_shell_value(test)?;
308                self.validate_ir_recursive(then_branch)?;
309                if let Some(else_b) = else_branch {
310                    self.validate_ir_recursive(else_b)?;
311                }
312            }
313            ShellIR::Sequence(irs) => {
314                for ir in irs {
315                    self.validate_ir_recursive(ir)?;
316                }
317            }
318            ShellIR::Function { body, .. } => {
319                self.validate_ir_recursive(body)?;
320            }
321            ShellIR::Exit { .. } | ShellIR::Noop => {}
322            ShellIR::Echo { value } => {
323                self.validate_shell_value(value)?;
324            }
325            ShellIR::For {
326                var,
327                start,
328                end,
329                body,
330            } => {
331                // Validate variable name
332                if var.is_empty() {
333                    return Err(RashError::ValidationError(
334                        "For loop variable name cannot be empty".to_string(),
335                    ));
336                }
337                // Validate range values
338                self.validate_shell_value(start)?;
339                self.validate_shell_value(end)?;
340                // Validate body
341                self.validate_ir_recursive(body)?;
342            }
343            ShellIR::While { condition, body } => {
344                // Validate condition
345                self.validate_shell_value(condition)?;
346                // Validate body
347                self.validate_ir_recursive(body)?;
348            }
349            ShellIR::Case { scrutinee, arms } => {
350                // Validate scrutinee
351                self.validate_shell_value(scrutinee)?;
352
353                // Validate each arm
354                for arm in arms {
355                    if let Some(guard) = &arm.guard {
356                        self.validate_shell_value(guard)?;
357                    }
358                    self.validate_ir_recursive(&arm.body)?;
359                }
360
361                // Check for at least one arm
362                if arms.is_empty() {
363                    return Err(RashError::ValidationError(
364                        "Match expression must have at least one arm".to_string(),
365                    ));
366                }
367            }
368            ShellIR::Break | ShellIR::Continue => {
369                // Break and continue are always valid in IR
370                // Context validation (must be inside loop) happens during AST validation
371            }
372        }
373        Ok(())
374    }

376    pub(crate) fn validate_shell_value(&self, value: &crate::ir::ShellValue) -> RashResult<()> {
377        use crate::ir::ShellValue;
378
379        match value {
380            ShellValue::Variable(_name) => {
381                if self.level >= ValidationLevel::Minimal {
382                    // Variables should generally be quoted in shell
383                    // This is a simplified check - real implementation would check context
384                }
385            }
386            ShellValue::CommandSubst(cmd) => {
387                if cmd.program.contains('`') && self.level >= ValidationLevel::Minimal {
388                    return Err(RashError::ValidationError(
389                        "Use $(...) instead of backticks (SC2006)".to_string(),
390                    ));
391                }
392            }
393            ShellValue::Concat(parts) => {
394                for part in parts {
395                    self.validate_shell_value(part)?;
396                }
397            }
398            _ => {}
399        }
400        Ok(())
401    }

403    #[allow(dead_code)]
404    pub(crate) fn validate_expression(&self, expr: &crate::ir::ShellExpression) -> RashResult<()> {
405        use crate::ir::ShellExpression;
406
407        match expr {
408            ShellExpression::Variable(name, quoted) => {
409                if !quoted && self.level >= ValidationLevel::Minimal {
410                    return Err(RashError::ValidationError(format!(
411                        "Unquoted variable ${name} (SC2086)"
412                    )));
413                }
414            }
415            ShellExpression::Command(cmd) => {
416                if cmd.contains('`') && self.level >= ValidationLevel::Minimal {
417                    return Err(RashError::ValidationError(
418                        "Use $(...) instead of backticks (SC2006)".to_string(),
419                    ));
420                }
421            }
422            ShellExpression::String(_) => {}
423            ShellExpression::Arithmetic(_) => {}
424        }
425        Ok(())
426    }

433    pub fn report_error(&self, error: &ValidationError) -> String {
434        if self.strict_mode && error.severity == super::Severity::Error {
435            format!("ERROR: {error}")
436        } else {
437            format!("{error}")

441    pub fn should_fail(&self, errors: &[ValidationError]) -> bool {
442        if self.strict_mode {
443            !errors.is_empty()
444        } else {
445            errors.iter().any(|e| e.severity == super::Severity::Error)
446        }
