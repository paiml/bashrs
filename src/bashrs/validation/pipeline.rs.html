20    pub fn validate_ast(&self, ast: &RestrictedAst) -> RashResult<()> {
21        if self.level == ValidationLevel::None {
22            return Ok(());
23        }
24
25        // Validate all functions
26        for function in &ast.functions {
27            // Validate function name is not a shell builtin
28            self.validate_function_name(&function.name)?;
29
30            // Validate function body
31            for stmt in &function.body {
32                self.validate_stmt(stmt)?;
33            }
34        }
35        Ok(())
36    }

38    pub fn validate_ir(&self, ir: &ShellIR) -> RashResult<()> {
39        if self.level == ValidationLevel::None {
40            return Ok(());
41        }
42
43        self.validate_ir_recursive(ir)
44    }

46    pub fn validate_output(&self, _shell_script: &str) -> RashResult<()> {
47        if self.level == ValidationLevel::None {
48            return Ok(());
49        }
50
51        #[cfg(debug_assertions)]
52        self.verify_with_embedded_rules(_shell_script)?;
53
54        Ok(())
55    }

57    fn validate_stmt(&self, stmt: &crate::ast::Stmt) -> RashResult<()> {
58        use crate::ast::Stmt;
59
60        match stmt {
61            Stmt::Let { value, .. } => {
62                self.validate_expr(value)?;
63            }
64            Stmt::Expr(expr) => {
65                self.validate_expr(expr)?;
66            }
67            Stmt::If {
68                condition,
69                then_block,
70                else_block,
71            } => {
72                self.validate_expr(condition)?;
73                for stmt in then_block {
74                    self.validate_stmt(stmt)?;
75                }
76                if let Some(else_stmts) = else_block {
77                    for stmt in else_stmts {
78                        self.validate_stmt(stmt)?;
79                    }
80                }
81            }
82            _ => {}
83        }
84        Ok(())
85    }

87    fn validate_expr(&self, expr: &crate::ast::Expr) -> RashResult<()> {
88        use crate::ast::Expr;
89
90        match expr {
91            Expr::Literal(lit) => self.validate_literal(lit),
92            Expr::Variable(name) => self.validate_variable_name(name),
93            Expr::Binary { left, right, .. } => self.validate_binary_expr(left, right),
94            Expr::Unary { operand, .. } => self.validate_expr(operand),
95            Expr::FunctionCall { name, args } => self.validate_function_call(name, args),
96            Expr::MethodCall {
97                receiver,
98                method,
99                args,
100            } => self.validate_method_call(receiver, method, args),
101            Expr::Array(items) => self.validate_array_items(items),
102            Expr::Index { object, index } => self.validate_index_expr(object, index),
103            Expr::Try { expr } => self.validate_expr(expr),
104            Expr::Block(stmts) => self.validate_block_statements(stmts),
105            Expr::Range { start, end, .. } => {
106                self.validate_expr(start)?;
107                self.validate_expr(end)
108            }
109            Expr::PositionalArgs => {
110                // Positional arguments are valid - will be handled by emitter
111                Ok(())
112            }
113        }
114    }

117    fn validate_literal(&self, lit: &crate::ast::restricted::Literal) -> RashResult<()> {
118        use crate::ast::restricted::Literal;
119
120        match lit {
121            Literal::Str(s) => self.validate_string_literal(s),
122            Literal::Bool(_) | Literal::U32(_) | Literal::I32(_) => Ok(()),
123        }

126    fn validate_string_literal(&self, s: &str) -> RashResult<()> {
127        if self.level == ValidationLevel::None {
128            return Ok(());
129        }
130
131        // Special handling for shellshock-style attacks
132        if s.contains("() { :; }") {
133            return Err(RashError::ValidationError(
134                "Shellshock-style function definition detected in string literal".to_string(),
135            ));
136        }
137
138        // Issue #94: Skip pipe check for table/formatting strings
139        // Strings that look like table borders (multiple |) are not command injection
140        let is_formatting_string = s
141            .chars()
142            .filter(|c| *c == '|')
143            .count() > 1
144            && !s.contains(";")
145            && !s.contains("$(")
146            && !s.contains("&&");
147
148        // Check for command injection patterns
149        let dangerous_patterns = [
150            (
151                "; ",
152                "Semicolon command separator detected in string literal",
153            ),
154            ("$(", "Command substitution detected in string literal"),
155            (
156                "`",
157                "Backtick command substitution detected in string literal (SC2006)",
158            ),
159            ("&& ", "AND operator detected in string literal"),
160            ("|| ", "OR operator detected in string literal"),
161            (
162                "'; ",
163                "Quote escape with semicolon detected in string literal",
164            ),
165            (
166                "\"; ",
167                "Quote escape with semicolon detected in string literal",
168            ),
169            ("<<", "Here-doc syntax detected in string literal"),
170            ("eval ", "eval command detected in string literal"),
171            ("exec ", "exec command detected in string literal"),
172        ];
173
174        for (pattern, message) in &dangerous_patterns {
175            if s.contains(pattern) {
176                return Err(RashError::ValidationError(format!(
177                    "{}: '{}'",
178                    message,
179                    s.chars().take(50).collect::<String>()
180                )));
181            }
182        }
183
184        // Issue #94: Only check for pipe operator if not a formatting string
185        // Pipes in table formatting (like "| col1 | col2 |") are not command injection
186        if !is_formatting_string && s.contains("| ") {
187            // Check if this looks like a command pipe (e.g., "| cmd" pattern)
188            // Skip if it's at the start or end (likely table border)
189            let trimmed = s.trim();
190            if !trimmed.starts_with('|') && !trimmed.ends_with('|') {
191                // Only flag if there's a command-like word after the pipe
192                if let Some(pos) = s.find("| ") {
193                    let after_pipe = &s[pos + 2..].trim_start();
194                    // Check if what follows looks like a command (alphanumeric word)
195                    if !after_pipe.is_empty()
196                        && after_pipe.chars().next().is_some_and(|c| c.is_alphabetic())
197                    {
198                        return Err(RashError::ValidationError(format!(
199                            "Pipe operator detected in string literal: '{}'",
200                            s.chars().take(50).collect::<String>()
201                        )));
202                    }
203                }
204            }
205        }
206
207        // Check for newlines/carriage returns followed by dangerous commands
208        if s.contains('\n') || s.contains('\r') {
209            let lines: Vec<&str> = s.split(&['\n', '\r'][..]).collect();
210            for line in lines {
211                let trimmed = line.trim();
212                // Check if any line starts with a shell command that could be dangerous
213                let dangerous_starts = ["rm ", "curl ", "wget ", "eval ", "exec ", "bash ", "sh "];
214                for start in &dangerous_starts {
215                    if trimmed.starts_with(start) {
216                        return Err(RashError::ValidationError(format!(
217                            "Newline followed by dangerous command detected: '{}'",
218                            trimmed.chars().take(50).collect::<String>()
219                        )));
220                    }
221                }
222            }
223        }
224
225        Ok(())
226    }

228    fn validate_function_name(&self, name: &str) -> RashResult<()> {
229        if name.is_empty() {
230            return Err(RashError::ValidationError(
231                "Empty function name".to_string(),
232            ));
233        }
234
235        // Check for reserved shell builtins that cannot be redefined
236        // These are POSIX special builtins and common builtins that cause errors
237        let reserved_builtins = [
238            "break", "continue", "exit", "return", "shift", "trap", "unset", "export", "readonly",
239            "set", "times", "exec", "eval", ".", ":", "true", "false", "test", "[",
240        ];
241
242        if reserved_builtins.contains(&name) {
243            return Err(RashError::ValidationError(format!(
244                "Function name '{}' is a reserved shell builtin and cannot be redefined",
245                name
246            )));
247        }
248
249        Ok(())
250    }

252    fn validate_variable_name(&self, name: &str) -> RashResult<()> {
253        if name.is_empty() {
254            return Err(RashError::ValidationError(
255                "Empty variable name".to_string(),
256            ));
257        }
258        if name.contains(char::is_whitespace) {
259            return Err(RashError::ValidationError(format!(
260                "Variable name '{name}' contains whitespace"
261            )));
262        }
263        Ok(())
264    }

328    fn validate_ir_recursive(&self, ir: &ShellIR) -> RashResult<()> {
329        match ir {
330            ShellIR::Let { value, .. } => {
331                self.validate_shell_value(value)?;
332            }
333            ShellIR::Exec { cmd, .. } => {
334                for arg in &cmd.args {
335                    self.validate_shell_value(arg)?;
336                }
337            }
338            ShellIR::If {
339                test,
340                then_branch,
341                else_branch,
342            } => {
343                self.validate_shell_value(test)?;
344                self.validate_ir_recursive(then_branch)?;
345                if let Some(else_b) = else_branch {
346                    self.validate_ir_recursive(else_b)?;
347                }
348            }
349            ShellIR::Sequence(irs) => {
350                for ir in irs {
351                    self.validate_ir_recursive(ir)?;
352                }
353            }
354            ShellIR::Function { body, .. } => {
355                self.validate_ir_recursive(body)?;
356            }
357            ShellIR::Exit { .. } | ShellIR::Noop => {}
358            ShellIR::Echo { value } => {
359                self.validate_shell_value(value)?;
360            }
361            ShellIR::For {
362                var,
363                start,
364                end,
365                body,
366            } => {
367                // Validate variable name
368                if var.is_empty() {
369                    return Err(RashError::ValidationError(
370                        "For loop variable name cannot be empty".to_string(),
371                    ));
372                }
373                // Validate range values
374                self.validate_shell_value(start)?;
375                self.validate_shell_value(end)?;
376                // Validate body
377                self.validate_ir_recursive(body)?;
378            }
379            ShellIR::While { condition, body } => {
380                // Validate condition
381                self.validate_shell_value(condition)?;
382                // Validate body
383                self.validate_ir_recursive(body)?;
384            }
385            ShellIR::Case { scrutinee, arms } => {
386                // Validate scrutinee
387                self.validate_shell_value(scrutinee)?;
388
389                // Validate each arm
390                for arm in arms {
391                    if let Some(guard) = &arm.guard {
392                        self.validate_shell_value(guard)?;
393                    }
394                    self.validate_ir_recursive(&arm.body)?;
395                }
396
397                // Check for at least one arm
398                if arms.is_empty() {
399                    return Err(RashError::ValidationError(
400                        "Match expression must have at least one arm".to_string(),
401                    ));
402                }
403            }
404            ShellIR::Break | ShellIR::Continue => {
405                // Break and continue are always valid in IR
406                // Context validation (must be inside loop) happens during AST validation
407            }
408        }
409        Ok(())
410    }

412    pub(crate) fn validate_shell_value(&self, value: &crate::ir::ShellValue) -> RashResult<()> {
413        use crate::ir::ShellValue;
414
415        match value {
416            ShellValue::Variable(_name) => {
417                if self.level >= ValidationLevel::Minimal {
418                    // Variables should generally be quoted in shell
419                    // This is a simplified check - real implementation would check context
420                }
421            }
422            ShellValue::CommandSubst(cmd) => {
423                if cmd.program.contains('`') && self.level >= ValidationLevel::Minimal {
424                    return Err(RashError::ValidationError(
425                        "Use $(...) instead of backticks (SC2006)".to_string(),
426                    ));
427                }
428            }
429            ShellValue::Concat(parts) => {
430                for part in parts {
431                    self.validate_shell_value(part)?;
432                }
433            }
434            _ => {}
435        }
436        Ok(())
437    }

440    pub(crate) fn validate_expression(&self, expr: &crate::ir::ShellExpression) -> RashResult<()> {
441        use crate::ir::ShellExpression;
442
443        match expr {
444            ShellExpression::Variable(name, quoted) => {
445                if !quoted && self.level >= ValidationLevel::Minimal {
446                    return Err(RashError::ValidationError(format!(
447                        "Unquoted variable ${name} (SC2086)"
448                    )));
449                }
450            }
451            ShellExpression::Command(cmd) => {
452                if cmd.contains('`') && self.level >= ValidationLevel::Minimal {
453                    return Err(RashError::ValidationError(
454                        "Use $(...) instead of backticks (SC2006)".to_string(),
455                    ));
456                }
457            }
458            ShellExpression::String(_) => {}
459            ShellExpression::Arithmetic(_) => {}
460        }
461        Ok(())
462    }

469    pub fn report_error(&self, error: &ValidationError) -> String {
470        if self.strict_mode && error.severity == super::Severity::Error {
471            format!("ERROR: {error}")
472        } else {
473            format!("{error}")
474        }
475    }
476
477    pub fn should_fail(&self, errors: &[ValidationError]) -> bool {
478        if self.strict_mode {
479            !errors.is_empty()
480        } else {
481            errors.iter().any(|e| e.severity == super::Severity::Error)
482        }
483    }
