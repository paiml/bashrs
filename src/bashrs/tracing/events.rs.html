1// Trace Events - Core data structures for bashrs tracing
2//
3// Task: TRACE-001-B - TraceEvent enum and sub-events
4// Source: bashrs-tracing-while-purify-lint-spec.md (Section 3.1)
5// Optimization: bashrs-tracing-architectural-refinements.md (Refinement 5 - Diff-based storage)
6//
7// Design Philosophy:
8// - Memory-efficient: Store nodes by reference, diffs as patches (99.25% savings)
9// - Semantic information: Full AST awareness (not text-based like shellcheck)
10// - Significance-aware: Every event has importance ranking
11//
12// Performance Target: <10% overhead (OOPSLA2 2024 standard)

18/// Unique identifier for AST nodes (cheap to copy, 8 bytes)
19///
20/// Optimization: Instead of cloning full AST nodes (can be >10 KB),
21/// we store a reference ID and reconstruct from registry on demand.
22pub type AstNodeId = u64;
23
24/// Unique identifier for transformations (for dependency tracking)
25pub type TransformationId = u64;
26
27/// Rule identifier (e.g., IDEM001, DET003, SEC002)
28pub type RuleId = String;

30/// Source code span (line:column range)
31#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
32pub struct Span {
33    pub line_start: usize,
34    pub col_start: usize,
35    pub line_end: usize,
36    pub col_end: usize,
37}
38
39impl Span {
40    #[must_use]
41    pub const fn new(line_start: usize, col_start: usize, line_end: usize, col_end: usize) -> Self {
42        Self {
43            line_start,
44            col_start,
45            line_end,
46            col_end,
47        }
48    }
49
50    #[must_use]
51    pub const fn single_line(line: usize, col_start: usize, col_end: usize) -> Self {
52        Self::new(line, col_start, line, col_end)
53    }
54}

56/// Minimal diff representation (not full clone)
57///
58/// Optimization: Instead of storing full node_before + node_after,
59/// store only the diff. Typical size: <100 bytes vs 20 KB.
60///
61/// Memory savings: 99.25% (20 KB → 150 bytes per event)
62#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
63pub enum AstNodePatch {
64    /// Added a flag to command (e.g., mkdir → mkdir -p)
65    AddedFlag { flag: String },
66
67    /// Removed a flag from command
68    RemovedFlag { flag: String },
69
70    /// Replaced an argument at index
71    ReplacedArgument {
72        index: usize,
73        old: String,
74        new: String,
75    },
76
77    /// Replaced an expression
78    ReplacedExpression { old_expr: String, new_expr: String },
79
80    /// Added quotes around variable (e.g., $foo → "$foo")
81    AddedQuotes { variable: String },
82
83    /// Removed $RANDOM variable
84    RemovedRandomVar,
85
86    /// Replaced timestamp with fixed value
87    ReplacedTimestamp {
88        old_pattern: String,
89        new_value: String,
90    },
91
92    /// Generic transformation (fallback)
93    Generic { description: String },
94}

96/// Core trace event enum
97///
98/// Every event during bash parsing, purification, linting, and generation
99/// is recorded as one of these events.
100#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
101pub enum TraceEvent {
102    // ===== PARSING EVENTS =====
103    Parse(ParseEvent),
104
105    // ===== PURIFICATION EVENTS =====
106    Purify(PurifyEvent),
107
108    // ===== LINTING EVENTS =====
109    Lint(LintEvent),
110
111    // ===== GENERATION EVENTS =====
112    Generate(GenerateEvent),
113}
114
115impl TraceEvent {
116    /// Calculate significance of this trace event
117    ///
118    /// Implements Refinement 1 (Trace Significance) from architectural spec.
119    /// Higher significance events bubble to the top in UI.
120    #[must_use]
121    pub fn significance(&self) -> TraceSignificance {
122        match self {
123            // CRITICAL: Conflicts and security violations
124            Self::Purify(PurifyEvent::TransformationConflict { .. }) => TraceSignificance::Critical,
125            Self::Lint(LintEvent::RuleEvaluated {
126                violation: Some(v), ..
127            }) if v.is_security => TraceSignificance::Critical,
128
129            // HIGH: Transformations applied, parse/generation errors
130            Self::Purify(PurifyEvent::TransformationApplied { .. }) => TraceSignificance::High,
131            Self::Parse(ParseEvent::ParseError { .. }) => TraceSignificance::High,
132            Self::Generate(GenerateEvent::GenerateError { .. }) => TraceSignificance::High,
133
134            // MEDIUM: Transformations skipped, rule evaluations
135            Self::Purify(PurifyEvent::TransformationSkipped { .. }) => TraceSignificance::Medium,
136            Self::Lint(LintEvent::RuleEvaluated { .. }) => TraceSignificance::Medium,
137
138            // LOW: Parse nodes, generation steps
139            Self::Parse(ParseEvent::ParseNode { .. }) => TraceSignificance::Low,
140            Self::Generate(GenerateEvent::GenerateCode { .. }) => TraceSignificance::Low,
141
142            // TRACE: Start/complete events (structural only)
143            Self::Parse(ParseEvent::ParseStart { .. }) => TraceSignificance::Trace,
144            Self::Parse(ParseEvent::ParseComplete { .. }) => TraceSignificance::Trace,
145            Self::Purify(PurifyEvent::PurifyStart { .. }) => TraceSignificance::Trace,
146            Self::Purify(PurifyEvent::PurifyComplete { .. }) => TraceSignificance::Trace,
147            Self::Lint(LintEvent::LintStart { .. }) => TraceSignificance::Trace,
148            Self::Lint(LintEvent::LintComplete { .. }) => TraceSignificance::Trace,
149            Self::Generate(GenerateEvent::GenerateStart { .. }) => TraceSignificance::Trace,
150            Self::Generate(GenerateEvent::GenerateComplete { .. }) => TraceSignificance::Trace,
151        }
152    }
153
154    /// Get a human-readable description of this event
155    #[must_use]
156    pub fn description(&self) -> String {
157        match self {
158            Self::Parse(e) => e.description(),
159            Self::Purify(e) => e.description(),
160            Self::Lint(e) => e.description(),
161            Self::Generate(e) => e.description(),
162        }
163    }
164}

166/// Parsing events
167#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
168pub enum ParseEvent {
169    ParseStart {
170        source: String,
171        line: usize,
172        col: usize,
173    },
174    ParseNode {
175        node_type: String,
176        span: Span,
177    },
178    ParseComplete {
179        node_count: usize,
180        duration: Duration,
181    },
182    ParseError {
183        error: String,
184        span: Span,
185    },
186}
187
188impl ParseEvent {
189    #[must_use]
190    pub fn description(&self) -> String {
191        match self {
192            Self::ParseStart { source, .. } => format!("Parse started: {source}"),
193            Self::ParseNode { node_type, span } => {
194                format!(
195                    "Parsed {node_type} at {}:{}",
196                    span.line_start, span.col_start
197                )
198            }
199            Self::ParseComplete {
200                node_count,
201                duration,
202            } => format!("Parse complete: {node_count} nodes in {duration:?}"),
203            Self::ParseError { error, span } => {
204                format!(
205                    "Parse error at {}:{}: {error}",
206                    span.line_start, span.col_start
207                )
208            }
209        }
210    }
211}

213/// Purification events
214#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
215pub enum PurifyEvent {
216    PurifyStart {
217        node_count: usize,
218    },
219    TransformationApplied {
220        id: TransformationId,
221        rule_id: RuleId,
222        node_id: AstNodeId,
223        patch: AstNodePatch,
224        reason: String,
225        span: Span,
226    },
227    TransformationSkipped {
228        rule_id: RuleId,
229        node_id: AstNodeId,
230        reason: String,
231        span: Span,
232    },
233    TransformationConflict {
234        id1: TransformationId,
235        rule1: RuleId,
236        id2: TransformationId,
237        rule2: RuleId,
238        node_id: AstNodeId,
239        resolution: String,
240        span: Span,
241    },
242    PurifyComplete {
243        transformations_applied: usize,
244        transformations_skipped: usize,
245        conflicts: usize,
246        duration: Duration,
247    },
248}
249
250impl PurifyEvent {
251    #[must_use]
252    pub fn description(&self) -> String {
253        match self {
254            Self::PurifyStart { node_count } => format!("Purify started: {node_count} nodes"),
255            Self::TransformationApplied {
256                rule_id, reason, ..
257            } => format!("Applied {rule_id}: {reason}"),
258            Self::TransformationSkipped {
259                rule_id, reason, ..
260            } => format!("Skipped {rule_id}: {reason}"),
261            Self::TransformationConflict {
262                rule1, rule2, resolution, ..
263            } => format!("Conflict: {rule1} vs {rule2} → {resolution}"),
264            Self::PurifyComplete {
265                transformations_applied,
266                transformations_skipped,
267                conflicts,
268                duration,
269            } => format!(
270                "Purify complete: {transformations_applied} applied, {transformations_skipped} skipped, {conflicts} conflicts in {duration:?}"
271            ),
272        }
273    }
274}

276/// Linting events
277#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
278pub enum LintEvent {
279    LintStart {
280        node_count: usize,
281    },
282    RuleEvaluated {
283        rule_id: RuleId,
284        node_id: AstNodeId,
285        passed: bool,
286        violation: Option<Violation>,
287        span: Span,
288    },
289    LintComplete {
290        rules_evaluated: usize,
291        violations: usize,
292        duration: Duration,
293    },
294}
295
296impl LintEvent {
297    #[must_use]
298    pub fn description(&self) -> String {
299        match self {
300            Self::LintStart { node_count } => format!("Lint started: {node_count} nodes"),
301            Self::RuleEvaluated {
302                rule_id,
303                passed,
304                violation,
305                ..
306            } => {
307                if *passed {
308                    format!("{rule_id}: passed")
309                } else if let Some(v) = violation {
310                    format!("{rule_id}: {}", v.message)
311                } else {
312                    format!("{rule_id}: failed")
313                }
314            }
315            Self::LintComplete {
316                rules_evaluated,
317                violations,
318                duration,
319            } => format!(
320                "Lint complete: {rules_evaluated} rules, {violations} violations in {duration:?}"
321            ),
322        }
323    }
324}
