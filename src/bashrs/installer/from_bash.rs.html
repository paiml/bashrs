1//! Convert legacy bash scripts to installer.toml format (#115)
2#![allow(clippy::indexing_slicing)] // Safe: bounds checked in while loop
3//!
4//! This module implements `bashrs installer from-bash` which converts
5//! existing bash installation scripts to the declarative installer.toml format.
6//!
7//! # Handled Patterns
8//!
9//! - Array syntax (#103) → converted to TOML lists
10//! - Case statements (#99) → converted to step conditions
11//! - Heredocs (#96) → converted to template files
12//! - sudo patterns (#100, #101) → converted to privileged actions
13//! - inline if/then (#93) → converted to step preconditions
14//!
15//! # Example
16//!
17//! ```bash
18//! # Input: install.sh
19//! if [ "$EUID" -ne 0 ]; then echo "Run as root"; exit 1; fi
20//! apt-get update
21//! apt-get install -y docker-ce
22//! ```
23//!
24//! ```toml
25//! # Output: installer.toml
26//! [installer.requirements]
27//! privileges = "root"
28//!
29//! [[step]]
30//! id = "update-packages"
31//! action = "apt-update"
32//!
33//! [[step]]
34//! id = "install-docker"
35//! action = "apt-install"
36//! packages = ["docker-ce"]
37//! ```
38

42/// Result of converting a bash script to installer format
43#[derive(Debug, Clone)]
44pub struct ConversionResult {
45    /// Generated installer.toml content
46    pub installer_toml: String,
47    /// Template files extracted from heredocs
48    pub templates: Vec<TemplateFile>,
49    /// Warnings about patterns that couldn't be converted
50    pub warnings: Vec<String>,
51    /// Statistics about the conversion
52    pub stats: ConversionStats,
53}

55/// A template file extracted from a heredoc
56#[derive(Debug, Clone)]
57pub struct TemplateFile {
58    /// Filename for the template
59    pub name: String,
60    /// Content of the template
61    pub content: String,
62}

64/// Statistics about the conversion process
65#[derive(Debug, Clone, Default)]
66pub struct ConversionStats {
67    /// Number of steps generated
68    pub steps_generated: usize,
69    /// Number of apt-install commands found
70    pub apt_installs: usize,
71    /// Number of heredocs converted to templates
72    pub heredocs_converted: usize,
73    /// Number of sudo patterns converted
74    pub sudo_patterns: usize,
75    /// Number of conditionals converted to preconditions
76    pub conditionals_converted: usize,
77}

79/// Detected bash pattern that can be converted
80#[derive(Debug, Clone, PartialEq)]
81pub(crate) enum BashPattern {
82    /// Root check: if [ "$EUID" -ne 0 ]
83    RootCheck,
84    /// apt-get update
85    AptUpdate,
86    /// apt-get install -y packages...
87    AptInstall { packages: Vec<String> },
88    /// mkdir -p directory
89    MkdirP { path: String },
90    /// curl/wget download
91    Download { url: String, output: Option<String> },
92    /// Heredoc content
93    Heredoc { delimiter: String, content: String },
94    /// sudo command
95    SudoCommand { command: String },
96    /// Generic script line
97    Script { content: String },
98}
99
100/// Convert a bash script to installer.toml format
101///
102/// # Arguments
103/// * `script` - The bash script content
104/// * `name` - Name for the installer
105///
106/// # Returns
107/// * `ConversionResult` with the generated installer.toml and any templates
108pub fn convert_bash_to_installer(script: &str, name: &str) -> Result<ConversionResult> {
109    let patterns = extract_patterns(script)?;
110    let (toml, templates, stats) = generate_installer_toml(&patterns, name)?;
111
112    let warnings = generate_warnings(&patterns);
113
114    Ok(ConversionResult {
115        installer_toml: toml,
116        templates,
117        warnings,
118        stats,
119    })
120}

122/// Extract recognizable patterns from a bash script
123fn extract_patterns(script: &str) -> Result<Vec<BashPattern>> {
124    let mut patterns = Vec::new();
125    let lines: Vec<&str> = script.lines().collect();
126    let mut i = 0;
127
128    while i < lines.len() {
129        let line = lines[i].trim();
130
131        // Skip empty lines and comments
132        if line.is_empty() || line.starts_with('#') {
133            i += 1;
134            continue;
135        }
136
137        // Check for root/EUID check
138        if line.contains("EUID") && line.contains("-ne 0") {
139            patterns.push(BashPattern::RootCheck);
140            i += 1;
141            continue;
142        }
143
144        // Check for apt-get update
145        if line.contains("apt-get update") || line.contains("apt update") {
146            patterns.push(BashPattern::AptUpdate);
147            i += 1;
148            continue;
149        }
150
151        // Check for apt-get install
152        if let Some(packages) = parse_apt_install(line) {
153            patterns.push(BashPattern::AptInstall { packages });
154            i += 1;
155            continue;
156        }
157
158        // Check for mkdir -p
159        if let Some(path) = parse_mkdir_p(line) {
160            patterns.push(BashPattern::MkdirP { path });
161            i += 1;
162            continue;
163        }
164
165        // Check for curl/wget download
166        if let Some((url, output)) = parse_download(line) {
167            patterns.push(BashPattern::Download { url, output });
168            i += 1;
169            continue;
170        }
171
172        // Check for heredoc
173        if let Some((delimiter, content, lines_consumed)) = parse_heredoc(&lines, i) {
174            patterns.push(BashPattern::Heredoc { delimiter, content });
175            i += lines_consumed;
176            continue;
177        }
178
179        // Check for sudo command
180        if let Some(command) = parse_sudo(line) {
181            patterns.push(BashPattern::SudoCommand { command });
182            i += 1;
183            continue;
184        }
185
186        // Default: treat as generic script line
187        patterns.push(BashPattern::Script {
188            content: line.to_string(),
189        });
190        i += 1;
191    }
192
193    Ok(patterns)
194}

196/// Parse apt-get install command and extract packages
197fn parse_apt_install(line: &str) -> Option<Vec<String>> {
198    // Match: apt-get install -y pkg1 pkg2 ... or apt install -y pkg1 pkg2
199    let line = line.trim();
200
201    // Remove sudo prefix if present
202    let line = line.strip_prefix("sudo ").unwrap_or(line);
203
204    if line.starts_with("apt-get install") || line.starts_with("apt install") {
205        let parts: Vec<&str> = line.split_whitespace().collect();
206        let packages: Vec<String> = parts
207            .iter()
208            .skip(2) // Skip "apt-get" and "install" or "apt" and "install"
209            .filter(|p| !p.starts_with('-')) // Skip flags like -y
210            .map(|p| p.to_string())
211            .collect();
212
213        if !packages.is_empty() {
214            return Some(packages);
215        }
216    }
217
218    None
219}

221/// Parse mkdir -p command
222fn parse_mkdir_p(line: &str) -> Option<String> {
223    let line = line.strip_prefix("sudo ").unwrap_or(line);
224
225    if line.starts_with("mkdir -p ") {
226        let path = line.strip_prefix("mkdir -p ")?.trim();
227        return Some(path.to_string());
228    }
229
230    None
231}

233/// Parse curl/wget download command
234fn parse_download(line: &str) -> Option<(String, Option<String>)> {
235    let line = line.strip_prefix("sudo ").unwrap_or(line);
236
237    // curl -fsSL URL -o OUTPUT or curl -fsSL URL > OUTPUT
238    if line.starts_with("curl ") {
239        // Extract URL (simplified - looks for http/https)
240        let parts: Vec<&str> = line.split_whitespace().collect();
241        for (i, part) in parts.iter().enumerate() {
242            if part.starts_with("http://") || part.starts_with("https://") {
243                let url = part.to_string();
244                // Check for -o flag
245                let output = parts
246                    .get(i + 2)
247                    .filter(|_| parts.get(i + 1) == Some(&"-o"))
248                    .map(|s| s.to_string());
249                return Some((url, output));
250            }
251        }
252    }
253
254    // wget URL -O OUTPUT
255    if line.starts_with("wget ") {
256        let parts: Vec<&str> = line.split_whitespace().collect();
257        for (i, part) in parts.iter().enumerate() {
258            if part.starts_with("http://") || part.starts_with("https://") {
259                let url = part.to_string();
260                let output = parts
261                    .get(i + 2)
262                    .filter(|_| parts.get(i + 1) == Some(&"-O"))
263                    .map(|s| s.to_string());
264                return Some((url, output));
265            }
266        }
267    }
268
269    None
270}

272/// Parse heredoc and extract content
273fn parse_heredoc(lines: &[&str], start: usize) -> Option<(String, String, usize)> {
274    let line = lines[start].trim();
275
276    // Match: cat << EOF or cat << 'EOF' or cat <<- EOF
277    if !line.contains("<<") {
278        return None;
279    }
280
281    // Extract delimiter
282    let after_heredoc = line.split("<<").nth(1)?;
283    let delimiter = after_heredoc
284        .trim()
285        .trim_start_matches('-')
286        .trim()
287        .trim_matches('\'')
288        .trim_matches('"')
289        .split_whitespace()
290        .next()?
291        .to_string();
292
293    // Collect content until we find the delimiter
294    let mut content = String::new();
295    let mut lines_consumed = 1;
296
297    for line in lines.iter().skip(start + 1) {
298        lines_consumed += 1;
299        if line.trim() == delimiter {
300            break;
301        }
302        content.push_str(line);
303        content.push('\n');
304    }
305
306    Some((delimiter, content, lines_consumed))
307}

309/// Parse sudo command
310fn parse_sudo(line: &str) -> Option<String> {
311    if line.starts_with("sudo ") {
312        let command = line.strip_prefix("sudo ")?.to_string();
313        // Don't return if it's already handled by other parsers
314        if !command.starts_with("apt") && !command.starts_with("mkdir") {
315            return Some(command);
316        }
317    }
318    None
319}

321/// Generate installer.toml from extracted patterns
322fn generate_installer_toml(
323    patterns: &[BashPattern],
324    name: &str,
325) -> Result<(String, Vec<TemplateFile>, ConversionStats)> {
326    let mut toml = String::new();
327    let mut templates = Vec::new();
328    let mut stats = ConversionStats::default();
329    let mut step_id = 0;
330
331    // Check for root requirement
332    let requires_root = patterns.iter().any(|p| matches!(p, BashPattern::RootCheck));
333
334    // Generate installer header
335    toml.push_str(&format!(
336        r#"# Installer specification converted from bash script
337# Generated by bashrs installer from-bash
338
339[installer]
340name = "{name}"
341version = "1.0.0"
342description = "Converted from legacy bash script"
343
344[installer.requirements]
345privileges = "{}"
346
347"#,
348        if requires_root { "root" } else { "user" }
349    ));
350
351    // Generate steps
352    for pattern in patterns {
353        match pattern {
354            BashPattern::RootCheck => {
355                // Already handled in requirements
356                stats.conditionals_converted += 1;
357            }
358            BashPattern::AptUpdate => {
359                step_id += 1;
360                stats.steps_generated += 1;
361                toml.push_str(&format!(
362                    r#"[[step]]
363id = "step-{step_id}-apt-update"
364name = "Update Package Lists"
365action = "script"
366
367[step.script]
368interpreter = "sh"
369content = "apt-get update"
370
371"#
372                ));
373            }
374            BashPattern::AptInstall { packages } => {
375                step_id += 1;
376                stats.steps_generated += 1;
377                stats.apt_installs += 1;
378                let packages_list = packages
379                    .iter()
380                    .map(|p| format!("\"{}\"", p))
381                    .collect::<Vec<_>>()
382                    .join(", ");
383                toml.push_str(&format!(
384                    r#"[[step]]
385id = "step-{step_id}-install"
386name = "Install Packages"
387action = "script"
388
389[step.script]
390interpreter = "sh"
391content = "apt-get install -y {}"
392
393# Packages: [{}]
394
395"#,
396                    packages.join(" "),
397                    packages_list
398                ));
399            }
400            BashPattern::MkdirP { path } => {
401                step_id += 1;
402                stats.steps_generated += 1;
403                toml.push_str(&format!(
404                    r#"[[step]]
405id = "step-{step_id}-mkdir"
406name = "Create Directory"
407action = "script"
408
409[step.script]
410interpreter = "sh"
411content = "mkdir -p {path}"
412
413"#
414                ));
415            }
416            BashPattern::Download { url, output } => {
417                step_id += 1;
418                stats.steps_generated += 1;
419                let output_str = output.as_deref().unwrap_or("downloaded-file");
420                toml.push_str(&format!(
421                    r#"[[step]]
422id = "step-{step_id}-download"
423name = "Download File"
424action = "script"
425
426[step.script]
427interpreter = "sh"
428content = "curl -fsSL {url} -o {output_str}"
429
430"#
431                ));
432            }
433            BashPattern::Heredoc { delimiter, content } => {
434                step_id += 1;
435                stats.steps_generated += 1;
436                stats.heredocs_converted += 1;
437                let template_name = format!("template-{}.txt", step_id);
438                templates.push(TemplateFile {
439                    name: template_name.clone(),
440                    content: content.clone(),
441                });
442                toml.push_str(&format!(
443                    r#"[[step]]
444id = "step-{step_id}-heredoc"
445name = "Write Template File"
446action = "script"
447
448# Original heredoc delimiter: {delimiter}
449# Template extracted to: templates/{template_name}
450
451[step.script]
452interpreter = "sh"
453content = "cat templates/{template_name}"
454
455"#
456                ));
457            }
458            BashPattern::SudoCommand { command } => {
459                step_id += 1;
460                stats.steps_generated += 1;
461                stats.sudo_patterns += 1;
462                toml.push_str(&format!(
463                    r#"[[step]]
464id = "step-{step_id}-sudo"
465name = "Execute Privileged Command"
466action = "script"
467
468[step.script]
469interpreter = "sh"
470content = "{command}"
471
472[step.checkpoint]
473enabled = true
474
475"#
476                ));
477            }
478            BashPattern::Script { content } => {
479                step_id += 1;
480                stats.steps_generated += 1;
481                toml.push_str(&format!(
482                    r#"[[step]]
483id = "step-{step_id}-script"
484name = "Execute Script"
485action = "script"
486
487[step.script]
488interpreter = "sh"
489content = "{content}"
490
491"#
492                ));
493            }
494        }
495    }
496
497    Ok((toml, templates, stats))
498}

500/// Generate warnings for patterns that might need manual review
501fn generate_warnings(patterns: &[BashPattern]) -> Vec<String> {
502    let mut warnings = Vec::new();
503
504    for pattern in patterns {
505        match pattern {
506            BashPattern::SudoCommand { command } => {
507                warnings.push(format!("Sudo command may need manual review: {}", command));
508            }
509            BashPattern::Script { content } => {
510                if content.contains("eval") {
511                    warnings.push(format!(
512                        "eval usage detected - potential security risk: {}",
513                        content
514                    ));
515                }
516                if content.contains("$RANDOM") || content.contains("$$") {
517                    warnings.push(format!("Non-deterministic pattern detected: {}", content));
518                }
519            }
520            _ => {}
521        }
522    }
523
524    warnings
525}

527/// Convert a bash script file to installer project
528pub fn convert_file_to_project(input: &Path, output_dir: &Path) -> Result<ConversionResult> {
529    // Read the bash script
530    let script = std::fs::read_to_string(input).map_err(|e| {
531        Error::Io(std::io::Error::new(
532            e.kind(),
533            format!("Failed to read {}: {}", input.display(), e),
534        ))
535    })?;
536
537    // Get project name from output directory
538    let name = output_dir
539        .file_name()
540        .and_then(|n| n.to_str())
541        .unwrap_or("converted-installer");
542
543    // Convert the script
544    let result = convert_bash_to_installer(&script, name)?;
545
546    // Create output directory structure
547    std::fs::create_dir_all(output_dir).map_err(|e| {
548        Error::Io(std::io::Error::new(
549            e.kind(),
550            format!("Failed to create output directory: {}", e),
551        ))
552    })?;
553
554    std::fs::create_dir_all(output_dir.join("templates")).map_err(|e| {
555        Error::Io(std::io::Error::new(
556            e.kind(),
557            format!("Failed to create templates directory: {}", e),
558        ))
559    })?;
560
561    std::fs::create_dir_all(output_dir.join("tests")).map_err(|e| {
562        Error::Io(std::io::Error::new(
563            e.kind(),
564            format!("Failed to create tests directory: {}", e),
565        ))
566    })?;
567
568    // Write installer.toml
569    std::fs::write(output_dir.join("installer.toml"), &result.installer_toml).map_err(|e| {
570        Error::Io(std::io::Error::new(
571            e.kind(),
572            format!("Failed to write installer.toml: {}", e),
573        ))
574    })?;
575
576    // Write template files
577    for template in &result.templates {
578        std::fs::write(
579            output_dir.join("templates").join(&template.name),
580            &template.content,
581        )
582        .map_err(|e| {
583            Error::Io(std::io::Error::new(
584                e.kind(),
585                format!("Failed to write template {}: {}", template.name, e),
586            ))
587        })?;
588    }
589
590    Ok(result)
591}
