1//! TDD-First Installer Framework (v2.0.0)
2//!
3//! This module provides a declarative installer framework that generates
4//! pure Rust installers with:
5//!
6//! - TDD by default - Tests exist before implementation
7//! - Checkpointed - Resume from any failure point
8//! - Observable - Visual progress, structured logging, tracing
9//! - Deterministic - Same inputs always produce same outputs
10//! - Falsifiable - Every claim can be empirically tested
11//! - Cryptographically Verified - Ed25519 signatures on all artifacts
12//! - Hermetically Reproducible - Bit-for-bit identical builds across machines
13//! - Container-Native - First-class multi-distro testing in isolation
14//!
15//! # Philosophy
16//!
17//! Apply Toyota Production System (TPS) principles and Karl Popper's
18//! falsificationism to installer engineering.
19//!
20//! # Usage
21//!
22//! ```bash
23//! bashrs installer init my-app-installer
24//! bashrs installer validate ./my-installer
25//! bashrs installer run ./my-installer --dry-run --diff
26//! ```

45/// Initialize a new installer project with TDD-first test harness.
46///
47/// Creates the following structure:
48/// ```text
49/// <name>/
50/// ├── installer.toml     # Declarative installer specification
51/// ├── tests/
52/// │   ├── mod.rs         # Test module
53/// │   └── falsification.rs # Popper-style falsification tests
54/// └── templates/         # Optional template files
55/// ```
56pub fn init_project(name: &Path, description: Option<&str>) -> Result<InstallerProject> {
57    // Validate project name
58    let project_name = name
59        .file_name()
60        .and_then(|n| n.to_str())
61        .ok_or_else(|| Error::Validation("Invalid project name".to_string()))?;
62
63    if project_name.is_empty() {
64        return Err(Error::Validation("Project name cannot be empty".to_string()));
65    }
66
67    // Create project directory
68    std::fs::create_dir_all(name)
69        .map_err(|e| Error::Io(std::io::Error::new(e.kind(), format!("Failed to create project directory: {e}"))))?;
70
71    // Create tests directory
72    std::fs::create_dir_all(name.join("tests"))
73        .map_err(|e| Error::Io(std::io::Error::new(e.kind(), format!("Failed to create tests directory: {e}"))))?;
74
75    // Create templates directory
76    std::fs::create_dir_all(name.join("templates"))
77        .map_err(|e| Error::Io(std::io::Error::new(e.kind(), format!("Failed to create templates directory: {e}"))))?;
78
79    // Generate installer.toml
80    let installer_toml = generate_installer_toml(project_name, description);
81    std::fs::write(name.join("installer.toml"), installer_toml)
82        .map_err(|e| Error::Io(std::io::Error::new(e.kind(), format!("Failed to write installer.toml: {e}"))))?;
83
84    // Generate test files
85    let test_mod = generate_test_mod(project_name);
86    std::fs::write(name.join("tests").join("mod.rs"), test_mod)
87        .map_err(|e| Error::Io(std::io::Error::new(e.kind(), format!("Failed to write tests/mod.rs: {e}"))))?;
88
89    let falsification_tests = generate_falsification_tests(project_name);
90    std::fs::write(name.join("tests").join("falsification.rs"), falsification_tests)
91        .map_err(|e| Error::Io(std::io::Error::new(e.kind(), format!("Failed to write tests/falsification.rs: {e}"))))?;
92
93    Ok(InstallerProject {
94        name: project_name.to_string(),
95        path: name.to_path_buf(),
96        description: description.map(|s| s.to_string()),
97    })
98}

100/// Validate an installer specification without executing it.
101///
102/// Checks:
103/// - TOML syntax is valid
104/// - All required fields are present
105/// - Step dependencies form a valid DAG (no cycles)
106/// - All referenced artifacts exist
107/// - All preconditions/postconditions are well-formed
108pub fn validate_installer(path: &Path) -> Result<ValidationResult> {
109    // Find installer.toml
110    let installer_toml = if path.is_dir() {
111        path.join("installer.toml")
112    } else {
113        path.to_path_buf()
114    };
115
116    if !installer_toml.exists() {
117        return Err(Error::Validation(format!(
118            "installer.toml not found at {}",
119            installer_toml.display()
120        )));
121    }
122
123    // Parse the TOML
124    let content = std::fs::read_to_string(&installer_toml)
125        .map_err(|e| Error::Io(std::io::Error::new(e.kind(), format!("Failed to read installer.toml: {e}"))))?;
126
127    let spec = InstallerSpec::parse(&content)?;
128
129    // Validate the spec
130    let plan = InstallerPlan::from_spec(spec)?;
131
132    Ok(ValidationResult {
133        valid: true,
134        steps: plan.steps().len(),
135        artifacts: plan.artifacts().len(),
136        warnings: Vec::new(),
137        errors: Vec::new(),
138    })
139}

167/// Generate initial installer.toml content
168fn generate_installer_toml(name: &str, description: Option<&str>) -> String {
169    let desc = description.unwrap_or("A TDD-first installer generated by bashrs");
170    format!(
171        r#"# Installer specification for {name}
172# Generated by bashrs installer init
173
174[installer]
175name = "{name}"
176version = "1.0.0"
177description = "{desc}"
178author = ""
179
180[installer.requirements]
181# Supported operating systems
182os = ["ubuntu >= 20.04", "debian >= 11"]
183# Required architectures
184arch = ["x86_64", "aarch64"]
185# Required privileges ("root" or "user")
186privileges = "user"
187# Network access required
188network = false
189
190[installer.environment]
191# Environment variables with defaults
192# EXAMPLE_VAR = {{ default = "value", validate = "string" }}
193
194[installer.security]
195# Trust model: "keyring" (explicit) or "tofu" (Trust On First Use)
196trust_model = "tofu"
197# Require signatures for all external artifacts
198require_signatures = false
199
200# =============================================================================
201# Artifacts: Externally-sourced files with verification
202# =============================================================================
203
204# [[artifact]]
205# id = "example-artifact"
206# url = "https://example.com/file.tar.gz"
207# sha256 = "..."
208# signature = "https://example.com/file.tar.gz.sig"
209# signed_by = "example-key"
210
211# =============================================================================
212# Steps: Each step is atomic, idempotent, and testable
213# =============================================================================
214
215[[step]]
216id = "hello-world"
217name = "Hello World Example Step"
218action = "script"
219
220[step.script]
221interpreter = "sh"
222content = '''
223echo "Hello from {name} installer!"
224'''
225
226[step.preconditions]
227# Preconditions that must be true before this step runs
228
229[step.postconditions]
230# Postconditions that must be true after this step completes
231
232[step.checkpoint]
233enabled = true
234"#
235    )
236}
