15/// A validated, executable installer plan.
16///
17/// The plan has been validated to ensure:
18/// - All step dependencies form a valid DAG (no cycles)
19/// - All referenced artifacts exist
20/// - All preconditions/postconditions are well-formed
21#[derive(Debug, Clone)]
22pub struct InstallerPlan {
23    /// Installer name
24    name: String,
25
26    /// Installer version
27    version: String,
28
29    /// Description
30    description: String,
31
32    /// Ordered steps (topologically sorted)
33    steps: Vec<ValidatedStep>,
34
35    /// Artifacts (validated)
36    artifacts: Vec<ValidatedArtifact>,
37
38    /// Execution waves (steps that can run in parallel)
39    waves: Vec<Vec<usize>>,
40}
41
42impl InstallerPlan {
43    /// Create a validated plan from an installer specification.
44    ///
45    /// This performs validation including:
46    /// - Dependency cycle detection
47    /// - Artifact reference validation
48    /// - Step ordering (topological sort)
49    pub fn from_spec(spec: InstallerSpec) -> Result<Self> {
50        // Validate steps and build dependency graph
51        let (steps, waves) = validate_and_order_steps(&spec.step)?;
52
53        // Validate artifacts
54        let artifacts = validate_artifacts(&spec.artifact, &spec.step)?;
55
56        Ok(Self {
57            name: spec.installer.name,
58            version: spec.installer.version,
59            description: spec.installer.description,
60            steps,
61            artifacts,
62            waves,
63        })
64    }
65
66    /// Get the installer name
67    pub fn name(&self) -> &str {
68        &self.name
69    }
70
71    /// Get the installer version
72    pub fn version(&self) -> &str {
73        &self.version
74    }
75
76    /// Get the installer description
77    pub fn description(&self) -> &str {
78        &self.description
79    }
80
81    /// Get all steps in execution order
82    pub fn steps(&self) -> &[ValidatedStep] {
83        &self.steps
84    }
85
86    /// Get all artifacts
87    pub fn artifacts(&self) -> &[ValidatedArtifact] {
88        &self.artifacts
89    }
90
91    /// Get execution waves (groups of steps that can run in parallel)
92    pub fn waves(&self) -> &[Vec<usize>] {
93        &self.waves
94    }
95
96    /// Get a step by ID
97    pub fn get_step(&self, id: &str) -> Option<&ValidatedStep> {
98        self.steps.iter().find(|s| s.id == id)
99    }
100
101    /// Get an artifact by ID
102    pub fn get_artifact(&self, id: &str) -> Option<&ValidatedArtifact> {
103        self.artifacts.iter().find(|a| a.id == id)
104    }
105}

148/// Validate steps and produce a topologically sorted order.
149///
150/// Returns (steps in order, execution waves)
151fn validate_and_order_steps(steps: &[Step]) -> Result<(Vec<ValidatedStep>, Vec<Vec<usize>>)> {
152    // Build ID to index map
153    let id_to_idx: HashMap<&str, usize> = steps
154        .iter()
155        .enumerate()
156        .map(|(i, s)| (s.id.as_str(), i))
157        .collect();
158
159    // Check for duplicate IDs
160    if id_to_idx.len() != steps.len() {
161        return Err(Error::Validation("Duplicate step IDs found".to_string()));
162    }
163
164    // Validate dependencies exist
165    for step in steps {
166        for dep in &step.depends_on {
167            if !id_to_idx.contains_key(dep.as_str()) {
168                return Err(Error::Validation(format!(
169                    "Step '{}' depends on unknown step '{}'",
170                    step.id, dep
171                )));
172            }
173        }
174    }
175
176    // Detect cycles using DFS
177    detect_cycles(steps, &id_to_idx)?;
178
179    // Topological sort using Kahn's algorithm
180    let (order, waves) = topological_sort(steps, &id_to_idx)?;
181
182    // Build validated steps in order
183    let validated_steps: Vec<ValidatedStep> = order
184        .iter()
185        .map(|&idx| {
186            let step = &steps[idx];
187            ValidatedStep {
188                id: step.id.clone(),
189                name: step.name.clone(),
190                action: step.action.clone(),
191                depends_on: step.depends_on.clone(),
192                original: step.clone(),
193            }
194        })
195        .collect();
196
197    Ok((validated_steps, waves))
198}

200/// Detect cycles in the dependency graph using DFS
201fn detect_cycles(steps: &[Step], id_to_idx: &HashMap<&str, usize>) -> Result<()> {
202    #[derive(Clone, Copy, PartialEq)]
203    enum Color {
204        White, // Unvisited
205        Gray,  // In progress
206        Black, // Completed
207    }
208
209    let n = steps.len();
210    let mut colors = vec![Color::White; n];
211
212    fn dfs(
213        idx: usize,
214        steps: &[Step],
215        id_to_idx: &HashMap<&str, usize>,
216        colors: &mut [Color],
217        path: &mut Vec<String>,
218    ) -> Result<()> {
219        colors[idx] = Color::Gray;
220        path.push(steps[idx].id.clone());
221
222        for dep in &steps[idx].depends_on {
223            if let Some(&dep_idx) = id_to_idx.get(dep.as_str()) {
224                match colors[dep_idx] {
225                    Color::Gray => {
226                        // Found a cycle
227                        let cycle_start = path.iter().position(|x| x == dep).unwrap_or(0);
228                        let cycle: Vec<_> = path[cycle_start..].to_vec();
229                        return Err(Error::Validation(format!(
230                            "Dependency cycle detected: {} -> {}",
231                            cycle.join(" -> "),
232                            dep
233                        )));
234                    }
235                    Color::White => {
236                        dfs(dep_idx, steps, id_to_idx, colors, path)?;
237                    }
238                    Color::Black => {
239                        // Already processed, no cycle through this node
240                    }
241                }
242            }
243        }
244
245        colors[idx] = Color::Black;
246        path.pop();
247        Ok(())
248    }
249
250    for i in 0..n {
251        if colors[i] == Color::White {
252            let mut path = Vec::new();
253            dfs(i, steps, id_to_idx, &mut colors, &mut path)?;
254        }
255    }
256
257    Ok(())
258}

260/// Topological sort using Kahn's algorithm.
261///
262/// Returns (ordered indices, execution waves)
263fn topological_sort(
264    steps: &[Step],
265    id_to_idx: &HashMap<&str, usize>,
266) -> Result<(Vec<usize>, Vec<Vec<usize>>)> {
267    let n = steps.len();
268
269    // Compute in-degrees
270    let mut in_degree = vec![0usize; n];
271    for step in steps {
272        for dep in &step.depends_on {
273            if let Some(&idx) = id_to_idx.get(dep.as_str()) {
274                in_degree[id_to_idx[step.id.as_str()]] += 1;
275                let _ = idx; // Use idx to avoid unused warning
276            }
277        }
278    }
279
280    // Find all nodes with in-degree 0 (reverse mapping for dependencies)
281    // Re-compute: for each step, what other steps depend on it?
282    let mut dependents: Vec<Vec<usize>> = vec![Vec::new(); n];
283    for (step_idx, step) in steps.iter().enumerate() {
284        for dep in &step.depends_on {
285            if let Some(&dep_idx) = id_to_idx.get(dep.as_str()) {
286                dependents[dep_idx].push(step_idx);
287            }
288        }
289    }
290
291    // Re-compute in-degrees correctly
292    let mut in_degree = vec![0usize; n];
293    for step in steps {
294        in_degree[id_to_idx[step.id.as_str()]] = step.depends_on.len();
295    }
296
297    let mut order = Vec::with_capacity(n);
298    let mut waves = Vec::new();
299
300    // Process in waves
301    loop {
302        // Find all nodes with in-degree 0
303        let wave: Vec<usize> = (0..n)
304            .filter(|&i| in_degree[i] == 0)
305            .filter(|i| !order.contains(i))
306            .collect();
307
308        if wave.is_empty() {
309            break;
310        }
311
312        waves.push(wave.clone());
313        order.extend(&wave);
314
315        // Reduce in-degrees
316        for &idx in &wave {
317            for &dependent in &dependents[idx] {
318                if in_degree[dependent] > 0 {
319                    in_degree[dependent] -= 1;
320                }
321            }
322        }
323
324        // Mark processed
325        for &idx in &wave {
326            in_degree[idx] = usize::MAX; // Mark as processed
327        }
328    }
329
330    if order.len() != n {
331        return Err(Error::Validation(
332            "Could not complete topological sort (cycle detected)".to_string(),
333        ));
334    }
335
336    Ok((order, waves))
337}

339/// Validate artifacts and track which steps use them
340fn validate_artifacts(
341    artifacts: &[Artifact],
342    steps: &[Step],
343) -> Result<Vec<ValidatedArtifact>> {
344    // Build artifact ID set
345    let artifact_ids: HashSet<&str> = artifacts.iter().map(|a| a.id.as_str()).collect();
346
347    // Check for duplicate artifact IDs
348    if artifact_ids.len() != artifacts.len() {
349        return Err(Error::Validation("Duplicate artifact IDs found".to_string()));
350    }
351
352    // Track which steps use each artifact
353    let mut artifact_usage: HashMap<&str, Vec<String>> = HashMap::new();
354    for step in steps {
355        for artifact_id in &step.uses_artifacts.artifacts {
356            if !artifact_ids.contains(artifact_id.as_str()) {
357                return Err(Error::Validation(format!(
358                    "Step '{}' references unknown artifact '{}'",
359                    step.id, artifact_id
360                )));
361            }
362            artifact_usage
363                .entry(artifact_id.as_str())
364                .or_default()
365                .push(step.id.clone());
366        }
367    }
368
369    // Build validated artifacts
370    let validated: Vec<ValidatedArtifact> = artifacts
371        .iter()
372        .map(|a| ValidatedArtifact {
373            id: a.id.clone(),
374            url: a.url.clone(),
375            sha256: a.sha256.clone(),
376            signature: a.signature.clone(),
377            signed_by: a.signed_by.clone(),
378            used_by: artifact_usage
379                .get(a.id.as_str())
380                .cloned()
381                .unwrap_or_default(),
382        })
383        .collect();
384
385    Ok(validated)
386}
