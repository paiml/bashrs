17#[derive(Debug, Clone, Serialize, Deserialize)]
18pub enum ShellIR {
19    /// Variable assignment: readonly NAME=VALUE
20    Let {
21        name: String,
22        value: ShellValue,
23        effects: EffectSet,
24    },
25
26    /// Command execution
27    Exec { cmd: Command, effects: EffectSet },
28
29    /// Conditional execution
30    If {
31        test: ShellValue,
32        then_branch: Box<ShellIR>,
33        else_branch: Option<Box<ShellIR>>,
34    },
35
36    /// Exit with code
37    Exit { code: u8, message: Option<String> },
38
39    /// Sequence of operations
40    Sequence(Vec<ShellIR>),
41
42    /// No-op
43    Noop,
44
45    /// Function definition
46    Function {
47        name: String,
48        params: Vec<String>,
49        body: Box<ShellIR>,
50    },
51
52    /// Echo a value (for function returns)
53    Echo { value: ShellValue },
54
55    /// For loop with range
56    For {
57        var: String,
58        start: ShellValue,
59        end: ShellValue,
60        body: Box<ShellIR>,
61    },
62
63    /// Case statement (for match expressions)
64    Case {
65        scrutinee: ShellValue,
66        arms: Vec<CaseArm>,
67    },
68
69    /// While loop
70    While {
71        condition: ShellValue,
72        body: Box<ShellIR>,
73    },
74
75    /// Break statement
76    Break,
77
78    /// Continue statement
79    Continue,
80}

82impl ShellIR {
83    /// Get all effects from this IR node and its children
84    pub fn effects(&self) -> EffectSet {
85        match self {
86            ShellIR::Let { effects, .. } | ShellIR::Exec { effects, .. } => effects.clone(),
87            ShellIR::If {
88                then_branch,
89                else_branch,
90                ..
91            } => {
92                let mut combined = then_branch.effects();
93                if let Some(else_ir) = else_branch {
94                    combined = combined.union(&else_ir.effects());
95                }
96                combined
97            }
98            ShellIR::Sequence(items) => items
99                .iter()
100                .fold(EffectSet::pure(), |acc, item| acc.union(&item.effects())),
101            ShellIR::Exit { .. } | ShellIR::Noop | ShellIR::Echo { .. } => EffectSet::pure(),
102            ShellIR::Function { body, .. } => body.effects(),
103            ShellIR::For { body, .. } => body.effects(),
104            ShellIR::While { body, .. } => body.effects(),
105            ShellIR::Case { arms, .. } => arms
106                .iter()
107                .fold(EffectSet::pure(), |acc, arm| acc.union(&arm.body.effects())),
108            ShellIR::Break | ShellIR::Continue => EffectSet::pure(),
109        }
110    }
111
112    /// Check if this IR node is pure (has no side effects)
113    pub fn is_pure(&self) -> bool {
114        self.effects().is_pure()
115    }
116}

249impl ShellValue {
250    /// Check if this value is a constant (doesn't depend on variables or commands)
251    pub fn is_constant(&self) -> bool {
252        match self {
253            ShellValue::String(_) | ShellValue::Bool(_) => true,
254            ShellValue::Variable(_)
255            | ShellValue::CommandSubst(_)
256            | ShellValue::EnvVar { .. }
257            | ShellValue::Arg { .. }
258            | ShellValue::ArgWithDefault { .. }
259            | ShellValue::ArgCount
260            | ShellValue::ExitCode => false,
261            ShellValue::Concat(parts) => parts.iter().all(|p| p.is_constant()),
262            ShellValue::Comparison { left, right, .. }
263            | ShellValue::Arithmetic { left, right, .. }
264            | ShellValue::LogicalAnd { left, right }
265            | ShellValue::LogicalOr { left, right } => left.is_constant() && right.is_constant(),
266            ShellValue::LogicalNot { operand } => operand.is_constant(),
267        }
268    }
269
270    /// Get the string representation for constant values
271    pub fn as_constant_string(&self) -> Option<String> {
272        match self {
273            ShellValue::String(s) => Some(s.clone()),
274            ShellValue::Bool(b) => Some(if *b {
275                "true".to_string()
276            } else {
277                "false".to_string()
278            }),
279            ShellValue::Concat(parts) => {
280                if parts.iter().all(|p| p.is_constant()) {
281                    let mut result = String::new();
282                    for part in parts {
283                        if let Some(s) = part.as_constant_string() {
284                            result.push_str(&s);
285                        } else {
286                            return None;
287                        }
288                    }
289                    Some(result)
290                } else {
291                    None
292                }
293            }
294            _ => None,
295        }
296    }
297}

307impl ShellExpression {
308    pub fn is_quoted(&self) -> bool {
309        match self {
310            ShellExpression::String(s) => s.starts_with('"') && s.ends_with('"'),
311            ShellExpression::Variable(_, quoted) => *quoted,
312            ShellExpression::Command(_) => false,
313            ShellExpression::Arithmetic(_) => true,
314        }
