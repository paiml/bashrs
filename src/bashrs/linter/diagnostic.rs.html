7/// A source code location span
8#[derive(Debug, Clone, Copy, PartialEq, Eq)]
9pub struct Span {
10    /// Start line (1-indexed)
11    pub start_line: usize,
12    /// Start column (1-indexed)
13    pub start_col: usize,
14    /// End line (1-indexed)
15    pub end_line: usize,
16    /// End column (1-indexed)
17    pub end_col: usize,
18}
19
20impl Span {
21    /// Create a new span
22    pub fn new(start_line: usize, start_col: usize, end_line: usize, end_col: usize) -> Self {
23        Self {
24            start_line,
25            start_col,
26            end_line,
27            end_col,
28        }
29    }
30
31    /// Create a span for a single point
32    pub fn point(line: usize, col: usize) -> Self {
33        Self::new(line, col, line, col)
34    }
35}
36
37impl fmt::Display for Span {
38    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
39        if self.start_line == self.end_line {
40            write!(f, "{}:{}-{}", self.start_line, self.start_col, self.end_col)
41        } else {
42            write!(
43                f,
44                "{}:{}-{}:{}",
45                self.start_line, self.start_col, self.end_line, self.end_col
46            )
47        }
48    }
49}

51/// Severity level of a diagnostic
52#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
53pub enum Severity {
54    /// Informational message (style, best practices)
55    Info,
56    /// Suggestion or note
57    Note,
58    /// Performance anti-pattern (not critical)
59    Perf,
60    /// Risk: potential runtime failure (context-dependent)
61    Risk,
62    /// Warning (likely bug, should fix)
63    Warning,
64    /// Error (definite syntax or semantic error, must fix)
65    Error,
66}
67
68impl fmt::Display for Severity {
69    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
70        match self {
71            Severity::Info => write!(f, "info"),
72            Severity::Note => write!(f, "note"),
73            Severity::Perf => write!(f, "perf"),
74            Severity::Risk => write!(f, "risk"),
75            Severity::Warning => write!(f, "warning"),
76            Severity::Error => write!(f, "error"),
77        }
78    }
79}

81/// Fix safety level (following APR research best practices)
82///
83/// Based on peer-reviewed research in Automated Program Repair:
84/// - Le et al. (2017): S3: Syntax- and Semantic-Guided Repair Synthesis
85/// - Monperrus (2018): Automatic Software Repair: A Bibliography
86///
87/// Key insight: Plausible patches â‰  Correct patches
88#[derive(Debug, Clone, Copy, PartialEq, Eq)]
89pub enum FixSafetyLevel {
90    /// SAFE: Semantic preservation guaranteed
91    ///
92    /// Criteria:
93    /// - No change to control flow
94    /// - No change to data flow
95    /// - No change to observable side effects
96    /// - Equivalent AST modulo formatting/style
97    ///
98    /// Examples: Quoting variables (SC2086), formatting
99    ///
100    /// Applied by: `--fix` (default)
101    Safe,
102
103    /// SAFE-WITH-ASSUMPTIONS: Semantic preservation under documented assumptions
104    ///
105    /// Criteria:
106    /// - Semantics preserved for 95%+ of real-world usage
107    /// - Edge cases are well-documented
108    /// - Failure mode is fail-safe (errors become explicit, not silent)
109    ///
110    /// Examples: `rm -f` (IDEM002), `mkdir -p` (IDEM001)
111    ///
112    /// Applied by: `--fix --fix-assumptions` (explicit opt-in)
113    SafeWithAssumptions,
114
115    /// UNSAFE: Semantic transformation required
116    ///
117    /// Criteria:
118    /// - Changes control flow or data flow
119    /// - Adds or removes operations
120    /// - Requires understanding of developer intent
121    ///
122    /// Examples: IDEM003 (adds `rm -f`), DET001 (needs intent)
123    ///
124    /// Applied by: NEVER (human-only, provides suggestions)
125    Unsafe,
126}
127
128impl fmt::Display for FixSafetyLevel {
129    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
130        match self {
131            FixSafetyLevel::Safe => write!(f, "safe"),
132            FixSafetyLevel::SafeWithAssumptions => write!(f, "safe-with-assumptions"),
133            FixSafetyLevel::Unsafe => write!(f, "unsafe"),
134        }
135    }
136}

138/// A suggested fix for a diagnostic
139#[derive(Debug, Clone, PartialEq, Eq)]
140pub struct Fix {
141    /// The replacement text
142    pub replacement: String,
143    /// Safety level of this fix
144    pub safety_level: FixSafetyLevel,
145    /// Assumptions required for this fix (for SafeWithAssumptions)
146    pub assumptions: Vec<String>,
147    /// Alternative suggested fixes (for Unsafe - human must choose)
148    pub suggested_alternatives: Vec<String>,
149}
150
151impl Fix {
152    /// Create a new SAFE fix (default)
153    pub fn new(replacement: impl Into<String>) -> Self {
154        Self {
155            replacement: replacement.into(),
156            safety_level: FixSafetyLevel::Safe,
157            assumptions: Vec::new(),
158            suggested_alternatives: Vec::new(),
159        }
160    }
161
162    /// Create a SAFE-WITH-ASSUMPTIONS fix
163    pub fn new_with_assumptions(
164        replacement: impl Into<String>,
165        assumptions: Vec<String>,
166    ) -> Self {
167        Self {
168            replacement: replacement.into(),
169            safety_level: FixSafetyLevel::SafeWithAssumptions,
170            assumptions,
171            suggested_alternatives: Vec::new(),
172        }
173    }
174
175    /// Create an UNSAFE fix (provides suggestions, no automatic replacement)
176    pub fn new_unsafe(suggested_alternatives: Vec<String>) -> Self {
177        Self {
178            replacement: String::new(), // No automatic replacement
179            safety_level: FixSafetyLevel::Unsafe,
180            assumptions: Vec::new(),
181            suggested_alternatives,
182        }
183    }
184
185    /// Check if this fix can be applied with `--fix`
186    pub fn is_safe(&self) -> bool {
187        self.safety_level == FixSafetyLevel::Safe
188    }
189
190    /// Check if this fix can be applied with `--fix --fix-assumptions`
191    pub fn is_safe_with_assumptions(&self) -> bool {
192        matches!(
193            self.safety_level,
194            FixSafetyLevel::Safe | FixSafetyLevel::SafeWithAssumptions
195        )
196    }
197
198    /// Check if this fix should never be auto-applied
199    pub fn is_unsafe(&self) -> bool {
200        self.safety_level == FixSafetyLevel::Unsafe
201    }
202}

204/// A lint diagnostic
205#[derive(Debug, Clone, PartialEq, Eq)]
206pub struct Diagnostic {
207    /// Diagnostic code (e.g., "SC2086", "BP1001")
208    pub code: String,
209    /// Severity level
210    pub severity: Severity,
211    /// Human-readable message
212    pub message: String,
213    /// Source location
214    pub span: Span,
215    /// Optional suggested fix
216    pub fix: Option<Fix>,
217}
218
219impl Diagnostic {
220    /// Create a new diagnostic
221    pub fn new(
222        code: impl Into<String>,
223        severity: Severity,
224        message: impl Into<String>,
225        span: Span,
226    ) -> Self {
227        Self {
228            code: code.into(),
229            severity,
230            message: message.into(),
231            span,
232            fix: None,
233        }
234    }
235
236    /// Add a suggested fix
237    pub fn with_fix(mut self, fix: Fix) -> Self {
238        self.fix = Some(fix);
239        self
240    }
241}
242
243impl fmt::Display for Diagnostic {
244    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
245        write!(
246            f,
247            "{} [{}] {}: {}",
248            self.span, self.severity, self.code, self.message
249        )
250    }
251}

253/// Collection of lint diagnostics
254#[derive(Debug, Clone, Default)]
255pub struct LintResult {
256    /// All diagnostics found
257    pub diagnostics: Vec<Diagnostic>,
258}
259
260impl LintResult {
261    /// Create an empty result
262    pub fn new() -> Self {
263        Self {
264            diagnostics: Vec::new(),
265        }
266    }
267
268    /// Add a diagnostic
269    pub fn add(&mut self, diagnostic: Diagnostic) {
270        self.diagnostics.push(diagnostic);
271    }
272
273    /// Merge another result into this one
274    pub fn merge(&mut self, other: LintResult) {
275        self.diagnostics.extend(other.diagnostics);
276    }
277
278    /// Check if there are any errors
279    pub fn has_errors(&self) -> bool {
280        self.diagnostics
281            .iter()
282            .any(|d| d.severity == Severity::Error)
283    }
284
285    /// Check if there are any warnings
286    pub fn has_warnings(&self) -> bool {
287        self.diagnostics
288            .iter()
289            .any(|d| d.severity == Severity::Warning)
290    }
291
292    /// Count diagnostics by severity
293    pub fn count_by_severity(&self, severity: Severity) -> usize {
294        self.diagnostics
295            .iter()
296            .filter(|d| d.severity == severity)
297            .count()
298    }
299
300    /// Get the highest severity level present
301    pub fn max_severity(&self) -> Option<Severity> {
302        self.diagnostics.iter().map(|d| d.severity).max()
303    }
304}
