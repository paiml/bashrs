181/// Platform-specific constraints
182#[derive(Debug, Clone, Copy, PartialEq, Eq)]
183pub enum PlatformProfile {
184    /// No specific platform constraints
185    Standard,
186    /// Coursera Labs constraints (10GB limit, 4GB RAM, 1-minute startup)
187    Coursera,
188}

190impl PlatformProfile {
191    /// Maximum image size in bytes
192    pub fn max_size_bytes(&self) -> u64 {
193        match self {
194            PlatformProfile::Standard => u64::MAX,
195            PlatformProfile::Coursera => 10 * 1024 * 1024 * 1024, // 10GB
196        }
197    }
198
199    /// Maximum memory in bytes
200    pub fn max_memory_bytes(&self) -> u64 {
201        match self {
202            PlatformProfile::Standard => u64::MAX,
203            PlatformProfile::Coursera => 4 * 1024 * 1024 * 1024, // 4GB
204        }
205    }
206
207    /// Maximum startup time in milliseconds
208    pub fn max_startup_ms(&self) -> u64 {
209        match self {
210            PlatformProfile::Standard => u64::MAX,
211            PlatformProfile::Coursera => 60_000, // 1 minute
212        }
213    }
214
215    /// Warning threshold for size (percentage of max)
216    pub fn size_warning_threshold(&self) -> f64 {
217        0.80 // Warn at 80% of limit
218    }

221/// Estimate image size from Dockerfile content
222pub fn estimate_size(dockerfile: &str) -> SizeEstimate {
223    let mut base_image = String::new();
224    let mut base_image_size: u64 = 0;
225    let mut layer_estimates = Vec::new();
226    let mut bloat_patterns = Vec::new();
227    let mut warnings = Vec::new();
228    let mut layer_num = 0;
229    let mut apt_update_without_clean = false;
230    let mut apt_update_line = 0;
231
232    for (line_idx, line) in dockerfile.lines().enumerate() {
233        let line_num = line_idx + 1;
234        let trimmed = line.trim();
235
236        // Skip comments and empty lines
237        if trimmed.is_empty() || trimmed.starts_with('#') {
238            continue;
239        }
240
241        let upper = trimmed.to_uppercase();
242
243        // Parse FROM instruction
244        if upper.starts_with("FROM ") {
245            layer_num += 1;
246            let image = trimmed[5..].split_whitespace().next().unwrap_or("");
247            base_image = image.to_string();
248            base_image_size = lookup_base_image_size(image);
249
250            if base_image_size == 0 {
251                warnings.push(format!(
252                    "Unknown base image '{}' - size estimate may be inaccurate",
253                    image
254                ));
255                // Default estimate for unknown images
256                base_image_size = 500_000_000; // 500MB default
257            }
258
259            layer_estimates.push(LayerEstimate {
260                layer_num,
261                instruction: "FROM".to_string(),
262                content: trimmed.to_string(),
263                line: line_num,
264                estimated_size: base_image_size,
265                cached: true, // Base images are typically cached
266                notes: Some(format!("Base image: {}", image)),
267            });
268        }
269        // Parse RUN instruction
270        else if upper.starts_with("RUN ") {
271            layer_num += 1;
272            let cmd = &trimmed[4..].trim();
273            let (estimated, notes, bloat) = estimate_run_layer_size(cmd, line_num);
274
275            // Track apt-get update without clean
276            if cmd.contains("apt-get update") && !cmd.contains("rm -rf /var/lib/apt/lists") {
277                apt_update_without_clean = true;
278                apt_update_line = line_num;
279            }
280
281            // Check for cleanup in same command
282            if cmd.contains("rm -rf /var/lib/apt/lists") {
283                apt_update_without_clean = false;
284            }
285
286            layer_estimates.push(LayerEstimate {
287                layer_num,
288                instruction: "RUN".to_string(),
289                content: trimmed.to_string(),
290                line: line_num,
291                estimated_size: estimated,
292                cached: false,
293                notes,
294            });
295
296            bloat_patterns.extend(bloat);
297        }
298        // Parse COPY/ADD instructions
299        else if upper.starts_with("COPY ") || upper.starts_with("ADD ") {
300            layer_num += 1;
301            let instruction = if upper.starts_with("COPY") {
302                "COPY"
303            } else {
304                "ADD"
305            };
306
307            // Can't estimate COPY/ADD size without build context
308            layer_estimates.push(LayerEstimate {
309                layer_num,
310                instruction: instruction.to_string(),
311                content: trimmed.to_string(),
312                line: line_num,
313                estimated_size: 0, // Unknown
314                cached: false,
315                notes: Some("Size depends on build context".to_string()),
316            });
317        }
318        // Other instructions that don't add layers
319        else if upper.starts_with("ENV ")
320            || upper.starts_with("WORKDIR ")
321            || upper.starts_with("EXPOSE ")
322            || upper.starts_with("USER ")
323            || upper.starts_with("ARG ")
324            || upper.starts_with("LABEL ")
325            || upper.starts_with("CMD ")
326            || upper.starts_with("ENTRYPOINT ")
327            || upper.starts_with("HEALTHCHECK ")
328        {
329            // These create metadata layers, minimal size
330            layer_estimates.push(LayerEstimate {
331                layer_num: layer_num + 1,
332                instruction: upper.split_whitespace().next().unwrap_or("").to_string(),
333                content: trimmed.to_string(),
334                line: line_num,
335                estimated_size: 0, // Metadata only
336                cached: true,
337                notes: Some("Metadata layer".to_string()),
338            });
339        }
340    }
341
342    // Add bloat pattern for apt-get update without cleanup
343    if apt_update_without_clean {
344        bloat_patterns.push(BloatPattern {
345            code: "SIZE001".to_string(),
346            description: "apt cache not cleaned after install".to_string(),
347            line: apt_update_line,
348            wasted_bytes: 200_000_000, // ~200MB
349            remediation: "Add '&& rm -rf /var/lib/apt/lists/*' after apt-get install".to_string(),
350        });
351    }
352
353    // Calculate total
354    let total_estimated: u64 = layer_estimates.iter().map(|l| l.estimated_size).sum();
355
356    SizeEstimate {
357        base_image_size,
358        base_image,
359        layer_estimates,
360        total_estimated,
361        bloat_patterns,
362        warnings,
363    }
364}

366/// Look up base image size from known sizes
367fn lookup_base_image_size(image: &str) -> u64 {
368    // Try exact match first
369    for (name, size) in BASE_IMAGE_SIZES {
370        if image == *name {
371            return *size;
372        }
373    }
374
375    // Try prefix match (for tagged versions)
376    let image_base = image.split(':').next().unwrap_or(image);
377    for (name, size) in BASE_IMAGE_SIZES {
378        let name_base = name.split(':').next().unwrap_or(name);
379        if image_base == name_base {
380            return *size;
381        }
382    }
383
384    0
385}

387/// Estimate size of a RUN layer
388fn estimate_run_layer_size(cmd: &str, line: usize) -> (u64, Option<String>, Vec<BloatPattern>) {
389    let mut total: u64 = 0;
390    let mut notes = Vec::new();
391    let mut bloat = Vec::new();
392
393    // Check for apt-get install
394    if cmd.contains("apt-get install") || cmd.contains("apt install") {
395        // Extract package names
396        let packages = extract_apt_packages(cmd);
397        for pkg in &packages {
398            let pkg_size = lookup_package_size(pkg);
399            if pkg_size > 0 {
400                total += pkg_size;
401                notes.push(format!("{}: ~{}MB", pkg, pkg_size / 1_000_000));
402            }
403        }
404
405        // Check for missing --no-install-recommends
406        if !cmd.contains("--no-install-recommends") && !packages.is_empty() {
407            bloat.push(BloatPattern {
408                code: "SIZE002".to_string(),
409                description: "apt-get install without --no-install-recommends".to_string(),
410                line,
411                wasted_bytes: 100_000_000, // ~100MB typically
412                remediation: "Add '--no-install-recommends' to apt-get install".to_string(),
413            });
414        }
415    }
416
417    // Check for pip install
418    if cmd.contains("pip install") || cmd.contains("pip3 install") {
419        let packages = extract_pip_packages(cmd);
420        for pkg in &packages {
421            let pkg_size = lookup_package_size(pkg);
422            if pkg_size > 0 {
423                total += pkg_size;
424                notes.push(format!("{}: ~{}MB", pkg, pkg_size / 1_000_000));
425            }
426        }
427
428        // Check for missing --no-cache-dir
429        if !cmd.contains("--no-cache-dir") {
430            bloat.push(BloatPattern {
431                code: "SIZE003".to_string(),
432                description: "pip install without --no-cache-dir".to_string(),
433                line,
434                wasted_bytes: 50_000_000, // ~50MB typically
435                remediation: "Add '--no-cache-dir' to pip install".to_string(),
436            });
437        }
438    }
439
440    // Check for npm install
441    if cmd.contains("npm install") || cmd.contains("npm i ") {
442        // NPM packages can be large
443        total += 200_000_000; // ~200MB base estimate
444        notes.push("npm dependencies".to_string());
445
446        // Check for node_modules
447        if !cmd.contains("--production") && !cmd.contains("ci") {
448            bloat.push(BloatPattern {
449                code: "SIZE004".to_string(),
450                description: "npm install includes dev dependencies".to_string(),
451                line,
452                wasted_bytes: 100_000_000,
453                remediation: "Use 'npm ci --only=production' for smaller image".to_string(),
454            });
455        }
456    }
457
458    // Default minimum estimate for RUN commands with no detected packages
459    if total == 0 {
460        total = 10_000_000; // 10MB default for misc RUN commands
461    }
462
463    let notes_str = if notes.is_empty() {
464        None
465    } else {
466        Some(notes.join(", "))
467    };
468
469    (total, notes_str, bloat)
470}

472/// Extract package names from apt-get install command
473fn extract_apt_packages(cmd: &str) -> Vec<String> {
474    let mut packages = Vec::new();
475
476    // Find the install command and extract packages
477    if let Some(idx) = cmd.find("install") {
478        let after_install = &cmd[idx + 7..];
479        for word in after_install.split_whitespace() {
480            // Skip flags
481            if word.starts_with('-') || word.starts_with('\\') {
482                continue;
483            }
484            // Skip operators
485            if word == "&&" || word == "||" || word == ";" {
486                break;
487            }
488            // Skip -y which might come after install
489            if word == "-y" {
490                continue;
491            }
492            packages.push(word.to_string());
493        }
494    }
495
496    packages
497}

499/// Extract package names from pip install command
500fn extract_pip_packages(cmd: &str) -> Vec<String> {
501    let mut packages = Vec::new();
502
503    // Find pip install and extract packages
504    let install_patterns = ["pip install", "pip3 install"];
505    for pattern in &install_patterns {
506        if let Some(idx) = cmd.find(pattern) {
507            let after_install = &cmd[idx + pattern.len()..];
508            for word in after_install.split_whitespace() {
509                // Skip flags
510                if word.starts_with('-') || word.starts_with('\\') {
511                    continue;
512                }
513                // Skip operators
514                if word == "&&" || word == "||" || word == ";" {
515                    break;
516                }
517                // Skip requirement files
518                if word.ends_with(".txt") {
519                    continue;
520                }
521                packages.push(word.to_string());
522            }
523        }
524    }
525
526    packages
527}

529/// Look up package size from known sizes
530fn lookup_package_size(package: &str) -> u64 {
531    let package_lower = package.to_lowercase();
532
533    for (name, size) in PACKAGE_SIZES {
534        if package_lower == *name || package_lower.contains(name) {
535            return *size;
536        }
537    }

539    0

551/// Get actual image size from Docker
552pub fn get_docker_image_size(image_name: &str) -> Option<u64> {
553    let output = Command::new("docker")
554        .args(["images", image_name, "--format", "{{.Size}}"])
555        .output()
556        .ok()?;
557
558    if !output.status.success() {
559        return None;
560    }
561
562    let size_str = String::from_utf8_lossy(&output.stdout);
563    parse_docker_size(size_str.trim())
564}

566/// Parse Docker size string (e.g., "1.5GB", "500MB")
567fn parse_docker_size(size_str: &str) -> Option<u64> {
568    let size_str = size_str.to_uppercase();
569
570    // Try to parse number and unit
571    let (num_str, multiplier) = if size_str.ends_with("GB") {
572        (&size_str[..size_str.len() - 2], 1_000_000_000u64)
573    } else if size_str.ends_with("MB") {
574        (&size_str[..size_str.len() - 2], 1_000_000u64)
575    } else if size_str.ends_with("KB") {
576        (&size_str[..size_str.len() - 2], 1_000u64)
577    } else if size_str.ends_with('B') {
578        (&size_str[..size_str.len() - 1], 1u64)
579    } else {
580        return None;
581    };
582
583    num_str
584        .trim()
585        .parse::<f64>()
586        .ok()
587        .map(|n| (n * multiplier as f64) as u64)
588}

590/// Generate lint result from size estimate
591pub fn size_estimate_to_lint_result(
592    estimate: &SizeEstimate,
593    profile: PlatformProfile,
594    strict: bool,
595) -> LintResult {
596    let mut result = LintResult::new();
597
598    // Add warnings from analysis
599    for warning in &estimate.warnings {
600        let span = Span::new(1, 1, 1, 1);
601        result.add(Diagnostic::new("SIZE-INFO", Severity::Info, warning, span));
602    }
603
604    // Add bloat patterns as warnings
605    for bloat in &estimate.bloat_patterns {
606        let span = Span::new(bloat.line, 1, bloat.line, 1);
607        let mut diag = Diagnostic::new(
608            bloat.code.clone(),
609            Severity::Warning,
610            format!(
611                "{} (~{}MB wasted)",
612                bloat.description,
613                bloat.wasted_bytes / 1_000_000
614            ),
615            span,
616        );
617        diag.fix = Some(Fix::new(bloat.remediation.clone()));
618        result.add(diag);
619    }
620
621    // Check against platform limits
622    let max_size = profile.max_size_bytes();
623    let warning_threshold = (max_size as f64 * profile.size_warning_threshold()) as u64;
624
625    if estimate.total_estimated > max_size {
626        let severity = if strict {
627            Severity::Error
628        } else {
629            Severity::Warning
630        };
631        let span = Span::new(1, 1, 1, 1);
632        let mut diag = Diagnostic::new(
633            "SIZE-LIMIT",
634            severity,
635            format!(
636                "Estimated image size ({:.1}GB) exceeds platform limit ({:.1}GB)",
637                estimate.total_estimated as f64 / 1_000_000_000.0,
638                max_size as f64 / 1_000_000_000.0
639            ),
640            span,
641        );
642        diag.fix = Some(Fix::new(
643            "Consider using a smaller base image or multi-stage build",
644        ));
645        result.add(diag);
646    } else if estimate.total_estimated > warning_threshold {
647        let span = Span::new(1, 1, 1, 1);
648        let mut diag = Diagnostic::new(
649            "SIZE-WARNING",
650            Severity::Warning,
651            format!(
652                "Estimated image size ({:.1}GB) approaching platform limit ({:.1}GB)",
653                estimate.total_estimated as f64 / 1_000_000_000.0,
654                max_size as f64 / 1_000_000_000.0
655            ),
656            span,
657        );
658        diag.fix = Some(Fix::new("Consider optimizations to reduce image size"));
659        result.add(diag);
660    }
661
662    result
663}

665/// Format size estimate as human-readable output
666pub fn format_size_estimate(estimate: &SizeEstimate, verbose: bool) -> String {
667    let mut output = String::new();
668
669    output.push_str("Image Size Analysis\n");
670    output.push_str("===================\n\n");
671
672    // Base image
673    output.push_str(&format!(
674        "Base image: {} (~{:.1}GB)\n\n",
675        estimate.base_image,
676        estimate.base_image_size as f64 / 1_000_000_000.0
677    ));
678
679    // Layer breakdown
680    if verbose {
681        output.push_str("Layer Breakdown:\n");
682        for layer in &estimate.layer_estimates {
683            let size_str = if layer.estimated_size == 0 {
684                "unknown".to_string()
685            } else {
686                format!("~{:.1}MB", layer.estimated_size as f64 / 1_000_000.0)
687            };
688
689            output.push_str(&format!(
690                "  [{}] {} ({}) - line {}\n",
691                layer.layer_num, layer.instruction, size_str, layer.line
692            ));
693
694            if let Some(notes) = &layer.notes {
695                output.push_str(&format!("      {}\n", notes));
696            }
697        }
698        output.push('\n');
699    }
700
701    // Total
702    output.push_str(&format!(
703        "Estimated total: {:.2}GB\n\n",
704        estimate.total_estimated as f64 / 1_000_000_000.0
705    ));
706
707    // Bloat patterns
708    if !estimate.bloat_patterns.is_empty() {
709        output.push_str("Optimization Opportunities:\n");
710        for bloat in &estimate.bloat_patterns {
711            output.push_str(&format!(
712                "  {} [line {}]: {} (~{}MB)\n",
713                bloat.code,
714                bloat.line,
715                bloat.description,
716                bloat.wasted_bytes / 1_000_000
717            ));
718            output.push_str(&format!("    Fix: {}\n", bloat.remediation));
719        }
720        output.push('\n');
721    }
722
723    // Warnings
724    if !estimate.warnings.is_empty() {
725        output.push_str("Warnings:\n");
726        for warning in &estimate.warnings {
727            output.push_str(&format!("  - {}\n", warning));
728        }
729    }
730
731    output
732}

734/// Format size estimate as JSON
735pub fn format_size_estimate_json(estimate: &SizeEstimate) -> String {
736    let layers: Vec<HashMap<&str, serde_json::Value>> = estimate
737        .layer_estimates
738        .iter()
739        .map(|l| {
740            let mut map = HashMap::new();
741            map.insert("layer_num", serde_json::json!(l.layer_num));
742            map.insert("instruction", serde_json::json!(l.instruction));
743            map.insert("line", serde_json::json!(l.line));
744            map.insert("estimated_bytes", serde_json::json!(l.estimated_size));
745            map.insert("notes", serde_json::json!(l.notes));
746            map
747        })
748        .collect();
749
750    let bloat: Vec<HashMap<&str, serde_json::Value>> = estimate
751        .bloat_patterns
752        .iter()
753        .map(|b| {
754            let mut map = HashMap::new();
755            map.insert("code", serde_json::json!(b.code));
756            map.insert("description", serde_json::json!(b.description));
757            map.insert("line", serde_json::json!(b.line));
758            map.insert("wasted_bytes", serde_json::json!(b.wasted_bytes));
759            map.insert("remediation", serde_json::json!(b.remediation));
760            map
761        })
762        .collect();
763
764    let json = serde_json::json!({
765        "base_image": estimate.base_image,
766        "base_image_bytes": estimate.base_image_size,
767        "total_estimated_bytes": estimate.total_estimated,
768        "total_estimated_gb": estimate.total_estimated as f64 / 1_000_000_000.0,
769        "layers": layers,
770        "bloat_patterns": bloat,
771        "warnings": estimate.warnings,
772    });
773
774    serde_json::to_string_pretty(&json).unwrap_or_else(|_| "{}".to_string())
775}
