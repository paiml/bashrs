1//! MAKE010: Missing error handling (|| exit 1)
2//!
3//! **Rule**: Detect commands without error handling in recipes
4//!
5//! **Why this matters**:
6//! By default, Make only stops on error if the recipe command returns non-zero.
7//! However, some commands may fail silently or have side effects that should
8//! stop the build. Adding `|| exit 1` ensures the build stops on failure.
9//!
10//! **Auto-fix**: Add `|| exit 1` to commands that should fail the build
11//!
12//! ## Examples
13//!
14//! ❌ **BAD** (no error handling):
15//! ```makefile
16//! install:
17//!     cp app /usr/bin/app
18//!     chmod +x /usr/bin/app
19//! ```
20//!
21//! ✅ **GOOD** (with error handling):
22//! ```makefile
23//! install:
24//!     cp app /usr/bin/app || exit 1
25//!     chmod +x /usr/bin/app || exit 1
26//! ```
27
28use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
29
30/// Commands that should have error handling
31const CRITICAL_COMMANDS: &[&str] = &[
32    "cp", "mv", "rm", "install", "chmod", "chown", "ln", "mkdir", "curl", "wget", "git",
33];
34
35/// Check for missing error handling in recipe commands
36pub fn check(source: &str) -> LintResult {
37    let mut result = LintResult::new();
38
39    for (line_num, line) in source.lines().enumerate() {
40        // Only check recipe lines (start with tab)
41        if !line.starts_with('\t') {
42            continue;
43        }
44
45        let recipe = line.trim();
46
47        // Skip if already has error handling
48        if has_error_handling(recipe) {
49            continue;
50        }
51
52        // Check if line contains a critical command
53        if let Some(cmd) = find_critical_command(recipe) {
54            let cmd_pos = line.find(cmd).unwrap_or(0);
55            let span = Span::new(
56                line_num + 1,
57                cmd_pos + 1,
58                line_num + 1,
59                cmd_pos + cmd.len() + 1,
60            );
61
62            // Create fix by adding || exit 1
63            let fix_replacement = format!("{} || exit 1", line.trim_start());
64
65            let diag = Diagnostic::new(
66                "MAKE010",
67                Severity::Warning,
68                format!(
69                    "Command '{}' missing error handling - consider adding '|| exit 1'",
70                    cmd
71                ),
72                span,
73            )
74            .with_fix(Fix::new(&fix_replacement));
75
76            result.add(diag);
77        }
78    }
79
80    result
81}
82
83/// Check if a recipe line already has error handling
84fn has_error_handling(recipe: &str) -> bool {
85    recipe.contains("|| exit") || recipe.contains("set -e") || recipe.contains("&&")
86}
87
88/// Find if the recipe contains a critical command
89///
90/// Returns None if the command keyword is inside:
91/// - Quoted strings (echo "install", printf 'rm -rf')
92/// - Variable assignments (MSG="install here")
93/// - Heredocs
94/// - Comments
95fn find_critical_command(recipe: &str) -> Option<&'static str> {
96    // Skip if this is an echo/printf/cat command with quoted strings
97    let trimmed = recipe.trim_start_matches('@').trim_start();
98
99    // Check if line starts with echo, printf, or cat (output commands)
100    if trimmed.starts_with("echo ") || trimmed.starts_with("printf ") || trimmed.starts_with("cat ")
101    {
102        return None;
103    }
104
105    // Check if this is a variable assignment (VAR="..." or VAR='...')
106    if is_variable_assignment(trimmed) {
107        return None;
108    }
109
110    // Check if we're in a heredoc context
111    if trimmed.contains("<<") {
112        return None;
113    }
114
115    // Now check for actual critical commands
116    CRITICAL_COMMANDS
117        .iter()
118        .find(|&cmd| {
119            recipe.split_whitespace().any(|word| {
120                word == *cmd
121                    || word.starts_with(&format!("{}@", cmd))
122                    || word.starts_with(&format!("{}-", cmd))
123            })
124        })
125        .map(|v| v as _)
126}
127
128/// Check if a line is a variable assignment
129fn is_variable_assignment(line: &str) -> bool {
130    // Pattern: VAR="..." or VAR='...'
131    if let Some(eq_pos) = line.find('=') {
132        let before_eq = &line[..eq_pos];
133        // Variable name should be alphanumeric + underscore only
134        let is_valid_var_name = before_eq
135            .chars()
136            .all(|c| c.is_alphanumeric() || c == '_' || c == '$');
137
138        if is_valid_var_name {
139            let after_eq = &line[eq_pos + 1..];
140            // Check if value is quoted
141            return after_eq.starts_with('"') || after_eq.starts_with('\'');
142        }
143    }
144    false
145}
