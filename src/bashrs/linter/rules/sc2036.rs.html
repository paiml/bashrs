1// SC2036: Quotes in backticks need escaping. Use $( ) instead or escape quotes.
2//
3// Backticks are the old-style command substitution. Inside backticks, quotes need
4// to be escaped with backslashes. Unescaped quotes can cause unexpected behavior.
5//
6// Examples:
7// Bad:
8//   result=`echo "hello world"`    // Quotes not escaped
9//   out=`grep "pattern" file`      // Can fail
10//
11// Good:
12//   result=`echo \"hello world\"`  // Escaped quotes
13//   result=$(echo "hello world")   // Modern syntax (preferred)
14//
15// Note: $( ) is preferred over backticks because it handles nesting better
16// and doesn't require escaping.
17
18use crate::linter::{Diagnostic, LintResult, Severity, Span};
19use once_cell::sync::Lazy;
20use regex::Regex;
21
22static BACKTICK_WITH_UNESCAPED_QUOTES: Lazy<Regex> = Lazy::new(|| {
23    // Match: `...unescaped "...'`
24    // Look for backticks containing unescaped double or single quotes
25    Regex::new(r#"`([^`]*[^\\])?(["'])([^`]*)`"#).unwrap()
26});
27
28/// Check if line should be analyzed (has backticks and quotes)
29fn should_check_line(line: &str) -> bool {
30    line.contains('`') && (line.contains('"') || line.contains('\''))
31}
32
33/// Check if character at position is a quote
34fn is_quote(c: char) -> bool {
35    c == '"' || c == '\''
36}
37
38/// Check if quote at position is escaped
39fn is_escaped_quote(chars: &[char], pos: usize) -> bool {
40    pos > 0 && chars[pos - 1] == '\\'
41}
42
43/// Check if character at position is an unescaped quote
44fn is_unescaped_quote(chars: &[char], pos: usize) -> bool {
45    is_quote(chars[pos]) && !is_escaped_quote(chars, pos)
46}
47
48/// Find unescaped quote position inside backtick expression
49fn find_unescaped_quote_in_backticks(chars: &[char], start: usize) -> Option<usize> {
50    let mut i = start + 1; // Skip opening backtick
51
52    while i < chars.len() && chars[i] != '`' {
53        if is_unescaped_quote(chars, i) {
54            return Some(i);
55        }
56        i += 1;
57    }
58
59    None
60}
61
62/// Create diagnostic for unescaped quote in backticks
63fn create_backtick_quote_diagnostic(
64    line_num: usize,
65    backtick_start: usize,
66    quote_pos: usize,
67) -> Diagnostic {
68    let start_col = backtick_start + 1;
69    let end_col = quote_pos + 2; // +2 to include quote
70
71    Diagnostic::new(
72        "SC2036",
73        Severity::Warning,
74        "Quotes in backticks need escaping. Use $( ) instead or escape with \\\"".to_string(),
75        Span::new(line_num, start_col, line_num, end_col),
76    )
77}

79pub fn check(source: &str) -> LintResult {
80    let mut result = LintResult::new();
81
82    for (line_num, line) in source.lines().enumerate() {
83        let line_num = line_num + 1;
84
85        if line.trim_start().starts_with('#') || !should_check_line(line) {
86            continue;
87        }
88
89        let chars: Vec<char> = line.chars().collect();
90        let mut i = 0;
91
92        while i < chars.len() {
93            if chars[i] == '`' {
94                let backtick_start = i;
95
96                // Check for unescaped quotes inside backticks
97                if let Some(quote_pos) = find_unescaped_quote_in_backticks(&chars, backtick_start) {
98                    let diagnostic = create_backtick_quote_diagnostic(
99                        line_num,
100                        backtick_start,
101                        quote_pos,
102                    );
103                    result.add(diagnostic);
104
105                    // Skip to end of this backtick expression
106                    i = quote_pos + 1;
107                    while i < chars.len() && chars[i] != '`' {
108                        i += 1;
109                    }
110                }
111
112                if i < chars.len() && chars[i] == '`' {
113                    i += 1; // Skip closing backtick
114                }
115            } else {
116                i += 1;
117            }
118        }
119    }
120
121    result
122}

124#[cfg(test)]
125mod tests {
126    use super::*;
127
128    #[test]
129    fn test_sc2036_unescaped_double_quotes() {
130        let code = r#"result=`echo "hello"`"#;
131        let result = check(code);
132        assert_eq!(result.diagnostics.len(), 1);
133        assert_eq!(result.diagnostics[0].code, "SC2036");
134        assert_eq!(result.diagnostics[0].severity, Severity::Warning);
135        assert!(result.diagnostics[0].message.contains("escaping"));
136    }
137
138    #[test]
139    fn test_sc2036_unescaped_single_quotes() {
140        let code = r#"result=`echo 'hello'`"#;
141        let result = check(code);
142        assert_eq!(result.diagnostics.len(), 1);
143    }
144
145    #[test]
146    fn test_sc2036_escaped_quotes_ok() {
147        let code = r#"result=`echo \"hello\"`"#;
148        let result = check(code);
149        // Escaped quotes are OK
150        assert_eq!(result.diagnostics.len(), 0);
151    }
