1//! DOCKER001: Missing USER directive (security risk)
2//!
3//! **Rule**: Detect Dockerfiles running as root user
4//!
5//! **Why this matters**:
6//! Running containers as root is a security risk. If the container is compromised,
7//! the attacker has root privileges. Always use a non-root USER.
8//!
9//! **Exceptions**:
10//! - `FROM scratch` images (no users exist in scratch)
11//! - Build stages (intermediate stages can run as root)
12//!
13//! ## Examples
14//!
15//! ❌ **BAD** (runs as root):
16//! ```dockerfile
17//! FROM debian:12-slim
18//! COPY app /app
19//! CMD ["/app"]
20//! ```
21//!
22//! ✅ **GOOD** (non-root user):
23//! ```dockerfile
24//! FROM debian:12-slim
25//! RUN useradd -m appuser
26//! USER appuser
27//! COPY app /app
28//! CMD ["/app"]
29//! ```
30
31use crate::linter::{Diagnostic, LintResult, Severity, Span};
32
33/// Check for missing USER directive in non-scratch Dockerfiles
34pub fn check(source: &str) -> LintResult {
35    let mut result = LintResult::new();
36
37    let lines: Vec<&str> = source.lines().collect();
38
39    // Parse Dockerfile to find FROM and USER directives
40    let mut stages = Vec::new();
41    let mut current_stage: Option<DockerStage> = None;
42
43    for (line_num, line) in lines.iter().enumerate() {
44        let trimmed = line.trim();
45
46        // Skip comments and empty lines
47        if trimmed.is_empty() || trimmed.starts_with('#') {
48            continue;
49        }
50
51        // Detect FROM directive (new stage)
52        if let Some(stripped) = trimmed.strip_prefix("FROM ") {
53            // Save previous stage
54            if let Some(stage) = current_stage.take() {
55                stages.push(stage);
56            }
57
58            // Start new stage
59            let from_image = stripped.trim();
60            let is_scratch = from_image.starts_with("scratch");
61            let is_named_stage = from_image.contains(" AS ");
62
63            current_stage = Some(DockerStage {
64                line: line_num + 1,
65                from_image: from_image.to_string(),
66                is_scratch,
67                is_named_stage,
68                has_user: false,
69            });
70        }
71
72        // Detect USER directive
73        if trimmed.starts_with("USER ") {
74            if let Some(ref mut stage) = current_stage {
75                stage.has_user = true;
76            }
77        }
78    }
79
80    // Save last stage
81    if let Some(stage) = current_stage {
82        stages.push(stage);
83    }
84
85    // Check the final stage (runtime stage)
86    if let Some(final_stage) = stages.last() {
87        // Only warn for non-scratch final stages
88        if !final_stage.is_scratch && !final_stage.has_user {
89            let span = Span::new(final_stage.line, 1, final_stage.line, 5);
90
91            let diag = Diagnostic::new(
92                "DOCKER001",
93                Severity::Warning,
94                "Missing USER directive - container runs as root (security risk). Add USER directive before CMD/ENTRYPOINT".to_string(),
95                span,
96            );
97
98            result.add(diag);
99        }
100    }
101
102    result
103}
104
105#[derive(Debug)]
106struct DockerStage {
107    line: usize,
108    from_image: String,
109    is_scratch: bool,
110    is_named_stage: bool,
111    has_user: bool,
112}

114#[cfg(test)]
115mod tests {
116    use super::*;
117
118    #[test]
119    fn test_DOCKER001_missing_user_directive() {
120        let dockerfile = "FROM debian:12-slim\nCOPY app /app\n";
121        let result = check(dockerfile);
122
123        assert_eq!(result.diagnostics.len(), 1);
124        let diag = &result.diagnostics[0];
125        assert_eq!(diag.code, "DOCKER001");
126        assert!(diag.message.contains("USER"));
127    }
128
129    #[test]
130    fn test_DOCKER001_scratch_no_warning() {
131        let dockerfile = "FROM scratch\nCOPY app /app\n";
132        let result = check(dockerfile);
133
134        assert_eq!(result.diagnostics.len(), 0);
135    }
136
137    #[test]
138    fn test_DOCKER001_user_present() {
139        let dockerfile = "FROM debian:12-slim\nUSER appuser\nCOPY app /app\n";
140        let result = check(dockerfile);
141
142        assert_eq!(result.diagnostics.len(), 0);
143    }
144
145    #[test]
146    fn test_DOCKER001_multi_stage_final_no_user() {
147        let dockerfile =
148            "FROM debian AS builder\nRUN build\n\nFROM debian\nCOPY --from=builder /app /app\n";
149        let result = check(dockerfile);
150
151        // Should warn about final stage
152        assert_eq!(result.diagnostics.len(), 1);
153    }
154
155    #[test]
156    fn test_DOCKER001_multi_stage_final_scratch() {
157        let dockerfile =
158            "FROM debian AS builder\nRUN build\n\nFROM scratch\nCOPY --from=builder /app /app\n";
159        let result = check(dockerfile);
160
161        // Should NOT warn (final stage is scratch)
162        assert_eq!(result.diagnostics.len(), 0);
163    }
164
165    // Property-based tests
166    #[cfg(test)]
167    mod property_tests {
168        use super::*;
169        use proptest::prelude::*;
170
171        proptest! {
172        #![proptest_config(proptest::test_runner::Config::with_cases(10))]
173            #[test]
174            fn prop_never_panics(dockerfile in ".*") {
175                let _ = check(&dockerfile);
176            }
177
178            #[test]
179            fn prop_scratch_never_warns(
180                commands in prop::collection::vec("(RUN|COPY|CMD|ENTRYPOINT) .*", 0..10)
181            ) {
182                let dockerfile = format!("FROM scratch\n{}", commands.join("\n"));
183                let result = check(&dockerfile);
184                prop_assert_eq!(result.diagnostics.len(), 0);
185            }
186
187            #[test]
188            fn prop_with_user_never_warns(
189                base_image in "[a-z]+:[0-9.]+",
190                username in "[a-z][a-z0-9_]*"
191            ) {
192                let dockerfile = format!("FROM {}\nUSER {}\nCMD [\"app\"]", base_image, username);
193                let result = check(&dockerfile);
194                prop_assert_eq!(result.diagnostics.len(), 0);
195            }
196
197            #[test]
198            fn prop_without_user_warns(
199                base_image in "[a-z]+:[0-9.]+"
200            ) {
201                let dockerfile = format!("FROM {}\nCMD [\"app\"]", base_image);
202                let result = check(&dockerfile);
203                prop_assert_eq!(result.diagnostics.len(), 1);
204            }
205        }
206    }
