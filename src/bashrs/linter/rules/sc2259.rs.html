1// SC2259: Prefer [[ ]] over grep for simple string checks
2use crate::linter::{Diagnostic, LintResult, Severity, Span};
3use once_cell::sync::Lazy;
4use regex::Regex;
5
6static GREP_FOR_SUBSTRING: Lazy<Regex> =
7    Lazy::new(|| Regex::new(r#"echo\s+[^|]+\|\s*grep\s+-q"#).unwrap());
8
9pub fn check(source: &str) -> LintResult {
10    let mut result = LintResult::new();
11    for (line_num, line) in source.lines().enumerate() {
12        let line_num = line_num + 1;
13        if line.trim_start().starts_with('#') {
14            continue;
15        }
16
17        if GREP_FOR_SUBSTRING.is_match(line) {
18            let diagnostic = Diagnostic::new(
19                "SC2259",
20                Severity::Info,
21                "Consider using [[ $var == *substring* ]] instead of echo/grep for substring matching".to_string(),
22                Span::new(line_num, 1, line_num, line.len() + 1),
23            );
24            result.add(diagnostic);
25        }
26    }
27    result
28}
