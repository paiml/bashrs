1// SC2030: Variable modified in subshell. Use var=$(cmd) or avoid subshell.
2//
3// Variables assigned inside subshells (parentheses) don't affect the parent shell.
4// The assignment is local to the subshell and lost when it exits.
5//
6// Examples:
7// Bad:
8//   (foo=bar)
9//   echo "$foo"  # Empty, modification was in subshell
10//
11//   result=$(x=5; echo "$x")  # x assignment lost
12//
13// Good:
14//   foo=bar      # Direct assignment in current shell
15//   echo "$foo"
16//
17//   x=5; echo "$x"  # No subshell
18//
19// Note: This rule detects variable assignments inside ( ) subshells.
20
21use crate::linter::{Diagnostic, LintResult, Severity, Span};
22use once_cell::sync::Lazy;
23use regex::Regex;
24
25static SUBSHELL_ASSIGNMENT: Lazy<Regex> = Lazy::new(|| {
26    // Match: ( ... var=value ... )
27    // Look for variable assignments inside parentheses
28    Regex::new(r"\(\s*([a-zA-Z_][a-zA-Z0-9_]*)=").unwrap()
29});
30
31pub fn check(source: &str) -> LintResult {
32    let mut result = LintResult::new();
33
34    for (line_num, line) in source.lines().enumerate() {
35        let line_num = line_num + 1;
36
37        if line.trim_start().starts_with('#') {
38            continue;
39        }
40
41        // Look for variable assignments in subshells
42        for cap in SUBSHELL_ASSIGNMENT.captures_iter(line) {
43            let var_name = cap.get(1).unwrap().as_str();
44
45            // Skip if inside quotes
46            let full_match = cap.get(0).unwrap().as_str();
47            let pos = line.find(full_match).unwrap_or(0);
48            let before = &line[..pos];
49            let quote_count = before.matches('"').count() + before.matches('\'').count();
50            if quote_count % 2 == 1 {
51                continue;
52            }
53
54            // Skip command substitution $( ... ) - not a subshell
55            if pos > 0 && line.chars().nth(pos - 1) == Some('$') {
56                continue;
57            }
58
59            let start_col = pos + 1;
60            let end_col = start_col + full_match.len();
61
62            let diagnostic = Diagnostic::new(
63                "SC2030",
64                Severity::Warning,
65                format!(
66                    "Variable '{}' modified in subshell. Use var=$(cmd) to capture output or remove subshell",
67                    var_name
68                ),
69                Span::new(line_num, start_col, line_num, end_col),
70            );
71
72            result.add(diagnostic);
73        }
74    }
75
76    result
77}

79#[cfg(test)]
80mod tests {
81    use super::*;
82
83    #[test]
84    fn test_sc2030_simple_subshell_assignment() {
85        let code = r#"(foo=bar)"#;
86        let result = check(code);
87        assert_eq!(result.diagnostics.len(), 1);
88        assert_eq!(result.diagnostics[0].code, "SC2030");
89        assert_eq!(result.diagnostics[0].severity, Severity::Warning);
90        assert!(result.diagnostics[0].message.contains("foo"));
91    }
92
93    #[test]
94    fn test_sc2030_subshell_with_command() {
95        let code = r#"(x=5; echo "$x")"#;
96        let result = check(code);
97        assert_eq!(result.diagnostics.len(), 1);
98        assert!(result.diagnostics[0].message.contains("x"));
99    }
100
101    #[test]
102    fn test_sc2030_direct_assignment_ok() {
103        let code = r#"foo=bar"#;
104        let result = check(code);
105        assert_eq!(result.diagnostics.len(), 0);
106    }
107
108    #[test]
109    fn test_sc2030_command_subst_ok() {
110        let code = r#"result=$(x=5; echo "$x")"#;
111        let result = check(code);
112        // Command substitution is OK (output captured)
113        assert_eq!(result.diagnostics.len(), 0);
114    }
115
116    #[test]
117    fn test_sc2030_nested_parens() {
118        let code = r#"((foo=bar))"#;
119        let result = check(code);
120        // Arithmetic (( )) or nested subshell
121        assert_eq!(result.diagnostics.len(), 1);
122    }
123
124    #[test]
125    fn test_sc2030_in_quotes_ok() {
126        let code = r#"echo "(foo=bar)""#;
127        let result = check(code);
128        // Inside quotes, not actual subshell
129        assert_eq!(result.diagnostics.len(), 0);
130    }
131
132    #[test]
133    fn test_sc2030_multiple_assignments() {
134        let code = r#"
135(a=1)
136(b=2)
137"#;
138        let result = check(code);
139        assert_eq!(result.diagnostics.len(), 2);
140    }
141
142    #[test]
143    fn test_sc2030_with_spaces() {
144        let code = r#"( foo=bar )"#;
145        let result = check(code);
146        assert_eq!(result.diagnostics.len(), 1);
147    }
148
149    #[test]
150    fn test_sc2030_comment_ok() {
151        let code = r#"# (foo=bar)"#;
152        let result = check(code);
153        assert_eq!(result.diagnostics.len(), 0);
154    }
155
156    #[test]
157    fn test_sc2030_export_in_subshell() {
158        let code = r#"(export PATH=/usr/bin)"#;
159        let result = check(code);
160        // export is not caught by simple assignment pattern (would need separate check)
161        assert_eq!(result.diagnostics.len(), 0);
162    }
