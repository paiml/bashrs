1// SC2059: Don't use variables in the printf format string. Use printf '..%s..' "$foo"
2//
3// Using variables in printf format strings can lead to format string injection vulnerabilities.
4// If the variable contains format specifiers like %s, %d, or %n, they will be interpreted
5// by printf, potentially causing crashes, information leaks, or arbitrary code execution.
6//
7// Examples:
8// Bad:
9//   printf "$format" "$value"        // Format string injection
10//   printf "Value: $var\n"           // Variable expansion in format
11//   printf "$msg"                    // Direct variable as format
12//
13// Good:
14//   printf '%s\n' "$value"           // Literal format string
15//   printf 'Value: %s\n' "$var"      // Literal format with %s
16//   printf '%s' "$msg"               // Safe variable output
17//
18// Security Impact:
19//   - Format string vulnerabilities (arbitrary memory read/write)
20//   - Information disclosure
21//   - Denial of service (crashes)
22//   - Potential code execution in some implementations
23//
24// Note: Always use literal format strings with printf. Use %s to safely output variables.
25
26use crate::linter::{Diagnostic, LintResult, Severity, Span};
27use once_cell::sync::Lazy;
28use regex::Regex;
29
30static PRINTF_WITH_VAR: Lazy<Regex> = Lazy::new(|| {
31    // Match: printf "$var" or printf "...$var..."
32    Regex::new(r#"printf\s+(['"]?)(\$[a-zA-Z_][a-zA-Z0-9_]*|\$\{[a-zA-Z_][a-zA-Z0-9_]*\})"#)
33        .unwrap()
34});
35
36static PRINTF_WITH_EXPANSION: Lazy<Regex> = Lazy::new(|| {
37    // Match: printf "...$var..." (variable in format string)
38    Regex::new(r#"printf\s+"[^"]*\$[a-zA-Z_][a-zA-Z0-9_]*"#).unwrap()
39});
40
41pub fn check(source: &str) -> LintResult {
42    let mut result = LintResult::new();
43
44    for (line_num, line) in source.lines().enumerate() {
45        let line_num = line_num + 1;
46
47        if line.trim_start().starts_with('#') {
48            continue;
49        }
50
51        // Check for printf with variable as format string
52        if let Some(mat) = PRINTF_WITH_VAR.find(line) {
53            let start_col = mat.start() + 1;
54            let end_col = mat.end() + 1;
55
56            let diagnostic = Diagnostic::new(
57                "SC2059",
58                Severity::Error,
59                "Don't use variables in the printf format string. Use printf '..%s..' \"$foo\""
60                    .to_string(),
61                Span::new(line_num, start_col, line_num, end_col),
62            );
63
64            result.add(diagnostic);
65        }
66
67        // Check for printf with variable expansion in format string
68        if let Some(mat) = PRINTF_WITH_EXPANSION.find(line) {
69            // Skip if already caught by first pattern
70            if !PRINTF_WITH_VAR.is_match(line) {
71                let start_col = mat.start() + 1;
72                let end_col = mat.end() + 1;
73
74                let diagnostic = Diagnostic::new(
75                    "SC2059",
76                    Severity::Error,
77                    "Don't use variables in the printf format string. Use printf '..%s..' \"$foo\""
78                        .to_string(),
79                    Span::new(line_num, start_col, line_num, end_col),
80                );
81
82                result.add(diagnostic);
83            }
84        }
85    }
86
87    result
88}

94    #[test]
95    fn test_sc2059_variable_as_format() {
96        let code = r#"printf "$format" "value""#;
97        let result = check(code);
98        assert_eq!(result.diagnostics.len(), 1);
99        assert_eq!(result.diagnostics[0].code, "SC2059");
100        assert_eq!(result.diagnostics[0].severity, Severity::Error);
101    }

108    }
