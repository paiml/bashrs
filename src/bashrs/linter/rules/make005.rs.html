1//! MAKE005: Recursive variable assignment in Makefiles
2//!
3//! **Rule**: Detect `=` (recursive expansion) used with `$(shell ...)` that should use `:=` (immediate expansion)
4//!
5//! **Why this matters**:
6//! Using `=` with `$(shell ...)` causes the shell command to be re-executed
7//! every time the variable is referenced, leading to non-deterministic behavior
8//! and performance issues. Use `:=` for immediate, one-time expansion.
9//!
10//! **Auto-fix**: Change `=` to `:=` for shell-based assignments
11//!
12//! ## Examples
13//!
14//! ❌ **BAD** (recursive expansion):
15//! ```makefile
16//! VERSION = $(shell git describe)
17//! TIMESTAMP = $(shell date +%s)
18//! ```
19//!
20//! ✅ **GOOD** (immediate expansion):
21//! ```makefile
22//! VERSION := $(shell git describe)
23//! TIMESTAMP := $(shell date +%s)
24//! ```
25
26use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
27
28/// Check if a line should be skipped (comment, empty, or recipe)
29fn should_skip_line(line: &str) -> bool {
30    line.trim_start().starts_with('#') || line.trim().is_empty() || line.starts_with('\t')
31}
32
33/// Get the character before equals sign
34fn get_char_before_eq(line: &str, eq_pos: usize) -> Option<char> {
35    if eq_pos > 0 {
36        line.chars().nth(eq_pos - 1)
37    } else {
38        None
39    }
40}
41
42/// Check if assignment is a special operator (:=, !=, +=, ?=)
43fn is_special_operator(before_eq: Option<char>) -> bool {
44    before_eq == Some(':')
45        || before_eq == Some('!')
46        || before_eq == Some('+')
47        || before_eq == Some('?')
48}
49
50/// Check if value contains $(shell ...)
51fn contains_shell_command(value: &str) -> bool {
52    value.contains("$(shell")
53}
54
55/// Create diagnostic for recursive shell assignment
56fn create_shell_diagnostic(
57    var_name: &str,
58    after_eq: &str,
59    line_num: usize,
60    eq_pos: usize,
61) -> Diagnostic {
62    let span = Span::new(line_num + 1, eq_pos + 1, line_num + 1, eq_pos + 2);
63    let fix_replacement = format!("{}:={}", var_name, after_eq);
64
65    Diagnostic::new(
66        "MAKE005",
67        Severity::Warning,
68        format!(
69            "Variable '{}' uses recursive expansion (=) with $(shell ...) - use := for immediate expansion",
70            var_name
71        ),
72        span,
73    )
74    .with_fix(Fix::new(&fix_replacement))
75}
76
77/// Check for recursive variable assignments with $(shell ...)
78pub fn check(source: &str) -> LintResult {
79    let mut result = LintResult::new();
80
81    for (line_num, line) in source.lines().enumerate() {
82        if should_skip_line(line) {
83            continue;
84        }
85
86        // Check for variable assignment with =
87        if let Some(eq_pos) = line.find('=') {
88            let before_eq = get_char_before_eq(line, eq_pos);
89
90            if is_special_operator(before_eq) {
91                continue;
92            }
93
94            // Check if the value contains $(shell ...)
95            let after_eq = &line[eq_pos + 1..];
96            if contains_shell_command(after_eq) {
97                let var_name = line[..eq_pos].trim();
98                let diag = create_shell_diagnostic(var_name, after_eq, line_num, eq_pos);
99                result.add(diag);
100            }
101        }
102    }
103
104    result
105}
