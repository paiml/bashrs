1//! SC2086: Double quote to prevent globbing and word splitting
2//!
3//! Detects unquoted variable expansions that could cause:
4//! - Word splitting on IFS characters (space, tab, newline)
5//! - Glob expansion of *, ?, [...] patterns
6//!
7//! References:
8//! - https://www.shellcheck.net/wiki/SC2086
9//! - POSIX Shell Command Language Section 2.6.2
10
11use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
12use regex::Regex;
13
14/// Check if line should be skipped (comments or assignments)
15fn should_skip_line(line: &str) -> bool {
16    // Skip comments
17    if line.trim_start().starts_with('#') {
18        return true;
19    }
20
21    // Skip variable assignments (VAR=value)
22    if line.contains('=') && !line.contains("if [") && !line.contains("[ ") {
23        if let Some(eq_pos) = line.find('=') {
24            if let Some(first_space) = line.find(' ') {
25                if eq_pos < first_space {
26                    return true; // Assignment, not command
27                }
28            }
29        }
30    }
31
32    false
33}

35/// Find the position of $ character before a variable
36fn find_dollar_position(line: &str, var_start: usize) -> usize {
37    line[..var_start].rfind('$').unwrap_or(var_start)
38}
39
40/// Calculate end column for variable span, including closing brace if present
41fn calculate_end_column(line: &str, var_end: usize, is_braced: bool) -> usize {
42    if is_braced {
43        let after_var = &line[var_end..];
44        if let Some(brace_pos) = after_var.find('}') {
45            var_end + brace_pos + 2 // +1 for }, +1 for 1-indexing
46        } else {
47            var_end + 1 // Fallback
48        }
49    } else {
50        var_end + 1 // Simple $VAR case
51    }
52}
53
54/// Check if variable is in arithmetic context (inside $(( )))
55fn is_in_arithmetic_context(line: &str, dollar_pos: usize, var_end: usize) -> bool {
56    let before = &line[..dollar_pos];
57    let after = &line[var_end..];
58    before.contains("$((") && after.contains("))")
59}
60
61/// Check if variable is already quoted
62fn is_already_quoted(line: &str, dollar_pos: usize, var_end: usize) -> bool {
63    let before_context = &line[..dollar_pos];
64    let after_context = &line[var_end..];
65    before_context.ends_with('"') && after_context.starts_with('"')
66}

68/// Build diagnostic for unquoted variable
69fn build_diagnostic(
70    line_num: usize,
71    col: usize,
72    end_col: usize,
73    var_name: &str,
74    is_braced: bool,
75) -> Diagnostic {
76    let span = Span::new(line_num, col, line_num, end_col);
77    let var_text = if is_braced {
78        format!("${{{}}}", var_name)
79    } else {
80        format!("${}", var_name)
81    };
82
83    let fix = Fix::new(format!("\"{}\"", var_text));
84
85    Diagnostic::new(
86        "SC2086",
87        Severity::Warning,
88        format!(
89            "Double quote to prevent globbing and word splitting on {}",
90            var_text
91        ),
92        span,
93    )
94    .with_fix(fix)
95}

97/// Check for unquoted variable expansions (SC2086)
98pub fn check(source: &str) -> LintResult {
99    let mut result = LintResult::new();
100
101    // Regex to find unquoted variables in command contexts
102    let var_pattern = Regex::new(r#"(?m)(?P<pre>[^"']|^)\$(?:\{(?P<brace>[A-Za-z_][A-Za-z0-9_]*)\}|(?P<simple>[A-Za-z_][A-Za-z0-9_]*))"#).unwrap();
103
104    for (line_num, line) in source.lines().enumerate() {
105        let line_num = line_num + 1; // 1-indexed
106
107        if should_skip_line(line) {
108            continue;
109        }
110
111        let is_arithmetic = line.contains("$((") || line.contains("(( ");
112
113        for cap in var_pattern.captures_iter(line) {
114            let var_capture = match cap.name("brace").or_else(|| cap.name("simple")) {
115                Some(v) => v,
116                None => continue,
117            };
118
119            let var_name = var_capture.as_str();
120            let dollar_pos = find_dollar_position(line, var_capture.start());
121            let col = dollar_pos + 1; // 1-indexed
122
123            let is_braced = cap.name("brace").is_some();
124            let end_col = calculate_end_column(line, var_capture.end(), is_braced);
125
126            // Skip if in arithmetic context or already quoted
127            if is_arithmetic && is_in_arithmetic_context(line, dollar_pos, var_capture.end()) {
128                continue;
129            }
130
131            if is_already_quoted(line, dollar_pos, var_capture.end()) {
132                continue;
133            }
134
135            let diag = build_diagnostic(line_num, col, end_col, var_name, is_braced);
136            result.add(diag);
137        }
138    }
139
140    result
141}

780                var_name in "[a-z]{1,10}",
781                leading_spaces in 0usize..20
782            ) {
783                // PROPERTY: Column positions must always be >= 1 (1-indexed)
784                // Catches: +  * mutations (would produce 0), +  - mutations
785                let spaces = " ".repeat(leading_spaces);
786                let bash_code = format!("{}echo ${}", spaces, var_name);
787                let result = check(&bash_code);
788
789                if result.diagnostics.len() > 0 {
790                    let span = &result.diagnostics[0].span;
791                    // INVARIANT: Columns are 1-indexed, never 0 or negative
792                    prop_assert!(span.start_col >= 1, "Start column must be >= 1, got {}", span.start_col);
793                    prop_assert!(span.end_col >= 1, "End column must be >= 1, got {}", span.end_col);
794                    // INVARIANT: End must be after start
795                    prop_assert!(span.end_col > span.start_col,
796                        "End col ({}) must be > start col ({})", span.end_col, span.start_col);
797                }
798            }
