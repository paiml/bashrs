1//! SC2086: Double quote to prevent globbing and word splitting
2//!
3//! Detects unquoted variable expansions that could cause:
4//! - Word splitting on IFS characters (space, tab, newline)
5//! - Glob expansion of *, ?, [...] patterns
6//!
7//! References:
8//! - https://www.shellcheck.net/wiki/SC2086
9//! - POSIX Shell Command Language Section 2.6.2
10
11use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
12use regex::Regex;
13
14/// Check for unquoted variable expansions (SC2086)
15pub fn check(source: &str) -> LintResult {
16    let mut result = LintResult::new();
17
18    // Regex to find unquoted variables in command contexts
19    // Matches: $VAR or ${VAR} not within quotes
20    let var_pattern = Regex::new(r#"(?m)(?P<pre>[^"']|^)\$(?:\{(?P<brace>[A-Za-z_][A-Za-z0-9_]*)\}|(?P<simple>[A-Za-z_][A-Za-z0-9_]*))"#).unwrap();
21
22    for (line_num, line) in source.lines().enumerate() {
23        let line_num = line_num + 1; // 1-indexed
24
25        // Skip comments
26        if line.trim_start().starts_with('#') {
27            continue;
28        }
29
30        // Skip variable assignments (VAR=value)
31        if line.contains('=') && !line.contains("if [") && !line.contains("[ ") {
32            // Simple heuristic: if line has = before any command, it's likely an assignment
33            if let Some(eq_pos) = line.find('=') {
34                if let Some(first_space) = line.find(' ') {
35                    if eq_pos < first_space {
36                        continue; // Assignment, not command
37                    }
38                }
39            }
40        }
41
42        // Check for unquoted variables in arithmetic contexts (false positive prevention)
43        let is_arithmetic = line.contains("$((") || line.contains("(( ");
44
45        for cap in var_pattern.captures_iter(line) {
46            // Get the actual variable match (not including 'pre' capture group)
47            let var_capture = cap.name("brace").or_else(|| cap.name("simple"));
48
49            if var_capture.is_none() {
50                continue;
51            }
52
53            let var_match = var_capture.unwrap();
54            let var_name = var_match.as_str();
55
56            // Column positions for the actual $VAR or ${VAR} (including $)
57            // We need to go back one character from the capture to include the $
58            let dollar_pos = line[..var_match.start()].rfind('$').unwrap_or(var_match.start());
59            let col = dollar_pos + 1; // 1-indexed
60            let end_col = var_match.end() + 1; // 1-indexed (after last char)
61
62            // Skip if in arithmetic context
63            if is_arithmetic {
64                // Check if this variable is inside $(( ))
65                let before = &line[..dollar_pos];
66                let after = &line[var_match.end()..];
67                if before.contains("$((") && after.contains("))") {
68                    continue;
69                }
70            }
71
72            // Check if already quoted
73            let before_context = &line[..dollar_pos];
74            let after_context = &line[var_match.end()..];
75
76            // Simple quote detection
77            let has_opening_quote = before_context.ends_with('"');
78            let has_closing_quote = after_context.starts_with('"');
79
80            if has_opening_quote && has_closing_quote {
81                continue; // Already quoted
82            }
83
84            // Create diagnostic
85            let span = Span::new(line_num, col, line_num, end_col);
86            let var_text = if cap.name("brace").is_some() {
87                format!("${{{}}}", var_name)
88            } else {
89                format!("${}", var_name)
90            };
91
92            let fix = Fix::new(format!("\"{}\"", var_text));
93
94            let diag = Diagnostic::new(
95                "SC2086",
96                Severity::Warning,
97                format!("Double quote to prevent globbing and word splitting on {}", var_text),
98                span,
99            )
100            .with_fix(fix);
101
102            result.add(diag);
103        }
104    }
105
106    result
107}

109#[cfg(test)]
110mod tests {
111    use super::*;
112
113    #[test]
114    fn test_sc2086_basic_detection() {
115        let bash_code = r#"
116#!/bin/bash
117FILES=$1
118ls $FILES
119"#;
120
121        let result = check(bash_code);
122        assert_eq!(result.diagnostics.len(), 1, "Should detect one unquoted variable");
123        assert_eq!(result.diagnostics[0].code, "SC2086");
124        assert!(result.diagnostics[0].message.contains("Double quote"));
125        assert!(result.diagnostics[0].message.contains("$FILES"));
126    }
127
128    #[test]
129    fn test_sc2086_autofix() {
130        let bash_code = "ls $FILES";
131        let result = check(bash_code);
132
133        assert_eq!(result.diagnostics.len(), 1);
134        assert!(result.diagnostics[0].fix.is_some());
135        assert_eq!(result.diagnostics[0].fix.as_ref().unwrap().replacement, "\"$FILES\"");
136    }
137
138    #[test]
139    fn test_sc2086_no_false_positive_arithmetic() {
140        let bash_code = "result=$(( $x + $y ))";
141        let result = check(bash_code);
142
143        // Should NOT trigger SC2086 in arithmetic context
144        assert_eq!(result.diagnostics.len(), 0, "Should not trigger in arithmetic context");
145    }
146
147    #[test]
148    fn test_sc2086_multiple_violations() {
149        let bash_code = r#"
150rm -rf $DIR
151cat $FILE1 $FILE2
152"#;
153
154        let result = check(bash_code);
155        assert_eq!(result.diagnostics.len(), 3, "Should detect three unquoted variables");
156
157        let codes: Vec<&str> = result.diagnostics.iter().map(|d| d.code.as_str()).collect();
158        assert_eq!(codes, vec!["SC2086", "SC2086", "SC2086"]);
159    }
160
161    #[test]
162    fn test_sc2086_braced_variables() {
163        let bash_code = r#"echo ${VAR}"#;
164        let result = check(bash_code);
165
166        assert_eq!(result.diagnostics.len(), 1);
167        assert!(result.diagnostics[0].message.contains("${VAR}"));
168    }
169
170    #[test]
171    fn test_sc2086_skip_comments() {
172        let bash_code = r#"
173# This is a comment with $VAR
174echo $ACTUAL
175"#;
176
177        let result = check(bash_code);
178        assert_eq!(result.diagnostics.len(), 1, "Should only detect variable in echo, not comment");
179        assert!(result.diagnostics[0].message.contains("$ACTUAL"));
180    }
181
182    #[test]
183    fn test_sc2086_skip_quoted() {
184        let bash_code = r#"echo "$VAR""#;
185        let result = check(bash_code);
186
187        // Should NOT trigger - already quoted
188        assert_eq!(result.diagnostics.len(), 0, "Should not trigger on already-quoted variables");
189    }
190
191    #[test]
192    fn test_sc2086_mixed_quoted_unquoted() {
193        let bash_code = r#"
194echo "$VAR1"
195echo $VAR2
196echo "$VAR3"
197"#;
198
199        let result = check(bash_code);
200        assert_eq!(result.diagnostics.len(), 1, "Should only detect unquoted $VAR2");
201        assert!(result.diagnostics[0].message.contains("$VAR2"));
202    }
203
204    #[test]
205    fn test_sc2086_severity() {
206        let bash_code = "ls $FILES";
207        let result = check(bash_code);
208
209        assert_eq!(result.diagnostics[0].severity, Severity::Warning);
210    }
211
212    #[test]
213    fn test_sc2086_span_accuracy() {
214        let bash_code = "ls $FILES";
215        let result = check(bash_code);
216
217        let span = result.diagnostics[0].span;
218        assert_eq!(span.start_line, 1);
219        assert_eq!(span.end_line, 1);
220        // Column positions should point to $FILES
221        assert!(span.start_col <= 4); // "ls " is 3 chars
222        assert!(span.end_col >= span.start_col);
223    }
