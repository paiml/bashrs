1//! SC2086: Double quote to prevent globbing and word splitting
2//!
3//! Detects unquoted variable expansions that could cause:
4//! - Word splitting on IFS characters (space, tab, newline)
5//! - Glob expansion of *, ?, [...] patterns
6//!
7//! References:
8//! - https://www.shellcheck.net/wiki/SC2086
9//! - POSIX Shell Command Language Section 2.6.2
10
11use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
12use regex::Regex;

14/// Check if line should be skipped (comments or assignments)
15fn should_skip_line(line: &str) -> bool {
16    // Skip comments
17    if line.trim_start().starts_with('#') {
18        return true;
19    }
20
21    // Skip variable assignments (VAR=value)
22    if line.contains('=') && !line.contains("if [") && !line.contains("[ ") {
23        if let Some(eq_pos) = line.find('=') {
24            if let Some(first_space) = line.find(' ') {
25                if eq_pos < first_space {
26                    return true; // Assignment, not command
27                }
28            }
29        }
30    }
31
32    false
33}

35/// Find the position of $ character before a variable
36fn find_dollar_position(line: &str, var_start: usize) -> usize {
37    line[..var_start].rfind('$').unwrap_or(var_start)
38}

40/// Calculate end column for variable span, including closing brace if present
41fn calculate_end_column(line: &str, var_end: usize, is_braced: bool) -> usize {
42    if is_braced {
43        let after_var = &line[var_end..];
44        if let Some(brace_pos) = after_var.find('}') {
45            var_end + brace_pos + 2 // +1 for }, +1 for 1-indexing
46        } else {
47            var_end + 1 // Fallback
48        }
49    } else {
50        var_end + 1 // Simple $VAR case
51    }
52}

54/// Check if variable is in arithmetic context (inside $(( )) or (( )))
55/// Issue #107: Also handles C-style for loops, standalone (( )), while/if (( ))
56fn is_in_arithmetic_context(line: &str, dollar_pos: usize, var_end: usize) -> bool {
57    let before = &line[..dollar_pos];
58    let after = &line[var_end..];
59
60    // Case 1: Command substitution arithmetic $(( ))
61    if before.contains("$((") && after.contains("))") {
62        return true;
63    }
64
65    // Case 2: Standalone arithmetic (( )) - for loops, while, if, statements
66    // Look for (( that is NOT preceded by $ (to distinguish from $(( ))
67    if let Some(paren_pos) = before.rfind("((") {
68        // Verify it's standalone (( not $((
69        let is_standalone = if paren_pos > 0 {
70            !before[..paren_pos].ends_with('$')
71        } else {
72            true
73        };
74
75        if is_standalone && after.contains("))") {
76            return true;
77        }
78    }
79
80    false
81}

83/// F048: Extract C-style for loop variable names from source
84/// C-style for loops: for ((i=0; i<n; i++)) define numeric loop variables
85/// These variables are always numeric, so SC2086 should not flag them
86fn get_cstyle_for_loop_vars(source: &str) -> std::collections::HashSet<String> {
87    use once_cell::sync::Lazy;
88    use regex::Regex;
89
90    #[allow(clippy::unwrap_used)] // Compile-time regex
91    static CSTYLE_FOR: Lazy<Regex> =
92        Lazy::new(|| Regex::new(r"\bfor\s*\(\(\s*([A-Za-z_][A-Za-z0-9_]*)\s*=").unwrap());
93
94    let mut vars = std::collections::HashSet::new();
95    for cap in CSTYLE_FOR.captures_iter(source) {
96        if let Some(m) = cap.get(1) {
97            vars.insert(m.as_str().to_string());
98        }
99    }
100    vars
101}

103/// Issue #105: Check if variable is inside [[ ]] (bash extended test)
104/// In [[ ]], word splitting and glob expansion do NOT occur on unquoted variables
105/// This is safe: [[ -n $var ]] (no word splitting inside [[ ]])
106/// This is NOT safe: [ -n $var ] (word splitting occurs in [ ])
107fn is_in_double_bracket_context(line: &str, dollar_pos: usize, var_end: usize) -> bool {
108    let before = &line[..dollar_pos];
109    let after = &line[var_end..];
110
111    // Check for [[ before and ]] after
112    // Must be [[ not [ (single bracket still has word splitting)
113    if let Some(open_pos) = before.rfind("[[") {
114        // Make sure it's not a single bracket by checking the character before
115        let is_double = if open_pos > 0 {
116            // Check there's no [ immediately before (would be [[[)
117            !before[..open_pos].ends_with('[')
118        } else {
119            true
120        };
121
122        if is_double && after.contains("]]") {
123            return true;
124        }
125    }
126
127    false
128}

130/// Check if variable is already quoted
131fn is_already_quoted(line: &str, dollar_pos: usize, var_end: usize) -> bool {
132    let before_context = &line[..dollar_pos];
133    let after_context = &line[var_end..];
134
135    // Simple case: "$VAR" (immediately surrounded by quotes)
136    if before_context.ends_with('"') && after_context.starts_with('"') {
137        return true;
138    }
139
140    // Braced case: "${VAR}" (immediately surrounded by quotes)
141    if after_context.starts_with('}') {
142        if let Some(brace_pos) = after_context.find('}') {
143            let after_brace = &after_context[brace_pos + 1..];
144            if before_context.ends_with('"') && after_brace.starts_with('"') {
145                return true;
146            }
147        }
148    }
149
150    // Check if variable is inside a quoted string (e.g., "${VAR1}text${VAR2}")
151    // Count unescaped quotes before the variable
152    let mut quote_count = 0;
153    let mut escaped = false;
154    for ch in before_context.chars() {
155        if escaped {
156            escaped = false;
157            continue;
158        }
159        if ch == '\\' {
160            escaped = true;
161            continue;
162        }
163        if ch == '"' {
164            quote_count += 1;
165        }
166    }
167
168    // If odd number of quotes, we're inside a quoted string
169    // Check if there's a closing quote after the variable
170    if quote_count % 2 == 1 {
171        // For braced variables, check after the closing brace
172        if after_context.starts_with('}') {
173            if let Some(brace_pos) = after_context.find('}') {
174                let after_brace = &after_context[brace_pos + 1..];
175                // Look for closing quote (could be immediately or after more content)
176                if after_brace.contains('"') {
177                    return true;
178                }
179            }
180        } else {
181            // For simple variables, check after the variable name
182            if after_context.contains('"') {
183                return true;
184            }
185        }
186    }
187
188    false
189}

191/// Build diagnostic for unquoted variable
192fn build_diagnostic(
193    line_num: usize,
194    col: usize,
195    end_col: usize,
196    var_name: &str,
197    is_braced: bool,
198) -> Diagnostic {
199    let span = Span::new(line_num, col, line_num, end_col);
200    let var_text = if is_braced {
201        format!("${{{}}}", var_name)
202    } else {
203        format!("${}", var_name)
204    };
205
206    let fix = Fix::new(format!("\"{}\"", var_text));
207
208    Diagnostic::new(
209        "SC2086",
210        Severity::Warning,
211        format!(
212            "Double quote to prevent globbing and word splitting on {}",
213            var_text
214        ),
215        span,
216    )
217    .with_fix(fix)
218}

220/// Check for unquoted variable expansions (SC2086)
221pub fn check(source: &str) -> LintResult {
222    let mut result = LintResult::new();
223
224    // F048: Get C-style for loop variables (always numeric, skip quoting check)
225    let cstyle_loop_vars = get_cstyle_for_loop_vars(source);
226
227    // Regex to find unquoted variables in command contexts
228    let var_pattern = Regex::new(r#"(?m)(?P<pre>[^"']|^)\$(?:\{(?P<brace>[A-Za-z_][A-Za-z0-9_]*)\}|(?P<simple>[A-Za-z_][A-Za-z0-9_]*))"#).unwrap();
229
230    for (line_num, line) in source.lines().enumerate() {
231        let line_num = line_num + 1; // 1-indexed
232
233        if should_skip_line(line) {
234            continue;
235        }
236
237        // Issue #107: Check for any arithmetic context - $(( )) or standalone (( ))
238        let is_arithmetic = line.contains("$((") || line.contains("((");
239
240        for cap in var_pattern.captures_iter(line) {
241            let var_capture = match cap.name("brace").or_else(|| cap.name("simple")) {
242                Some(v) => v,
243                None => continue,
244            };
245
246            let var_name = var_capture.as_str();
247            let dollar_pos = find_dollar_position(line, var_capture.start());
248            let col = dollar_pos + 1; // 1-indexed
249
250            let is_braced = cap.name("brace").is_some();
251            let end_col = calculate_end_column(line, var_capture.end(), is_braced);
252
253            // Skip if in arithmetic context or already quoted
254            if is_arithmetic && is_in_arithmetic_context(line, dollar_pos, var_capture.end()) {
255                continue;
256            }
257
258            if is_already_quoted(line, dollar_pos, var_capture.end()) {
259                continue;
260            }
261
262            // Issue #105: Skip if in [[ ]] context (no word splitting in bash extended test)
263            if is_in_double_bracket_context(line, dollar_pos, var_capture.end()) {
264                continue;
265            }
266
267            // F048: Skip C-style for loop variables (always numeric)
268            if cstyle_loop_vars.contains(var_name) {
269                continue;
270            }
271
272            let diag = build_diagnostic(line_num, col, end_col, var_name, is_braced);
273            result.add(diag);
274        }
275    }
276
277    result
278}
