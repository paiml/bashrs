1// SC2194: This word is constant - consider using the command name directly
2//
3// When a variable is assigned a constant command name and then immediately executed,
4// it's clearer and more efficient to just run the command directly.
5//
6// Examples:
7// Bad:
8//   cmd="ls"
9//   $cmd                   // Just use 'ls' directly
10//
11//   command="grep"
12//   $command pattern file  // Just use 'grep pattern file'
13//
14// Good:
15//   ls                     // Direct command
16//   grep pattern file      // Direct command
17//
18// When variable command execution is OK:
19//   # When command varies based on condition
20//   if [ condition ]; then
21//     cmd="ls"
22//   else
23//     cmd="find"
24//   fi
25//   $cmd                  // Variable command is meaningful here
26//
27// Note: This rule catches simple cases where a constant is assigned and
28// immediately used, making the code unnecessarily indirect.
29
30use crate::linter::{Diagnostic, LintResult, Severity, Span};
31use once_cell::sync::Lazy;
32use regex::Regex;
33
34static COMMAND_VAR_ASSIGNMENT: Lazy<Regex> = Lazy::new(|| {
35    // Match: cmd="command" or command='command' or cmd=command
36    Regex::new(r##"^([a-zA-Z_][a-zA-Z0-9_]*)=["']?([a-z_][a-z0-9_-]*)["']?\s*$"##).unwrap()
37});
38
39static COMMAND_VAR_USAGE: Lazy<Regex> = Lazy::new(|| {
40    // Match: $cmd or ${cmd} at start of command
41    Regex::new(r"^\s*\$(\{)?([a-zA-Z_][a-zA-Z0-9_]*)(\})?").unwrap()
42});
43
44pub fn check(source: &str) -> LintResult {
45    let mut result = LintResult::new();
46    let lines: Vec<&str> = source.lines().collect();
47
48    #[allow(clippy::needless_range_loop)]
49    for i in 0..lines.len() {
50        let line = lines[i];
51        let line_num = i + 1;
52
53        if line.trim_start().starts_with('#') {
54            continue;
55        }
56
57        // Check if this line assigns a constant to a variable
58        if let Some(assign_cap) = COMMAND_VAR_ASSIGNMENT.captures(line.trim()) {
59            let var_name = assign_cap.get(1).unwrap().as_str();
60            let command_name = assign_cap.get(2).unwrap().as_str();
61
62            // Look at the next non-empty, non-comment line
63            #[allow(clippy::needless_range_loop)]
64            for j in (i + 1)..lines.len() {
65                let next_line = lines[j];
66                let next_trimmed = next_line.trim();
67
68                if next_trimmed.is_empty() || next_trimmed.starts_with('#') {
69                    continue;
70                }
71
72                // Check if next line uses this variable as a command
73                if let Some(usage_cap) = COMMAND_VAR_USAGE.captures(next_trimmed) {
74                    let used_var = usage_cap.get(2).unwrap().as_str();
75
76                    if used_var == var_name {
77                        let start_col = next_line.find('$').unwrap_or(0) + 1;
78                        let end_col = start_col + usage_cap.get(0).unwrap().as_str().len();
79
80                        let diagnostic = Diagnostic::new(
81                            "SC2194",
82                            Severity::Info,
83                            format!(
84                                "This variable '{}' is constant (assigned '{}' on line {}). Consider using '{}' directly",
85                                var_name, command_name, line_num, command_name
86                            ),
87                            Span::new(j + 1, start_col, j + 1, end_col),
88                        );
89
90                        result.add(diagnostic);
91                    }
92                }
93
94                // Stop at first non-comment line
95                break;
96            }
97        }
98    }
99
100    result
101}

103#[cfg(test)]
104mod tests {
105    use super::*;
106
107    #[test]
108    fn test_sc2194_constant_command() {
109        let code = r#"
110cmd="ls"
111$cmd
112"#;
113        let result = check(code);
114        assert_eq!(result.diagnostics.len(), 1);
115        assert_eq!(result.diagnostics[0].code, "SC2194");
116        assert_eq!(result.diagnostics[0].severity, Severity::Info);
117        assert!(result.diagnostics[0].message.contains("constant"));
118    }
119
120    #[test]
121    fn test_sc2194_constant_grep() {
122        let code = r#"
123command="grep"
124$command pattern file
125"#;
126        let result = check(code);
127        assert_eq!(result.diagnostics.len(), 1);
128    }
