1//! SEC019: Injection Safety - Unquoted Variables
2//!
3//! **Rule**: Detect unquoted variable expansions that may lead to injection attacks
4//!
5//! **Why this matters**:
6//! Unquoted variables can be exploited for command injection attacks. When user-controlled
7//! data is expanded without quotes, attackers can inject shell metacharacters (`;`, `|`, `&`, etc.)
8//! to execute arbitrary commands.
9//!
10//! **Auto-fix**: Add quotes around variable expansions
11//!
12//! ## Examples
13//!
14//! ❌ **UNSAFE** (Injection Risk):
15//! ```bash
16//! echo $user_input       # Attacker: user_input="; rm -rf /"
17//! rm -rf $dir            # Attacker: dir="/; cat /etc/passwd"
18//! eval $command          # Critical: arbitrary code execution
19//! ```
20//!
21//! ✅ **SAFE**:
22//! ```bash
23//! echo "$user_input"     # Quoted: safe from injection
24//! rm -rf "$dir"          # Quoted: safe from injection
25//! # Avoid eval entirely, or use arrays
26//! ```
27//!
28//! ## Security Properties (Taint Tracking)
29//!
30//! This rule implements basic taint tracking:
31//! - Variables from external sources are considered **Tainted**
32//! - Unquoted tainted variables → **Injection Risk**
33//! - Quoted variables → **Sanitized** (safe)
34//!
35//! ## Exceptions
36//!
37//! - Variables in `[[ ]]` test contexts (safe by design)
38//! - Arithmetic expressions `$(( ))` (safe by design)
39//! - Some control flow constructs where quoting not needed
40
41use crate::linter::{Diagnostic, LintResult, Severity, Span};

43/// Check for unquoted variable expansions (injection risk)
44pub fn check(source: &str) -> LintResult {
45    let mut result = LintResult::new();
46
47    for (line_num, line) in source.lines().enumerate() {
48        check_line(line, line_num, &mut result);
49    }
50
51    result
52}
53
54fn check_line(line: &str, line_num: usize, result: &mut LintResult) {
55    let chars: Vec<char> = line.chars().collect();
56    let mut i = 0;
57
58    while i < chars.len() {
59        // Look for $VAR or ${VAR}
60        if chars[i] == '$' && i + 1 < chars.len() {
61            let start_col = i;
62
63            // Skip $$ (process ID) - special case, not user data
64            if chars[i + 1] == '$' {
65                i += 2;
66                continue;
67            }
68
69            // Skip $? $# $@ $* (special variables)
70            if matches!(chars[i + 1], '?' | '#' | '@' | '*' | '!' | '-') {
71                i += 2;
72                continue;
73            }
74
75            // Skip $((  (arithmetic expansion - safe)
76            if i + 2 < chars.len() && chars[i + 1] == '(' && chars[i + 2] == '(' {
77                i = skip_arithmetic(&chars, i);
78                continue;
79            }
80
81            // Skip $(  (command substitution - check inside)
82            if chars[i + 1] == '(' {
83                i = skip_command_substitution(&chars, i);
84                continue;
85            }
86
87            // Check if variable is quoted
88            let is_quoted = is_variable_quoted(&chars, start_col);
89
90            // Check if in safe context
91            let in_safe_context = is_in_safe_context(line, start_col);
92
93            if !is_quoted && !in_safe_context {
94                // Unquoted variable expansion - potential injection
95                let var_end = find_variable_end(&chars, i);
96                let var_name = extract_variable_name(&chars, i, var_end);
97
98                let span = Span::new(line_num + 1, start_col + 1, line_num + 1, var_end + 1);
99
100                let message = format!(
101                    "Unquoted variable expansion ${} - injection risk. Use \"${}\" instead",
102                    var_name, var_name
103                );
104
105                let diag = Diagnostic::new("SEC019", Severity::Warning, &message, span);
106                result.add(diag);
107            }
108
109            i = find_variable_end(&chars, i);
110        } else {
111            i += 1;
112        }
113    }
114}

116/// Check if variable is quoted (inside "..." or '...')
117fn is_variable_quoted(chars: &[char], var_pos: usize) -> bool {
118    let mut in_double_quotes = false;
119    let mut in_single_quotes = false;
120
121    for ch in chars.iter().take(var_pos) {
122        match ch {
123            '"' if !in_single_quotes => in_double_quotes = !in_double_quotes,
124            '\'' if !in_double_quotes => in_single_quotes = !in_single_quotes,
125            _ => {}
126        }
127    }
128
129    // Double quotes protect against injection, single quotes don't expand vars
130    in_double_quotes || in_single_quotes
131}

133/// Check if variable is in a safe context (e.g., [[ ]], arithmetic)
134fn is_in_safe_context(line: &str, _var_pos: usize) -> bool {
135    // Check for [[ ]] test context (safe by design)
136    if line.contains("[[") && line.contains("]]") {
137        return true;
138    }
139
140    false
141}

143/// Find the end position of a variable expansion
144fn find_variable_end(chars: &[char], start: usize) -> usize {
145    if start + 1 >= chars.len() {
146        return start + 1;
147    }
148
149    let mut i = start + 1; // Skip $
150
151    if chars[i] == '{' {
152        // ${VAR} form
153        i += 1;
154        while i < chars.len() && chars[i] != '}' {
155            i += 1;
156        }
157        if i < chars.len() {
158            i += 1; // Include closing }
159        }
160    } else {
161        // $VAR form
162        while i < chars.len() && (chars[i].is_alphanumeric() || chars[i] == '_') {
163            i += 1;
164        }
165    }
166
167    i
168}

170/// Extract variable name from expansion
171fn extract_variable_name(chars: &[char], start: usize, end: usize) -> String {
172    let mut name_start = start + 1; // Skip $
173
174    if name_start < chars.len() && chars[name_start] == '{' {
175        name_start += 1; // Skip {
176    }
177
178    let mut name_end = end;
179    if name_end > name_start && chars.get(name_end - 1) == Some(&'}') {
180        name_end -= 1; // Exclude }
181    }
182
183    chars[name_start..name_end.min(chars.len())]
184        .iter()
185        .collect()
186}

188/// Skip arithmetic expansion $(( ... ))
189fn skip_arithmetic(chars: &[char], start: usize) -> usize {
190    let mut i = start + 3; // Skip $((
191    let mut depth = 1;
192
193    while i + 1 < chars.len() && depth > 0 {
194        if chars[i] == '(' && chars[i + 1] == '(' {
195            depth += 1;
196            i += 2;
197        } else if chars[i] == ')' && chars[i + 1] == ')' {
198            depth -= 1;
199            i += 2;
200        } else {
201            i += 1;
202        }
203    }
204
205    i
206}
207
208/// Skip command substitution $( ... )
209fn skip_command_substitution(chars: &[char], start: usize) -> usize {
210    let mut i = start + 2; // Skip $(
211    let mut depth = 1;
212
213    while i < chars.len() && depth > 0 {
214        if chars[i] == '$' && i + 1 < chars.len() && chars[i + 1] == '(' {
215            depth += 1;
216            i += 2;
217        } else if chars[i] == ')' {
218            depth -= 1;
219            i += 1;
220        } else {
221            i += 1;
222        }
223    }
224
225    i
226}
