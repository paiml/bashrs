1// SC2024: sudo doesn't affect redirects. Use '| sudo tee' instead
2//
3// Redirections (>, >>) are processed by the current shell before sudo runs.
4// The file is opened with your permissions, not root's.
5//
6// Examples:
7// Bad:
8//   sudo echo "text" > /root/file   // Redirect runs as you, not root
9//   sudo cat data >> /etc/hosts     // Append runs as you, fails
10//   sudo cmd > /var/log/app.log     // Log write uses your permissions
11//
12// Good:
13//   echo "text" | sudo tee /root/file >/dev/null    // tee runs as root
14//   cat data | sudo tee -a /etc/hosts >/dev/null    // -a for append
15//   sudo sh -c 'cmd > /var/log/app.log'             // sh -c works too
16//
17// Note: The shell processes redirects before executing sudo, so the file
18// operations happen with your UID, not root's.
19
20use crate::linter::{Diagnostic, LintResult, Severity, Span};
21use once_cell::sync::Lazy;
22use regex::Regex;
23
24#[allow(clippy::expect_used)] // Compile-time regex, panic on invalid pattern is acceptable
25static SUDO_WITH_REDIRECT: Lazy<Regex> = Lazy::new(|| {
26    // Match: sudo command > file or sudo command >> file
27    Regex::new(r"\bsudo\s+[^>|]+\s*(>>?)\s*[^\s]+").expect("valid regex for sudo redirect")
28});
29
30/// Issue #101: Detect sudo sh -c or sudo bash -c patterns
31/// These patterns wrap the redirect inside the shell, so sudo DOES affect it
32#[allow(clippy::expect_used)] // Compile-time regex, panic on invalid pattern is acceptable
33static SUDO_SHELL_WRAPPER: Lazy<Regex> = Lazy::new(|| {
34    // Match: sudo [flags] sh/bash/dash/ash/zsh -c 'content' or "content"
35    // Uses ['"] to match either single or double quote
36    Regex::new(r#"\bsudo\s+(?:-\S+\s+)*(?:sh|bash|dash|ash|zsh)\s+-c\s+['"]"#)
37        .expect("valid regex for sudo shell wrapper")
38});
39
40/// Issue #101: Check if this is a sudo sh -c or sudo bash -c pattern
41fn is_sudo_shell_wrapper(line: &str) -> bool {
42    SUDO_SHELL_WRAPPER.is_match(line)
43}
44
45/// Issue #100: Check if this is a piped sudo tee pattern
46/// cmd | sudo tee [file] is the correct way to write to root-owned files
47fn is_piped_sudo_tee(line: &str) -> bool {
48    // Look for pipe followed by sudo tee
49    line.contains("| sudo tee") || line.contains("|sudo tee")
50}
51
52/// F004: Check if redirect target is a user-writable location
53/// User-writable locations like /tmp, /var/tmp, /dev/null don't need sudo for writing
54fn is_redirect_to_user_writable(line: &str) -> bool {
55    // Common user-writable paths where sudo redirect warning is not useful
56    const USER_WRITABLE_PATHS: &[&str] = &[
57        "/tmp/",
58        "/tmp",
59        "/var/tmp/",
60        "/var/tmp",
61        "/dev/null",
62        "/dev/zero",
63        "/dev/stdout",
64        "/dev/stderr",
65    ];
66
67    // Extract the redirect target from the line
68    // Look for > or >> followed by the path
69    if let Some(redirect_pos) = line.find('>') {
70        let after_redirect = &line[redirect_pos..];
71        // Skip the > or >> and any whitespace
72        let target = after_redirect
73            .trim_start_matches('>')
74            .trim_start();
75
76        // Check if target starts with any user-writable path
77        for path in USER_WRITABLE_PATHS {
78            if target.starts_with(path) {
79                return true;
80            }
81        }
82    }
83
84    false
85}

87/// Check if a line is a comment
88fn is_comment_line(line: &str) -> bool {
89    line.trim_start().starts_with('#')
90}
91
92/// Check if line has sudo with output redirect
93fn has_sudo_with_redirect(line: &str) -> bool {
94    line.contains("sudo") && (line.contains('>') && !line.contains("2>") && !line.contains("&>"))
95}
96
97/// Check if match is inside quotes
98fn is_inside_quotes(line: &str, match_pos: usize) -> bool {
99    let before = &line[..match_pos];
100    let quote_count = before.matches('"').count() + before.matches('\'').count();
101    quote_count % 2 == 1
102}

104/// Get tee flag for redirect operator
105fn get_tee_flag(redirect_op: &str) -> &str {
106    if redirect_op == ">>" {
107        "-a "
108    } else {
109        ""
110    }
111}
112
113/// Create diagnostic for sudo with redirect
114fn create_sudo_redirect_diagnostic(
115    line_num: usize,
116    start_col: usize,
117    end_col: usize,
118    redirect_op: &str,
119) -> Diagnostic {
120    let tee_flag = get_tee_flag(redirect_op);
121    Diagnostic::new(
122        "SC2024",
123        Severity::Warning,
124        format!(
125            "sudo doesn't affect redirects. Use '| sudo tee {}file' instead of 'sudo cmd {} file'",
126            tee_flag, redirect_op
127        ),
128        Span::new(line_num, start_col, line_num, end_col),
129    )
130}

132#[allow(clippy::expect_used)] // Capture groups 0 and 1 always exist when regex matches
133pub fn check(source: &str) -> LintResult {
134    let mut result = LintResult::new();
135
136    for (line_num, line) in source.lines().enumerate() {
137        let line_num = line_num + 1;
138
139        if is_comment_line(line) {
140            continue;
141        }
142
143        // Issue #101: Skip if sudo sh -c or sudo bash -c pattern
144        // The redirect is inside the shell wrapper, so sudo DOES affect it
145        if is_sudo_shell_wrapper(line) {
146            continue;
147        }
148
149        // Issue #100: Skip if piped sudo tee pattern
150        // cmd | sudo tee file is the correct way to write to root-owned files
151        if is_piped_sudo_tee(line) {
152            continue;
153        }
154
155        // F004: Skip if redirect target is user-writable (e.g., /tmp, /dev/null)
156        // These locations don't need sudo for writing, so the warning is not useful
157        if is_redirect_to_user_writable(line) {
158            continue;
159        }
160
161        // Look for sudo with output redirection
162        if has_sudo_with_redirect(line) {
163            for cap in SUDO_WITH_REDIRECT.captures_iter(line) {
164                let redirect_op = cap
165                    .get(1)
166                    .expect("capture group 1 exists for redirect operator")
167                    .as_str();
168
169                // Skip if inside quotes
170                let full_match = cap
171                    .get(0)
172                    .expect("capture group 0 exists for full match")
173                    .as_str();
174                let pos = line.find(full_match).unwrap_or(0);
175
176                if is_inside_quotes(line, pos) {
177                    continue;
178                }
179
180                let start_col = pos + 1;
181                let end_col = start_col + full_match.len();
182
183                let diagnostic =
184                    create_sudo_redirect_diagnostic(line_num, start_col, end_col, redirect_op);
185
186                result.add(diagnostic);
187            }
188        }
189    }
190
191    result
192}
