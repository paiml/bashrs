1//! SC2027: Wrong quoting in printf format strings
2//!
3//! # Examples
4//!
5//! Bad:
6//! ```bash
7//! printf "Hello $name\n"
8//! ```
9//!
10//! Good:
11//! ```bash
12//! printf "Hello %s\n" "$name"
13//! printf 'Hello %s\n' "$name"
14//! ```
15//!
16//! # Rationale
17//!
18//! Printf format strings should use format specifiers, not direct variable expansion:
19//! - Prevents injection attacks
20//! - Handles special characters correctly
21//! - Follows printf conventions
22//!
23//! # Auto-fix
24//!
25//! Suggest using printf format specifiers
26
27use crate::linter::{Diagnostic, LintResult, Severity, Span};
28use regex::Regex;
29
30/// Check for wrong quoting in printf format strings
31pub fn check(source: &str) -> LintResult {
32    let mut result = LintResult::new();
33
34    // Pattern: printf "...$var..." or printf '...$var...'
35    let pattern =
36        Regex::new(r#"printf\s+["']([^"']*\$\{?[A-Za-z_][A-Za-z0-9_]*\}?[^"']*)["']"#).unwrap();
37
38    for (line_num, line) in source.lines().enumerate() {
39        let line_num = line_num + 1;
40
41        if line.trim_start().starts_with('#') {
42            continue;
43        }
44
45        for cap in pattern.captures_iter(line) {
46            let full_match = cap.get(0).unwrap();
47            let format_str = cap.get(1).unwrap().as_str();
48
49            // Skip if it's already using format specifiers (%s, %d, etc.)
50            if format_str.contains("%s") || format_str.contains("%d") || format_str.contains("%f") {
51                continue;
52            }
53
54            let start_col = full_match.start() + 1;
55            let end_col = full_match.end() + 1;
56
57            let diagnostic = Diagnostic::new(
58                "SC2027",
59                Severity::Warning,
60                "The surrounding quotes actually unquote this. Use printf '%s\\n' \"$var\" instead",
61                Span::new(line_num, start_col, line_num, end_col),
62            );
63
64            result.add(diagnostic);
65        }
66    }
67
68    result
69}
