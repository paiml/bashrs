1//! SC2046: Quote command substitutions to prevent word splitting
2//!
3//! Detects unquoted command substitutions like $(cmd) or `cmd` that could
4//! cause word splitting on the output.
5//!
6//! References:
7//! - https://www.shellcheck.net/wiki/SC2046
8
9use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
10use regex::Regex;
11
12/// Check for unquoted command substitutions (SC2046)
13pub fn check(source: &str) -> LintResult {
14    let mut result = LintResult::new();
15
16    // Pattern for command substitution: $(...)
17    let cmd_sub_pattern = Regex::new(r#"(?m)(?P<pre>[^"']|^)\$\((?P<cmd>[^)]+)\)"#).unwrap();
18
19    // Pattern for backtick command substitution: `...`
20    let backtick_pattern = Regex::new(r#"(?m)(?P<pre>[^"']|^)`(?P<cmd>[^`]+)`"#).unwrap();
21
22    for (line_num, line) in source.lines().enumerate() {
23        let line_num = line_num + 1;
24
25        // Skip comments
26        if line.trim_start().starts_with('#') {
27            continue;
28        }
29
30        // Check $(...) substitutions
31        for cap in cmd_sub_pattern.captures_iter(line) {
32            // Find the actual $( position (not including 'pre' capture)
33            let cmd_match = cap.name("cmd").unwrap();
34            let dollar_paren_pos = line[..cmd_match.start()]
35                .rfind("$(")
36                .unwrap_or(cmd_match.start());
37
38            let col = dollar_paren_pos + 1; // 1-indexed
39            let end_col = cmd_match.end() + 2; // +1 for ) and +1 for 1-indexing
40
41            // Check if already quoted
42            if dollar_paren_pos > 0 && line.chars().nth(dollar_paren_pos - 1) == Some('"') {
43                continue;
44            }
45
46            let span = Span::new(line_num, col, line_num, end_col);
47            let cmd_text = format!("$({})", cmd_match.as_str());
48            let fix = Fix::new(format!("\"{}\"", cmd_text));
49
50            let diag = Diagnostic::new(
51                "SC2046",
52                Severity::Warning,
53                format!("Quote this to prevent word splitting: {}", cmd_text),
54                span,
55            )
56            .with_fix(fix);
57
58            result.add(diag);
59        }
60
61        // Check backtick substitutions
62        for cap in backtick_pattern.captures_iter(line) {
63            // Find the actual backtick position (not including 'pre' capture)
64            let cmd_match = cap.name("cmd").unwrap();
65            let backtick_pos = line[..cmd_match.start()]
66                .rfind('`')
67                .unwrap_or(cmd_match.start());
68
69            let col = backtick_pos + 1; // 1-indexed
70            let end_col = cmd_match.end() + 2; // +1 for closing ` and +1 for 1-indexing
71
72            let span = Span::new(line_num, col, line_num, end_col);
73            let cmd = cmd_match.as_str();
74            let fix = Fix::new(format!("\"$({})\"", cmd));
75
76            let diag = Diagnostic::new(
77                "SC2046",
78                Severity::Warning,
79                "Quote this and use $(...) instead of backticks".to_string(),
80                span,
81            )
82            .with_fix(fix);
83
84            result.add(diag);
85        }
86    }
87
88    result
89}

91#[cfg(test)]
92mod tests {
93    use super::*;
94
95    #[test]
96    fn test_sc2046_basic_detection() {
97        let bash_code = "files=$(find . -name '*.txt')";
98        let result = check(bash_code);
99
100        assert_eq!(result.diagnostics.len(), 1);
101        assert_eq!(result.diagnostics[0].code, "SC2046");
102        assert!(result.diagnostics[0].message.contains("Quote this"));
103    }
104
105    #[test]
106    fn test_sc2046_autofix() {
107        let bash_code = "files=$(ls)";
108        let result = check(bash_code);
109
110        assert!(result.diagnostics[0].fix.is_some());
111        assert_eq!(
112            result.diagnostics[0].fix.as_ref().unwrap().replacement,
113            "\"$(ls)\""
114        );
115    }
116
117    #[test]
118    fn test_sc2046_backtick_detection() {
119        let bash_code = "files=`ls *.txt`";
120        let result = check(bash_code);
121
122        assert_eq!(result.diagnostics.len(), 1);
123        assert_eq!(result.diagnostics[0].code, "SC2046");
124        assert!(result.diagnostics[0].message.contains("backticks"));
125    }
126
127    #[test]
128    fn test_sc2046_backtick_autofix() {
129        let bash_code = "files=`ls`";
130        let result = check(bash_code);
131
132        assert!(result.diagnostics[0].fix.is_some());
133        assert_eq!(
134            result.diagnostics[0].fix.as_ref().unwrap().replacement,
135            "\"$(ls)\""
136        );
137    }
138
139    #[test]
140    fn test_sc2046_skip_quoted() {
141        let bash_code = r#"files="$(find . -name '*.txt')""#;
142        let result = check(bash_code);
143
144        // Should NOT trigger - already quoted
145        assert_eq!(result.diagnostics.len(), 0);
146    }
147
148    #[test]
149    fn test_sc2046_multiple_substitutions() {
150        let bash_code = r#"
151result=$(echo $(cat file.txt))
152"#;
153        let result = check(bash_code);
154
155        // Should detect nested unquoted substitutions
156        assert!(!result.diagnostics.is_empty());
157    }
158
159    #[test]
160    fn test_sc2046_severity() {
161        let bash_code = "files=$(ls)";
162        let result = check(bash_code);
163
164        assert_eq!(result.diagnostics[0].severity, Severity::Warning);
165    }
