405/// Lint a shell script with shell-type detection based on file path.
406///
407/// Automatically detects the shell type (POSIX sh, bash, etc.) from the file extension
408/// and shebang, then applies only compatible rules. This prevents false positives when
409/// linting POSIX sh scripts with bash-specific rules.
410///
411/// # Arguments
412///
413/// * `path` - File path used for shell type detection (`.sh`, `.bash`, etc.)
414/// * `source` - The shell script source code to lint
415///
416/// # Returns
417///
418/// A [`LintResult`] containing diagnostics appropriate for the detected shell type.
419///
420/// # Examples
421///
422/// ## POSIX sh script
423///
424/// ```
425/// use bashrs::linter::lint_shell_with_path;
426/// use std::path::Path;
427///
428/// let path = Path::new("script.sh");
429/// let script = "#!/bin/sh\necho hello";
430/// let result = lint_shell_with_path(path, script);
431/// // Should detect POSIX sh and skip bash-only rules
432/// // Note: May have style suggestions, but no critical errors
433/// let errors = result.diagnostics.iter()
434///     .filter(|d| d.severity == bashrs::linter::Severity::Error)
435///     .count();
436/// assert_eq!(errors, 0);
437/// ```
438///
439/// ## Bash-specific script
440///
441/// ```
442/// use bashrs::linter::lint_shell_with_path;
443/// use std::path::Path;
444///
445/// let path = Path::new("script.bash");
446/// let script = "#!/bin/bash\necho hello";
447/// let result = lint_shell_with_path(path, script);
448/// // Bash-specific rules are applied
449/// // Should complete successfully with minimal issues
450/// let errors = result.diagnostics.iter()
451///     .filter(|d| d.severity == bashrs::linter::Severity::Error)
452///     .count();
453/// assert_eq!(errors, 0);
454/// ```
455pub fn lint_shell_with_path(path: &std::path::Path, source: &str) -> LintResult {
456    use crate::linter::shell_type::detect_shell_type;
457
458    // Detect shell type from path and content
459    let shell_type = detect_shell_type(path, source);
460
461    // Run rules with shell-specific filtering
462    lint_shell_filtered(source, shell_type)
463}

465/// Lint a shell script with shell-specific rule filtering
466///
467/// Conditionally applies rules based on shell type compatibility.
468/// For example, bash-only array rules are skipped for POSIX sh files.
469///
470/// # Arguments
471/// * `source` - Shell script source code
472/// * `shell_type` - Detected or explicit shell type
473///
474/// # Returns
475/// LintResult with filtered diagnostics
476fn lint_shell_filtered(
477    source: &str,
478    shell_type: crate::linter::shell_type::ShellType,
479) -> LintResult {
480    use crate::linter::rule_registry::should_apply_rule;
481    use crate::linter::suppression::SuppressionManager;
482
483    let mut result = LintResult::new();
484
485    // Helper macro to conditionally apply rules
486    macro_rules! apply_rule {
487        ($rule_id:expr, $check_fn:expr) => {
488            if should_apply_rule($rule_id, shell_type) {
489                result.merge($check_fn(source));
490            }
491        };
492    }
493
494    // Run ShellCheck-equivalent rules with filtering
495    apply_rule!("SC2001", sc2001::check);
496    apply_rule!("SC2002", sc2002::check);

505    apply_rule!("SC2018", sc2018::check);
506    apply_rule!("SC2019", sc2019::check);
507    apply_rule!("SC2020", sc2020::check);
508    apply_rule!("SC2021", sc2021::check);
509    apply_rule!("SC2022", sc2022::check);
510    apply_rule!("SC2023", sc2023::check);
511    apply_rule!("SC2024", sc2024::check);
512    apply_rule!("SC2025", sc2025::check);
513    apply_rule!("SC2026", sc2026::check);
514    apply_rule!("SC2027", sc2027::check);
515    apply_rule!("SC2028", sc2028::check); // Universal - echo may not expand \\n (use printf)
516    apply_rule!("SC2029", sc2029::check);
517    apply_rule!("SC2030", sc2030::check);
518    apply_rule!("SC2031", sc2031::check); // Universal - subshell scope
519    apply_rule!("SC2032", sc2032::check); // Universal - variable in shebang script
520
521    // Add classified rules (SC2039 and SC2198-2201)
522    apply_rule!("SC2039", sc2039::check); // NotSh - bash/zsh features
523
524    // Batch 2: Arithmetic and quoting rules (Universal)
525    apply_rule!("SC2079", sc2079::check); // Universal - decimals in arithmetic
526    apply_rule!("SC2080", sc2080::check); // Universal - octal numbers
527    apply_rule!("SC2084", sc2084::check); // Universal - arithmetic as command
528    apply_rule!("SC2085", sc2085::check); // Universal - local with arithmetic
529    apply_rule!("SC2087", sc2087::check); // Universal - quote in sh -c
530    apply_rule!("SC2088", sc2088::check); // Universal - tilde expansion
531    apply_rule!("SC2089", sc2089::check); // Universal - quotes in assignment
532    apply_rule!("SC2090", sc2090::check); // Universal - quotes in expansion
533    apply_rule!("SC2091", sc2091::check); // Universal - remove $() execution
534    apply_rule!("SC2092", sc2092::check); // Universal - remove backticks execution
535    apply_rule!("SC2093", sc2093::check); // Universal - remove exec
536
537    // Batch 2: [[ ]] test syntax rules (NotSh - bash/zsh/ksh only)
538    apply_rule!("SC2108", sc2108::check); // NotSh - [[ ]] use && not -a
539    apply_rule!("SC2109", sc2109::check); // NotSh - [[ ]] use || not -o
540    apply_rule!("SC2110", sc2110::check); // NotSh - don't mix && || with -a -o
541
542    // Batch 2: function keyword rules (NotSh - bash/ksh only)
543    apply_rule!("SC2111", sc2111::check); // NotSh - function keyword in sh
544    apply_rule!("SC2112", sc2112::check); // NotSh - function keyword non-standard
545    apply_rule!("SC2113", sc2113::check); // NotSh - function with () redundant
546
547    // Batch 2: More arithmetic rules (Universal)
548    apply_rule!("SC2133", sc2133::check); // Universal - unexpected tokens in arithmetic
549    apply_rule!("SC2134", sc2134::check); // Universal - use (( )) for numeric tests
550    apply_rule!("SC2137", sc2137::check); // Universal - unnecessary braces in arithmetic
551
552    // Array rules (NotSh)
553    apply_rule!("SC2198", sc2198::check); // NotSh - arrays
554    apply_rule!("SC2199", sc2199::check); // NotSh - arrays
555    apply_rule!("SC2200", sc2200::check); // NotSh - arrays
556    apply_rule!("SC2201", sc2201::check); // NotSh - arrays
557
558    // Batch 3: Loop and iteration safety (Universal - all shells)
559    apply_rule!("SC2038", sc2038::check); // Universal - find loop safety
560    apply_rule!("SC2040", sc2040::check); // Universal - avoid -o confusion
561    apply_rule!("SC2041", sc2041::check); // Universal - read in for loop
562    apply_rule!("SC2042", sc2042::check); // Universal - echo vs printf
563    apply_rule!("SC2043", sc2043::check); // Universal - loop runs once

580    apply_rule!("SC2056", sc2056::check); // Universal - deprecated -o
581    apply_rule!("SC2057", sc2057::check); // Universal - unknown binary operator

585    apply_rule!("SC2059", sc2059::check); // Universal - CRITICAL printf format injection
586    apply_rule!("SC2060", sc2060::check); // Universal - unquoted tr params
587    apply_rule!("SC2061", sc2061::check); // Universal - quote tr params
588    apply_rule!("SC2062", sc2062::check); // Universal - grep pattern glob expansion

591    apply_rule!("SC2063", sc2063::check); // Universal - grep regex vs literal
592    apply_rule!("SC2064", sc2064::check); // Universal - CRITICAL trap timing bug
593    apply_rule!("SC2065", sc2065::check); // Universal - shell redirection interpretation
594    apply_rule!("SC2066", sc2066::check); // Universal - missing semicolon before done

597    apply_rule!("SC2067", sc2067::check); // Universal - missing $ on array lookup

599    apply_rule!("SC2069", sc2069::check); // Universal - redirect stdout+stderr correctly
600    apply_rule!("SC2070", sc2070::check); // Universal - -n doesn't work unquoted
601    apply_rule!("SC2071", sc2071::check); // Universal - arithmetic operators in [ ]
602    apply_rule!("SC2072", sc2072::check); // Universal - lexicographic vs numeric comparison

604    apply_rule!("SC2074", sc2074::check); // Universal - can't use =~ in [ ]
605
606    // Batch 4: Quote and expansion safety (Universal)
607    apply_rule!("SC2075", sc2075::check); // Universal - escaping quotes in single quotes
608    apply_rule!("SC2076", sc2076::check); // Universal - don't quote RHS of =~ in [[ ]]
609    apply_rule!("SC2077", sc2077::check); // Universal - quote regex to prevent word splitting
610    apply_rule!("SC2078", sc2078::check); // Universal - constant expression (forgot $?)
611    apply_rule!("SC2081", sc2081::check); // Universal - escape [ in globs
612    apply_rule!("SC2082", sc2082::check); // Universal - variable indirection with $$
613    apply_rule!("SC2083", sc2083::check); // Universal - don't add spaces after shebang
614
615    // Batch 4: Command and redirection safety (Universal - CRITICAL)
616    apply_rule!("SC2094", sc2094::check); // Universal - same file for input/output (truncate)
617    apply_rule!("SC2095", sc2095::check); // Universal - ssh in loops may consume stdin
618    apply_rule!("SC2096", sc2096::check); // Universal - use #! shebang
619    apply_rule!("SC2097", sc2097::check); // Universal - assign and use variable separately
620    apply_rule!("SC2098", sc2098::check); // Universal - variable assignment vs redirection
621    apply_rule!("SC2103", sc2103::check); // Universal - cd without error check
622
623    // Batch 4: Test and conditional safety (Universal)
624    apply_rule!("SC2104", sc2104::check); // Universal - == is literal in [[ ]]
625    apply_rule!("SC2105", sc2105::check); // Universal - break outside loop
626    apply_rule!("SC2107", sc2107::check); // Universal - use [ a ] || [ b ] not [ a -o b ]
627
628    // Batch 4: Function and scope safety (Universal - CRITICAL dangerous rm)
629    apply_rule!("SC2114", sc2114::check); // Universal - CRITICAL dangerous rm -rf without validation
630    apply_rule!("SC2115", sc2115::check); // Universal - CRITICAL use ${var:?} before rm -rf
631    apply_rule!("SC2116", sc2116::check); // Universal - useless echo $(cmd)
632
633    // Batch 4: Bash-specific function analysis (NotSh - bash/zsh/ksh only)
634    // apply_rule!("SC2120", sc2120::check); // NotSh - function references $1 but none passed (TODO: has false positives)
635    apply_rule!("SC2128", sc2128::check); // NotSh - array expansion without index
636
637    // Batch 5: CRITICAL word splitting (Universal - HIGHEST PRIORITY)
638    apply_rule!("SC2086", sc2086::check); // Universal - CRITICAL: Quote to prevent word splitting and globbing
639
640    // TODO: Add remaining SC2xxx rules (~237 rules remaining, was ~257)
641    // For now, fall back to lint_shell() for unclassified rules
642    // This ensures backward compatibility while we incrementally classify
643

645    result.merge(det001::check(source));
646    result.merge(det002::check(source));
647    result.merge(det003::check(source));
648
649    // Idempotency rules (Universal - always apply)
650    result.merge(idem001::check(source));
651    result.merge(idem002::check(source));
652    result.merge(idem003::check(source));
653
654    // Best practice rules
655    result.merge(bash001::check(source));
656    result.merge(bash002::check(source));
657    result.merge(bash003::check(source));
658    result.merge(bash005::check(source));
659    result.merge(bash006::check(source));
660    result.merge(bash007::check(source));
661    result.merge(bash008::check(source));
662    result.merge(bash009::check(source));
663    result.merge(bash010::check(source));
