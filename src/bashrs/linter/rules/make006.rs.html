1//! MAKE006: Missing target dependencies
2//!
3//! **Rule**: Detect when targets don't declare necessary dependencies
4//!
5//! **Why this matters**:
6//! Targets should explicitly declare their dependencies to ensure correct
7//! build order and enable parallel builds. Missing dependencies can lead to
8//! race conditions and incorrect builds.
9//!
10//! **Auto-fix**: Suggest common dependencies based on recipe commands
11//!
12//! ## Examples
13//!
14//! ❌ **BAD** (missing dependencies):
15//! ```makefile
16//! app:
17//!     gcc main.c utils.c -o app
18//! ```
19//!
20//! ✅ **GOOD** (dependencies declared):
21//! ```makefile
22//! app: main.c utils.c
23//!     gcc main.c utils.c -o app
24//! ```
25
26use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
27use std::collections::HashSet;
28
29/// Check for targets with missing dependencies
30pub fn check(source: &str) -> LintResult {
31    let mut result = LintResult::new();
32
33    let phony_targets = find_phony_targets(source);
34    let lines: Vec<&str> = source.lines().collect();
35    let mut i = 0;
36
37    while i < lines.len() {
38        if let Some((target_info, next_line)) = parse_target_line(&lines, i, &phony_targets) {
39            if let Some(diag) = check_target_dependencies(&target_info, &lines, i) {
40                result.add(diag);
41            }
42            i = next_line;
43        } else {
44            i += 1;
45        }
46    }
47
48    result
49}

51/// Information about a parsed target line
52struct TargetInfo {
53    name: String,
54    declared_deps: HashSet<String>,
55}
56
57/// Parse a target line and extract target name and dependencies
58fn parse_target_line(
59    lines: &[&str],
60    line_idx: usize,
61    phony_targets: &HashSet<String>,
62) -> Option<(TargetInfo, usize)> {
63    let line = lines[line_idx];
64
65    // Skip lines that aren't target definitions
66    if should_skip_line(line) {
67        return None;
68    }
69
70    // Find colon position
71    let colon_pos = line.find(':')?;
72
73    // Skip recipe lines (start with tab)
74    if line.starts_with('\t') {
75        return None;
76    }
77
78    // Extract target name
79    let target = line[..colon_pos].trim();
80
81    // Skip .PHONY targets
82    if phony_targets.contains(target) {
83        return None;
84    }
85
86    // Extract declared dependencies
87    let deps_part = &line[colon_pos + 1..];
88    let declared_deps: HashSet<String> = deps_part
89        .split_whitespace()
90        .map(|s| s.to_string())
91        .collect();
92
93    Some((
94        TargetInfo {
95            name: target.to_string(),
96            declared_deps,
97        },
98        line_idx + 1,
99    ))
100}

102/// Check if a line should be skipped (not a target definition)
103fn should_skip_line(line: &str) -> bool {
104    line.trim().is_empty()
105        || line.trim_start().starts_with('#')
106        || line.trim_start().starts_with('.')
107        || !line.contains(':')
108        || line.contains('=')
109}

111/// Check if a target is missing dependencies
112fn check_target_dependencies(
113    target_info: &TargetInfo,
114    lines: &[&str],
115    start_idx: usize,
116) -> Option<Diagnostic> {
117    // Collect source files from recipe lines
118    let source_files = collect_recipe_source_files(lines, start_idx + 1);
119
120    // Find missing dependencies
121    let missing_deps = find_missing_dependencies(&source_files, &target_info.declared_deps);
122
123    if missing_deps.is_empty() {
124        return None;
125    }
126
127    // Create diagnostic with fix suggestion
128    Some(create_missing_deps_diagnostic(
129        &target_info.name,
130        &target_info.declared_deps,
131        &missing_deps,
132        start_idx,
133    ))
134}
135
136/// Collect source files from recipe lines
137fn collect_recipe_source_files(lines: &[&str], start_idx: usize) -> HashSet<String> {
138    let mut source_files = HashSet::new();
139    let mut i = start_idx;
140
141    while i < lines.len() && lines[i].starts_with('\t') {
142        extract_source_files(lines[i], &mut source_files);
143        i += 1;
144    }
145
146    source_files
147}

149/// Find dependencies that are missing from declared dependencies
150fn find_missing_dependencies(
151    source_files: &HashSet<String>,
152    declared_deps: &HashSet<String>,
153) -> Vec<String> {
154    let mut missing: Vec<String> = source_files
155        .iter()
156        .filter(|f| !declared_deps.contains(*f))
157        .cloned()
158        .collect();
159
160    missing.sort();
161    missing
162}

164/// Create a diagnostic for missing dependencies
165fn create_missing_deps_diagnostic(
166    target: &str,
167    declared_deps: &HashSet<String>,
168    missing_deps: &[String],
169    line_idx: usize,
170) -> Diagnostic {
171    let span = Span::new(line_idx + 1, 1, line_idx + 1, target.len() + 1);
172
173    // Create fix suggestion with all dependencies
174    let mut all_deps: Vec<String> = declared_deps.iter().cloned().collect();
175    all_deps.extend_from_slice(missing_deps);
176    all_deps.sort();
177    all_deps.dedup();
178
179    let fix_replacement = format!("{}: {}", target, all_deps.join(" "));
180
181    Diagnostic::new(
182        "MAKE006",
183        Severity::Warning,
184        format!(
185            "Target '{}' may be missing dependencies: {}",
186            target,
187            missing_deps.join(", ")
188        ),
189        span,
190    )
191    .with_fix(Fix::new(&fix_replacement))
192}

194/// Find all .PHONY targets in the Makefile
195fn find_phony_targets(source: &str) -> HashSet<String> {
196    let mut phony = HashSet::new();
197
198    for line in source.lines() {
199        if line.trim_start().starts_with(".PHONY:") {
200            if let Some(targets_str) = line.split(':').nth(1) {
201                for target in targets_str.split_whitespace() {
202                    phony.insert(target.to_string());
203                }
204            }
205        }
206    }
207
208    phony
209}

211/// Extract source files (.c, .cpp, .h, .rs, etc.) from a recipe line
212fn extract_source_files(recipe: &str, files: &mut HashSet<String>) {
213    // Common source file extensions
214    let extensions = [".c", ".cpp", ".cc", ".h", ".hpp", ".rs", ".go"];
215
216    for word in recipe.split_whitespace() {
217        for ext in &extensions {
218            if word.ends_with(ext) {
219                // Remove any flags or options attached to the filename
220                let clean = word.trim_start_matches('-');
221                files.insert(clean.to_string());
222            }
223        }
224    }
225}

234    fn test_MAKE006_detects_missing_dependencies_basic() {
235        let makefile = r#"app:
236	gcc main.c -o app"#;
237        let result = check(makefile);
238
239        assert_eq!(result.diagnostics.len(), 1);
240        let diag = &result.diagnostics[0];
241        assert_eq!(diag.code, "MAKE006");
242        assert_eq!(diag.severity, Severity::Warning);
243        assert!(diag.message.contains("dependencies"));
244    }
