1// SC2037: To assign command output to variable, use var=$(cmd), not cmd > $var
2//
3// Redirecting to a variable like `cmd > $var` doesn't assign output to the variable.
4// It redirects to a FILE whose name is the value of $var. To capture output, use
5// command substitution: var=$(cmd)
6//
7// Examples:
8// Bad:
9//   echo "result" > $VAR    // Redirects to file named by $VAR
10//   cmd > $output           // Creates file named by $output
11//   date > $timestamp       // Writes to file, doesn't assign
12//
13// Good:
14//   VAR=$(echo "result")    // Assigns output to VAR
15//   output=$(cmd)           // Captures command output
16//   timestamp=$(date)       // Assigns date output
17//
18// Note: This is a common mistake for beginners.
19
20use crate::linter::{Diagnostic, LintResult, Severity, Span};
21use once_cell::sync::Lazy;
22use regex::Regex;
23
24static REDIRECT_TO_VAR: Lazy<Regex> = Lazy::new(|| {
25    // Match: command > $VAR or command >> $VAR
26    Regex::new(r"(>>?)\s*\$([a-zA-Z_][a-zA-Z0-9_]*)").unwrap()
27});
28
29/// Check if a line is a comment
30fn is_comment_line(line: &str) -> bool {
31    line.trim_start().starts_with('#')
32}
33
34/// Check if position is inside quotes
35fn is_inside_quotes(line: &str, pos: usize) -> bool {
36    let before = &line[..pos];
37    let quote_count = before.matches('"').count() + before.matches('\'').count();
38    quote_count % 2 == 1
39}
40
41/// Check if redirect is stderr or combined (&>, 2>)
42fn is_stderr_or_combined_redirect(line: &str, pos: usize) -> bool {
43    if pos > 0 {
44        let prev_char = line.chars().nth(pos - 1);
45        prev_char == Some('2') || prev_char == Some('&')
46    } else {
47        false
48    }
49}
50
51/// Check if variable name is a common log file pattern
52fn is_log_file_pattern(var_name: &str) -> bool {
53    var_name == "LOGFILE"
54        || var_name == "LOG"
55        || var_name.ends_with("_FILE")
56        || var_name.ends_with("_LOG")
57}
58
59/// Create diagnostic for redirect to variable
60fn create_redirect_diagnostic(
61    var_name: &str,
62    redirect_op: &str,
63    line_num: usize,
64    start_col: usize,
65    end_col: usize,
66) -> Diagnostic {
67    Diagnostic::new(
68        "SC2037",
69        Severity::Warning,
70        format!(
71            "To assign command output, use {}=$(cmd), not cmd {} ${}",
72            var_name.to_lowercase(),
73            redirect_op,
74            var_name
75        ),
76        Span::new(line_num, start_col, line_num, end_col),
77    )
78}

80pub fn check(source: &str) -> LintResult {
81    let mut result = LintResult::new();
82
83    for (line_num, line) in source.lines().enumerate() {
84        let line_num = line_num + 1;
85
86        if is_comment_line(line) {
87            continue;
88        }
89
90        // Look for redirects to variables
91        for cap in REDIRECT_TO_VAR.captures_iter(line) {
92            let redirect_op = cap.get(1).unwrap().as_str();
93            let var_name = cap.get(2).unwrap().as_str();
94            let full_match = cap.get(0).unwrap().as_str();
95            let pos = line.find(full_match).unwrap_or(0);
96
97            // Skip if inside quotes
98            if is_inside_quotes(line, pos) {
99                continue;
100            }
101
102            // Skip stderr/combined redirects (2>, &>)
103            if is_stderr_or_combined_redirect(line, pos) {
104                continue;
105            }
106
107            // Skip common log file patterns
108            if is_log_file_pattern(var_name) {
109                continue;
110            }
111
112            let start_col = pos + 1;
113            let end_col = start_col + full_match.len();
114            let diagnostic =
115                create_redirect_diagnostic(var_name, redirect_op, line_num, start_col, end_col);
116
117            result.add(diagnostic);
118        }
119    }
120
121    result
122}

131    fn prop_sc2037_comments_never_diagnosed() {
132        // Property: Comment lines should never produce diagnostics
133        let test_cases = vec![

136            "\t# date > $timestamp",
