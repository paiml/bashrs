1//! MAKE007: Silent recipe errors (missing @ prefix)
2//!
3//! **Rule**: Detect echo/printf commands without @ prefix for silent output
4//!
5//! **Why this matters**:
6//! By default, Make prints every command before executing it. For echo/printf
7//! commands, this creates duplicate output (the command itself + its output).
8//! Using @ prefix silences the command printing, showing only the output.
9//!
10//! **Auto-fix**: Add @ prefix to echo/printf commands
11//!
12//! ## Examples
13//!
14//! ❌ **BAD** (without @ prefix - duplicate output):
15//! ```makefile
16//! build:
17//! \techo "Building..."
18//! # Output:
19//! # echo "Building..."
20//! # Building...
21//! ```
22//!
23//! ✅ **GOOD** (with @ prefix - clean output):
24//! ```makefile
25//! build:
26//! \t@echo "Building..."
27//! # Output:
28//! # Building...
29//! ```
30
31use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
32
33/// Commands that should typically be silent
34const SILENT_COMMANDS: &[&str] = &["echo", "printf"];
35
36/// Check for echo/printf commands without @ prefix
37pub fn check(source: &str) -> LintResult {
38    let mut result = LintResult::new();
39
40    for (line_num, line) in source.lines().enumerate() {
41        // Only check recipe lines (start with tab)
42        if !line.starts_with('\t') {
43            continue;
44        }
45
46        // Check if line contains echo or printf without @ prefix
47        if let Some(diag) = check_recipe_line(line, line_num) {
48            result.add(diag);
49        }
50    }
51
52    result
53}
54
55/// Check a recipe line for echo/printf without @ prefix
56fn check_recipe_line(line: &str, line_num: usize) -> Option<Diagnostic> {
57    let trimmed = line.trim_start_matches('\t').trim_start();
58
59    // Already has @ prefix - OK
60    if trimmed.starts_with('@') {
61        return None;
62    }
63
64    // Check if this is an echo or printf command
65    for cmd in SILENT_COMMANDS {
66        if is_command(trimmed, cmd) {
67            let span = Span::new(line_num + 1, 1, line_num + 1, line.len() + 1);
68            let fix_replacement = line.replacen('\t', "\t@", 1);
69
70            return Some(
71                Diagnostic::new(
72                    "MAKE007",
73                    Severity::Warning,
74                    &format!(
75                        "Command '{}' without @ prefix - will show duplicate output",
76                        cmd
77                    ),
78                    span,
79                )
80                .with_fix(Fix::new(&fix_replacement)),
81            );
82        }
83    }
84
85    None
86}
87
88/// Check if trimmed line starts with the given command
89fn is_command(line: &str, cmd: &str) -> bool {
90    // Check if line starts with the command (as whole word)
91    if line.starts_with(cmd) {
92        // Ensure it's a whole word (followed by space, tab, or nothing)
93        if line.len() == cmd.len() {
94            return true;
95        }
96        let next_char = line.chars().nth(cmd.len());
97        matches!(next_char, Some(' ') | Some('\t'))
98    } else {
99        false
100    }
101}

103#[cfg(test)]

109    #[test]
110    fn test_MAKE007_detects_echo_without_at() {
111        let makefile = "build:\n\techo Building...";
112        let result = check(makefile);
113
114        assert_eq!(result.diagnostics.len(), 1);
115        let diag = &result.diagnostics[0];
116        assert_eq!(diag.code, "MAKE007");
117        assert_eq!(diag.severity, Severity::Warning);
118        assert!(diag.message.contains("@"));
119    }
120
121    #[test]
122    fn test_MAKE007_no_warning_with_at_prefix() {
123        let makefile = "build:\n\t@echo Building...";
124        let result = check(makefile);
125
126        assert_eq!(result.diagnostics.len(), 0);
127    }
128
129    #[test]
130    fn test_MAKE007_provides_fix() {
131        let makefile = "build:\n\techo Building...";
132        let result = check(makefile);
133
134        assert!(result.diagnostics[0].fix.is_some());
135        let fix = result.diagnostics[0].fix.as_ref().unwrap();
136        assert!(fix.replacement.contains("@echo"));
137    }

140    fn test_MAKE007_detects_printf_without_at() {
141        let makefile = "test:\n\tprintf \"Testing...\\n\"";

143
144        assert_eq!(result.diagnostics.len(), 1);
145    }
