1//! MAKE007: Silent recipe errors (missing @ prefix)
2//!
3//! **Rule**: Detect echo/printf commands without @ prefix for silent output
4//!
5//! **Why this matters**:
6//! By default, Make prints every command before executing it. For echo/printf
7//! commands, this creates duplicate output (the command itself + its output).
8//! Using @ prefix silences the command printing, showing only the output.
9//!
10//! **Auto-fix**: Add @ prefix to echo/printf commands
11//!
12//! ## Examples
13//!
14//! ❌ **BAD** (without @ prefix - duplicate output):
15//! ```makefile
16//! build:
17//! \techo "Building..."
18//! # Output:
19//! # echo "Building..."
20//! # Building...
21//! ```
22//!
23//! ✅ **GOOD** (with @ prefix - clean output):
24//! ```makefile
25//! build:
26//! \t@echo "Building..."
27//! # Output:
28//! # Building...
29//! ```
30
31use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
32
33/// Commands that should typically be silent
34const SILENT_COMMANDS: &[&str] = &["echo", "printf"];
35
36/// Check for echo/printf commands without @ prefix
37pub fn check(source: &str) -> LintResult {
38    let mut result = LintResult::new();
39    let lines: Vec<&str> = source.lines().collect();
40
41    for (line_num, line) in lines.iter().enumerate() {
42        // Only check recipe lines (start with tab)
43        if !line.starts_with('\t') {
44            continue;
45        }
46
47        // Skip if this is a shell continuation line (previous line ended with \)
48        if line_num > 0 && is_continuation_line(&lines, line_num) {
49            continue;
50        }
51
52        // Check if line contains echo or printf without @ prefix
53        if let Some(diag) = check_recipe_line(line, line_num) {
54            result.add(diag);
55        }
56    }
57
58    result
59}
60
61/// Check if this line is a shell script continuation
62fn is_continuation_line(lines: &[&str], line_num: usize) -> bool {
63    if line_num == 0 {
64        return false;
65    }
66
67    let prev_line = lines[line_num - 1].trim_end();
68
69    // Previous line ends with backslash = continuation
70    if prev_line.ends_with('\\') {
71        return true;
72    }
73
74    false
75}
76
77/// Check a recipe line for echo/printf without @ prefix
78fn check_recipe_line(line: &str, line_num: usize) -> Option<Diagnostic> {
79    let trimmed = line.trim_start_matches('\t').trim_start();
80
81    // Already has @ prefix - OK
82    if trimmed.starts_with('@') {
83        return None;
84    }
85
86    // Check if this is an echo or printf command
87    for cmd in SILENT_COMMANDS {
88        if is_command(trimmed, cmd) {
89            let span = Span::new(line_num + 1, 1, line_num + 1, line.len() + 1);
90            let fix_replacement = line.replacen('\t', "\t@", 1);
91
92            return Some(
93                Diagnostic::new(
94                    "MAKE007",
95                    Severity::Warning,
96                    format!(
97                        "Command '{}' without @ prefix - will show duplicate output",
98                        cmd
99                    ),
100                    span,
101                )
102                .with_fix(Fix::new(&fix_replacement)),
103            );
104        }
105    }
106
107    None
108}
109
110/// Check if trimmed line starts with the given command
111fn is_command(line: &str, cmd: &str) -> bool {
112    // Check if line starts with the command (as whole word)
113    if line.starts_with(cmd) {
114        // Ensure it's a whole word (followed by space, tab, or nothing)
115        if line.len() == cmd.len() {
116            return true;
117        }
118        let next_char = line.chars().nth(cmd.len());
119        matches!(next_char, Some(' ') | Some('\t'))
120    } else {
121        false
122    }
123}

131    #[test]
132    fn test_MAKE007_detects_echo_without_at() {
133        let makefile = "build:\n\techo Building...";
134        let result = check(makefile);
135
136        assert_eq!(result.diagnostics.len(), 1);
137        let diag = &result.diagnostics[0];
138        assert_eq!(diag.code, "MAKE007");
139        assert_eq!(diag.severity, Severity::Warning);
140        assert!(diag.message.contains("@"));
141    }

149    }
