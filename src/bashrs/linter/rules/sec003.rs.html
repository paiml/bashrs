1//! SEC003: Unquoted find -exec {} Pattern
2//!
3//! **Rule**: Detect unquoted `{}` in `find -exec` commands
4//!
5//! **Why this matters**:
6//! Filenames with spaces or special characters can break or execute unintended
7//! commands when {} is not quoted in find -exec.
8//!
9//! **Auto-fix**: Safe (add quotes)
10//!
11//! ## Examples
12//!
13//! ❌ **UNSAFE**:
14//! ```bash
15//! find . -name "*.sh" -exec chmod +x {} \;
16//! find /tmp -type f -exec rm {} \;
17//! ```
18//!
19//! ✅ **SAFE** (auto-fixable):
20//! ```bash
21//! find . -name "*.sh" -exec chmod +x "{}" \;
22//! find /tmp -type f -exec rm "{}" \;
23//! ```
24
25use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
26
27/// Check for unquoted {} in find -exec
28pub fn check(source: &str) -> LintResult {
29    let mut result = LintResult::new();
30
31    for (line_num, line) in source.lines().enumerate() {
32        // Look for find -exec pattern with unquoted {}
33        if line.contains("find ") && line.contains("-exec") {
34            // Check for {} that is NOT "{}"
35            if let Some(col) = line.find(" {} ") {
36                // Make sure it's not already quoted
37                let before = &line[..col];
38                if !before.ends_with('"') {
39                    let span = Span::new(
40                        line_num + 1,
41                        col + 2, // Space before {}
42                        line_num + 1,
43                        col + 4, // {} is 2 chars
44                    );
45
46                    let diag = Diagnostic::new(
47                        "SEC003",
48                        Severity::Error,
49                        "Unquoted {} in find -exec - filenames with spaces will break",
50                        span,
51                    )
52                    .with_fix(Fix::new("\"{}\""));
53
54                    result.add(diag);
55                }
56            }
57        }
58    }
59
60    result
61}

63#[cfg(test)]

67    #[test]
68    fn test_SEC003_detects_unquoted_find_exec() {
69        let script = r#"find . -name "*.sh" -exec chmod +x {} \;"#;
70        let result = check(script);
71
72        assert_eq!(result.diagnostics.len(), 1);
73        let diag = &result.diagnostics[0];
74        assert_eq!(diag.code, "SEC003");
75        assert_eq!(diag.severity, Severity::Error);
76        assert!(diag.message.contains("Unquoted"));
77    }

81        let script = "find /tmp -type f -exec rm {} \\;";
82        let result = check(script);
83
84        assert_eq!(result.diagnostics.len(), 1);
85    }
