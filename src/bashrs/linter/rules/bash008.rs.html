1//! BASH008: Missing Error Messages Before Exit
2//!
3//! **Rule**: Detect `exit` without preceding error message
4//!
5//! **Why this matters**:
6//! Silent failures are hard to debug:
7//! - Users don't know what went wrong
8//! - No context for troubleshooting
9//! - Automation pipelines fail mysteriously
10//! - Wastes developer time investigating
11//!
12//! **Examples**:
13//!
14//! ❌ **BAD** (silent failure):
15//! ```bash
16//! if [ ! -f "$CONFIG" ]; then
17//!   exit 1  # Why did it fail? User has no idea!
18//! fi
19//!
20//! command || exit 1  # Silent failure
21//! ```
22//!
23//! ✅ **GOOD** (informative error):
24//! ```bash
25//! if [ ! -f "$CONFIG" ]; then
26//!   echo "Error: Config file not found: $CONFIG" >&2
27//!   exit 1
28//! fi
29//!
30//! command || { echo "Error: Command failed" >&2; exit 1; }
31//! ```
32//!
33//! ## Detection Logic
34//!
35//! This rule detects:
36//! - `exit 1` or `exit <non-zero>` without preceding error message
37//! - `exit` without explicit code (defaults to last command's exit code)
38//!
39//! Does NOT flag:
40//! - `exit 0` - success, no error message needed
41//! - Lines with both error message and exit: `echo "Error" >&2; exit 1`
42//! - Error message on previous line
43//!
44//! ## Auto-fix
45//!
46//! Suggests:
47//! ```bash
48//! echo "Error: [description]" >&2
49//! exit 1
50//! ```
51
52use crate::linter::LintResult;
53use crate::linter::{Diagnostic, Severity, Span};
54
55/// Check for exit without error message
56pub fn check(source: &str) -> LintResult {
57    let mut result = LintResult::new();
58
59    let lines: Vec<&str> = source.lines().collect();
60
61    for (line_num, line) in lines.iter().enumerate() {
62        let trimmed = line.trim();
63
64        // Skip comments
65        if trimmed.starts_with('#') {
66            continue;
67        }
68
69        // Strip comments from code
70        let code_only = if let Some(pos) = trimmed.find('#') {
71            &trimmed[..pos]
72        } else {
73            trimmed
74        };
75        let code_only = code_only.trim();
76
77        // Detect exit statements (non-zero exit codes)
78        if has_exit_statement(code_only) {
79            // Check if exit code is 0 (success - no error message needed)
80            if code_only.contains("exit 0") {
81                continue;
82            }
83
84            // Check if error message is on same line (e.g., echo "Error" >&2; exit 1)
85            if code_only.contains("echo") && code_only.contains(">&2") {
86                continue;
87            }
88
89            // Check if previous line has error message
90            if line_num > 0 && has_error_message_on_line(lines[line_num - 1]) {
91                continue;
92            }
93
94            let span = Span::new(line_num + 1, 1, line_num + 1, line.len());
95
96            let diag = Diagnostic::new(
97                "BASH008",
98                Severity::Info,
99                "Exit without error message - add 'echo \"Error: [description]\" >&2' before exit for better debugging",
100                span,
101            );
102            result.add(diag);
103        }
104    }
105
106    result
107}

109/// Check if line has exit statement (non-zero)
110fn has_exit_statement(line: &str) -> bool {
111    // Match: exit, exit 1, exit <number>
112    // But not: exit 0
113    if line.contains("exit") {
114        // Exclude "exit 0"
115        if line.contains("exit 0") {
116            return false;
117        }
118        // Match standalone "exit" or "exit <number>"
119        return true;
120    }
121    false
122}
123
124/// Check if line has error message to stderr
125fn has_error_message_on_line(line: &str) -> bool {
126    let trimmed = line.trim();
127
128    // Strip comments
129    let code_only = if let Some(pos) = trimmed.find('#') {
130        &trimmed[..pos]
131    } else {
132        trimmed
133    };
134
135    // Look for echo/printf to stderr
136    (code_only.contains("echo") || code_only.contains("printf")) && code_only.contains(">&2")
137}
