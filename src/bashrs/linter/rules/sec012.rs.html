1//! SEC012: Unsafe Deserialization
2//!
3//! **Rule**: Detect unsafe deserialization of data from untrusted sources
4//!
5//! **Why this matters**:
6//! Deserializing untrusted data without validation can lead to:
7//! - Remote code execution via eval
8//! - Arbitrary command execution via source
9//! - Variable injection and environment poisoning
10//! - Configuration tampering
11//!
12//! **Examples**:
13//!
14//! ❌ **DANGEROUS** (eval of JSON):
15//! ```bash
16//! # Attacker controls JSON, could inject malicious code
17//! eval $(echo "$USER_JSON" | jq -r '. | to_entries[] | "\\(.key)=\\(.value)"')
18//! ```
19//!
20//! ❌ **DANGEROUS** (source from remote):
21//! ```bash
22//! # Downloads and executes arbitrary code
23//! source <(curl https://example.com/config.sh)
24//! . <(wget -qO- https://example.com/setup.sh)
25//! ```
26//!
27//! ✅ **SAFE** (validate before use):
28//! ```bash
29//! # Download, verify checksum, then source
30//! curl -o config.sh https://example.com/config.sh
31//! echo "$EXPECTED_SHA256  config.sh" | sha256sum -c || exit 1
32//! source config.sh
33//! ```
34//!
35//! ## Detection Patterns
36//!
37//! This rule detects:
38//! - `eval $(... jq ...)` - JSON deserialization via eval
39//! - `source <(curl ...)` - Remote code execution via source
40//! - `eval $(curl ...)` - Direct eval of remote content
41//! - `. <(wget ...)` - Alternative source syntax with wget
42//!
43//! ## Auto-fix
44//!
45//! This rule provides **warnings** but not automatic fixes, because:
46//! - Context-dependent validation requirements
47//! - Different security requirements per use case
48//! - Requires understanding of data source trust model
49
50use crate::linter::LintResult;
51use crate::linter::{Diagnostic, Severity, Span};

53/// Check for unsafe deserialization patterns
54pub fn check(source: &str) -> LintResult {
55    let mut result = LintResult::new();
56
57    for (line_num, line) in source.lines().enumerate() {
58        let trimmed = line.trim();
59
60        // Strip comments
61        let code_only = if let Some(pos) = trimmed.find('#') {
62            &trimmed[..pos]
63        } else {
64            trimmed
65        };
66        let code_only = code_only.trim();
67
68        // Pattern 1: eval $(... jq ...) - JSON deserialization
69        if code_only.contains("eval") && code_only.contains("jq") {
70            let span = Span::new(line_num + 1, 1, line_num + 1, line.len());
71            let diag = Diagnostic::new(
72                "SEC012",
73                Severity::Error,
74                "Unsafe deserialization: eval with jq can execute arbitrary code from JSON - validate data before eval or use safer parsing",
75                span,
76            );
77            result.add(diag);
78        }
79
80        // Pattern 2: source <(curl ...) - Remote code execution
81        if (code_only.contains("source") || code_only.starts_with('.'))
82            && code_only.contains("<(curl")
83        {
84            let span = Span::new(line_num + 1, 1, line_num + 1, line.len());
85            let diag = Diagnostic::new(
86                "SEC012",
87                Severity::Error,
88                "Unsafe deserialization: sourcing remote content without verification - download, verify checksum, then source",
89                span,
90            );
91            result.add(diag);
92        }
93
94        // Pattern 3: source <(wget ...) - Remote code execution
95        if (code_only.contains("source") || code_only.starts_with('.'))
96            && code_only.contains("<(wget")
97        {
98            let span = Span::new(line_num + 1, 1, line_num + 1, line.len());
99            let diag = Diagnostic::new(
100                "SEC012",
101                Severity::Error,
102                "Unsafe deserialization: sourcing remote content without verification - download, verify checksum, then source",
103                span,
104            );
105            result.add(diag);
106        }
107
108        // Pattern 4: eval $(curl ...) - Direct eval of remote content
109        if code_only.contains("eval") && code_only.contains("$(curl") {
110            let span = Span::new(line_num + 1, 1, line_num + 1, line.len());
111            let diag = Diagnostic::new(
112                "SEC012",
113                Severity::Error,
114                "Unsafe deserialization: eval of remote content without verification - download, verify, validate, then execute",
115                span,
116            );
117            result.add(diag);
118        }
119
120        // Pattern 5: eval $(wget ...) - Direct eval of remote content
121        if code_only.contains("eval") && code_only.contains("$(wget") {
122            let span = Span::new(line_num + 1, 1, line_num + 1, line.len());
123            let diag = Diagnostic::new(
124                "SEC012",
125                Severity::Error,
126                "Unsafe deserialization: eval of remote content without verification - download, verify, validate, then execute",
127                span,
128            );
129            result.add(diag);
130        }
131
132        // Pattern 6: eval with yq (YAML deserialization)
133        if code_only.contains("eval") && code_only.contains("yq") {
134            let span = Span::new(line_num + 1, 1, line_num + 1, line.len());
135            let diag = Diagnostic::new(
136                "SEC012",
137                Severity::Error,
138                "Unsafe deserialization: eval with yq can execute arbitrary code from YAML - validate data before eval or use safer parsing",
139                span,
140            );
141            result.add(diag);
142        }
143    }
144
145    result
146}
