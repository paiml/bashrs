1// SC2104: Missing space before ]
2//
3// Detects test expressions missing required whitespace before closing bracket.
4// In POSIX shell, [ is a command and ] is its final argument, so spaces are required.
5//
6// Examples:
7// Bad:
8//   if [ "$var" = "value"]; then
9//
10// Good:
11//   if [ "$var" = "value" ]; then
12
13use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
14use once_cell::sync::Lazy;
15use regex::Regex;
16
17static MISSING_SPACE_BEFORE_BRACKET: Lazy<Regex> = Lazy::new(|| {
18    // Match: anything followed by ] without space before it
19    Regex::new(r"[^\s\[]\]").unwrap()
20});
21
22static TEST_COMMAND: Lazy<Regex> = Lazy::new(|| Regex::new(r"\[\s+").unwrap());
23
24pub fn check(source: &str) -> LintResult {
25    let mut result = LintResult::new();
26
27    for (i, line) in source.lines().enumerate() {
28        let line_num = i + 1;
29
30        // Skip comments
31        if line.trim_start().starts_with('#') {
32            continue;
33        }
34
35        // Only check lines with test commands
36        if !TEST_COMMAND.is_match(line) {
37            continue;
38        }
39
40        // Skip double brackets [[...]]
41        if line.contains("[[") {
42            continue;
43        }
44
45        // Find missing spaces before ]
46        for mat in MISSING_SPACE_BEFORE_BRACKET.find_iter(line) {
47            let match_str = mat.as_str();
48
49            // Skip if next char is ] (this is ]])
50            if mat.end() < line.len() && line.chars().nth(mat.end()) == Some(']') {
51                continue;
52            }
53
54            let start_col = mat.start() + 1;
55            let end_col = mat.end() + 1;
56
57            // Auto-fix: insert space before ]
58            // Match is like "value]" - we need to insert space before ]
59            let fixed_match = format!("{} ]", &match_str[..match_str.len() - 1]);
60            let fixed_line = format!(
61                "{}{}{}",
62                &line[..mat.start()],
63                fixed_match,
64                &line[mat.end()..]
65            );
66
67            let diagnostic = Diagnostic::new(
68                "SC2104",
69                Severity::Error,
70                "Missing space before ]",
71                Span::new(line_num, start_col, line_num, end_col),
72            )
73            .with_fix(Fix::new(fixed_line));
74
75            result.add(diagnostic);
76        }
77    }
78
79    result
80}

82#[cfg(test)]
83mod tests {
84    use super::*;
85
86    #[test]
87    fn test_sc2104_missing_space_basic() {
88        let code = r#"if [ "$var" = "value"]; then"#;
89        let result = check(code);
90        assert_eq!(result.diagnostics.len(), 1);
91        assert_eq!(result.diagnostics[0].code, "SC2104");
92        assert_eq!(result.diagnostics[0].severity, Severity::Error);
93        assert!(result.diagnostics[0].fix.is_some());
94    }
95
96    #[test]
97    fn test_sc2104_autofix() {
98        let code = r#"if [ "$var" = "value"]; then"#;
99        let result = check(code);
100        let fix = result.diagnostics[0].fix.as_ref().unwrap();
101        assert!(fix.replacement.contains(" ]"));
102        assert!(!fix.replacement.contains("\"]\"")); // Should not have "]" without space
103        assert!(fix.replacement.contains("\" ]")); // Should have " ]" with space
104    }
105
106    #[test]
107    fn test_sc2104_correct_spacing_ok() {
108        let code = r#"if [ "$var" = "value" ]; then"#;
109        let result = check(code);
110        assert_eq!(result.diagnostics.len(), 0);
111    }
112
113    #[test]
114    fn test_sc2104_double_bracket_ok() {
115        let code = r#"if [[ "$var" = "value"]]; then"#;
116        let result = check(code);
117        // Should not trigger on [[...]] (bash extended test)
118        assert_eq!(result.diagnostics.len(), 0);
119    }
120
121    #[test]
122    fn test_sc2104_numeric_comparison() {
123        let code = r#"if [ "$count" -eq 10]; then"#;
124        let result = check(code);

126    }
127
128    #[test]
129    fn test_sc2104_string_comparison() {
130        let code = r#"if [ "$str" != "test"]; then"#;
131        let result = check(code);
132        assert_eq!(result.diagnostics.len(), 1);
133    }
134
135    #[test]
136    fn test_sc2104_file_test() {
137        let code = r#"if [ -f "$file"]; then"#;
138        let result = check(code);

140    }
141
142    #[test]
143    fn test_sc2104_multiple_conditions() {
144        let code = r#"if [ "$a" = "1"] && [ "$b" = "2" ]; then"#;
145        let result = check(code);
146        assert_eq!(result.diagnostics.len(), 1);
147    }
148
149    #[test]
150    fn test_sc2104_no_test_command() {

154    }
