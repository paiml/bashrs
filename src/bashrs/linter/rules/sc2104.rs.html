1// SC2104: Missing space before ]
2//
3// Detects test expressions missing required whitespace before closing bracket.
4// In POSIX shell, [ is a command and ] is its final argument, so spaces are required.
5//
6// Examples:
7// Bad:
8//   if [ "$var" = "value"]; then
9//
10// Good:
11//   if [ "$var" = "value" ]; then
12
13use crate::linter::{Diagnostic, Fix, LintResult, Severity, Span};
14use once_cell::sync::Lazy;
15use regex::Regex;
16
17static MISSING_SPACE_BEFORE_BRACKET: Lazy<Regex> = Lazy::new(|| {
18    // Match: anything followed by ] without space before it
19    Regex::new(r"[^\s\[]\]").unwrap()
20});
21
22static TEST_COMMAND: Lazy<Regex> = Lazy::new(|| Regex::new(r"\[\s+").unwrap());
23
24/// Check if a position is inside a parameter expansion ${...}
25fn is_inside_param_expansion(line: &str, pos: usize) -> bool {
26    let chars: Vec<char> = line.chars().collect();
27    let mut depth = 0;
28
29    for i in 0..pos.min(chars.len()) {
30        if i + 1 < chars.len() && chars[i] == '$' && chars[i + 1] == '{' {
31            depth += 1;
32        } else if chars[i] == '}' && depth > 0 {
33            depth -= 1;
34        }
35    }
36
37    depth > 0
38}
39
40pub fn check(source: &str) -> LintResult {
41    let mut result = LintResult::new();
42
43    for (i, line) in source.lines().enumerate() {
44        let line_num = i + 1;
45
46        // Skip comments
47        if line.trim_start().starts_with('#') {
48            continue;
49        }
50
51        // Only check lines with test commands
52        if !TEST_COMMAND.is_match(line) {
53            continue;
54        }
55
56        // Skip double brackets [[...]]
57        if line.contains("[[") {
58            continue;
59        }
60
61        // Find missing spaces before ]
62        for mat in MISSING_SPACE_BEFORE_BRACKET.find_iter(line) {
63            let match_str = mat.as_str();
64
65            // Skip if next char is ] (this is ]])
66            if mat.end() < line.len() && line.chars().nth(mat.end()) == Some(']') {
67                continue;
68            }
69
70            // Issue #88: Skip ] inside parameter expansions ${...}
71            // e.g., ${#array[@]}, ${var[$key]}, ${var:-default}
72            if is_inside_param_expansion(line, mat.end() - 1) {
73                continue;
74            }
75
76            let start_col = mat.start() + 1;
77            let end_col = mat.end() + 1;
78
79            // Auto-fix: insert space before ]
80            // Match is like "value]" - we need to insert space before ]
81            let fixed_match = format!("{} ]", &match_str[..match_str.len() - 1]);
82            let fixed_line = format!(
83                "{}{}{}",
84                &line[..mat.start()],
85                fixed_match,
86                &line[mat.end()..]
87            );
88
89            let diagnostic = Diagnostic::new(
90                "SC2104",
91                Severity::Error,
92                "Missing space before ]",
93                Span::new(line_num, start_col, line_num, end_col),
94            )
95            .with_fix(Fix::new(fixed_line));
96
97            result.add(diagnostic);
98        }
99    }
100
101    result
102}

104#[cfg(test)]
105mod tests {

108    #[test]
109    fn test_sc2104_missing_space_basic() {
110        let code = r#"if [ "$var" = "value"]; then"#;
111        let result = check(code);
112        assert_eq!(result.diagnostics.len(), 1);
113        assert_eq!(result.diagnostics[0].code, "SC2104");
114        assert_eq!(result.diagnostics[0].severity, Severity::Error);
115        assert!(result.diagnostics[0].fix.is_some());
116    }
117
118    #[test]
119    fn test_sc2104_autofix() {
120        let code = r#"if [ "$var" = "value"]; then"#;
121        let result = check(code);
122        let fix = result.diagnostics[0].fix.as_ref().unwrap();
123        assert!(fix.replacement.contains(" ]"));
124        assert!(!fix.replacement.contains("\"]\"")); // Should not have "]" without space
125        assert!(fix.replacement.contains("\" ]")); // Should have " ]" with space
126    }
