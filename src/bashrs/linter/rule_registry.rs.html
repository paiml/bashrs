4use crate::linter::shell_compatibility::ShellCompatibility;
5use std::collections::HashMap;
6
7/// Rule metadata including shell compatibility
8#[derive(Debug, Clone)]
9pub struct RuleMetadata {
10    pub id: &'static str,
11    pub name: &'static str,
12    pub compatibility: ShellCompatibility,
13}
14
15/// Get compatibility for a specific rule ID
16pub fn get_rule_compatibility(rule_id: &str) -> Option<ShellCompatibility> {
17    RULE_REGISTRY.get(rule_id).map(|meta| meta.compatibility)
18}
19
20/// Check if a rule should be applied for the given shell type
21pub fn should_apply_rule(rule_id: &str, shell: crate::linter::shell_type::ShellType) -> bool {
22    if let Some(compat) = get_rule_compatibility(rule_id) {
23        compat.applies_to(shell)
24    } else {
25        // If rule not in registry, assume universal (conservative approach)
26        true
27    }
28}
