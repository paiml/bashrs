1//! Shell compatibility rules for linter.
2//!
3//! Defines which shells each linter rule applies to, allowing bashrs to skip
4//! rules that don't apply to a specific shell type.
5//!
6//! # Examples
7//!
8//! ## Checking if a rule applies to a shell
9//!
10//! ```
11//! use bashrs::linter::{ShellCompatibility, ShellType};
12//!
13//! let compat = ShellCompatibility::BashOnly;
14//! assert!(compat.applies_to(ShellType::Bash));
15//! assert!(!compat.applies_to(ShellType::Sh));
16//! ```
17//!
18//! ## Getting human-readable descriptions
19//!
20//! ```
21//! use bashrs::linter::ShellCompatibility;
22//!
23//! let compat = ShellCompatibility::Universal;
24//! assert_eq!(compat.description(), "all shells (bash, zsh, sh, ksh)");
25//! ```
26//!
27//! ## Using with rule registry
28//!
29//! ```
30//! use bashrs::linter::{get_rule_compatibility, ShellType};
31//!
32//! // Check if SEC001 applies to POSIX sh
33//! if let Some(compat) = get_rule_compatibility("SEC001") {
34//!     if compat.applies_to(ShellType::Sh) {
35//!         // Apply SEC001 to this sh script
36//!     }
37//! }
38//! ```

42/// Shell compatibility level for linter rules.
43///
44/// Specifies which shell types a linter rule applies to. Used by the rule registry
45/// to filter rules based on the detected shell type.
46///
47/// # Variant Descriptions
48///
49/// - **Universal**: Rule applies to all shells (bash, zsh, sh, ksh)
50/// - **BashOnly**: Rule only applies to bash scripts
51/// - **ZshOnly**: Rule only applies to zsh scripts
52/// - **ShOnly**: Rule only applies to POSIX sh scripts (strict)
53/// - **BashZsh**: Rule applies to bash and zsh (modern shells)
54/// - **NotSh**: Rule applies to bash/zsh/ksh but not POSIX sh
55///
56/// # Examples
57///
58/// ## Checking shell compatibility
59///
60/// ```
61/// use bashrs::linter::{ShellCompatibility, ShellType};
62///
63/// let compat = ShellCompatibility::BashZsh;
64/// assert!(compat.applies_to(ShellType::Bash));
65/// assert!(compat.applies_to(ShellType::Zsh));
66/// assert!(!compat.applies_to(ShellType::Sh));
67/// ```
68///
69/// ## Using universal compatibility
70///
71/// ```
72/// use bashrs::linter::{ShellCompatibility, ShellType};
73///
74/// let compat = ShellCompatibility::Universal;
75/// // Universal rules apply to ALL shells
76/// assert!(compat.applies_to(ShellType::Bash));
77/// assert!(compat.applies_to(ShellType::Zsh));
78/// assert!(compat.applies_to(ShellType::Sh));
79/// assert!(compat.applies_to(ShellType::Ksh));
80/// ```
81///
82/// ## Shell-specific rules
83///
84/// ```
85/// use bashrs::linter::{ShellCompatibility, ShellType};
86///
87/// // BashOnly rules only apply to bash
88/// let bash_rule = ShellCompatibility::BashOnly;
89/// assert!(bash_rule.applies_to(ShellType::Bash));
90/// assert!(!bash_rule.applies_to(ShellType::Zsh));
91///
92/// // ShOnly rules only apply to POSIX sh
93/// let sh_rule = ShellCompatibility::ShOnly;
94/// assert!(sh_rule.applies_to(ShellType::Sh));
95/// assert!(!sh_rule.applies_to(ShellType::Bash));
96/// ```
97#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
98pub enum ShellCompatibility {
99    /// Works in all shells (bash, zsh, sh, ksh)
100    Universal,
101    /// Bash-specific features only
102    BashOnly,
103    /// Zsh-specific features only
104    ZshOnly,
105    /// POSIX sh only (strict)
106    ShOnly,
107    /// Works in bash and zsh
108    BashZsh,
109    /// Works in bash/zsh/ksh but not sh
110    NotSh,
111}
112
113impl ShellCompatibility {
114    /// Checks if this compatibility level applies to the given shell type.
115    ///
116    /// Returns `true` if a rule with this compatibility should be applied to the specified shell.
117    ///
118    /// # Arguments
119    ///
120    /// * `shell` - The shell type to check compatibility for
121    ///
122    /// # Returns
123    ///
124    /// `true` if the rule applies to this shell, `false` otherwise
125    ///
126    /// # Special Cases
127    ///
128    /// - `ShellType::Auto` defaults to bash behavior for compatibility purposes
129    /// - `Universal` applies to all shells including `Auto`
130    ///
131    /// # Examples
132    ///
133    /// ## Universal compatibility
134    ///
135    /// ```
136    /// use bashrs::linter::{ShellCompatibility, ShellType};
137    ///
138    /// let compat = ShellCompatibility::Universal;
139    /// assert!(compat.applies_to(ShellType::Bash));
140    /// assert!(compat.applies_to(ShellType::Zsh));
141    /// assert!(compat.applies_to(ShellType::Sh));
142    /// assert!(compat.applies_to(ShellType::Auto));
143    /// ```
144    ///
145    /// ## Bash-only rules
146    ///
147    /// ```
148    /// use bashrs::linter::{ShellCompatibility, ShellType};
149    ///
150    /// let compat = ShellCompatibility::BashOnly;
151    /// assert!(compat.applies_to(ShellType::Bash));
152    /// assert!(!compat.applies_to(ShellType::Zsh));
153    /// assert!(!compat.applies_to(ShellType::Sh));
154    /// // Auto defaults to bash
155    /// assert!(compat.applies_to(ShellType::Auto));
156    /// ```
157    ///
158    /// ## POSIX sh-only rules
159    ///
160    /// ```
161    /// use bashrs::linter::{ShellCompatibility, ShellType};
162    ///
163    /// let compat = ShellCompatibility::ShOnly;
164    /// assert!(!compat.applies_to(ShellType::Bash));
165    /// assert!(compat.applies_to(ShellType::Sh));
166    /// assert!(!compat.applies_to(ShellType::Auto));
167    /// ```
168    ///
169    /// ## NotSh compatibility (bash/zsh/ksh but not sh)
170    ///
171    /// ```
172    /// use bashrs::linter::{ShellCompatibility, ShellType};
173    ///
174    /// let compat = ShellCompatibility::NotSh;
175    /// assert!(compat.applies_to(ShellType::Bash));
176    /// assert!(compat.applies_to(ShellType::Zsh));
177    /// assert!(compat.applies_to(ShellType::Ksh));
178    /// assert!(!compat.applies_to(ShellType::Sh));
179    /// ```
180    pub fn applies_to(&self, shell: ShellType) -> bool {
181        match (self, shell) {
182            (ShellCompatibility::Universal, _) => true,
183            (ShellCompatibility::BashOnly, ShellType::Bash) => true,
184            (ShellCompatibility::ZshOnly, ShellType::Zsh) => true,
185            (ShellCompatibility::ShOnly, ShellType::Sh) => true,
186            (ShellCompatibility::BashZsh, ShellType::Bash | ShellType::Zsh) => true,
187            (ShellCompatibility::NotSh, ShellType::Bash | ShellType::Zsh | ShellType::Ksh) => true,
188            (ShellCompatibility::NotSh, ShellType::Auto) => true, // Auto defaults to bash
189            (ShellCompatibility::BashOnly, ShellType::Auto) => true, // Auto defaults to bash
190            (ShellCompatibility::BashZsh, ShellType::Auto) => true, // Auto defaults to bash
191            _ => false,
192        }
193    }
194
195    /// Returns a human-readable description of the shell compatibility level.
196    ///
197    /// Provides a concise description of which shells this compatibility level applies to,
198    /// useful for displaying in error messages, documentation, or user interfaces.
199    ///
200    /// # Returns
201    ///
202    /// A static string describing the shell compatibility level
203    ///
204    /// # Examples
205    ///
206    /// ## Display compatibility descriptions
207    ///
208    /// ```
209    /// use bashrs::linter::ShellCompatibility;
210    ///
211    /// assert_eq!(
212    ///     ShellCompatibility::Universal.description(),
213    ///     "all shells (bash, zsh, sh, ksh)"
214    /// );
215    ///
216    /// assert_eq!(
217    ///     ShellCompatibility::BashOnly.description(),
218    ///     "bash only"
219    /// );
220    ///
221    /// assert_eq!(
222    ///     ShellCompatibility::BashZsh.description(),
223    ///     "bash and zsh"
224    /// );
225    /// ```
226    ///
227    /// ## Use in error messages
228    ///
229    /// ```
230    /// use bashrs::linter::ShellCompatibility;
231    ///
232    /// let compat = ShellCompatibility::NotSh;
233    /// let msg = format!(
234    ///     "This rule applies to {}",
235    ///     compat.description()
236    /// );
237    /// assert_eq!(msg, "This rule applies to bash, zsh, ksh (not POSIX sh)");
238    /// ```
239    pub fn description(&self) -> &'static str {
240        match self {
241            ShellCompatibility::Universal => "all shells (bash, zsh, sh, ksh)",
242            ShellCompatibility::BashOnly => "bash only",
243            ShellCompatibility::ZshOnly => "zsh only",
244            ShellCompatibility::ShOnly => "POSIX sh only",
245            ShellCompatibility::BashZsh => "bash and zsh",
246            ShellCompatibility::NotSh => "bash, zsh, ksh (not POSIX sh)",
247        }
248    }
249}
