1//! Bash Lexer
2//!
3//! Tokenizes bash scripts into a stream of tokens for parsing.
4//! Handles shell-specific quirks like variable expansion, quoting, etc.
5
6use std::fmt;
7use thiserror::Error;
8
9#[derive(Debug, Clone, PartialEq, Eq)]
10pub enum Token {
11    // Keywords
12    If,
13    Then,
14    Elif,
15    Else,
16    Fi,
17    For,
18    While,
19    Do,
20    Done,
21    Case,
22    Esac,
23    In,
24    Function,
25    Return,
26    Export,
27    Local,
28
29    // Identifiers and literals
30    Identifier(String),
31    String(String),
32    Number(i64),
33
34    // Operators
35    Assign,             // =
36    Eq,                 // ==
37    Ne,                 // !=
38    Lt,                 // <
39    Le,                 // <=
40    Gt,                 // >
41    Ge,                 // >=
42    And,                // &&
43    Or,                 // ||
44    Not,                // !
45    Pipe,               // |
46    Semicolon,          // ;
47    Ampersand,          // &
48    Dollar,             // $
49    LeftParen,          // (
50    RightParen,         // )
51    LeftBrace,          // {
52    RightBrace,         // }
53    LeftBracket,        // [
54    RightBracket,       // ]
55    DoubleLeftBracket,  // [[
56    DoubleRightBracket, // ]]
57
58    // Special
59    Variable(String),            // $VAR
60    ArithmeticExpansion(String), // $((expr))
61    Comment(String),
62    Newline,
63    Eof,
64}
65
66impl fmt::Display for Token {
67    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
68        match self {
69            Token::If => write!(f, "if"),
70            Token::Then => write!(f, "then"),
71            Token::Identifier(s) => write!(f, "Identifier({})", s),
72            Token::String(s) => write!(f, "String({})", s),
73            Token::Number(n) => write!(f, "Number({})", n),
74            Token::Variable(v) => write!(f, "${}", v),
75            Token::ArithmeticExpansion(e) => write!(f, "$(({})", e),
76            Token::Comment(c) => write!(f, "#{}", c),
77            Token::Eof => write!(f, "EOF"),
78            _ => write!(f, "{:?}", self),
79        }
80    }
81}

83#[derive(Error, Debug)]
84pub enum LexerError {
85    #[error("Unexpected character '{0}' at line {1}, column {2}")]
86    UnexpectedChar(char, usize, usize),
87
88    #[error("Unterminated string at line {0}, column {1}")]
89    UnterminatedString(usize, usize),
90
91    #[error("Invalid number format: {0}")]
92    InvalidNumber(String),
93}
94
95pub struct Lexer {
96    input: Vec<char>,
97    position: usize,
98    line: usize,
99    column: usize,
100}
101
102impl Lexer {
103    pub fn new(input: &str) -> Self {
104        Self {
105            input: input.chars().collect(),
106            position: 0,
107            line: 1,
108            column: 1,
109        }
110    }
111
112    pub fn tokenize(&mut self) -> Result<Vec<Token>, LexerError> {
113        let mut tokens = Vec::new();
114
115        loop {
116            self.skip_whitespace_except_newline();
117
118            if self.is_at_end() {
119                tokens.push(Token::Eof);
120                break;
121            }
122
123            let token = self.next_token()?;
124            tokens.push(token.clone());
125
126            if token == Token::Eof {
127                break;
128            }
129        }
130
131        Ok(tokens)
132    }
133
134    fn next_token(&mut self) -> Result<Token, LexerError> {
135        if self.is_at_end() {
136            return Ok(Token::Eof);
137        }
138
139        let ch = self.current_char();
140
141        // Comments
142        if ch == '#' {
143            return Ok(self.read_comment());
144        }
145
146        // Newlines
147        if ch == '\n' {
148            self.advance();
149            return Ok(Token::Newline);
150        }
151
152        // Variables
153        if ch == '$' {
154            return self.read_variable();
155        }
156
157        // Strings
158        if ch == '"' || ch == '\'' {
159            return self.read_string(ch);
160        }
161
162        // Numbers
163        if ch.is_ascii_digit() {
164            return self.read_number();
165        }
166
167        // Identifiers and keywords
168        if ch.is_alphabetic() || ch == '_' {
169            return Ok(self.read_identifier_or_keyword());
170        }
171
172        // Bare words (paths, globs, etc) - must come before operators
173        // These are unquoted strings that can contain /  * . - : + % etc
174        // Note: ':' is included for bash builtin no-op command (BUILTIN-001)
175        // Note: '+' and '%' are included for flags like date +%FORMAT (PARSER-ENH-001)
176        if ch == '/'
177            || ch == '.'
178            || ch == '-'
179            || ch == '*'
180            || ch == '~'
181            || ch == ':'
182            || ch == '+'
183            || ch == '%'
184        {
185            return Ok(self.read_bare_word());
186        }
187
188        // Operators and symbols
189        self.read_operator()
190    }
191
192    fn current_char(&self) -> char {
193        self.input[self.position]
194    }
195
196    fn peek_char(&self, offset: usize) -> Option<char> {
197        self.input.get(self.position + offset).copied()
198    }
199
200    fn advance(&mut self) -> char {
201        let ch = self.current_char();
202        self.position += 1;
203        if ch == '\n' {
204            self.line += 1;
205            self.column = 1;
206        } else {
207            self.column += 1;
208        }
209        ch
210    }
211
212    fn is_at_end(&self) -> bool {
213        self.position >= self.input.len()
214    }
215
216    fn skip_whitespace_except_newline(&mut self) {
217        while !self.is_at_end() {
218            let ch = self.current_char();
219            if ch == ' ' || ch == '\t' || ch == '\r' {
220                self.advance();
221            } else {
222                break;
223            }
224        }
225    }
226
227    fn read_comment(&mut self) -> Token {
228        self.advance(); // skip '#'
229        let mut comment = String::new();
230        while !self.is_at_end() && self.current_char() != '\n' {
231            comment.push(self.advance());
232        }
233        Token::Comment(comment)
234    }
235
236    fn read_variable(&mut self) -> Result<Token, LexerError> {
237        self.advance(); // skip '$'
238
239        // Check for arithmetic expansion $((...))
240        if !self.is_at_end() && self.current_char() == '(' {
241            if let Some('(') = self.peek_char(1) {
242                return self.read_arithmetic_expansion();
243            }
244            // Note: $(command) command substitution handled separately
245            // For now, fall through to general variable parsing
246        }
247
248        // Check for $$ (process ID special variable)
249        if !self.is_at_end() && self.current_char() == '$' {
250            self.advance(); // skip second '$'
251            // Return special variable name for process ID
252            // Using "$" as the variable name to represent $$
253            return Ok(Token::Variable("$".to_string()));
254        }
255
256        let mut var_name = String::new();
257
258        // Handle ${VAR} syntax
259        if !self.is_at_end() && self.current_char() == '{' {
260            self.advance();
261            while !self.is_at_end() && self.current_char() != '}' {
262                var_name.push(self.advance());
263            }
264            if !self.is_at_end() {
265                self.advance(); // skip '}'
266            }
267        } else {
268            // Handle $VAR syntax
269            while !self.is_at_end() {
270                let ch = self.current_char();
271                if ch.is_alphanumeric() || ch == '_' {
272                    var_name.push(self.advance());
273                } else {
274                    break;
275                }
276            }
277        }
278
279        Ok(Token::Variable(var_name))
280    }
281
282    fn read_arithmetic_expansion(&mut self) -> Result<Token, LexerError> {
283        // Skip '(('
284        self.advance(); // skip first '('
285        self.advance(); // skip second '('
286
287        let mut expr = String::new();
288        let mut paren_depth = 0;
289
290        while !self.is_at_end() {
291            let ch = self.current_char();
292
293            // Handle nested parentheses
294            if ch == '(' {
295                paren_depth += 1;
296                expr.push(self.advance());
297            } else if ch == ')' {
298                // Check if this closes the arithmetic expansion
299                if paren_depth == 0 && self.peek_char(1) == Some(')') {
300                    self.advance(); // skip first ')'
301                    self.advance(); // skip second ')'
302                    break;
303                } else {
304                    paren_depth -= 1;
305                    expr.push(self.advance());
306                }
307            } else {
308                expr.push(self.advance());
309            }
310        }
311
312        Ok(Token::ArithmeticExpansion(expr))
313    }
314
315    fn read_string(&mut self, quote: char) -> Result<Token, LexerError> {
316        let start_line = self.line;
317        let start_col = self.column;
318
319        self.advance(); // skip opening quote
320        let mut string = String::new();
321
322        while !self.is_at_end() && self.current_char() != quote {
323            let ch = self.advance();
324            if ch == '\\' && !self.is_at_end() {
325                // Handle escape sequences
326                let escaped = self.advance();
327                match escaped {
328                    'n' => string.push('\n'),
329                    't' => string.push('\t'),
330                    'r' => string.push('\r'),
331                    '\\' => string.push('\\'),
332                    _ => {
333                        string.push('\\');
334                        string.push(escaped);
335                    }
336                }
337            } else {
338                string.push(ch);
339            }
340        }
341
342        if self.is_at_end() {
343            return Err(LexerError::UnterminatedString(start_line, start_col));
344        }
345
346        self.advance(); // skip closing quote
347        Ok(Token::String(string))
348    }
349
350    fn read_number(&mut self) -> Result<Token, LexerError> {
351        let mut num_str = String::new();
352
353        while !self.is_at_end() && self.current_char().is_ascii_digit() {
354            num_str.push(self.advance());
355        }
356
357        num_str
358            .parse::<i64>()
359            .map(Token::Number)
360            .map_err(|_| LexerError::InvalidNumber(num_str))
361    }
362
363    fn read_identifier_or_keyword(&mut self) -> Token {
364        let mut ident = String::new();
365
366        while !self.is_at_end() {
367            let ch = self.current_char();
368            if ch.is_alphanumeric() || ch == '_' {
369                ident.push(self.advance());
370            } else {
371                break;
372            }
373        }
374
375        // Check for keywords
376        match ident.as_str() {
377            "if" => Token::If,
378            "then" => Token::Then,
379            "elif" => Token::Elif,
380            "else" => Token::Else,
381            "fi" => Token::Fi,
382            "for" => Token::For,
383            "while" => Token::While,
384            "do" => Token::Do,
385            "done" => Token::Done,
386            "case" => Token::Case,
387            "esac" => Token::Esac,
388            "in" => Token::In,
389            "function" => Token::Function,
390            "return" => Token::Return,
391            "export" => Token::Export,
392            "local" => Token::Local,
393            _ => Token::Identifier(ident),
394        }
395    }
396
397    fn read_bare_word(&mut self) -> Token {
398        let mut word = String::new();
399
400        while !self.is_at_end() {
401            let ch = self.current_char();
402            // Bare words can contain alphanumeric, path separators, globs, dots, dashes, plus signs, percent signs
403            // Note: '+' and '%' added for date +%FORMAT support (PARSER-ENH-001)
404            if ch.is_alphanumeric()
405                || ch == '/'
406                || ch == '.'
407                || ch == '-'
408                || ch == '_'
409                || ch == '*'
410                || ch == '?'
411                || ch == '~'
412                || ch == ':'
413                || ch == '+'
414                || ch == '%'
415            {
416                word.push(self.advance());
417            } else {
418                break;
419            }
420        }
421
422        Token::Identifier(word)
423    }
424
425    fn read_operator(&mut self) -> Result<Token, LexerError> {
426        let ch = self.current_char();
427        let next_ch = self.peek_char(1);
428
429        let token = match (ch, next_ch) {
430            ('=', Some('=')) => {
431                self.advance();
432                self.advance();
433                Token::Eq
434            }
435            ('!', Some('=')) => {
436                self.advance();
437                self.advance();
438                Token::Ne
439            }
440            ('<', Some('=')) => {
441                self.advance();
442                self.advance();
443                Token::Le
444            }
445            ('>', Some('=')) => {
446                self.advance();
447                self.advance();
448                Token::Ge
449            }
450            ('&', Some('&')) => {
451                self.advance();
452                self.advance();
453                Token::And
454            }
455            ('|', Some('|')) => {
456                self.advance();
457                self.advance();
458                Token::Or
459            }
460            ('[', Some('[')) => {
461                self.advance();
462                self.advance();
463                Token::DoubleLeftBracket
464            }
465            (']', Some(']')) => {
466                self.advance();
467                self.advance();
468                Token::DoubleRightBracket
469            }
470            ('=', _) => {
471                self.advance();
472                Token::Assign
473            }
474            ('<', _) => {
475                self.advance();
476                Token::Lt
477            }
478            ('>', _) => {
479                self.advance();
480                Token::Gt
481            }
482            ('!', _) => {
483                self.advance();
484                Token::Not
485            }
486            ('|', _) => {
487                self.advance();
488                Token::Pipe
489            }
490            (';', _) => {
491                self.advance();
492                Token::Semicolon
493            }
494            ('&', _) => {
495                self.advance();
496                Token::Ampersand
497            }
498            ('(', _) => {
499                self.advance();
500                Token::LeftParen
501            }
502            (')', _) => {
503                self.advance();
504                Token::RightParen
505            }
506            ('{', _) => {
507                self.advance();
508                Token::LeftBrace
509            }
510            ('}', _) => {
511                self.advance();
512                Token::RightBrace
513            }
514            ('[', _) => {
515                self.advance();
516                Token::LeftBracket
517            }
518            (']', _) => {
519                self.advance();
520                Token::RightBracket
521            }
522            _ => {
523                return Err(LexerError::UnexpectedChar(ch, self.line, self.column));
524            }
525        };
526
527        Ok(token)
528    }
