1//! Bash Parser
2//!
3//! Parses token stream from lexer into bash AST.
4//! Implements recursive descent parsing for bash syntax.
5
6use super::ast::*;
7use super::lexer::{Lexer, LexerError, Token};
8use thiserror::Error;
9
10#[derive(Error, Debug)]
11pub enum ParseError {
12    #[error("Lexer error: {0}")]
13    LexerError(#[from] LexerError),
14
15    #[error("Unexpected token: expected {expected}, found {found} at line {line}")]
16    UnexpectedToken {
17        expected: String,
18        found: String,
19        line: usize,
20    },
21
22    #[error("Unexpected end of file")]
23    UnexpectedEof,
24
25    #[error("Invalid syntax: {0}")]
26    InvalidSyntax(String),
27}
28
29pub type ParseResult<T> = Result<T, ParseError>;
30
31pub struct BashParser {
32    tokens: Vec<Token>,
33    position: usize,
34    current_line: usize,
35}
36
37impl BashParser {
38    pub fn new(input: &str) -> ParseResult<Self> {
39        let mut lexer = Lexer::new(input);
40        let tokens = lexer.tokenize()?;
41
42        Ok(Self {
43            tokens,
44            position: 0,
45            current_line: 1,
46        })
47    }
48
49    pub fn parse(&mut self) -> ParseResult<BashAst> {
50        let start_time = std::time::Instant::now();
51        let mut statements = Vec::new();
52
53        while !self.is_at_end() {
54            self.skip_newlines();
55            if self.is_at_end() {
56                break;
57            }
58
59            let stmt = self.parse_statement()?;
60            statements.push(stmt);
61
62            self.skip_newlines();
63        }
64
65        let parse_time_ms = start_time.elapsed().as_millis() as u64;
66
67        Ok(BashAst {
68            statements,
69            metadata: AstMetadata {
70                source_file: None,
71                line_count: self.current_line,
72                parse_time_ms,
73            },
74        })
75    }
76
77    fn parse_statement(&mut self) -> ParseResult<BashStmt> {
78        // Skip comments and collect them
79        if let Some(Token::Comment(text)) = self.peek() {
80            let comment = text.clone();
81            self.advance();
82            return Ok(BashStmt::Comment {
83                text: comment,
84                span: Span::dummy(),
85            });
86        }
87
88        match self.peek() {
89            Some(Token::If) => self.parse_if(),
90            Some(Token::While) => self.parse_while(),
91            Some(Token::For) => self.parse_for(),
92            Some(Token::Function) => self.parse_function(),
93            Some(Token::Return) => self.parse_return(),
94            Some(Token::Export) => self.parse_export(),
95            Some(Token::Identifier(_)) => {
96                // Could be assignment or command
97                if self.peek_ahead(1) == Some(&Token::Assign) {
98                    self.parse_assignment(false)
99                } else {
100                    self.parse_command()
101                }
102            }
103            _ => self.parse_command(),
104        }
105    }
106
107    fn parse_if(&mut self) -> ParseResult<BashStmt> {
108        self.expect(Token::If)?;
109
110        let condition = self.parse_test_expression()?;
111
112        // Skip optional semicolon before then
113        if self.check(&Token::Semicolon) {
114            self.advance();
115        }
116
117        self.skip_newlines();
118        self.expect(Token::Then)?;
119        self.skip_newlines();
120
121        let then_block = self.parse_block_until(&[Token::Elif, Token::Else, Token::Fi])?;
122
123        let mut elif_blocks = Vec::new();
124        while self.check(&Token::Elif) {
125            self.advance();
126            let elif_condition = self.parse_test_expression()?;
127
128            // Skip optional semicolon before then
129            if self.check(&Token::Semicolon) {
130                self.advance();
131            }
132
133            self.skip_newlines();
134            self.expect(Token::Then)?;
135            self.skip_newlines();
136            let elif_body = self.parse_block_until(&[Token::Elif, Token::Else, Token::Fi])?;
137            elif_blocks.push((elif_condition, elif_body));
138        }
139
140        let else_block = if self.check(&Token::Else) {
141            self.advance();
142            self.skip_newlines();
143            Some(self.parse_block_until(&[Token::Fi])?)
144        } else {
145            None
146        };
147
148        self.expect(Token::Fi)?;
149
150        Ok(BashStmt::If {
151            condition,
152            then_block,
153            elif_blocks,
154            else_block,
155            span: Span::dummy(),
156        })
157    }
158
159    fn parse_while(&mut self) -> ParseResult<BashStmt> {
160        self.expect(Token::While)?;
161
162        let condition = self.parse_test_expression()?;
163        self.skip_newlines();
164        self.expect(Token::Do)?;
165        self.skip_newlines();
166
167        let body = self.parse_block_until(&[Token::Done])?;
168        self.expect(Token::Done)?;
169
170        Ok(BashStmt::While {
171            condition,
172            body,
173            span: Span::dummy(),
174        })
175    }
176
177    fn parse_for(&mut self) -> ParseResult<BashStmt> {
178        self.expect(Token::For)?;
179
180        let variable = if let Some(Token::Identifier(name)) = self.peek() {
181            let var = name.clone();
182            self.advance();
183            var
184        } else {
185            return Err(ParseError::InvalidSyntax(
186                "Expected identifier after 'for'".to_string(),
187            ));
188        };
189
190        // Expect 'in'
191        if let Some(Token::Identifier(word)) = self.peek() {
192            if word == "in" {
193                self.advance();
194            } else {
195                return Err(ParseError::InvalidSyntax(
196                    "Expected 'in' after for variable".to_string(),
197                ));
198            }
199        }
200
201        let items = self.parse_expression()?;
202
203        // Skip optional semicolon before do
204        if self.check(&Token::Semicolon) {
205            self.advance();
206        }
207
208        self.skip_newlines();
209        self.expect(Token::Do)?;
210        self.skip_newlines();
211
212        let body = self.parse_block_until(&[Token::Done])?;
213        self.expect(Token::Done)?;
214
215        Ok(BashStmt::For {
216            variable,
217            items,
218            body,
219            span: Span::dummy(),
220        })
221    }
222
223    fn parse_function(&mut self) -> ParseResult<BashStmt> {
224        self.expect(Token::Function)?;
225
226        let name = if let Some(Token::Identifier(n)) = self.peek() {
227            let fn_name = n.clone();
228            self.advance();
229            fn_name
230        } else {
231            return Err(ParseError::InvalidSyntax(
232                "Expected function name".to_string(),
233            ));
234        };
235
236        // Optional () after function name
237        if self.check(&Token::LeftParen) {
238            self.advance();
239            self.expect(Token::RightParen)?;
240        }
241
242        self.skip_newlines();
243        self.expect(Token::LeftBrace)?;
244        self.skip_newlines();
245
246        let body = self.parse_block_until(&[Token::RightBrace])?;
247        self.expect(Token::RightBrace)?;
248
249        Ok(BashStmt::Function {
250            name,
251            body,
252            span: Span::dummy(),
253        })
254    }
255
256    fn parse_return(&mut self) -> ParseResult<BashStmt> {
257        self.expect(Token::Return)?;
258
259        let code = if self.check(&Token::Newline) || self.is_at_end() {
260            None
261        } else {
262            Some(self.parse_expression()?)
263        };
264
265        Ok(BashStmt::Return {
266            code,
267            span: Span::dummy(),
268        })
269    }
270
271    fn parse_export(&mut self) -> ParseResult<BashStmt> {
272        self.expect(Token::Export)?;
273        self.parse_assignment(true)
274    }
275
276    fn parse_assignment(&mut self, exported: bool) -> ParseResult<BashStmt> {
277        let name = if let Some(Token::Identifier(n)) = self.peek() {
278            let var_name = n.clone();
279            self.advance();
280            var_name
281        } else {
282            return Err(ParseError::InvalidSyntax(
283                "Expected variable name in assignment".to_string(),
284            ));
285        };
286
287        self.expect(Token::Assign)?;
288        let value = self.parse_expression()?;
289
290        Ok(BashStmt::Assignment {
291            name,
292            value,
293            exported,
294            span: Span::dummy(),
295        })
296    }
297
298    fn parse_command(&mut self) -> ParseResult<BashStmt> {
299        let name = match self.peek() {
300            Some(Token::Identifier(n)) => {
301                let cmd = n.clone();
302                self.advance();
303                cmd
304            }
305            Some(Token::String(s)) => {
306                let cmd = s.clone();
307                self.advance();
308                cmd
309            }
310            _ => {
311                return Err(ParseError::InvalidSyntax(
312                    "Expected command name".to_string(),
313                ))
314            }
315        };
316
317        let mut args = Vec::new();
318
319        // Parse arguments until newline or special token
320        while !self.is_at_end()
321            && !self.check(&Token::Newline)
322            && !self.check(&Token::Semicolon)
323            && !self.check(&Token::Pipe)
324        {
325            args.push(self.parse_expression()?);
326        }
327
328        Ok(BashStmt::Command {
329            name,
330            args,
331            span: Span::dummy(),
332        })
333    }
334
335    fn parse_expression(&mut self) -> ParseResult<BashExpr> {
336        match self.peek() {
337            Some(Token::String(s)) => {
338                let str = s.clone();
339                self.advance();
340                Ok(BashExpr::Literal(str))
341            }
342            Some(Token::Number(n)) => {
343                let num = *n;
344                self.advance();
345                Ok(BashExpr::Literal(num.to_string()))
346            }
347            Some(Token::Variable(v)) => {
348                let var = v.clone();
349                self.advance();
350                Ok(BashExpr::Variable(var))
351            }
352            Some(Token::Identifier(s)) => {
353                let ident = s.clone();
354                self.advance();
355                Ok(BashExpr::Literal(ident))
356            }
357            _ => Err(ParseError::InvalidSyntax("Expected expression".to_string())),
358        }
359    }
360
361    fn parse_test_expression(&mut self) -> ParseResult<BashExpr> {
362        // Handle [ ... ] test syntax
363        if self.check(&Token::LeftBracket) {
364            self.advance();
365            let expr = self.parse_test_condition()?;
366            self.expect(Token::RightBracket)?;
367            return Ok(BashExpr::Test(Box::new(expr)));
368        }
369
370        // Handle [[ ... ]] test syntax
371        if self.check(&Token::DoubleLeftBracket) {
372            self.advance();
373            let expr = self.parse_test_condition()?;
374            self.expect(Token::DoubleRightBracket)?;
375            return Ok(BashExpr::Test(Box::new(expr)));
376        }
377
378        // Fallback to regular expression
379        self.parse_expression()
380    }
381
382    fn parse_test_condition(&mut self) -> ParseResult<TestExpr> {
383        let left = self.parse_expression()?;
384
385        match self.peek() {
386            Some(Token::Eq) => {
387                self.advance();
388                let right = self.parse_expression()?;
389                Ok(TestExpr::StringEq(left, right))
390            }
391            Some(Token::Ne) => {
392                self.advance();
393                let right = self.parse_expression()?;
394                Ok(TestExpr::StringNe(left, right))
395            }
396            Some(Token::Lt) => {
397                self.advance();
398                let right = self.parse_expression()?;
399                Ok(TestExpr::IntLt(left, right))
400            }
401            Some(Token::Gt) => {
402                self.advance();
403                let right = self.parse_expression()?;
404                Ok(TestExpr::IntGt(left, right))
405            }
406            _ => Ok(TestExpr::StringNonEmpty(left)),
407        }
408    }
409
410    fn parse_block_until(&mut self, terminators: &[Token]) -> ParseResult<Vec<BashStmt>> {
411        let mut statements = Vec::new();
412
413        while !self.is_at_end() {
414            self.skip_newlines();
415
416            if terminators.iter().any(|t| self.check(t)) {
417                break;
418            }
419
420            if self.is_at_end() {
421                break;
422            }
423
424            statements.push(self.parse_statement()?);
425        }
426
427        Ok(statements)
428    }

430    // Helper methods
431    fn peek(&self) -> Option<&Token> {
432        self.tokens.get(self.position)
433    }
434
435    fn peek_ahead(&self, offset: usize) -> Option<&Token> {
436        self.tokens.get(self.position + offset)
437    }
438
439    fn advance(&mut self) -> Option<&Token> {
440        if !self.is_at_end() {
441            self.position += 1;
442        }
443        self.tokens.get(self.position - 1)
444    }
445
446    fn is_at_end(&self) -> bool {
447        matches!(self.peek(), Some(Token::Eof) | None)
448    }
449
450    fn check(&self, token: &Token) -> bool {
451        if let Some(current) = self.peek() {
452            std::mem::discriminant(current) == std::mem::discriminant(token)
453        } else {
454            false
455        }
456    }
457
458    fn expect(&mut self, expected: Token) -> ParseResult<()> {
459        if self.check(&expected) {
460            self.advance();
461            Ok(())
462        } else {
463            Err(ParseError::UnexpectedToken {
464                expected: format!("{:?}", expected),
465                found: format!("{:?}", self.peek()),
466                line: self.current_line,
467            })
468        }
469    }
470
471    fn skip_newlines(&mut self) {
472        while self.check(&Token::Newline) {
473            self.advance();
474            self.current_line += 1;
475        }
476    }
