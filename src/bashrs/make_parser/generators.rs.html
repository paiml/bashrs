1//! Code generators for Makefile AST
2//!
3//! Generates purified Makefiles from AST.
4//!
5//! This module implements the code generation phase of the purification pipeline:
6//! Parse → AST → Analyze → Purify → **Generate** → Purified Makefile
7//!
8//! ## Design Principles
9//!
10//! 1. **Correctness**: Generated Makefiles must be syntactically valid
11//! 2. **Determinism**: Same AST always produces same output
12//! 3. **POSIX Compliance**: Generated output passes shellcheck
13//! 4. **Readability**: Output is human-readable and maintainable
14//!
15//! ## Key Features
16//!
17//! - Variable generation with all 5 flavors (`:=`, `=`, `?=`, `+=`, `!=`)
18//! - Target generation with tab-indented recipes (REQUIRED by Make)
19//! - Comment preservation
20//! - .PHONY target support
21//! - Pattern rule support
22//! - Conditional block support
23

26/// Generate a purified Makefile from an AST
27///
28/// This function emits a complete Makefile from a parsed and purified AST.
29///
30/// # Examples
31///
32/// ```ignore
33/// use bashrs::make_parser::{MakeAst, MakeItem, MakeMetadata, VarFlavor, Span, generate_purified_makefile};
34///
35/// let ast = MakeAst {
36///     items: vec![
37///         MakeItem::Variable {
38///             name: "CC".to_string(),
39///             value: "gcc".to_string(),
40///             flavor: VarFlavor::Simple,
41///             span: Span::dummy(),
42///         }
43///     ],
44///     metadata: MakeMetadata::new(),
45/// };
46///
47/// let output = generate_purified_makefile(&ast);
48/// assert_eq!(output.trim(), "CC := gcc");
49/// ```
50pub fn generate_purified_makefile(ast: &MakeAst) -> String {
51    let mut output = String::new();
52
53    for item in &ast.items {
54        let item_output = generate_item(item);
55        output.push_str(&item_output);
56        output.push('\n');
57    }
58
59    output
60}

62/// Generate text for a single MakeItem
63fn generate_item(item: &MakeItem) -> String {
64    match item {
65        MakeItem::Variable {
66            name,
67            value,
68            flavor,
69            ..
70        } => generate_variable(name, value, flavor),
71
72        MakeItem::Target {
73            name,
74            prerequisites,
75            recipe,
76            phony,
77            ..
78        } => generate_target(name, prerequisites, recipe, *phony),
79
80        MakeItem::PatternRule {
81            target_pattern,
82            prereq_patterns,
83            recipe,
84            ..
85        } => generate_pattern_rule(target_pattern, prereq_patterns, recipe),
86
87        MakeItem::Conditional {
88            condition,
89            then_items,
90            else_items,
91            ..
92        } => generate_conditional(condition, then_items, else_items.as_deref()),
93
94        MakeItem::Include { path, optional, .. } => generate_include(path, *optional),
95
96        MakeItem::Comment { text, .. } => generate_comment(text),
97
98        MakeItem::FunctionCall { name, args, .. } => {
99            // Function calls are typically embedded in variables, not standalone
100            // But if we encounter one, generate it
101            format!("$({} {})", name, args.join(","))
102        }
103    }
104}

106/// Generate a variable assignment
107///
108/// # Examples
109///
110/// ```ignore
111/// # use bashrs::make_parser::ast::VarFlavor;
112/// # use bashrs::make_parser::generators::generate_variable;
113/// assert_eq!(generate_variable("CC", "gcc", &VarFlavor::Simple), "CC := gcc");
114/// assert_eq!(generate_variable("VAR", "val", &VarFlavor::Recursive), "VAR = val");
115/// ```
116fn generate_variable(name: &str, value: &str, flavor: &VarFlavor) -> String {
117    format!("{} {} {}", name, flavor, value)
118}

120/// Generate a target with prerequisites and recipe
121///
122/// # Arguments
123///
124/// * `name` - Target name
125/// * `prerequisites` - List of prerequisites
126/// * `recipe` - List of recipe lines (will be tab-indented)
127/// * `phony` - Whether this target should be marked as .PHONY
128///
129/// # Examples
130///
131/// ```ignore
132/// # use bashrs::make_parser::generators::generate_target;
133/// let output = generate_target("build", &vec!["main.c".to_string()], &vec!["gcc -o build main.c".to_string()], false);
134/// assert!(output.contains("build: main.c"));
135/// assert!(output.contains("\tgcc -o build main.c"));
136/// ```
137fn generate_target(name: &str, prerequisites: &[String], recipe: &[String], phony: bool) -> String {
138    let mut output = String::new();
139
140    // Add .PHONY declaration if needed
141    if phony {
142        output.push_str(&format!(".PHONY: {}\n", name));
143    }
144
145    // Generate target line
146    output.push_str(name);
147    output.push(':');
148
149    if !prerequisites.is_empty() {
150        output.push(' ');
151        output.push_str(&prerequisites.join(" "));
152    }
153
154    output.push('\n');
155
156    // Generate recipe lines (MUST use tabs)
157    for line in recipe {
158        output.push('\t');
159        output.push_str(line);
160        output.push('\n');
161    }
162
163    // Remove trailing newline (will be added by generate_item)
164    output.pop();
165
166    output
167}

169/// Generate a pattern rule
170fn generate_pattern_rule(
171    target_pattern: &str,
172    prereq_patterns: &[String],
173    recipe: &[String],

175    generate_target(target_pattern, prereq_patterns, recipe, false)

178/// Generate a conditional block
179fn generate_conditional(
180    condition: &MakeCondition,
181    then_items: &[MakeItem],
182    else_items: Option<&[MakeItem]>,
183) -> String {
184    let mut output = String::new();
185
186    // Generate condition
187    match condition {
188        MakeCondition::IfEq(left, right) => {
189            output.push_str(&format!("ifeq ({},{})\n", left, right));
190        }
191        MakeCondition::IfNeq(left, right) => {
192            output.push_str(&format!("ifneq ({},{})\n", left, right));
193        }
194        MakeCondition::IfDef(var) => {
195            output.push_str(&format!("ifdef {}\n", var));
196        }
197        MakeCondition::IfNdef(var) => {
198            output.push_str(&format!("ifndef {}\n", var));
199        }
200    }
201
202    // Generate then branch
203    for item in then_items {
204        output.push_str(&generate_item(item));
205        output.push('\n');
206    }
207
208    // Generate else branch if present
209    if let Some(else_items) = else_items {
210        output.push_str("else\n");
211        for item in else_items {
212            output.push_str(&generate_item(item));
213            output.push('\n');
214        }
215    }
216
217    // Close conditional
218    output.push_str("endif");
219
220    output
221}

223/// Generate an include directive
224fn generate_include(path: &str, optional: bool) -> String {
225    if optional {
226        format!("-include {}", path)
227    } else {
228        format!("include {}", path)
229    }
230}
