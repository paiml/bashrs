1//! Code generators for Makefile AST
2//!
3//! Generates purified Makefiles from AST.
4//!
5//! This module implements the code generation phase of the purification pipeline:
6//! Parse → AST → Analyze → Purify → **Generate** → Purified Makefile
7//!
8//! ## Design Principles
9//!
10//! 1. **Correctness**: Generated Makefiles must be syntactically valid
11//! 2. **Determinism**: Same AST always produces same output
12//! 3. **POSIX Compliance**: Generated output passes shellcheck
13//! 4. **Readability**: Output is human-readable and maintainable
14//!
15//! ## Key Features
16//!
17//! - Variable generation with all 5 flavors (`:=`, `=`, `?=`, `+=`, `!=`)
18//! - Target generation with tab-indented recipes (REQUIRED by Make)
19//! - Comment preservation
20//! - .PHONY target support
21//! - Pattern rule support
22//! - Conditional block support
23

26/// Options for controlling Makefile generation
27///
28/// These options allow users to customize the output format while maintaining
29/// correctness and determinism.
30///
31/// ## Added in v6.34.0 (Dogfooding Follow-up)
32///
33/// See: docs/dogfooding/makefile-purification.md
34#[derive(Debug, Clone, Default)]
35pub struct MakefileGeneratorOptions {
36    /// Preserve formatting (keep blank lines, multi-line format)
37    pub preserve_formatting: bool,
38
39    /// Maximum line length (None = unlimited)
40    pub max_line_length: Option<usize>,
41
42    /// Skip blank line removal transformation
43    pub skip_blank_line_removal: bool,
44
45    /// Skip multi-line consolidation transformation
46    pub skip_consolidation: bool,
47}

49/// Generate a purified Makefile from an AST
50///
51/// This function emits a complete Makefile from a parsed and purified AST.
52///
53/// # Examples
54///
55/// ```ignore
56/// use bashrs::make_parser::{MakeAst, MakeItem, MakeMetadata, VarFlavor, Span, generate_purified_makefile};
57///
58/// let ast = MakeAst {
59///     items: vec![
60///         MakeItem::Variable {
61///             name: "CC".to_string(),
62///             value: "gcc".to_string(),
63///             flavor: VarFlavor::Simple,
64///             span: Span::dummy(),
65///         }
66///     ],
67///     metadata: MakeMetadata::new(),
68/// };
69///
70/// let output = generate_purified_makefile(&ast);
71/// assert_eq!(output.trim(), "CC := gcc");
72/// ```
73pub fn generate_purified_makefile(ast: &MakeAst) -> String {
74    generate_purified_makefile_with_options(ast, &MakefileGeneratorOptions::default())
75}

77/// Generate a purified Makefile from an AST with custom formatting options
78///
79/// This function provides fine-grained control over the output format while
80/// maintaining correctness and determinism.
81///
82/// ## Options
83///
84/// - `preserve_formatting`: Keep blank lines and multi-line format (combines skip flags)
85/// - `max_line_length`: Break lines longer than this (None = unlimited)
86/// - `skip_blank_line_removal`: Keep blank lines between sections
87/// - `skip_consolidation`: Keep multi-line format (no single-line if/then/else)
88///
89/// ## Added in v6.34.0 (Dogfooding Follow-up)
90///
91/// See: docs/dogfooding/makefile-purification.md
92///
93/// # Examples
94///
95/// ```ignore
96/// use bashrs::make_parser::{MakeAst, MakefileGeneratorOptions, generate_purified_makefile_with_options};
97///
98/// let ast = parse_makefile("...")?;
99/// let options = MakefileGeneratorOptions {
100///     preserve_formatting: true,
101///     max_line_length: Some(120),
102///     ..Default::default()
103/// };
104///
105/// let output = generate_purified_makefile_with_options(&ast, &options);
106/// ```
107pub fn generate_purified_makefile_with_options(
108    ast: &MakeAst,
109    options: &MakefileGeneratorOptions,
110) -> String {
111    let mut output = String::new();
112    let mut prev_was_comment = false;
113
114    for (idx, item) in ast.items.iter().enumerate() {
115        let item_output = generate_item(item, options);
116
117        // Handle blank line preservation
118        let should_add_blank_line =
119            should_preserve_blank_line(item, idx > 0, prev_was_comment, options);
120
121        if should_add_blank_line && idx > 0 {
122            output.push('\n'); // Add blank line before item
123        }
124
125        // Apply line length limits
126        let formatted_output = if let Some(max_len) = options.max_line_length {
127            apply_line_length_limit(&item_output, max_len)
128        } else {
129            item_output
130        };
131
132        output.push_str(&formatted_output);
133        output.push('\n');
134
135        prev_was_comment = matches!(item, MakeItem::Comment { .. });
136    }
137
138    output
139}

141/// Determine if a blank line should be preserved before this item
142fn should_preserve_blank_line(
143    item: &MakeItem,
144    has_prev: bool,
145    prev_was_comment: bool,
146    options: &MakefileGeneratorOptions,
147) -> bool {
148    if !has_prev {
149        return false;
150    }
151
152    // If preserve_formatting is on, preserve blank lines before major sections
153    if options.preserve_formatting || options.skip_blank_line_removal {
154        match item {
155            MakeItem::Comment { .. } if !prev_was_comment => true,
156            MakeItem::Target { .. } => true,
157            _ => false,
158        }
159    } else {
160        false
161    }
162}

164/// Apply line length limit to output
165///
166/// Breaks long lines at reasonable boundaries (spaces, semicolons).
167fn apply_line_length_limit(text: &str, max_length: usize) -> String {
168    let mut result = String::new();
169
170    for line in text.lines() {
171        if line.len() <= max_length {
172            result.push_str(line);
173            result.push('\n');
174        } else {
175            // Break long line at spaces or semicolons
176            let mut current_line = String::new();
177
178            // Preserve leading tabs for recipe lines
179            let leading_tabs = line.chars().take_while(|c| *c == '\t').count();
180            let indent = "\t".repeat(leading_tabs);
181            current_line.push_str(&indent);
182            let mut current_len = indent.len();
183
184            let content = &line[leading_tabs..];
185
186            for word in content.split_whitespace() {
187                let word_len = word.len() + 1; // +1 for space
188
189                if current_len + word_len > max_length && current_len > indent.len() {
190                    // Line would be too long, break here
191                    result.push_str(&current_line);
192                    if !current_line.ends_with('\\') {
193                        result.push_str(" \\");
194                    }
195                    result.push('\n');
196
197                    // Start new line with indent
198                    current_line.clear();
199                    current_line.push_str(&indent);
200                    current_line.push(' '); // Continuation indent
201                    current_len = indent.len() + 1;
202                }
203
204                if !current_line.ends_with(&indent) && !current_line.ends_with(' ') {
205                    current_line.push(' ');
206                    current_len += 1;
207                }
208
209                current_line.push_str(word);
210                current_len += word.len();
211            }
212
213            if !current_line.trim().is_empty() {
214                result.push_str(&current_line);
215                result.push('\n');
216            }
217        }
218    }
219
220    result.trim_end_matches('\n').to_string()
221}

223/// Generate text for a single MakeItem
224fn generate_item(item: &MakeItem, options: &MakefileGeneratorOptions) -> String {
225    match item {
226        MakeItem::Variable {
227            name,
228            value,
229            flavor,
230            ..
231        } => generate_variable(name, value, flavor),
232
233        MakeItem::Target {
234            name,
235            prerequisites,
236            recipe,
237            phony,
238            recipe_metadata,
239            ..
240        } => generate_target(
241            name,
242            prerequisites,
243            recipe,
244            *phony,
245            recipe_metadata.as_ref(),
246            options,
247        ),
248
249        MakeItem::PatternRule {
250            target_pattern,
251            prereq_patterns,
252            recipe,
253            recipe_metadata,
254            ..
255        } => generate_pattern_rule(
256            target_pattern,
257            prereq_patterns,
258            recipe,
259            recipe_metadata.as_ref(),
260            options,
261        ),
262
263        MakeItem::Conditional {
264            condition,
265            then_items,
266            else_items,
267            ..
268        } => generate_conditional(condition, then_items, else_items.as_deref(), options),
269
270        MakeItem::Include { path, optional, .. } => generate_include(path, *optional),
271
272        MakeItem::Comment { text, .. } => generate_comment(text),
273
274        MakeItem::FunctionCall { name, args, .. } => {
275            // Function calls are typically embedded in variables, not standalone
276            // But if we encounter one, generate it
277            format!("$({} {})", name, args.join(","))
278        }
279    }
280}

296/// Generate a target with prerequisites and recipe
297///
298/// # Arguments
299///
300/// * `name` - Target name
301/// * `prerequisites` - List of prerequisites
302/// * `recipe` - List of recipe lines (will be tab-indented)
303/// * `phony` - Whether this target should be marked as .PHONY
304/// * `recipe_metadata` - Optional metadata about line continuations
305/// * `options` - Generator options (for preserve_formatting, skip_consolidation)
306///
307/// # Examples
308///
309/// ```ignore
310/// # use bashrs::make_parser::generators::generate_target;
311/// let options = MakefileGeneratorOptions::default();
312/// let output = generate_target("build", &vec!["main.c".to_string()], &vec!["gcc -o build main.c".to_string()], false, None, &options);
313/// assert!(output.contains("build: main.c"));
314/// assert!(output.contains("\tgcc -o build main.c"));
315/// ```
316fn generate_target(
317    name: &str,
318    prerequisites: &[String],
319    recipe: &[String],
320    phony: bool,
321    recipe_metadata: Option<&RecipeMetadata>,
322    options: &MakefileGeneratorOptions,
323) -> String {
324    let mut output = String::new();
325
326    // Add .PHONY declaration if needed
327    if phony {
328        output.push_str(&format!(".PHONY: {}\n", name));
329    }
330
331    // Generate target line
332    output.push_str(name);
333    output.push(':');
334
335    if !prerequisites.is_empty() {
336        output.push(' ');
337        output.push_str(&prerequisites.join(" "));
338    }
339
340    output.push('\n');
341
342    // Generate recipe lines
343    // If preserve_formatting or skip_consolidation is set AND we have metadata,
344    // reconstruct the original line breaks with backslash continuations
345    if let Some(metadata) = recipe_metadata {
346        if (options.preserve_formatting || options.skip_consolidation) && !recipe.is_empty() {
347            // Reconstruct with line breaks
348            for line in recipe {
349                output.push_str(&reconstruct_recipe_line_with_breaks(line, metadata));
350            }
351        } else {
352            // Default: Generate recipe lines as single lines (MUST use tabs)
353            for line in recipe {
354                output.push('\t');
355                output.push_str(line);
356                output.push('\n');
357            }
358        }
359    } else {
360        // No metadata: Generate recipe lines as single lines (MUST use tabs)
361        for line in recipe {
362            output.push('\t');
363            output.push_str(line);
364            output.push('\n');
365        }
366    }
367
368    // Remove trailing newline (will be added by generate_item)
369    output.pop();
370
371    output
372}

374/// Reconstruct a recipe line with backslash continuations based on metadata
375///
376/// Takes a consolidated single-line recipe and metadata about where line breaks
377/// occurred, and reconstructs the original multi-line format with backslashes.
378fn reconstruct_recipe_line_with_breaks(line: &str, metadata: &RecipeMetadata) -> String {
379    if metadata.line_breaks.is_empty() {
380        // No breaks, return as single line with tab
381        return format!("\t{}\n", line);
382    }
383
384    let mut output = String::new();
385    output.push('\t'); // Start with tab
386
387    let line_bytes = line.as_bytes();
388    let mut last_pos = 0;
389
390    for (break_pos, original_indent) in &metadata.line_breaks {
391        // Add text up to break position
392        if *break_pos <= line.len() {
393            // Add text up to break, trimming any trailing space
394            let text_segment = &line[last_pos..*break_pos];
395            let trimmed_segment = text_segment.trim_end();
396            output.push_str(trimmed_segment);
397            output.push_str(" \\");
398            output.push('\n');
399            output.push_str(original_indent);
400
401            // Move past the break position and skip the consolidation space
402            last_pos = *break_pos;
403            if last_pos < line_bytes.len() && line_bytes[last_pos] == b' ' {
404                last_pos += 1; // Skip the space we added during consolidation
405            }
406        }
407    }
408
409    // Add remaining text
410    if last_pos < line.len() {
411        output.push_str(&line[last_pos..]);
412    }
413    output.push('\n');
414
415    output
416}

418/// Generate a pattern rule
419fn generate_pattern_rule(
420    target_pattern: &str,
421    prereq_patterns: &[String],
422    recipe: &[String],
423    recipe_metadata: Option<&RecipeMetadata>,
424    options: &MakefileGeneratorOptions,
425) -> String {
426    generate_target(

429        recipe,

431        recipe_metadata,
432        options,

436/// Generate a conditional block
437fn generate_conditional(
438    condition: &MakeCondition,
439    then_items: &[MakeItem],
440    else_items: Option<&[MakeItem]>,
441    options: &MakefileGeneratorOptions,
442) -> String {
443    let mut output = String::new();
444
445    // Generate condition
446    match condition {
447        MakeCondition::IfEq(left, right) => {
448            output.push_str(&format!("ifeq ({},{})\n", left, right));
449        }
450        MakeCondition::IfNeq(left, right) => {
451            output.push_str(&format!("ifneq ({},{})\n", left, right));
452        }
453        MakeCondition::IfDef(var) => {
454            output.push_str(&format!("ifdef {}\n", var));
455        }
456        MakeCondition::IfNdef(var) => {
457            output.push_str(&format!("ifndef {}\n", var));
458        }
459    }
460
461    // Generate then branch
462    for item in then_items {
463        output.push_str(&generate_item(item, options));
464        output.push('\n');
465    }
466
467    // Generate else branch if present
468    if let Some(else_items) = else_items {
469        output.push_str("else\n");
470        for item in else_items {
471            output.push_str(&generate_item(item, options));
472            output.push('\n');
473        }
474    }
475
476    // Close conditional
477    output.push_str("endif");
478
479    output
480}
