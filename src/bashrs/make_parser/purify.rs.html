211/// Purify a Makefile AST by fixing non-deterministic patterns
212///
213/// # Example
214/// ```ignore
215/// let ast = parse_makefile("FILES := $(wildcard *.c)").unwrap();
216/// let result = purify_makefile(&ast);
217/// // result.ast contains: FILES := $(sort $(wildcard *.c))
218/// ```
219pub fn purify_makefile(ast: &MakeAst) -> PurificationResult {
220    // 1. Run semantic analysis to find issues
221    let issues = analyze_makefile(ast);
222
223    // 2. Plan transformations for each issue
224    let mut transformations = plan_transformations(ast, &issues);
225
226    // 3. Add parallel safety analysis (Sprint 83 - Days 2-3)
227    transformations.extend(analyze_parallel_safety(ast));
228
229    // 4. Add reproducible builds analysis (Sprint 83 - Day 4)
230    transformations.extend(analyze_reproducible_builds(ast));
231
232    // 5. Add performance optimization analysis (Sprint 83 - Day 5)
233    transformations.extend(analyze_performance_optimization(ast));
234
235    // 6. Add error handling analysis (Sprint 83 - Day 6)
236    transformations.extend(analyze_error_handling(ast));
237
238    // 7. Add portability analysis (Sprint 83 - Day 7)
239    transformations.extend(analyze_portability(ast));
240
241    // 8. Apply transformations
242    let purified_ast = apply_transformations(ast, &transformations);
243
244    // 8. Count results
245    let issues_fixed = transformations
246        .iter()
247        .filter(|t| is_safe_transformation(t))
248        .count();
249
250    let manual_fixes_needed = transformations
251        .iter()
252        .filter(|t| !is_safe_transformation(t))
253        .count();
254
255    // 9. Generate report
256    let report = generate_report(&transformations);
257
258    PurificationResult {
259        ast: purified_ast,
260        transformations_applied: transformations.len(),
261        issues_fixed,
262        manual_fixes_needed,
263        report,
264    }
265}

267/// Plan which transformations to apply for detected issues
268fn plan_transformations(_ast: &MakeAst, issues: &[SemanticIssue]) -> Vec<Transformation> {
269    let mut transformations = Vec::new();
270
271    for issue in issues {
272        // Extract variable name from issue message
273        let var_name = extract_variable_name(&issue.message);
274
275        match issue.rule.as_str() {
276            "NO_WILDCARD" => {
277                transformations.push(Transformation::WrapWithSort {
278                    variable_name: var_name,
279                    pattern: "$(wildcard".to_string(),
280                    safe: true,
281                });
282            }
283            "NO_UNORDERED_FIND" => {
284                transformations.push(Transformation::WrapWithSort {
285                    variable_name: var_name,
286                    pattern: "$(shell find".to_string(),
287                    safe: true,
288                });
289            }
290            "NO_TIMESTAMPS" => {
291                transformations.push(Transformation::AddComment {
292                    variable_name: var_name,
293                    rule: issue.rule.clone(),
294                    suggestion: issue.suggestion.clone().unwrap_or_default(),
295                    safe: false, // Manual fix required
296                });
297            }
298            "NO_RANDOM" => {
299                transformations.push(Transformation::AddComment {
300                    variable_name: var_name,
301                    rule: issue.rule.clone(),
302                    suggestion: issue.suggestion.clone().unwrap_or_default(),
303                    safe: false, // Manual fix required
304                });
305            }
306            _ => {}
307        }
308    }
309
310    transformations
311}

435/// Wrap pattern in specific variable with $(sort ...)
436fn wrap_variable_with_sort(ast: &mut MakeAst, variable_name: &str, pattern: &str) {
437    for item in &mut ast.items {
438        if let MakeItem::Variable { name, value, .. } = item {
439            if name == variable_name && value.contains(pattern) {
440                *value = wrap_pattern_with_sort(value, pattern);
441            }
442        }
443    }
444}

446/// Wrap specific pattern with $(sort ...)
447///
448/// Examples:
449/// - "$(wildcard *.c)" → "$(sort $(wildcard *.c))"
450/// - "$(filter %.o, $(wildcard *.c))" → "$(filter %.o, $(sort $(wildcard *.c)))"
451fn wrap_pattern_with_sort(value: &str, pattern: &str) -> String {
452    // Find pattern start
453    if let Some(start) = value.find(pattern) {
454        // Find matching closing paren
455        if let Some(end) = find_matching_paren(value, start) {
456            // Extract the complete pattern
457            let complete_pattern = &value[start..=end];
458
459            // Wrap with $(sort ...)
460            let wrapped = format!("$(sort {})", complete_pattern);
461
462            // Replace in original string
463            value.replace(complete_pattern, &wrapped)
464        } else {
465            // No matching paren found, return unchanged
466            value.to_string()
467        }
468    } else {
469        // Pattern not found, return unchanged
470        value.to_string()
471    }
472}

474/// Find matching closing parenthesis for pattern starting at position
475///
476/// Handles nested parentheses correctly.
477///
478/// # Arguments
479/// * `s` - The string to search
480/// * `start` - Starting position (should point to the pattern start, e.g., "$(" )
481///
482/// # Returns
483/// Position of the matching closing parenthesis
484fn find_matching_paren(s: &str, start: usize) -> Option<usize> {
485    let bytes = s.as_bytes();
486
487    // Find the opening '(' for the pattern at start
488    let mut depth = 0;
489    let mut found_opening = false;
490    let mut i = start;
491
492    // Skip to the opening paren
493    while i < bytes.len() {
494        if bytes.get(i) == Some(&b'(') && i > 0 && bytes.get(i - 1) == Some(&b'$') {
495            depth = 1;
496            found_opening = true;
497            i += 1;
498            break;
499        }
500        i += 1;
501    }
502
503    if !found_opening {
504        return None;
505    }
506
507    // Now find the matching closing paren
508    while i < bytes.len() {
509        match bytes.get(i) {
510            Some(&b'$') if bytes.get(i + 1) == Some(&b'(') => {
511                depth += 1;
512                i += 2; // Skip past $(
513                continue;
514            }
515            Some(&b')') => {
516                depth -= 1;
517                if depth == 0 {
518                    return Some(i);
519                }
520            }
521            _ => {}
522        }
523        i += 1;
524    }
525
526    None
527}

560/// Detect output file conflicts (multiple targets writing to same file)
561fn detect_output_file_conflicts(targets: &[(&String, &Vec<String>)]) -> Vec<Transformation> {
562    let mut transformations = Vec::new();
563    let mut output_files: std::collections::HashMap<String, Vec<String>> =
564        std::collections::HashMap::new();
565
566    for (target_name, recipes) in targets {
567        for recipe in *recipes {
568            // Detect output redirects: > filename or >> filename
569            if let Some(pos) = recipe.find(" > ") {
570                let after = &recipe[pos + 3..];
571                let filename = after.split_whitespace().next().unwrap_or("");
572                if !filename.is_empty() {
573                    output_files
574                        .entry(filename.to_string())
575                        .or_default()
576                        .push((*target_name).clone());
577                }
578            }
579            // Detect compiler output: -o filename
580            if let Some(pos) = recipe.find(" -o ") {
581                let after = &recipe[pos + 4..];
582                let filename = after.split_whitespace().next().unwrap_or("");
583                if !filename.is_empty() && filename != "$@" {
584                    output_files
585                        .entry(filename.to_string())
586                        .or_default()
587                        .push((*target_name).clone());
588                }
589            }
590        }
591    }
592
593    // Report conflicts
594    for (file, target_names) in output_files {
595        if target_names.len() > 1 {
596            transformations.push(Transformation::DetectRaceCondition {
597                target_names: target_names.clone(),
598                conflicting_file: file.clone(),
599                safe: false,
600            });
601
602            transformations.push(Transformation::DetectOutputConflict {
603                target_names,
604                output_file: file,
605                safe: false,
606            });
607        }
608    }
609
610    transformations
611}

616fn try_extract_output_redirect(recipe: &str) -> Option<String> {

619        let filename = after.split_whitespace().next()?;
620        if !filename.is_empty() {
621            return Some(filename.to_string());
622        }
623    }
624    None
625}

627/// Try to extract input filename from cat command: "cat filename"
628fn try_extract_cat_input(recipe: &str) -> Option<String> {
629    if recipe.contains("cat ") {
630        if let Some(pos) = recipe.find("cat ") {
631            let after = &recipe[pos + 4..];
632            let filename = after.split_whitespace().next()?;
633            if !filename.is_empty() && !is_automatic_variable(filename) {
634                return Some(filename.to_string());
635            }
636        }
637    }
638    None
639}

662/// Detect missing file dependencies (file usage without dependency)
663fn detect_missing_file_dependencies(
664    ast: &MakeAst,
665    targets: &[(&String, &Vec<String>)],
666) -> Vec<Transformation> {
667    let mut transformations = Vec::new();
668    let mut file_creators: std::collections::HashMap<String, String> =
669        std::collections::HashMap::new();
670    let mut file_users: Vec<(String, String)> = Vec::new();
671
672    for (target_name, recipes) in targets {
673        for recipe in *recipes {
674            // Detect file creation using helper
675            if let Some(filename) = try_extract_output_redirect(recipe) {
676                file_creators.insert(filename, (*target_name).clone());
677            }
678            // Detect file usage using helper
679            if let Some(filename) = try_extract_cat_input(recipe) {
680                file_users.push(((*target_name).clone(), filename));
681            }
682        }
683    }
684
685    // Report missing dependencies using helper
686    for (user_target, used_file) in file_users {
687        if let Some(provider_target) = file_creators.get(&used_file) {
688            if !target_has_prerequisite(ast, &user_target, provider_target) {
689                transformations.push(Transformation::DetectMissingDependency {
690                    target_name: user_target.clone(),
691                    missing_file: used_file,
692                    provider_target: provider_target.clone(),
693                    safe: false,
694                });
695            }
696        }
697    }
698
699    transformations
700}

702/// Detect recursive make calls
703fn detect_recursive_make_calls(targets: &[(&String, &Vec<String>)]) -> Vec<Transformation> {
704    let mut transformations = Vec::new();
705
706    for (target_name, recipes) in targets {
707        let mut subdirs = Vec::new();
708        for recipe in *recipes {
709            if recipe.contains("$(MAKE)") || recipe.contains("${MAKE}") {
710                if let Some(pos) = recipe.find("-C ") {
711                    let after = &recipe[pos + 3..];
712                    let subdir = after.split_whitespace().next().unwrap_or("");
713                    if !subdir.is_empty() {
714                        subdirs.push(subdir.to_string());
715                    }
716                }
717            }
718        }
719
720        if !subdirs.is_empty() {
721            transformations.push(Transformation::RecommendRecursiveMakeHandling {
722                target_name: (*target_name).clone(),
723                subdirs,
724                safe: false,
725            });
726        }
727    }
728
729    transformations
730}

732/// Detect shared directory creation races
733fn detect_directory_creation_races(targets: &[(&String, &Vec<String>)]) -> Vec<Transformation> {
734    let mut transformations = Vec::new();
735    let mut dir_creators: std::collections::HashMap<String, Vec<String>> =
736        std::collections::HashMap::new();
737
738    for (target_name, recipes) in targets {
739        for recipe in *recipes {
740            if recipe.contains("mkdir") {
741                if let Some(pos) = recipe.find("mkdir") {
742                    let after = &recipe[pos + 5..];
743                    let parts: Vec<&str> = after.split_whitespace().collect();
744                    for part in parts {
745                        if part != "-p" && !part.is_empty() {
746                            dir_creators
747                                .entry(part.to_string())
748                                .or_default()
749                                .push((*target_name).clone());
750                        }
751                    }
752                }
753            }
754        }
755    }
756
757    // Report directory creation races

894/// Analyze Makefile for performance optimization opportunities (Sprint 83 - Day 5)
895fn analyze_performance_optimization(ast: &MakeAst) -> Vec<Transformation> {
896    let mut transformations = Vec::new();
897
898    // Check if Makefile has .SUFFIXES directive
899    let has_suffixes = ast
900        .items
901        .iter()
902        .any(|item| matches!(item, MakeItem::Target { name, .. } if name == ".SUFFIXES"));
903
904    // Collect all targets and variables for analysis
905    let mut targets: Vec<(&String, &Vec<String>)> = Vec::new();
906    let mut variables: Vec<(&String, &String, &crate::make_parser::ast::VarFlavor)> = Vec::new();
907
908    for item in &ast.items {
909        if let MakeItem::Target { name, recipe, .. } = item {
910            targets.push((name, recipe));
911        } else if let MakeItem::Variable {
912            name,
913            value,
914            flavor,
915            ..
916        } = item
917        {
918            variables.push((name, value, flavor));
919        }
920    }
921
922    // Analysis 1: Detect variables using = with $(shell) that should use :=
923    for (var_name, value, flavor) in &variables {
924        if matches!(flavor, crate::make_parser::ast::VarFlavor::Recursive) {
925            // Check if value contains $(shell) - should use :=
926            if value.contains("$(shell") || value.contains("${shell") {
927                transformations.push(Transformation::SuggestSimpleExpansion {
928                    variable_name: (*var_name).clone(),
929                    reason: "Use := instead of = to avoid re-expanding $(shell) multiple times"
930                        .to_string(),
931                    safe: false,
932                });
933            }
934            // Check if value is simple (no variable references) - could use :=
935            else if !value.contains("$(") && !value.contains("${") {
936                transformations.push(Transformation::SuggestSimpleExpansion {
937                    variable_name: (*var_name).clone(),
938                    reason:
939                        "Use := instead of = for simple variables to avoid unnecessary re-expansion"
940                            .to_string(),
941                    safe: false,
942                });
943            }
944        }
945    }
946
947    // Analysis 3: Detect targets with multiple recipe lines that could be combined
948    for (target_name, recipes) in &targets {
949        if recipes.len() >= 3 {
950            // Check for sequential commands (not using && or ;)
951            let has_command_separator = recipes.iter().any(|r| r.contains("&&") || r.contains(";"));
952            if !has_command_separator {
953                transformations.push(Transformation::DetectSequentialRecipes {
954                    target_name: (*target_name).clone(),
955                    recipe_count: recipes.len(),
956                    safe: false,
957                });
958
959                transformations.push(Transformation::SuggestCombineShellInvocations {
960                    target_name: (*target_name).clone(),
961                    recipe_count: recipes.len(),
962                    safe: false,
963                });
964            }
965        }
966
967        // Analysis 4: Detect multiple rm commands that could be combined
968        let rm_count = recipes
969            .iter()
970            .filter(|r| r.trim().starts_with("rm "))
971            .count();
972        if rm_count >= 2 {
973            transformations.push(Transformation::DetectSequentialRecipes {
974                target_name: (*target_name).clone(),
975                recipe_count: rm_count,
976                safe: false,
977            });
978        }
979    }
980
981    // Analysis 5: Detect repeated explicit rules that could be pattern rules
982    // Group targets by their recipe pattern
983    let mut rule_patterns: std::collections::HashMap<String, Vec<String>> =
984        std::collections::HashMap::new();
985
986    for (target_name, recipes) in &targets {
987        // Look for .o: .c pattern
988        if target_name.ends_with(".o") && recipes.iter().any(|r| r.contains("-c")) {
989            let pattern = "%.o: %.c compilation".to_string();
990            rule_patterns
991                .entry(pattern.clone())
992                .or_default()
993                .push((*target_name).clone());
994        }
995    }
996
997    // Report if we found 3+ similar rules
998    for (_pattern, target_names) in rule_patterns {
999        if target_names.len() >= 3 {
1000            transformations.push(Transformation::SuggestPatternRule {
1001                pattern: "%.o: %.c".to_string(),
1002                target_count: target_names.len(),
1003                safe: false,
1004            });
1005        }
1006    }
1007
1008    // Final Analysis: Recommend .SUFFIXES: only if we found other performance issues
1009    // This ensures idempotency - we don't recommend .SUFFIXES: on already-clean Makefiles
1010    if !has_suffixes && !targets.is_empty() && !transformations.is_empty() {
1011        transformations.push(Transformation::RecommendSuffixes {
1012            reason: "Add .SUFFIXES: to disable builtin rules for better performance".to_string(),
1013            safe: false,
1014        });
1015    }
1016
1017    transformations
1018}

1453/// Generate human-readable report of transformations
1454fn generate_report(transformations: &[Transformation]) -> Vec<String> {
1455    transformations.iter().map(|t| match t {
1456        Transformation::WrapWithSort { variable_name, pattern, .. } => {
1457            format!("✅ Wrapped {} in variable '{}' with $(sort ...)", pattern, variable_name)
1458        }
1459        Transformation::AddComment { variable_name, rule, .. } => {
1460            format!("⚠️  Manual fix needed for variable '{}': {}", variable_name, rule)
1461        }
1462        Transformation::RecommendNotParallel { reason, .. } => {
1463            format!("⚠️  Parallel safety: {} (.NOTPARALLEL)", reason)
1464        }
1465        Transformation::DetectRaceCondition { target_names, conflicting_file, .. } => {
1466            format!("⚠️  Race condition detected: targets {:?} write to same file '{}'", target_names, conflicting_file)
1467        }
1468        Transformation::RecommendOrderOnlyPrereq { target_name, prereq_name, reason, .. } => {
1469            format!("⚠️  Recommend order-only prerequisite for '{}': add | {} ({})", target_name, prereq_name, reason)
1470        }
1471        Transformation::DetectMissingDependency { target_name, missing_file, provider_target, .. } => {
1472            format!("⚠️  Missing dependency: '{}' uses '{}' created by '{}', but '{}' is not in prerequisites", target_name, missing_file, provider_target, provider_target)
1473        }
1474        Transformation::DetectOutputConflict { output_file, .. } => {
1475            format!("⚠️  Output conflict: multiple targets write to same output file '{}'", output_file)
1476        }
1477        Transformation::RecommendRecursiveMakeHandling { target_name, subdirs, .. } => {
1478            format!("⚠️  Recursive make in '{}': consider dependencies for subdirs {:?} ($(MAKE))", target_name, subdirs)
1479        }
1480        Transformation::DetectDirectoryRace { target_names, directory, .. } => {
1481            format!("⚠️  Directory creation race: targets {:?} create directory '{}'", target_names, directory)
1482        }
1483        // Sprint 83 - Reproducible Builds (Day 4)
1484        Transformation::DetectTimestamp { variable_name, pattern, .. } => {
1485            format!("⚠️  Non-deterministic timestamp in '{}': {} - consider using SOURCE_DATE_EPOCH", variable_name, pattern)
1486        }
1487        Transformation::DetectRandom { variable_name, .. } => {
1488            format!("⚠️  Non-deterministic $RANDOM in '{}' - use fixed seed or version number", variable_name)
1489        }
1490        Transformation::DetectProcessId { variable_name, .. } => {
1491            format!("⚠️  Non-deterministic process ID ($$) in '{}' - use fixed temporary file name", variable_name)
1492        }
1493        Transformation::SuggestSourceDateEpoch { variable_name, .. } => {
1494            format!("💡 Suggestion: Use SOURCE_DATE_EPOCH for reproducible timestamps in '{}'", variable_name)
1495        }
1496        Transformation::DetectNonDeterministicCommand { variable_name, command, reason, .. } => {
1497            format!("⚠️  Non-deterministic command in '{}': {} - {}", variable_name, command, reason)
1498        }
1499        // Sprint 83 - Performance Optimization (Day 5)
1500        Transformation::SuggestCombineShellInvocations { target_name, recipe_count, .. } => {
1501            format!("💡 Performance: Combine {} shell invocations in '{}' using && or ; to reduce subshell spawns", recipe_count, target_name)
1502        }
1503        Transformation::SuggestSimpleExpansion { variable_name, reason, .. } => {
1504            format!("💡 Performance: {} for variable '{}'", reason, variable_name)
1505        }
1506        Transformation::RecommendSuffixes { reason, .. } => {
1507            format!("💡 Performance: {}", reason)
1508        }
1509        Transformation::DetectSequentialRecipes { target_name, recipe_count, .. } => {
1510            format!("⚠️  Performance: Target '{}' has {} sequential recipe lines - consider combining with && or ;", target_name, recipe_count)
1511        }
1512        Transformation::SuggestPatternRule { pattern, target_count, .. } => {
1513            format!("💡 Performance: {} explicit rules could use pattern rule '{}'", target_count, pattern)
1514        }
1515        // Sprint 83 - Error Handling (Day 6)
1516        Transformation::DetectMissingErrorHandling { target_name, command, .. } => {
1517            format!("⚠️  Error handling: Target '{}' has command without error handling: '{}' - consider adding || exit 1", target_name, command)
1518        }
1519        Transformation::DetectSilentFailure { target_name, command, .. } => {
1520            format!("⚠️  Error handling: Target '{}' has @ prefix that may hide errors: '{}' - consider removing @", target_name, command)
1521        }
1522        Transformation::RecommendDeleteOnError { reason, .. } => {
1523            format!("💡 Error handling: {}", reason)
1524        }

1526            format!("💡 Error handling: Target '{}' - {}", target_name, reason)

1531        Transformation::DetectLoopWithoutErrorHandling { target_name, loop_command, .. } => {
1532            format!("⚠️  Error handling: Target '{}' has loop without error handling: '{}' - add || exit 1 inside loop", target_name, loop_command)
1533        }
1534        // Sprint 83 - Portability (Day 7)
1535        Transformation::DetectBashism { target_name, construct, posix_alternative, .. } => {
1536            format!("⚠️  Portability: Target '{}' uses bashism '{}' - {}", target_name, construct, posix_alternative)
1537        }
1538        Transformation::DetectPlatformSpecific { target_name, command, reason, .. } => {
1539            format!("⚠️  Portability: Target '{}' uses platform-specific command '{}' - {}", target_name, command, reason)
1540        }
1541        Transformation::DetectShellSpecific { target_name, feature, posix_alternative, .. } => {
1542            format!("⚠️  Portability: Target '{}' uses shell-specific feature '{}' - {}", target_name, feature, posix_alternative)
1543        }

1545            format!("⚠️  Portability: Target '{}' uses non-portable flag '{}' in '{}' - {}", target_name, flag, command, reason)

1548            format!("⚠️  Portability: Target '{}' uses non-portable echo usage '{}' - use printf for portability", target_name, command)
1549        }
