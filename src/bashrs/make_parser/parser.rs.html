3//! Parses GNU Makefiles into AST representation.

15/// Preprocess Makefile input to handle line continuations
16///
17/// Line continuations in Makefiles use backslash (\) at the end of a line
18/// to concatenate with the next line. This function processes the input
19/// and joins continued lines with a single space.
20///
21/// # Arguments
22///
23/// * `input` - Raw Makefile source code
24///
25/// # Returns
26///
27/// * `String` - Preprocessed Makefile with continuations resolved
28///
29/// # Examples
30///
31/// ```ignore
32/// let input = "VAR = a \\\n    b";
33/// let output = preprocess_line_continuations(input);
34/// assert_eq!(output, "VAR = a b");
35/// ```
36fn preprocess_line_continuations(input: &str) -> String {
37    let mut result = String::new();
38    let lines: Vec<&str> = input.lines().collect();
39    let mut i = 0;
40
41    while i < lines.len() {
42        let mut line = lines[i].to_string();
43
44        // Check if this line ends with backslash (continuation)
45        while line.trim_end().ends_with('\\') && i + 1 < lines.len() {
46            // Remove the trailing backslash and any trailing whitespace
47            line = line.trim_end().strip_suffix('\\').unwrap().trim_end().to_string();
48
49            // Get the next line and trim leading whitespace
50            i += 1;
51            let next_line = lines[i].trim_start();
52
53            // Concatenate with a single space
54            line.push(' ');
55            line.push_str(next_line);
56        }
57
58        result.push_str(&line);
59        result.push('\n');
60        i += 1;
61    }
62
63    result
64}

66/// Parse a Makefile string into an AST
67///
68/// # Arguments
69///
70/// * `input` - Makefile source code as a string
71///
72/// # Returns
73///
74/// * `Ok(MakeAst)` - Successfully parsed AST
75/// * `Err(String)` - Parse error with description
76///
77/// # Examples
78///
79/// ```rust
80/// use bashrs::make_parser::parse_makefile;
81///
82/// let makefile = "test:\n\tcargo test";
83/// let ast = parse_makefile(makefile).unwrap();
84/// assert_eq!(ast.items.len(), 1);
85/// ```
86pub fn parse_makefile(input: &str) -> Result<MakeAst, String> {
87    // Preprocess: handle line continuations (backslash at end of line)
88    let preprocessed = preprocess_line_continuations(input);
89    let lines: Vec<&str> = preprocessed.lines().collect();
90    let line_count = lines.len();
91
92    let mut items = Vec::new();
93    let mut i = 0;
94
95    while i < lines.len() {
96        let line = lines[i];
97
98        // Skip empty lines
99        if line.trim().is_empty() {
100            i += 1;
101            continue;
102        }
103
104        // Parse comment lines
105        if line.trim_start().starts_with('#') {
106            let text = line.trim_start()
107                .strip_prefix('#')
108                .unwrap_or("")
109                .trim()
110                .to_string();
111
112            items.push(MakeItem::Comment {
113                text,
114                span: Span::new(0, line.len(), i + 1),
115            });
116
117            i += 1;
118            continue;
119        }
120
121        // Parse include directives
122        if line.trim_start().starts_with("include ") || line.trim_start().starts_with("-include ") || line.trim_start().starts_with("sinclude ") {
123            match parse_include(line, i + 1) {
124                Ok(include) => items.push(include),
125                Err(e) => return Err(e.to_detailed_string()),
126            }
127            i += 1;
128            continue;
129        }
130
131        // Parse conditional blocks (ifeq, ifdef, ifndef, ifneq)
132        if line.trim_start().starts_with("ifeq ") ||
133           line.trim_start().starts_with("ifdef ") ||
134           line.trim_start().starts_with("ifndef ") ||
135           line.trim_start().starts_with("ifneq ") {
136            match parse_conditional(&lines, &mut i) {
137                Ok(conditional) => items.push(conditional),
138                Err(e) => return Err(e.to_detailed_string()),
139            }
140            continue;
141        }
142
143        // Parse variable assignments (contains '=' but is not a target rule)
144        if is_variable_assignment(line) {
145            match parse_variable(line, i + 1) {
146                Ok(var) => items.push(var),
147                Err(e) => return Err(e.to_detailed_string()),
148            }
149            i += 1;
150            continue;
151        }
152
153        // Parse target rules (contains ':')
154        if line.contains(':') && !line.trim_start().starts_with('\t') {
155            match parse_target_rule(&lines, &mut i) {
156                Ok(target) => items.push(target),
157                Err(e) => return Err(e.to_detailed_string()),
158            }
159            continue;
160        }
161
162        // Unknown line - skip for now
163        i += 1;
164    }
165
166    // Second pass: Mark targets as .PHONY
167    // Collect all .PHONY declarations
168    let mut phony_targets: std::collections::HashSet<String> = std::collections::HashSet::new();
169    for item in &items {
170        if let MakeItem::Target { name, prerequisites, .. } = item {
171            if name == ".PHONY" {
172                for prereq in prerequisites {
173                    phony_targets.insert(prereq.clone());
174                }
175            }
176        }
177    }
178
179    // Update targets to mark them as phony if declared
180    items = items.into_iter().map(|item| {
181        if let MakeItem::Target { name, prerequisites, recipe, phony: _, span } = item {
182            MakeItem::Target {
183                phony: phony_targets.contains(&name),
184                name,
185                prerequisites,
186                recipe,
187                span,
188            }
189        } else {
190            item
191        }
192    }).collect();
193
194    Ok(MakeAst {
195        items,
196        metadata: MakeMetadata::with_line_count(line_count),
197    })
198}

200/// Check if a line is a variable assignment
201///
202/// Variable assignments contain '=' but are not target rules (which contain ':').
203/// We need to check for assignment operators (:=, ?=, +=, !=, =) to distinguish.
204///
205/// Examples:
206/// - "CC = gcc" -> true (variable)
207/// - "CC := gcc" -> true (variable, := is assignment operator)
208/// - "target: dep" -> false (target rule)
209/// - "target: VAR=value" -> false (target rule with variable in prerequisites)
210fn is_variable_assignment(line: &str) -> bool {
211    let trimmed = line.trim();
212
213    // Check for assignment operators (in order of specificity)
214    if trimmed.contains(":=") || trimmed.contains("?=") ||
215       trimmed.contains("+=") || trimmed.contains("!=") {
216        return true;
217    }
218
219    // Must contain '='
220    if !trimmed.contains('=') {
221        return false;
222    }
223
224    // If it contains ':', check if ':' comes before '='
225    // This distinguishes "CC = gcc" from "target: VAR=value"
226    if let Some(colon_pos) = trimmed.find(':') {
227        if let Some(equals_pos) = trimmed.find('=') {
228            // If ':' comes before '=', it's a target rule
229            if colon_pos < equals_pos {
230                return false;
231            }
232        }
233    }
234
235    true
236}

238/// Parse a variable assignment
239///
240/// Variable assignment syntax:
241/// ```makefile
242/// VAR = value      # Recursive (expanded when used)
243/// VAR := value     # Simple (expanded immediately)
244/// VAR ?= value     # Conditional (only if not defined)
245/// VAR += value     # Append
246/// VAR != command   # Shell (execute command)
247/// ```
248fn parse_variable(line: &str, line_num: usize) -> Result<MakeItem, MakeParseError> {
249    let trimmed = line.trim();
250
251    // Determine the flavor by finding the assignment operator
252    let (name_part, value_part, flavor) = if let Some(pos) = trimmed.find(":=") {
253        (&trimmed[..pos], &trimmed[pos + 2..], VarFlavor::Simple)
254    } else if let Some(pos) = trimmed.find("?=") {
255        (&trimmed[..pos], &trimmed[pos + 2..], VarFlavor::Conditional)
256    } else if let Some(pos) = trimmed.find("+=") {
257        (&trimmed[..pos], &trimmed[pos + 2..], VarFlavor::Append)
258    } else if let Some(pos) = trimmed.find("!=") {
259        (&trimmed[..pos], &trimmed[pos + 2..], VarFlavor::Shell)
260    } else if let Some(pos) = trimmed.find('=') {
261        (&trimmed[..pos], &trimmed[pos + 1..], VarFlavor::Recursive)
262    } else {
263        let location = SourceLocation::new(line_num)
264            .with_source_line(line.to_string());
265        return Err(MakeParseError::NoAssignmentOperator {
266            location,
267            found: trimmed.to_string(),
268        });
269    };
270
271    let name = name_part.trim();
272    if name.is_empty() {
273        let location = SourceLocation::new(line_num)
274            .with_source_line(line.to_string());
275        return Err(MakeParseError::EmptyVariableName { location });
276    }
277
278    let value = value_part.trim();
279
280    Ok(MakeItem::Variable {
281        name: name.to_string(),
282        value: value.to_string(),
283        flavor,
284        span: Span::new(0, line.len(), line_num),
285    })
286}

288/// Parse an include directive
289///
290/// Include directive syntax:
291/// ```makefile
292/// include common.mk
293/// -include optional.mk
294/// sinclude optional.mk
295/// ```
296///
297/// The `-include` and `sinclude` variants silently ignore missing files.
298fn parse_include(line: &str, line_num: usize) -> Result<MakeItem, MakeParseError> {
299    let trimmed = line.trim();
300
301    // Check if this is optional include (-include or sinclude)
302    let optional = trimmed.starts_with("-include ") || trimmed.starts_with("sinclude ");
303
304    // Extract the keyword and path
305    let path = if trimmed.starts_with("-include ") {
306        trimmed.strip_prefix("-include ")
307            .unwrap_or("")
308            .trim()
309            .to_string()
310    } else if trimmed.starts_with("sinclude ") {
311        trimmed.strip_prefix("sinclude ")
312            .unwrap_or("")
313            .trim()
314            .to_string()
315    } else if trimmed.starts_with("include ") {
316        trimmed.strip_prefix("include ")
317            .unwrap_or("")
318            .trim()
319            .to_string()
320    } else {
321        let location = SourceLocation::new(line_num)
322            .with_source_line(line.to_string());
323        return Err(MakeParseError::InvalidIncludeSyntax {
324            location,
325            found: trimmed.to_string(),
326        });
327    };
328
329    // Path can be empty (edge case handled in tests)
330    // Semantic validation should check for non-empty paths
331
332    Ok(MakeItem::Include {
333        path,
334        optional,
335        span: Span::new(0, line.len(), line_num),
336    })
337}

339/// Parse a conditional block starting at the given line index
340///
341/// Updates the index to point past the parsed conditional (past 'endif').
342///
343/// Conditional block syntax:
344/// ```makefile
345/// ifeq ($(VAR),value)
346///     CFLAGS = -g
347/// else
348///     CFLAGS = -O2
349/// endif
350/// ```
351///
352/// Supported directives: ifeq, ifneq, ifdef, ifndef
353fn parse_conditional(lines: &[&str], index: &mut usize) -> Result<MakeItem, MakeParseError> {
354    let start_line = lines[*index];
355    let start_line_num = *index + 1;
356    let trimmed = start_line.trim();
357
358    // Parse the condition type and expression
359    let condition = if trimmed.starts_with("ifeq ") {
360        // ifeq (arg1,arg2)
361        let rest = trimmed.strip_prefix("ifeq ").unwrap().trim();
362        if !rest.starts_with('(') || !rest.ends_with(')') {
363            let location = SourceLocation::new(start_line_num)
364                .with_source_line(start_line.to_string());
365            return Err(MakeParseError::InvalidConditionalSyntax {
366                location,
367                directive: "ifeq".to_string(),
368                found: rest.to_string(),
369            });
370        }
371        let inner = &rest[1..rest.len()-1];
372        let parts: Vec<&str> = inner.splitn(2, ',').collect();
373        if parts.len() != 2 {
374            let location = SourceLocation::new(start_line_num)
375                .with_source_line(start_line.to_string());
376            return Err(MakeParseError::MissingConditionalArguments {
377                location,
378                directive: "ifeq".to_string(),
379                expected_args: 2,
380                found_args: parts.len(),
381            });
382        }
383        MakeCondition::IfEq(parts[0].to_string(), parts[1].to_string())
384    } else if trimmed.starts_with("ifneq ") {
385        // ifneq (arg1,arg2)
386        let rest = trimmed.strip_prefix("ifneq ").unwrap().trim();
387        if !rest.starts_with('(') || !rest.ends_with(')') {
388            let location = SourceLocation::new(start_line_num)
389                .with_source_line(start_line.to_string());
390            return Err(MakeParseError::InvalidConditionalSyntax {
391                location,
392                directive: "ifneq".to_string(),
393                found: rest.to_string(),
394            });
395        }
396        let inner = &rest[1..rest.len()-1];
397        let parts: Vec<&str> = inner.splitn(2, ',').collect();
398        if parts.len() != 2 {
399            let location = SourceLocation::new(start_line_num)
400                .with_source_line(start_line.to_string());
401            return Err(MakeParseError::MissingConditionalArguments {
402                location,
403                directive: "ifneq".to_string(),
404                expected_args: 2,
405                found_args: parts.len(),
406            });
407        }
408        MakeCondition::IfNeq(parts[0].to_string(), parts[1].to_string())
409    } else if trimmed.starts_with("ifdef ") {
410        // ifdef VAR
411        let var_name = trimmed.strip_prefix("ifdef ").unwrap().trim().to_string();
412        if var_name.is_empty() {
413            let location = SourceLocation::new(start_line_num)
414                .with_source_line(start_line.to_string());
415            return Err(MakeParseError::MissingVariableName {
416                location,
417                directive: "ifdef".to_string(),
418            });
419        }
420        MakeCondition::IfDef(var_name)
421    } else if trimmed.starts_with("ifndef ") {
422        // ifndef VAR
423        let var_name = trimmed.strip_prefix("ifndef ").unwrap().trim().to_string();
424        if var_name.is_empty() {
425            let location = SourceLocation::new(start_line_num)
426                .with_source_line(start_line.to_string());
427            return Err(MakeParseError::MissingVariableName {
428                location,
429                directive: "ifndef".to_string(),
430            });
431        }
432        MakeCondition::IfNdef(var_name)
433    } else {
434        let location = SourceLocation::new(start_line_num)
435            .with_source_line(start_line.to_string());
436        return Err(MakeParseError::UnknownConditional {
437            location,
438            found: trimmed.to_string(),
439        });
440    };
441
442    // Move past the ifeq/ifdef/ifndef/ifneq line
443    *index += 1;
444
445    // Parse items in the 'then' branch until we hit 'else' or 'endif'
446    let mut then_items = Vec::new();
447    let mut else_items = None;
448    let mut depth = 1; // Track nested conditionals
449
450    while *index < lines.len() {
451        let line = lines[*index];
452        let trimmed = line.trim();
453
454        // Check for nested conditionals
455        if trimmed.starts_with("ifeq ") || trimmed.starts_with("ifneq ") ||
456           trimmed.starts_with("ifdef ") || trimmed.starts_with("ifndef ") {
457            depth += 1;
458        }
459
460        // Check for endif
461        if trimmed == "endif" {
462            depth -= 1;
463            if depth == 0 {
464                // This endif closes our conditional
465                *index += 1;
466                break;
467            }
468        }
469
470        // Check for else at our level
471        if trimmed == "else" && depth == 1 {
472            *index += 1;
473            // Parse items in the 'else' branch
474            let mut else_vec = Vec::new();
475            while *index < lines.len() {
476                let else_line = lines[*index];
477                let else_trimmed = else_line.trim();
478
479                // Check for nested conditionals in else branch
480                if else_trimmed.starts_with("ifeq ") || else_trimmed.starts_with("ifneq ") ||
481                   else_trimmed.starts_with("ifdef ") || else_trimmed.starts_with("ifndef ") {
482                    depth += 1;
483                }
484
485                if else_trimmed == "endif" {
486                    depth -= 1;
487                    if depth == 0 {
488                        *index += 1;
489                        break;
490                    }
491                }
492
493                // Parse item in else branch
494                match parse_conditional_item(lines, index) {
495                    Ok(Some(item)) => {
496                        else_vec.push(item);
497                        // Note: index was already incremented by parse_conditional_item
498                    }
499                    Ok(None) => {
500                        // Empty line or unrecognized - skip it
501                        *index += 1;
502                    }
503                    Err(e) => {
504                        let location = SourceLocation::new(*index + 1);
505                        return Err(MakeParseError::InvalidTargetRule {
506                            location,
507                            found: e,
508                        });
509                    }
510                }
511            }
512            else_items = Some(else_vec);
513            break;
514        }
515
516        // Parse item in then branch
517        match parse_conditional_item(lines, index) {
518            Ok(Some(item)) => {
519                then_items.push(item);
520                // Note: index was already incremented by parse_conditional_item
521            }
522            Ok(None) => {
523                // Empty line or unrecognized - skip it
524                *index += 1;
525            }
526            Err(e) => {
527                let location = SourceLocation::new(*index + 1);
528                return Err(MakeParseError::InvalidTargetRule {
529                    location,
530                    found: e,
531                });
532            }
533        }
534    }
535
536    Ok(MakeItem::Conditional {
537        condition,
538        then_items,
539        else_items,
540        span: Span::new(0, start_line.len(), start_line_num),
541    })
542}

544/// Parse a single item within a conditional block
545///
546/// Returns None for empty lines or lines that should be skipped
547///
548/// IMPORTANT: This function does NOT increment index for simple items like variables or comments,
549/// but parse_target_rule DOES increment index (it advances past recipes).
550/// The caller must increment index when Ok(None) is returned.
551fn parse_conditional_item(lines: &[&str], index: &mut usize) -> Result<Option<MakeItem>, String> {
552    let line = lines[*index];
553    let line_num = *index + 1;
554
555    // Skip empty lines
556    if line.trim().is_empty() {
557        return Ok(None);
558    }
559
560    // Don't parse conditional keywords here (handled by parent)
561    let trimmed = line.trim();
562    if trimmed == "else" || trimmed == "endif" ||
563       trimmed.starts_with("ifeq ") || trimmed.starts_with("ifneq ") ||
564       trimmed.starts_with("ifdef ") || trimmed.starts_with("ifndef ") {
565        return Ok(None);
566    }
567
568    // Parse variable assignment
569    if is_variable_assignment(line) {
570        let var = parse_variable(line, line_num).map_err(|e| e.to_string())?;
571        *index += 1; // Move past this variable line
572        return Ok(Some(var));
573    }
574
575    // Parse target rule (THIS INCREMENTS INDEX - it advances past recipes)
576    if line.contains(':') && !line.trim_start().starts_with('\t') {
577        let target = parse_target_rule(lines, index).map_err(|e| e.to_string())?;
578        // parse_target_rule already incremented index past the target and its recipe
579        // So we DON'T increment it again
580        return Ok(Some(target));
581    }
582
583    // Parse comment
584    if line.trim_start().starts_with('#') {
585        let text = line.trim_start()
586            .strip_prefix('#')
587            .unwrap_or("")
588            .trim()
589            .to_string();
590        *index += 1; // Move past this comment line
591        return Ok(Some(MakeItem::Comment {
592            text,
593            span: Span::new(0, line.len(), line_num),
594        }));
595    }
596
597    // Unknown item - skip
598    Ok(None)
599}

601/// Parse a target rule starting at the given line index
602///
603/// Updates the index to point past the parsed rule.
604///
605/// Target rule syntax:
606/// ```makefile
607/// target: prerequisites
608///     recipe line 1
609///     recipe line 2
610/// ```
611///
612/// Pattern rule syntax:
613/// ```makefile
614/// %.o: %.c
615///     $(CC) -c $< -o $@
616/// ```
617fn parse_target_rule(lines: &[&str], index: &mut usize) -> Result<MakeItem, MakeParseError> {
618    let line = lines[*index];
619    let line_num = *index + 1;
620
621    // Split on ':' to get target and prerequisites
622    let parts: Vec<&str> = line.splitn(2, ':').collect();
623    if parts.len() != 2 {
624        let location = SourceLocation::new(line_num)
625            .with_source_line(line.to_string());
626        return Err(MakeParseError::InvalidTargetRule {
627            location,
628            found: line.trim().to_string(),
629        });
630    }
631
632    let name = parts[0].trim().to_string();
633    if name.is_empty() {
634        let location = SourceLocation::new(line_num)
635            .with_source_line(line.to_string());
636        return Err(MakeParseError::EmptyTargetName { location });
637    }
638
639    // Parse prerequisites (space-separated)
640    let prerequisites: Vec<String> = parts[1]
641        .split_whitespace()
642        .map(|s| s.to_string())
643        .collect();
644
645    // Parse recipe lines (tab-indented lines following the target)
646    *index += 1;
647    let mut recipe = Vec::new();
648
649    while *index < lines.len() {
650        let recipe_line = lines[*index];
651
652        // Recipe lines must start with a tab
653        if recipe_line.starts_with('\t') {
654            recipe.push(recipe_line.trim().to_string());
655            *index += 1;
656        } else if recipe_line.trim().is_empty() {
657            // Empty line - could be end of recipe or just whitespace
658            *index += 1;
659            // Check if next line is also recipe
660            if *index < lines.len() && lines[*index].starts_with('\t') {
661                continue;
662            } else {
663                break;
664            }
665        } else {
666            // Non-tab-indented, non-empty line - end of recipe
667            break;
668        }
669    }
670
671    // Check if this is a pattern rule (target contains %)
672    if name.contains('%') {
673        Ok(MakeItem::PatternRule {
674            target_pattern: name,
675            prereq_patterns: prerequisites,
676            recipe,
677            span: Span::new(0, line.len(), line_num),
678        })
679    } else {
680        Ok(MakeItem::Target {
681            name,
682            prerequisites,
683            recipe,
684            phony: false, // Will be detected in semantic analysis
685            span: Span::new(0, line.len(), line_num),
686        })
687    }
688}
