1//! Dockerfile Quality Scoring (Issue #10)
2//!
3//! Dockerfile-specific quality metrics instead of general bash script metrics.
4//!
5//! ## Scoring Dimensions (Weighted)
6//!
7//! 1. **Safety (30%)**: set -euo pipefail usage, error handling
8//! 2. **Complexity (25%)**: RUN command simplicity, script length
9//! 3. **Layer Optimization (20%)**: combined commands, cache cleanup in same layer
10//! 4. **Determinism (15%)**: version pinning, specific tags (not :latest)
11//! 5. **Security (10%)**: non-root user, no credential exposure
12//!
13//! ## Grade Scale
14//!
15//! - A+ (9.5-10): Excellent
16//! - A  (9.0-9.5): Very Good
17//! - B+ (8.5-9.0): Good
18//! - B  (8.0-8.5): Above Average
19//! - C+ (7.5-8.0): Average
20//! - C  (7.0-7.5): Below Average
21//! - D  (6.0-7.0): Poor
22//! - F  (<6.0): Failing
23

26/// Dockerfile quality score
27#[derive(Debug, Clone, Serialize, Deserialize)]
28pub struct DockerfileQualityScore {
29    /// Overall grade (A+ to F)
30    pub grade: String,
31
32    /// Numeric score (0.0 - 10.0)
33    pub score: f64,
34
35    /// Safety score (0.0 - 10.0)
36    pub safety: f64,
37
38    /// Complexity score (0.0 - 10.0)
39    pub complexity: f64,
40
41    /// Layer optimization score (0.0 - 10.0)
42    pub layer_optimization: f64,
43
44    /// Determinism score (0.0 - 10.0)
45    pub determinism: f64,
46
47    /// Security score (0.0 - 10.0)
48    pub security: f64,
49
50    /// Improvement suggestions
51    pub suggestions: Vec<String>,
52}

76/// Score a Dockerfile for quality
77///
78/// Returns Dockerfile-specific quality score with grade, numeric score, and suggestions.
79pub fn score_dockerfile(source: &str) -> Result<DockerfileQualityScore, String> {
80    let mut score = DockerfileQualityScore::new();
81
82    // Calculate each dimension
83    score.safety = calculate_safety_score(source);
84    score.complexity = calculate_complexity_score(source);
85    score.layer_optimization = calculate_layer_optimization_score(source);
86    score.determinism = calculate_determinism_score(source);
87    score.security = calculate_security_score(source);
88
89    // Calculate overall score (weighted average per Issue #10 spec)
90    score.score = (score.safety * 0.30)
91        + (score.complexity * 0.25)
92        + (score.layer_optimization * 0.20)
93        + (score.determinism * 0.15)
94        + (score.security * 0.10);
95
96    // Assign grade
97    score.grade = calculate_grade(score.score);
98
99    // Generate suggestions
100    score.suggestions = generate_suggestions(source, &score);
101
102    Ok(score)
103}

105/// Calculate safety score (30% weight)
106///
107/// Checks for:
108/// - `set -euo pipefail` in RUN commands
109/// - Error handling (|| return, || exit)
110/// - Proper error propagation
111fn calculate_safety_score(source: &str) -> f64 {
112    if source.trim().is_empty() {
113        return 0.0;
114    }
115
116    let mut run_commands = 0;
117    let mut safe_run_commands = 0;
118    let mut has_error_handling = false;
119
120    for line in source.lines() {
121        let trimmed = line.trim();
122
123        // Track RUN commands
124        if trimmed.starts_with("RUN ") {
125            run_commands += 1;
126
127            // Check for set -euo pipefail
128            if trimmed.contains("set -euo pipefail")
129                || trimmed.contains("set -e") && trimmed.contains("set -o pipefail")
130            {
131                safe_run_commands += 1;
132                has_error_handling = true;
133            }
134
135            // Check for error handling
136            if trimmed.contains("|| exit") || trimmed.contains("|| return") {
137                has_error_handling = true;
138            }
139        }
140    }
141
142    if run_commands == 0 {
143        return 5.0; // Neutral score if no RUN commands
144    }
145
146    let safety_ratio = safe_run_commands as f64 / run_commands as f64;
147
148    let mut score: f64 = match safety_ratio {
149        r if r >= 0.8 => 10.0, // 80%+ safe commands
150        r if r >= 0.6 => 8.0,  // 60-79% safe
151        r if r >= 0.4 => 6.0,  // 40-59% safe
152        r if r >= 0.2 => 4.0,  // 20-39% safe
153        _ => 2.0,              // <20% safe
154    };
155
156    // Bonus for having any error handling
157    if has_error_handling {
158        score += 1.0;
159    }
160
161    score.min(10.0)
162}

164/// Calculate complexity score (25% weight)
165///
166/// Checks for:
167/// - Number of RUN commands (fewer is better - encourages layer optimization)
168/// - Length of RUN commands
169/// - Multi-line command complexity
170fn calculate_complexity_score(source: &str) -> f64 {
171    let lines: Vec<&str> = source.lines().collect();
172
173    if lines.is_empty() {
174        return 0.0;
175    }
176
177    let mut run_count = 0;
178    let mut longest_run = 0;
179    let mut current_run_lines = 0;
180    let mut in_run = false;
181
182    for line in &lines {
183        let trimmed = line.trim();
184
185        if trimmed.starts_with("RUN ") {
186            run_count += 1;
187            in_run = true;
188            current_run_lines = 1;
189        } else if in_run {
190            if trimmed.ends_with('\\') {
191                current_run_lines += 1;
192            } else {
193                if current_run_lines > 0 {
194                    current_run_lines += 1;
195                    if current_run_lines > longest_run {
196                        longest_run = current_run_lines;
197                    }
198                }
199                in_run = false;
200                current_run_lines = 0;
201            }
202        }
203    }
204
205    // Finalize if still in RUN
206    if in_run && current_run_lines > 0 && current_run_lines > longest_run {
207        longest_run = current_run_lines;
208    }
209
210    // Score based on RUN command count (fewer is better)
211    let run_score = match run_count {
212        0 => 5.0,       // Neutral
213        1..=3 => 10.0,  // Excellent (combined commands)
214        4..=6 => 8.0,   // Good
215        7..=10 => 6.0,  // Average
216        11..=15 => 4.0, // Below average
217        _ => 2.0,       // Poor (too many layers)
218    };
219
220    // Score based on longest RUN command
221    let length_score = match longest_run {
222        0 => 5.0,       // Neutral
223        1..=5 => 10.0,  // Simple
224        6..=10 => 8.0,  // Moderate
225        11..=20 => 6.0, // Complex
226        21..=30 => 4.0, // Very complex
227        _ => 2.0,       // Extremely complex
228    };
229
230    (run_score + length_score) / 2.0
231}

233/// Calculate layer optimization score (20% weight)
234///
235/// Checks for:
236/// - Combined commands with && instead of separate RUN statements
237/// - Cache cleanup in same layer (rm -rf /var/cache/apk/*, apt clean)
238/// - --no-cache flag for package managers
239/// - Multi-stage builds
240fn calculate_layer_optimization_score(source: &str) -> f64 {
241    if source.trim().is_empty() {
242        return 0.0;
243    }
244
245    let mut has_combined_commands = false;
246    let mut has_cache_cleanup = false;
247    let mut has_no_cache_flag = false;
248    let mut has_multistage = false;
249    let mut run_count = 0;
250
251    for line in source.lines() {
252        let trimmed = line.trim();
253
254        // Count RUN commands
255        if trimmed.starts_with("RUN ") {
256            run_count += 1;
257
258            // Check for combined commands
259            if trimmed.contains("&&") {
260                has_combined_commands = true;
261            }
262
263            // Check for cache cleanup in same layer
264            if trimmed.contains("rm -rf /var/cache/apk/*")
265                || trimmed.contains("apt-get clean")
266                || trimmed.contains("yum clean all")
267                || trimmed.contains("rm -rf /var/lib/apt/lists/*")
268            {
269                has_cache_cleanup = true;
270            }
271
272            // Check for --no-cache flag
273            if trimmed.contains("--no-cache") {
274                has_no_cache_flag = true;
275            }
276        }
277
278        // Check for multi-stage builds (FROM ... AS ...)
279        if trimmed.starts_with("FROM ") && trimmed.contains(" AS ") {
280            has_multistage = true;
281        }
282    }
283
284    let mut score: f64 = 0.0;
285
286    // Base score for having any optimizations
287    if run_count > 0 {
288        score += 2.0;
289    }
290
291    // Combined commands (4 points)
292    if has_combined_commands {
293        score += 4.0;
294    }
295
296    // Cache cleanup (3 points)
297    if has_cache_cleanup || has_no_cache_flag {
298        score += 3.0;
299    }
300
301    // Multi-stage builds (1 point bonus)
302    if has_multistage {
303        score += 1.0;
304    }
305
306    score.min(10.0)
307}

309/// Calculate determinism score (15% weight)
310///
311/// Checks for:
312/// - Version pinning in package installs (curl=8.2.1-r0)
313/// - Specific image tags (not :latest)
314/// - Pinned base images (alpine:3.18 not alpine:latest)
315fn calculate_determinism_score(source: &str) -> f64 {
316    if source.trim().is_empty() {
317        return 0.0;
318    }
319
320    let mut has_pinned_base_image = false;
321    let mut package_installs = 0;
322    let mut pinned_packages = 0;
323    let mut uses_latest_tag = false;
324
325    for line in source.lines() {
326        let trimmed = line.trim();
327
328        // Check FROM statement for tag specificity
329        if trimmed.starts_with("FROM ") {
330            if trimmed.contains(":latest") || (!trimmed.contains(':') && !trimmed.contains('@')) {
331                uses_latest_tag = true;
332            } else if trimmed.contains(':') {
333                has_pinned_base_image = true;
334            }
335        }
336
337        // Check RUN commands for package version pinning
338        if trimmed.starts_with("RUN ") {
339            // Count package install commands
340            if trimmed.contains("apk add")
341                || trimmed.contains("apt-get install")
342                || trimmed.contains("yum install")
343            {
344                package_installs += 1;
345
346                // Check for version pinning
347                // apk: curl=8.2.1-r0
348                // apt: curl=7.68.0-1
349                // yum: curl-7.68.0
350                if trimmed.contains('=') && (trimmed.contains("apk add") || trimmed.contains("apt"))
351                {
352                    pinned_packages += 1;
353                }
354            }
355        }
356    }
357
358    let mut score = 0.0;
359
360    // Base image pinning (5 points)
361    if has_pinned_base_image && !uses_latest_tag {
362        score += 5.0;
363    } else if has_pinned_base_image {
364        score += 3.0;
365    } else if !uses_latest_tag {
366        score += 1.0;
367    }
368
369    // Package version pinning (5 points)
370    if package_installs > 0 {
371        let pinning_ratio = pinned_packages as f64 / package_installs as f64;
372        score += pinning_ratio * 5.0;
373    } else {
374        score += 2.5; // Neutral if no packages
375    }
376
377    score.min(10.0)
378}

380/// Calculate security score (10% weight)
381///
382/// Checks for:
383/// - USER directive (non-root)
384/// - Avoiding ADD (use COPY instead)
385/// - No exposed secrets or credentials
386/// - Reasonable file permissions (not 777)
387fn calculate_security_score(source: &str) -> f64 {
388    if source.trim().is_empty() {
389        return 0.0;
390    }
391
392    let mut has_user_directive = false;
393    let mut uses_copy_not_add = true;
394    let mut has_bad_permissions = false;
395    let mut potential_secrets = false;
396
397    for line in source.lines() {
398        let trimmed = line.trim();
399
400        // Check for USER directive
401        if trimmed.starts_with("USER ") && !trimmed.contains("USER root") {
402            has_user_directive = true;
403        }
404
405        // Check for ADD (should prefer COPY)
406        if trimmed.starts_with("ADD ") && !trimmed.contains(".tar") {
407            uses_copy_not_add = false;
408        }
409
410        // Check for bad permissions (777, 666)
411        if trimmed.contains("chmod 777") || trimmed.contains("chmod 666") {
412            has_bad_permissions = true;
413        }
414
415        // Check for potential secrets
416        if trimmed.contains("PASSWORD")
417            || trimmed.contains("SECRET")
418            || trimmed.contains("API_KEY")
419            || trimmed.contains("TOKEN")
420        {
421            potential_secrets = true;
422        }
423    }
424
425    let mut score: f64 = 5.0; // Start at neutral
426
427    // USER directive (4 points)
428    if has_user_directive {
429        score += 4.0;
430    } else {
431        score -= 2.0; // Penalty for running as root
432    }
433
434    // Using COPY instead of ADD (1 point)
435    if uses_copy_not_add {
436        score += 1.0;
437    }
438
439    // No bad permissions (maintain score)
440    if has_bad_permissions {
441        score -= 2.0;
442    }
443
444    // No exposed secrets (maintain score)
445    if potential_secrets {
446        score -= 1.0;
447    }
448
449    score.clamp(0.0, 10.0)
450}

452/// Calculate grade from numeric score
453fn calculate_grade(score: f64) -> String {
454    match score {
455        s if s >= 9.5 => "A+".to_string(),
456        s if s >= 9.0 => "A".to_string(),
457        s if s >= 8.5 => "B+".to_string(),
458        s if s >= 8.0 => "B".to_string(),
459        s if s >= 7.5 => "C+".to_string(),
460        s if s >= 7.0 => "C".to_string(),
461        s if s >= 6.0 => "D".to_string(),
462        _ => "F".to_string(),
463    }
464}

466/// Generate improvement suggestions
467fn generate_suggestions(source: &str, score: &DockerfileQualityScore) -> Vec<String> {
468    let mut suggestions = Vec::new();
469
470    // Safety suggestions
471    if score.safety < 7.0 {
472        let mut has_pipefail = false;
473        for line in source.lines() {
474            if line.contains("set -euo pipefail") {
475                has_pipefail = true;
476                break;
477            }
478        }
479
480        if !has_pipefail {
481            suggestions.push(
482                "Add 'set -euo pipefail &&' at the beginning of RUN commands for better error handling".to_string()
483            );
484        }
485    }
486
487    // Layer optimization suggestions
488    if score.layer_optimization < 7.0 {
489        suggestions.push("Combine RUN commands with && to reduce image layers".to_string());
490        suggestions.push(
491            "Clean up package manager cache in the same layer (rm -rf /var/cache/apk/*)"
492                .to_string(),
493        );
494        suggestions.push("Consider using --no-cache flag for package managers".to_string());
495    }
496
497    // Determinism suggestions
498    if score.determinism < 7.0 {
499        let has_latest = source.contains(":latest");
500        let has_version_pinning = source.contains('=');
501
502        if has_latest || !has_version_pinning {
503            suggestions
504                .push("Pin package versions for reproducibility (e.g., curl=8.2.1-r0)".to_string());
505            suggestions
506                .push("Use specific image tags instead of :latest (e.g., alpine:3.18)".to_string());
507        }
508    }
509
510    // Security suggestions
511    if score.security < 7.0 {
512        let has_user = source.contains("USER ");
513
514        if !has_user || source.contains("USER root") {
515            suggestions.push("Add USER directive to run container as non-root user".to_string());
516            suggestions.push("Create a dedicated user with adduser/addgroup".to_string());
517        }
518    }
519
520    // Complexity suggestions
521    if score.complexity < 7.0 {
522        suggestions
523            .push("Reduce the number of separate RUN commands by combining them".to_string());
524        suggestions
525            .push("Consider using multi-stage builds to reduce final image size".to_string());
526    }
527
528    suggestions
529}
