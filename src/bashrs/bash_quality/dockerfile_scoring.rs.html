1//! Dockerfile Quality Scoring (Issue #10)
2//!
3//! Dockerfile-specific quality metrics instead of general bash script metrics.
4//!
5//! ## Scoring Dimensions (Weighted)
6//!
7//! 1. **Safety (30%)**: set -euo pipefail usage, error handling
8//! 2. **Complexity (25%)**: RUN command simplicity, script length
9//! 3. **Layer Optimization (20%)**: combined commands, cache cleanup in same layer
10//! 4. **Determinism (15%)**: version pinning, specific tags (not :latest)
11//! 5. **Security (10%)**: non-root user, no credential exposure
12//!
13//! ## Grade Scale
14//!
15//! - A+ (9.5-10): Excellent
16//! - A  (9.0-9.5): Very Good
17//! - B+ (8.5-9.0): Good
18//! - B  (8.0-8.5): Above Average
19//! - C+ (7.5-8.0): Average
20//! - C  (7.0-7.5): Below Average
21//! - D  (6.0-7.0): Poor
22//! - F  (<6.0): Failing
23

78    determinism: f64,
79    security: f64,
80    layer: f64,

83/// Calculate scoring penalties based on Dockerfile lint results (Issue #19)
84///
85/// Penalties are intentionally small to avoid overwhelming existing scoring.
86/// The main value is in the detailed suggestions shown to users.
87fn calculate_lint_penalty(lint_results: &crate::linter::LintResult) -> LintPenalty {
88    let mut penalty = LintPenalty {
89        determinism: 0.0,
90        security: 0.0,
91        layer: 0.0,
92    };
93
94    for diag in &lint_results.diagnostics {
95        match diag.code.as_str() {
96            "DOCKER001" => penalty.security += 0.5, // Missing USER (security issue, but scratch is OK)
97            "DOCKER002" => penalty.determinism += 0.2, // Unpinned base image (common, minor penalty)
98            "DOCKER003" => penalty.layer += 0.2,       // Missing apt cleanup
99            "DOCKER004" => penalty.security += 0.5,    // Invalid COPY --from
100            "DOCKER005" => penalty.layer += 0.1, // Missing --no-install-recommends (info level)
101            "DOCKER006" => penalty.security += 0.05, // Use COPY not ADD (very minor)
102            _ => {}
103        }
104    }
105
106    penalty
107}

109/// Score a Dockerfile for quality
110///
111/// Returns Dockerfile-specific quality score with grade, numeric score, and suggestions.
112/// Now integrates Issue #19 Dockerfile linting rules for accurate scoring.
113pub fn score_dockerfile(source: &str) -> Result<DockerfileQualityScore, String> {
114    let mut score = DockerfileQualityScore::new();
115
116    // Run Dockerfile linting (Issue #19) to get accurate diagnostics
117    use crate::linter::rules::lint_dockerfile;
118    let lint_results = lint_dockerfile(source);
119
120    // Apply lint penalties to scores
121    let lint_penalty = calculate_lint_penalty(&lint_results);
122
123    // Calculate each dimension
124    score.safety = calculate_safety_score(source);
125    score.complexity = calculate_complexity_score(source);
126    score.layer_optimization = calculate_layer_optimization_score(source);
127    score.determinism = calculate_determinism_score(source);
128    score.security = calculate_security_score(source);
129
130    // Apply lint penalties to relevant dimensions
131    score.determinism = (score.determinism - lint_penalty.determinism).max(0.0);
132    score.security = (score.security - lint_penalty.security).max(0.0);
133    score.layer_optimization = (score.layer_optimization - lint_penalty.layer).max(0.0);
134
135    // Calculate overall score (weighted average per Issue #10 spec)
136    score.score = (score.safety * 0.30)
137        + (score.complexity * 0.25)
138        + (score.layer_optimization * 0.20)
139        + (score.determinism * 0.15)
140        + (score.security * 0.10);
141
142    // Assign grade
143    score.grade = calculate_grade(score.score);
144
145    // Generate suggestions
146    score.suggestions = generate_suggestions(source, &score);
147
148    // Add lint-based suggestions (Issue #19)
149    for diag in &lint_results.diagnostics {
150        score
151            .suggestions
152            .push(format!("Line {}: {}", diag.span.start_line, diag.message));
153    }
154
155    Ok(score)
156}

158/// Calculate safety score (30% weight)
159///
160/// Checks for:
161/// - `set -euo pipefail` in RUN commands
162/// - Error handling (|| return, || exit)
163/// - Proper error propagation
164fn calculate_safety_score(source: &str) -> f64 {
165    if source.trim().is_empty() {
166        return 0.0;
167    }
168
169    let mut run_commands = 0;
170    let mut safe_run_commands = 0;
171    let mut has_error_handling = false;
172
173    for line in source.lines() {
174        let trimmed = line.trim();
175
176        // Track RUN commands
177        if trimmed.starts_with("RUN ") {
178            run_commands += 1;
179
180            // Check for set -euo pipefail
181            if trimmed.contains("set -euo pipefail")
182                || trimmed.contains("set -e") && trimmed.contains("set -o pipefail")
183            {
184                safe_run_commands += 1;
185                has_error_handling = true;
186            }
187
188            // Check for error handling
189            if trimmed.contains("|| exit") || trimmed.contains("|| return") {
190                has_error_handling = true;
191            }
192        }
193    }
194
195    if run_commands == 0 {
196        return 5.0; // Neutral score if no RUN commands
197    }
198
199    let safety_ratio = safe_run_commands as f64 / run_commands as f64;
200
201    let mut score: f64 = match safety_ratio {
202        r if r >= 0.8 => 10.0, // 80%+ safe commands
203        r if r >= 0.6 => 8.0,  // 60-79% safe
204        r if r >= 0.4 => 6.0,  // 40-59% safe
205        r if r >= 0.2 => 4.0,  // 20-39% safe
206        _ => 2.0,              // <20% safe
207    };
208
209    // Bonus for having any error handling
210    if has_error_handling {
211        score += 1.0;
212    }
213
214    score.min(10.0)
215}

217/// Calculate complexity score (25% weight)
218///
219/// Checks for:
220/// - Number of RUN commands (fewer is better - encourages layer optimization)
221/// - Length of RUN commands
222/// - Multi-line command complexity
223fn calculate_complexity_score(source: &str) -> f64 {
224    let lines: Vec<&str> = source.lines().collect();
225
226    if lines.is_empty() {
227        return 0.0;
228    }
229
230    let mut run_count = 0;
231    let mut longest_run = 0;
232    let mut current_run_lines = 0;
233    let mut in_run = false;
234
235    for line in &lines {
236        let trimmed = line.trim();
237
238        if trimmed.starts_with("RUN ") {
239            run_count += 1;
240            in_run = true;
241            current_run_lines = 1;
242        } else if in_run {
243            if trimmed.ends_with('\\') {
244                current_run_lines += 1;
245            } else {
246                if current_run_lines > 0 {
247                    current_run_lines += 1;
248                    if current_run_lines > longest_run {
249                        longest_run = current_run_lines;
250                    }
251                }
252                in_run = false;
253                current_run_lines = 0;
254            }
255        }
256    }
257
258    // Finalize if still in RUN
259    if in_run && current_run_lines > 0 && current_run_lines > longest_run {
260        longest_run = current_run_lines;
261    }
262
263    // Score based on RUN command count (fewer is better)
264    let run_score = match run_count {
265        0 => 5.0,       // Neutral
266        1..=3 => 10.0,  // Excellent (combined commands)
267        4..=6 => 8.0,   // Good
268        7..=10 => 6.0,  // Average
269        11..=15 => 4.0, // Below average
270        _ => 2.0,       // Poor (too many layers)
271    };
272
273    // Score based on longest RUN command
274    let length_score = match longest_run {
275        0 => 5.0,       // Neutral
276        1..=5 => 10.0,  // Simple
277        6..=10 => 8.0,  // Moderate
278        11..=20 => 6.0, // Complex
279        21..=30 => 4.0, // Very complex
280        _ => 2.0,       // Extremely complex
281    };
282
283    (run_score + length_score) / 2.0
284}

286/// Calculate layer optimization score (20% weight)
287///
288/// Checks for:
289/// - Combined commands with && instead of separate RUN statements
290/// - Cache cleanup in same layer (rm -rf /var/cache/apk/*, apt clean)
291/// - --no-cache flag for package managers
292/// - Multi-stage builds
293fn calculate_layer_optimization_score(source: &str) -> f64 {
294    if source.trim().is_empty() {
295        return 0.0;
296    }
297
298    let mut has_combined_commands = false;
299    let mut has_cache_cleanup = false;
300    let mut has_no_cache_flag = false;
301    let mut has_multistage = false;
302    let mut run_count = 0;
303
304    for line in source.lines() {
305        let trimmed = line.trim();
306
307        // Count RUN commands
308        if trimmed.starts_with("RUN ") {
309            run_count += 1;
310
311            // Check for combined commands
312            if trimmed.contains("&&") {
313                has_combined_commands = true;
314            }
315
316            // Check for cache cleanup in same layer
317            if trimmed.contains("rm -rf /var/cache/apk/*")
318                || trimmed.contains("apt-get clean")
319                || trimmed.contains("yum clean all")
320                || trimmed.contains("rm -rf /var/lib/apt/lists/*")
321            {
322                has_cache_cleanup = true;
323            }
324
325            // Check for --no-cache flag
326            if trimmed.contains("--no-cache") {
327                has_no_cache_flag = true;
328            }
329        }
330
331        // Check for multi-stage builds (FROM ... AS ...)
332        if trimmed.starts_with("FROM ") && trimmed.contains(" AS ") {
333            has_multistage = true;
334        }
335    }
336
337    let mut score: f64 = 0.0;
338
339    // Base score for having any optimizations
340    if run_count > 0 {
341        score += 2.0;
342    }
343
344    // Combined commands (4 points)
345    if has_combined_commands {
346        score += 4.0;
347    }
348
349    // Cache cleanup (3 points)
350    if has_cache_cleanup || has_no_cache_flag {
351        score += 3.0;
352    }
353
354    // Multi-stage builds (1 point bonus)
355    if has_multistage {
356        score += 1.0;
357    }
358
359    score.min(10.0)
360}

362/// Calculate determinism score (15% weight)
363///
364/// Checks for:
365/// - Version pinning in package installs (curl=8.2.1-r0)
366/// - Specific image tags (not :latest)
367/// - Pinned base images (alpine:3.18 not alpine:latest)
368fn calculate_determinism_score(source: &str) -> f64 {
369    if source.trim().is_empty() {
370        return 0.0;
371    }
372
373    let mut has_pinned_base_image = false;
374    let mut package_installs = 0;
375    let mut pinned_packages = 0;
376    let mut uses_latest_tag = false;
377
378    for line in source.lines() {
379        let trimmed = line.trim();
380
381        // Check FROM statement for tag specificity
382        if trimmed.starts_with("FROM ") {
383            if trimmed.contains(":latest") || (!trimmed.contains(':') && !trimmed.contains('@')) {
384                uses_latest_tag = true;
385            } else if trimmed.contains(':') {
386                has_pinned_base_image = true;
387            }
388        }
389
390        // Check RUN commands for package version pinning
391        if trimmed.starts_with("RUN ") {
392            // Count package install commands
393            if trimmed.contains("apk add")
394                || trimmed.contains("apt-get install")
395                || trimmed.contains("yum install")
396            {
397                package_installs += 1;
398
399                // Check for version pinning
400                // apk: curl=8.2.1-r0
401                // apt: curl=7.68.0-1
402                // yum: curl-7.68.0
403                if trimmed.contains('=') && (trimmed.contains("apk add") || trimmed.contains("apt"))
404                {
405                    pinned_packages += 1;
406                }
407            }
408        }
409    }
410
411    let mut score = 0.0;
412
413    // Base image pinning (5 points)
414    if has_pinned_base_image && !uses_latest_tag {
415        score += 5.0;
416    } else if has_pinned_base_image {
417        score += 3.0;
418    } else if !uses_latest_tag {
419        score += 1.0;
420    }
421
422    // Package version pinning (5 points)
423    if package_installs > 0 {
424        let pinning_ratio = pinned_packages as f64 / package_installs as f64;
425        score += pinning_ratio * 5.0;
426    } else {
427        score += 2.5; // Neutral if no packages
428    }
429
430    score.min(10.0)
431}

433/// Calculate security score (10% weight)
434///
435/// Checks for:
436/// - USER directive (non-root)
437/// - Avoiding ADD (use COPY instead)
438/// - No exposed secrets or credentials
439/// - Reasonable file permissions (not 777)
440/// - FROM scratch detection (Issue #13)
441fn calculate_security_score(source: &str) -> f64 {
442    if source.trim().is_empty() {
443        return 0.0;
444    }
445
446    // Issue #13: Detect if final stage is FROM scratch
447    let is_final_stage_scratch = detect_final_stage_scratch(source);
448
449    let mut has_user_directive = false;
450    let mut uses_copy_not_add = true;
451    let mut has_bad_permissions = false;
452    let mut potential_secrets = false;
453
454    for line in source.lines() {
455        let trimmed = line.trim();
456
457        // Check for USER directive
458        if trimmed.starts_with("USER ") && !trimmed.contains("USER root") {
459            has_user_directive = true;
460        }
461
462        // Check for ADD (should prefer COPY)
463        if trimmed.starts_with("ADD ") && !trimmed.contains(".tar") {
464            uses_copy_not_add = false;
465        }
466
467        // Check for bad permissions (777, 666)
468        if trimmed.contains("chmod 777") || trimmed.contains("chmod 666") {
469            has_bad_permissions = true;
470        }
471
472        // Check for potential secrets
473        if trimmed.contains("PASSWORD")
474            || trimmed.contains("SECRET")
475            || trimmed.contains("API_KEY")
476            || trimmed.contains("TOKEN")
477        {
478            potential_secrets = true;
479        }
480    }
481
482    let mut score: f64 = 5.0; // Start at neutral
483
484    // Issue #13: FROM scratch images get bonus for minimal attack surface
485    if is_final_stage_scratch {
486        score += 4.0; // Scratch = high security (no OS layer)
487    } else {
488        // Regular images: USER directive (4 points)
489        if has_user_directive {
490            score += 4.0;
491        } else {
492            score -= 2.0; // Penalty for running as root
493        }
494    }
495
496    // Using COPY instead of ADD (1 point)
497    if uses_copy_not_add {
498        score += 1.0;
499    }
500
501    // No bad permissions (maintain score)
502    if has_bad_permissions {
503        score -= 2.0;
504    }
505
506    // No exposed secrets (maintain score)
507    if potential_secrets {
508        score -= 1.0;
509    }
510
511    score.clamp(0.0, 10.0)
512}

514/// Detect if the final stage is FROM scratch (Issue #13)
515fn detect_final_stage_scratch(source: &str) -> bool {
516    let mut last_from_is_scratch = false;
517
518    for line in source.lines() {
519        let trimmed = line.trim();
520
521        // Skip comments and empty lines
522        if trimmed.is_empty() || trimmed.starts_with('#') {
523            continue;
524        }
525
526        // Detect FROM directive
527        if let Some(stripped) = trimmed.strip_prefix("FROM ") {
528            let from_image = stripped.trim();
529            // Check if this FROM is for scratch
530            last_from_is_scratch = from_image.starts_with("scratch");
531        }
532    }
533
534    last_from_is_scratch
535}

537/// Calculate grade from numeric score
538fn calculate_grade(score: f64) -> String {
539    match score {
540        s if s >= 9.5 => "A+".to_string(),
541        s if s >= 9.0 => "A".to_string(),
542        s if s >= 8.5 => "B+".to_string(),
543        s if s >= 8.0 => "B".to_string(),
544        s if s >= 7.5 => "C+".to_string(),
545        s if s >= 7.0 => "C".to_string(),
546        s if s >= 6.0 => "D".to_string(),
547        _ => "F".to_string(),
548    }
549}

551/// Generate improvement suggestions
552fn generate_suggestions(source: &str, score: &DockerfileQualityScore) -> Vec<String> {
553    let mut suggestions = Vec::new();
554
555    // Issue #13: Detect FROM scratch for appropriate suggestions
556    let is_final_stage_scratch = detect_final_stage_scratch(source);
557
558    // Safety suggestions
559    if score.safety < 7.0 {
560        let mut has_pipefail = false;
561        for line in source.lines() {
562            if line.contains("set -euo pipefail") {
563                has_pipefail = true;
564                break;
565            }
566        }
567
568        if !has_pipefail {
569            suggestions.push(
570                "Add 'set -euo pipefail &&' at the beginning of RUN commands for better error handling".to_string()
571            );
572        }
573    }
574
575    // Layer optimization suggestions
576    if score.layer_optimization < 7.0 {
577        suggestions.push("Combine RUN commands with && to reduce image layers".to_string());
578        suggestions.push(
579            "Clean up package manager cache in the same layer (rm -rf /var/cache/apk/*)"
580                .to_string(),
581        );
582        suggestions.push("Consider using --no-cache flag for package managers".to_string());
583    }
584
585    // Determinism suggestions
586    if score.determinism < 7.0 {
587        let has_latest = source.contains(":latest");
588        let has_version_pinning = source.contains('=');
589
590        if has_latest || !has_version_pinning {
591            suggestions
592                .push("Pin package versions for reproducibility (e.g., curl=8.2.1-r0)".to_string());
593            suggestions
594                .push("Use specific image tags instead of :latest (e.g., alpine:3.18)".to_string());
595        }
596    }
597
598    // Security suggestions (Issue #13: skip USER directive for FROM scratch)
599    if score.security < 7.0 {
600        let has_user = source.contains("USER ");
601
602        // Only suggest USER directive for non-scratch images
603        if !is_final_stage_scratch && (!has_user || source.contains("USER root")) {
604            suggestions.push("Add USER directive to run container as non-root user".to_string());
605            suggestions.push("Create a dedicated user with adduser/addgroup".to_string());
606        }
607    }
608
609    // Complexity suggestions
610    if score.complexity < 7.0 {
611        suggestions
612            .push("Reduce the number of separate RUN commands by combining them".to_string());
613        suggestions
614            .push("Consider using multi-stage builds to reduce final image size".to_string());
615    }
616
617    suggestions
618}
