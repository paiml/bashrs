48/// Check if a variable name should suppress SC2154 warnings
49pub fn should_suppress_sc2154(var_name: &str, context: &LintContext) -> bool {
50    if known_external_vars().contains(var_name) {
51        return true;
52    }
53    if let Some(func_name) = &context.current_function {
54        if func_name.starts_with("test_") {
55            return true;
56        }
57    }
58    if var_name.chars().any(|c| c.is_uppercase())
59        && var_name.chars().all(|c| c.is_uppercase() || c == '_')
60    {
61        return true;
62    }
63    false
64}
65/// Context for lint suppression decisions
66#[derive(Debug, Clone, Default)]
67pub struct LintContext {
68    pub current_function: Option<String>,
69    pub file_type: FileType,
70    pub line_number: usize,
71}
72/// File type affects suppression rules
73#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
74pub enum FileType {
75    #[default]
76    Script,
77    Config,
78    Library,
79}
80impl FileType {
81    /// Determine file type from path extension and filename
82    pub fn from_path(path: &std::path::Path) -> Self {
83        let filename = path.file_name().and_then(|n| n.to_str()).unwrap_or("");
84        if filename.ends_with("rc")
85            || filename.ends_with("profile")
86            || filename == ".bash_profile"
87            || filename == ".bash_login"
88            || filename == ".bash_logout"
89        {
90            return FileType::Config;
91        }
92        if path.extension().is_some_and(|e| e == "sh") {
93            return FileType::Script;
94        }
95        FileType::Library
96    }
97}
