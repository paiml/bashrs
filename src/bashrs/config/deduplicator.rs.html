1//! CONFIG-001: Deduplicate PATH entries
2//!
3//! This module detects and removes duplicate PATH entries in shell configuration
4//! files while preserving the order of first occurrence.
5
6use super::{ConfigIssue, PathEntry, Severity};
7use std::collections::HashMap;
8
9/// Analyze PATH entries in shell script source
10pub fn analyze_path_entries(source: &str) -> Vec<PathEntry> {
11    let mut entries = Vec::new();
12    let mut seen = HashMap::new();
13    let mut line_num = 0;
14
15    for line in source.lines() {
16        line_num += 1;
17
18        // Match: export PATH="/some/path:$PATH"
19        // Match: export PATH="/some/path:${PATH}"
20        // Match: PATH="/some/path:$PATH"
21        if let Some(path) = extract_path_addition(line) {
22            let is_duplicate = seen.contains_key(&path);
23            if is_duplicate {
24                // Track which line had the original
25                let _original_line = seen.get(&path).copied();
26            } else {
27                seen.insert(path.clone(), line_num);
28            }
29
30            entries.push(PathEntry {
31                line: line_num,
32                path,
33                is_duplicate,
34            });
35        }
36    }
37
38    entries
39}

41/// Extract the path being added from a PATH export line
42fn extract_path_addition(line: &str) -> Option<String> {
43    let line = line.trim();
44
45    // Skip comments
46    if line.starts_with('#') {
47        return None;
48    }
49
50    // Match patterns like:
51    // export PATH="/usr/local/bin:$PATH"
52    // export PATH="/usr/local/bin:${PATH}"
53    // PATH="/usr/local/bin:$PATH"
54
55    if !line.contains("PATH") || !line.contains('=') {
56        return None;
57    }
58
59    // Extract the part after = and before :$PATH or :${PATH}
60    let parts: Vec<&str> = line.split('=').collect();
61    if parts.len() < 2 {
62        return None;
63    }
64
65    let value = parts[1].trim();
66
67    // Remove quotes
68    let value = value.trim_matches('"').trim_matches('\'');
69
70    // Extract the path being added (before :$PATH or :${PATH})
71    if let Some(colon_pos) = value.find(':') {
72        let path = &value[..colon_pos];
73        Some(path.to_string())
74    } else {
75        None
76    }
77}

79/// Generate issues for duplicate PATH entries
80pub fn detect_duplicate_paths(entries: &[PathEntry]) -> Vec<ConfigIssue> {
81    let mut issues = Vec::new();
82
83    for entry in entries {
84        if entry.is_duplicate {
85            issues.push(ConfigIssue {
86                rule_id: "CONFIG-001".to_string(),
87                severity: Severity::Warning,
88                message: format!(
89                    "Duplicate PATH entry: '{}' (already added earlier)",
90                    entry.path
91                ),
92                line: entry.line,
93                column: 0,
94                suggestion: Some(format!(
95                    "Remove this line - '{}' is already in PATH",
96                    entry.path
97                )),
98            });
99        }
100    }
101
102    issues
103}

105/// Deduplicate PATH entries in source, preserving first occurrence
106pub fn deduplicate_path_entries(source: &str) -> String {
107    let entries = analyze_path_entries(source);
108    let mut seen_paths = HashMap::new();
109    let mut result = Vec::new();
110    let mut line_num = 0;
111
112    // First pass: identify which lines to keep
113    for entry in &entries {
114        if !entry.is_duplicate {
115            seen_paths.insert(entry.line, true);
116        }
117    }
118
119    // Second pass: rebuild source
120    for line in source.lines() {
121        line_num += 1;
122
123        // Check if this is a PATH line
124        let is_path_line = extract_path_addition(line).is_some();
125
126        if is_path_line {
127            // Only include if not a duplicate
128            if seen_paths.contains_key(&line_num) {
129                result.push(line.to_string());
130            }
131            // Duplicates are skipped
132        } else {
133            // Keep non-PATH lines as-is
134            result.push(line.to_string());
135        }
136    }
137
138    result.join("\n")
139}
