28/// Configuration analysis result
29#[derive(Serialize, Deserialize)]
30#[wasm_bindgen]
31pub struct ConfigAnalysisResult {
32    file_path: String,
33    line_count: usize,
34    complexity_score: u8,
35    issues: Vec<ConfigIssue>,
36}

38/// A single configuration issue
39#[derive(Serialize, Deserialize, Clone)]
40#[wasm_bindgen]
41pub struct ConfigIssue {
42    rule_id: String,
43    severity: String,
44    line: usize,
45    column: usize,
46    message: String,
47    suggestion: Option<String>,
48}
49
50#[wasm_bindgen]
51impl ConfigAnalysisResult {
52    /// Get the number of issues found
53    #[wasm_bindgen(getter)]
54    pub fn issue_count(&self) -> usize {
55        self.issues.len()
56    }
57
58    /// Get issues as JSON string
59    #[wasm_bindgen(getter)]
60    pub fn issues_json(&self) -> String {
61        serde_json::to_string(&self.issues).unwrap_or_else(|_| "[]".to_string())
62    }
63
64    /// Get line count
65    #[wasm_bindgen(getter)]
66    pub fn line_count(&self) -> usize {
67        self.line_count
68    }
69
70    /// Get complexity score (0-10)
71    #[wasm_bindgen(getter)]
72    pub fn complexity_score(&self) -> u8 {
73        self.complexity_score
74    }
75}

77/// Analyze a shell configuration file
78///
79/// Runs CONFIG-001 to CONFIG-004 analysis on the provided config content.
80///
81/// # Arguments
82///
83/// * `content` - The configuration file content (.bashrc, .zshrc, etc.)
84/// * `filename` - Optional filename for context (e.g., ".bashrc")
85///
86/// # Returns
87///
88/// A `ConfigAnalysisResult` containing detected issues
89///
90/// # Example (JavaScript)
91///
92/// ```js
93/// const bashrc = `
94/// export PATH="/usr/local/bin:$PATH"
95/// export PATH="/usr/local/bin:$PATH"  # Duplicate!
96/// export SESSION_ID=$RANDOM          # Non-deterministic!
97/// `;
98///
99/// const result = analyzeConfig(bashrc, ".bashrc");
100/// console.log(`Found ${result.issue_count} issues`);
101/// console.log(JSON.parse(result.issues_json));
102/// ```
103#[wasm_bindgen]
104pub fn analyze_config(content: &str, filename: Option<String>) -> Result<ConfigAnalysisResult, JsValue> {
105    use crate::config::analyzer;
106    use std::path::PathBuf;
107
108    let path = PathBuf::from(filename.unwrap_or_else(|| ".bashrc".to_string()));
109    let analysis = analyzer::analyze_config(content, path);
110
111    // Convert to WASM-friendly format
112    let issues = analysis
113        .issues
114        .iter()
115        .map(|issue| ConfigIssue {
116            rule_id: issue.rule_id.clone(),
117            severity: format!("{:?}", issue.severity),
118            line: issue.line,
119            column: issue.column,
120            message: issue.message.clone(),
121            suggestion: issue.suggestion.clone(),
122        })
123        .collect();
124
125    Ok(ConfigAnalysisResult {
126        file_path: analysis.file_path.display().to_string(),
127        line_count: analysis.line_count,
128        complexity_score: analysis.complexity_score,
129        issues,
130    })
131}

133/// Purify a shell configuration file
134///
135/// Applies automatic fixes for CONFIG-001 to CONFIG-004 issues.
136///
137/// # Arguments
138///
139/// * `content` - The configuration file content
140///
141/// # Returns
142///
143/// Purified configuration file content
144///
145/// # Example (JavaScript)
146///
147/// ```js
148/// const messy = `
149/// export PATH="/usr/local/bin:$PATH"
150/// export PATH="/usr/local/bin:$PATH"
151/// export SESSION_ID=$RANDOM
152/// `;
153///
154/// const clean = purifyConfig(messy);
155/// // Duplicates removed, non-determinism fixed
156/// ```
157#[wasm_bindgen]
158pub fn purify_config(content: &str) -> String {
159    use crate::config::purifier;
160
161    purifier::purify_config(content)
162}
