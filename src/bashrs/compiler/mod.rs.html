63    pub fn compile(&self, script: &str) -> Result<Vec<u8>> {
64        // Step 1: Compress script with Zstandard
65        let compressed = zstd::encode_all(script.as_bytes(), self.compression.level())
66            .map_err(|e| Error::Internal(format!("Compression failed: {e}")))?;
67
68        // Step 2: Load base runtime
69        let mut binary = self.load_runtime()?;
70
71        // Step 3: Inject script as ELF section
72        let section_offset = self.inject_section(&mut binary, ".rash_script", &compressed)?;
73
74        // Step 4: Patch entrypoint to our loader
75        self.patch_entrypoint(binary.as_mut_slice(), section_offset)?;
76
77        // Step 5: Strip unnecessary symbols
78        self.strip_binary(binary.as_mut_slice())?;
79
80        Ok(binary)
81    }

132/// Create a self-extracting shell script
133pub fn create_self_extracting_script(script: &str, output_path: &str) -> Result<()> {
134    let compressed = zstd::encode_all(script.as_bytes(), 11)
135        .map_err(|e| Error::Internal(format!("Compression failed: {e}")))?;
136
137    let encoded = base64::Engine::encode(&base64::engine::general_purpose::STANDARD, &compressed);
138
139    let self_extract = format!(
140        r#"#!/bin/sh
141# Self-extracting RASH script
142set -euf
143
144# Embedded compressed script
145PAYLOAD='{}'
146
147# Extract and decompress
148if command -v base64 >/dev/null && command -v zstd >/dev/null; then
149    echo "$PAYLOAD" | base64 -d | zstd -d | sh -euf
150elif command -v base64 >/dev/null && command -v gzip >/dev/null; then
151    # Fallback to gzip if zstd not available
152    echo "Warning: zstd not found, using gzip (larger size)" >&2
153    GZIP_PAYLOAD='{}'
154    echo "$GZIP_PAYLOAD" | base64 -d | gzip -d | sh -euf
155else
156    echo "Error: base64 and compression tools required" >&2
157    exit 1
158fi
159"#,
160        encoded,
161        base64::Engine::encode(
162            &base64::engine::general_purpose::STANDARD,
163            compress_gzip(script.as_bytes())
164        )
165    );
166
167    std::fs::write(output_path, self_extract)
168        .map_err(|e| Error::Internal(format!("Failed to write output: {e}")))?;
169
170    // Make executable
171    #[cfg(unix)]
172    {
173        use std::os::unix::fs::PermissionsExt;
174        let mut perms = std::fs::metadata(output_path)
175            .map_err(|e| Error::Internal(format!("Failed to get metadata: {e}")))?
176            .permissions();
177        perms.set_mode(0o755);
178        std::fs::set_permissions(output_path, perms)
179            .map_err(|e| Error::Internal(format!("Failed to set permissions: {e}")))?;
180    }
181
182    Ok(())
183}

185fn compress_gzip(data: &[u8]) -> Vec<u8> {
186    use flate2::write::GzEncoder;
187    use flate2::Compression;
188
189    let mut encoder = GzEncoder::new(Vec::new(), Compression::best());
190    encoder.write_all(data).unwrap();
191    encoder.finish().unwrap()
