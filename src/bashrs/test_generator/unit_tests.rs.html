28    /// Generate unit tests for all functions in AST
29    pub fn generate_tests(&self, ast: &BashAst) -> TestGenResult<Vec<UnitTest>> {
30        let mut tests = Vec::new();
31
32        for stmt in &ast.statements {
33            if let BashStmt::Function { name, body, .. } = stmt {
34                // Generate tests for this function
35                tests.extend(self.generate_function_tests(name, body)?);
36            }
37        }
38
39        // Add edge case tests
40        tests.extend(self.generate_edge_case_tests(ast)?);
41
42        // Add error case tests
43        tests.extend(self.generate_error_case_tests(ast)?);
44
45        Ok(tests)
46    }
47
48    /// Generate tests for a specific function
49    fn generate_function_tests(&self, name: &str, body: &[BashStmt]) -> TestGenResult<Vec<UnitTest>> {
50        let mut tests = Vec::new();
51
52        // 1. Generate branch coverage tests
53        tests.extend(self.generate_branch_tests(name, body)?);
54
55        // 2. Generate boundary value tests
56        tests.extend(self.generate_boundary_tests(name, body)?);
57
58        Ok(tests)
59    }

61    /// Generate tests for branch coverage (if/else, case, loops)
62    fn generate_branch_tests(&self, name: &str, body: &[BashStmt]) -> TestGenResult<Vec<UnitTest>> {
63        let mut tests = Vec::new();
64
65        for stmt in body {
66            match stmt {
67                BashStmt::If { elif_blocks, else_block, .. } => {
68                    // Test the "then" branch
69                    tests.push(UnitTest {
70                        name: format!("test_{}_if_then_branch", name),
71                        test_fn: format!("{}()", name),
72                        assertions: vec![
73                            Assertion::Comment("Test if-then branch".to_string()),
74                        ],
75                    });
76
77                    // Test elif branches
78                    for (i, _) in elif_blocks.iter().enumerate() {
79                        tests.push(UnitTest {
80                            name: format!("test_{}_elif_{}_branch", name, i),
81                            test_fn: format!("{}()", name),
82                            assertions: vec![
83                                Assertion::Comment(format!("Test elif {} branch", i)),
84                            ],
85                        });
86                    }
87
88                    // Test else branch if present
89                    if else_block.is_some() {
90                        tests.push(UnitTest {
91                            name: format!("test_{}_else_branch", name),
92                            test_fn: format!("{}()", name),
93                            assertions: vec![
94                                Assertion::Comment("Test else branch".to_string()),
95                            ],
96                        });
97                    }
98                }
99
100                BashStmt::While { .. } => {
101                    // Test while loop
102                    tests.push(UnitTest {
103                        name: format!("test_{}_while_loop", name),
104                        test_fn: format!("{}()", name),
105                        assertions: vec![
106                            Assertion::Comment("Test while loop execution".to_string()),
107                        ],
108                    });
109                }
110
111                BashStmt::For { .. } => {
112                    // Test for loop
113                    tests.push(UnitTest {
114                        name: format!("test_{}_for_loop", name),
115                        test_fn: format!("{}()", name),
116                        assertions: vec![
117                            Assertion::Comment("Test for loop iteration".to_string()),
118                        ],
119                    });
120                }
121
122                _ => {}
123            }
124        }
125
126        Ok(tests)
127    }

129    /// Generate boundary value tests
130    fn generate_boundary_tests(&self, name: &str, _body: &[BashStmt]) -> TestGenResult<Vec<UnitTest>> {
131        vec![
132            UnitTest {
133                name: format!("test_{}_boundary_zero", name),
134                test_fn: format!("{}(0)", name),
135                assertions: vec![
136                    Assertion::Comment("Test with zero value".to_string()),
137                ],
138            },
139            UnitTest {
140                name: format!("test_{}_boundary_one", name),
141                test_fn: format!("{}(1)", name),
142                assertions: vec![
143                    Assertion::Comment("Test with one value".to_string()),
144                ],
145            },
146        ].into_iter().map(Ok).collect()
147    }

149    /// Generate edge case tests (empty strings, null, max values)
150    fn generate_edge_case_tests(&self, ast: &BashAst) -> TestGenResult<Vec<UnitTest>> {
151        let mut tests = Vec::new();
152
153        for stmt in &ast.statements {
154            if let BashStmt::Function { name, .. } = stmt {
155                // Empty string test
156                tests.push(UnitTest {
157                    name: format!("test_{}_edge_case_empty_string", name),
158                    test_fn: format!("{}(\"\")", name),
159                    assertions: vec![
160                        Assertion::Comment("Test with empty string input".to_string()),
161                    ],
162                });
163
164                // Negative number test
165                tests.push(UnitTest {
166                    name: format!("test_{}_edge_case_negative", name),
167                    test_fn: format!("{}(-1)", name),
168                    assertions: vec![
169                        Assertion::Comment("Test with negative value".to_string()),
170                    ],
171                });
172
173                // Large number test
174                tests.push(UnitTest {
175                    name: format!("test_{}_edge_case_large_value", name),
176                    test_fn: format!("{}(i64::MAX)", name),
177                    assertions: vec![
178                        Assertion::Comment("Test with maximum value".to_string()),
179                    ],
180                });
181            }
182        }
183
184        Ok(tests)
185    }

187    /// Generate error case tests
188    fn generate_error_case_tests(&self, ast: &BashAst) -> TestGenResult<Vec<UnitTest>> {
189        let mut tests = Vec::new();
190
191        for stmt in &ast.statements {
192            if let BashStmt::Function { name, body, .. } = stmt {
193                // Check if function uses file operations
194                if self.uses_file_operations(body) {
195                    tests.push(UnitTest {
196                        name: format!("test_{}_error_file_not_found", name),
197                        test_fn: format!("{}(\"/nonexistent/file\")", name),
198                        assertions: vec![
199                            Assertion::ShouldPanic {
200                                expected_message: Some("File not found".to_string()),
201                            },
202                        ],
203                    });
204                }
205
206                // Check if function has numeric operations
207                if self.uses_arithmetic(body) {
208                    tests.push(UnitTest {
209                        name: format!("test_{}_error_invalid_input", name),
210                        test_fn: format!("{}(\"invalid\")", name),
211                        assertions: vec![
212                            Assertion::ShouldPanic {
213                                expected_message: Some("Invalid input".to_string()),
214                            },
215                        ],
216                    });
217                }
218            }
219        }
220
221        Ok(tests)
222    }

224    /// Generate targeted tests for specific uncovered paths
225    pub fn generate_targeted_tests(&self, uncovered: &[UncoveredPath]) -> TestGenResult<Vec<UnitTest>> {
226        let mut tests = Vec::new();
227
228        for path in uncovered {
229            match path {
230                UncoveredPath::Line(line) => {
231                    tests.push(UnitTest {
232                        name: format!("test_coverage_line_{}", line),
233                        test_fn: "target_line()".to_string(),
234                        assertions: vec![
235                            Assertion::Comment(format!("Cover line {}", line)),
236                        ],
237                    });
238                }
239                UncoveredPath::Branch(branch) => {
240                    tests.push(UnitTest {
241                        name: format!("test_coverage_branch_{}", branch.function),
242                        test_fn: format!("{}()", branch.function),
243                        assertions: vec![
244                            Assertion::Comment(format!("Cover branch {:?}", branch.branch_type)),
245                        ],
246                    });
247                }
248                UncoveredPath::Function(func) => {
249                    tests.push(UnitTest {
250                        name: format!("test_coverage_function_{}", func),
251                        test_fn: format!("{}()", func),
252                        assertions: vec![
253                            Assertion::Comment(format!("Cover function {}", func)),
254                        ],
255                    });
256                }
257            }
258        }
259
260        Ok(tests)
261    }

263    /// Check if function body uses file operations
264    fn uses_file_operations(&self, body: &[BashStmt]) -> bool {
265        for stmt in body {
266            if let BashStmt::Command { name, .. } = stmt {
267                if matches!(name.as_str(), "cat" | "ls" | "mkdir" | "rm" | "cp" | "mv") {
268                    return true;
269                }
270            }
271        }
272        false
273    }
274
275    /// Check if function body uses arithmetic operations
276    fn uses_arithmetic(&self, body: &[BashStmt]) -> bool {
277        for stmt in body {
278            if let BashStmt::Assignment { value, .. } = stmt {
279                if matches!(value, BashExpr::Arithmetic(_)) {
280                    return true;
281                }
282            }
283        }
284        false
285    }

288/// A single unit test
289#[derive(Debug, Clone)]
290pub struct UnitTest {
291    pub name: String,
292    pub test_fn: String,
293    pub assertions: Vec<Assertion>,
294}
295
296impl UnitTest {
297    /// Convert to Rust test code
298    pub fn to_rust_code(&self) -> String {
299        let mut code = "#[test]\n".to_string();
300
301        // Add #[should_panic] if needed
302        for assertion in &self.assertions {
303            if let Assertion::ShouldPanic { .. } = assertion {
304                if let Assertion::ShouldPanic { expected_message: Some(msg) } = assertion {
305                    code.push_str(&format!("#[should_panic(expected = \"{}\")]\n", msg));
306                } else {
307                    code.push_str("#[should_panic]\n");
308                }
309                break;
310            }
311        }
312
313        code.push_str(&format!("fn {}() {{\n", self.name));
314
315        for assertion in &self.assertions {
316            code.push_str(&format!("    {}\n", assertion.to_rust_code()));
317        }
318
319        code.push_str("}\n");
320        code
321    }
322}

324/// Test assertion types
325#[derive(Debug, Clone)]
326pub enum Assertion {
327    /// assert_eq!(actual, expected)
328    Equals { actual: String, expected: String },
329
330    /// assert_ne!(actual, expected)
331    NotEquals { actual: String, expected: String },
332
333    /// assert!(condition)
334    True { condition: String },
335
336    /// assert!(!condition)
337    False { condition: String },
338
339    /// #[should_panic]
340    ShouldPanic { expected_message: Option<String> },
341
342    /// Comment for documentation
343    Comment(String),
344}
345
346impl Assertion {
347    /// Convert to Rust assertion code
348    fn to_rust_code(&self) -> String {
349        match self {
350            Assertion::Equals { actual, expected } => {
351                format!("assert_eq!({}, {});", actual, expected)
352            }
353            Assertion::NotEquals { actual, expected } => {
354                format!("assert_ne!({}, {});", actual, expected)
355            }
356            Assertion::True { condition } => {
357                format!("assert!({});", condition)
358            }
359            Assertion::False { condition } => {
360                format!("assert!(!{});", condition)
361            }
362            Assertion::ShouldPanic { .. } => {
363                // This is handled at the function level
364                "// Should panic test".to_string()
365            }
366            Assertion::Comment(text) => {
367                format!("// {}", text)
368            }
369        }
370    }
371}
