6/// Check if a function name is a stdlib function
7pub fn is_stdlib_function(name: &str) -> bool {
8    matches!(
9        name,
10        // String module
11        "string_trim"
12            | "string_contains"
13            | "string_len"
14            | "string_split"
15            | "string_replace"
16            | "string_to_upper"
17            | "string_to_lower"
18            // File system module
19            | "fs_exists"
20            | "fs_read_file"
21            | "fs_write_file"
22            | "fs_copy"
23            | "fs_remove"
24            | "fs_is_file"
25            | "fs_is_dir"
26            // Array module
27            | "array_len"
28            | "array_join"
29            // Environment module (Sprint 27a)
30            | "env"
31            | "env_var_or"
32            // Arguments module (Sprint 27b)
33            | "arg"
34            | "args"
35            | "arg_count"
36            // Exit code module (Sprint 27c)
37            | "exit_code"
38    )
39}

41/// Get the shell function name for a stdlib function
42pub fn get_shell_function_name(name: &str) -> String {
43    format!("rash_{}", name)
44}

392/// RED TEST: Invalid variable names should be rejected
393/// Tests that env() rejects variable names with invalid characters
394#[test]
395fn test_env_rejects_invalid_var_names() {
396    // Valid var names: alphanumeric + underscore only
397    assert!(is_valid_var_name("HOME"));
398    assert!(is_valid_var_name("MY_VAR"));
399    assert!(is_valid_var_name("VAR123"));
400    assert!(is_valid_var_name("_PRIVATE"));
401
402    // Invalid var names (injection attempts)
403    assert!(!is_valid_var_name("'; rm -rf /; #"));
404    assert!(!is_valid_var_name("VAR; echo hack"));
405    assert!(!is_valid_var_name("$(whoami)"));
406    assert!(!is_valid_var_name("VAR`id`"));
407    assert!(!is_valid_var_name("VAR$OTHER"));
408    assert!(!is_valid_var_name("VAR-NAME")); // Dash not allowed
409    assert!(!is_valid_var_name("VAR.NAME")); // Dot not allowed
410}
