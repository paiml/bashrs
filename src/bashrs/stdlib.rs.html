1// ! Standard Library Support
2//!
3//! This module provides support for stdlib functions that are transpiled
4//! to POSIX shell runtime functions.
5
6/// Check if a function name is a stdlib function
7pub fn is_stdlib_function(name: &str) -> bool {
8    matches!(
9        name,
10        // String module
11        "string_trim"
12            | "string_contains"
13            | "string_len"
14            | "string_split"
15            | "string_replace"
16            | "string_to_upper"
17            | "string_to_lower"
18            // File system module
19            | "fs_exists"
20            | "fs_read_file"
21            | "fs_write_file"
22            | "fs_copy"
23            | "fs_remove"
24            | "fs_is_file"
25            | "fs_is_dir"
26            // Array module
27            | "array_len"
28            | "array_join"
29    )
30}
31
32/// Get the shell function name for a stdlib function
33pub fn get_shell_function_name(name: &str) -> String {
34    format!("rash_{}", name)
35}

130#[cfg(test)]
131mod tests {
132    use super::*;
133
134    #[test]
135    fn test_is_stdlib_function() {
136        // String functions
137        assert!(is_stdlib_function("string_trim"));
138        assert!(is_stdlib_function("string_replace"));
139        assert!(is_stdlib_function("string_to_upper"));
140        assert!(is_stdlib_function("string_to_lower"));
141
142        // File system functions
143        assert!(is_stdlib_function("fs_exists"));
144        assert!(is_stdlib_function("fs_copy"));
145        assert!(is_stdlib_function("fs_remove"));
146        assert!(is_stdlib_function("fs_is_file"));
147        assert!(is_stdlib_function("fs_is_dir"));
148
149        // Not stdlib functions
150        assert!(!is_stdlib_function("custom_function"));
151        assert!(!is_stdlib_function("println"));
152    }
153
154    #[test]
155    fn test_get_shell_function_name() {
156        assert_eq!(get_shell_function_name("string_trim"), "rash_string_trim");
157        assert_eq!(get_shell_function_name("fs_exists"), "rash_fs_exists");
158    }
