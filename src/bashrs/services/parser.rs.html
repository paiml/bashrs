11/// Parse Rust source code into a RestrictedAst
12pub fn parse(input: &str) -> Result<RestrictedAst> {
13    let file: File = syn::parse_str(input)?;
14
15    let mut functions = Vec::new();
16    let mut entry_point = None;
17
18    for item in file.items {
19        process_item(item, &mut functions, &mut entry_point)?;
20    }
21
22    let entry_point = entry_point
23        .ok_or_else(|| Error::Validation("No #[bashrs::main] function found".to_string()))?;
24
25    Ok(RestrictedAst {
26        functions,
27        entry_point,
28    })
29}

31fn process_item(
32    item: Item,
33    functions: &mut Vec<Function>,
34    entry_point: &mut Option<String>,
35) -> Result<()> {
36    let Item::Fn(item_fn) = item else {
37        return Err(Error::Validation(
38            "Only functions are allowed in Rash code".to_string(),
39        ));
40    };
41
42    let is_main = has_main_attribute(&item_fn) || item_fn.sig.ident == "main";
43    let function = convert_function(item_fn)?;
44
45    if is_main {
46        check_single_entry_point(entry_point, &function.name)?;
47        *entry_point = Some(function.name.clone());
48    }
49
50    functions.push(function);
51    Ok(())
52}

54fn has_main_attribute(item_fn: &ItemFn) -> bool {
55    item_fn.attrs.iter().any(is_main_attribute)
56}
57
58fn is_main_attribute(attr: &syn::Attribute) -> bool {
59    let path = attr.path();
60    path.segments.len() == 2
61        && (path.segments[0].ident == "bashrs" || path.segments[0].ident == "rash")
62        && path.segments[1].ident == "main"
63}

65fn check_single_entry_point(current: &Option<String>, _new_name: &str) -> Result<()> {
66    if current.is_some() {
67        return Err(Error::Validation(
68            "Multiple #[bashrs::main] functions found".to_string(),
69        ));
70    }
71    Ok(())
72}

74fn convert_function(item_fn: ItemFn) -> Result<Function> {
75    let name = item_fn.sig.ident.to_string();
76
77    // Convert parameters
78    let mut params = Vec::new();
79    for input in item_fn.sig.inputs {
80        match input {
81            FnArg::Typed(pat_type) => {
82                if let Pat::Ident(pat_ident) = &*pat_type.pat {
83                    let param_name = pat_ident.ident.to_string();
84                    let param_type = convert_type(&pat_type.ty)?;
85                    params.push(Parameter {
86                        name: param_name,
87                        param_type,
88                    });
89                } else {
90                    return Err(Error::Validation(
91                        "Complex parameter patterns not supported".to_string(),
92                    ));
93                }
94            }
95            FnArg::Receiver(_) => {
96                return Err(Error::Validation("Self parameters not allowed".to_string()));
97            }
98        }
99    }
100
101    // Convert return type
102    let return_type = match &item_fn.sig.output {
103        ReturnType::Default => Type::Void, // Default to void/unit type
104        ReturnType::Type(_, ty) => convert_type(ty)?,
105    };
106
107    // Convert function body
108    let body = convert_block(&item_fn.block)?;
109
110    Ok(Function {
111        name,
112        params,
113        return_type,
114        body,
115    })
116}

118fn convert_type(ty: &SynType) -> Result<Type> {
119    match ty {
120        SynType::Path(type_path) => {
121            let path_str = type_path
122                .path
123                .segments
124                .iter()
125                .map(|seg| seg.ident.to_string())
126                .collect::<Vec<_>>()
127                .join("::");
128
129            match path_str.as_str() {
130                "bool" => Ok(Type::Bool),
131                "u32" | "i32" => Ok(Type::U32), // Treat i32 as u32 for now
132                "str" | "String" => Ok(Type::Str),
133                path if path.starts_with("Result") => {
134                    // Simple Result type parsing - assumes Result<T, E>
135                    Ok(Type::Result {
136                        ok_type: Box::new(Type::Str), // Simplified
137                        err_type: Box::new(Type::Str),
138                    })
139                }
140                path if path.starts_with("Option") => {
141                    // Simple Option type parsing - assumes Option<T>
142                    Ok(Type::Option {
143                        inner_type: Box::new(Type::Str), // Simplified
144                    })
145                }
146                _ => Err(Error::Validation(format!("Unsupported type: {path_str}"))),
147            }
148        }
149        SynType::Reference(type_ref) => {
150            // Handle &str and other reference types
151            if let SynType::Path(path) = &*type_ref.elem {
152                let path_str = path
153                    .path
154                    .segments
155                    .iter()
156                    .map(|seg| seg.ident.to_string())
157                    .collect::<Vec<_>>()
158                    .join("::");
159
160                match path_str.as_str() {
161                    "str" => Ok(Type::Str),
162                    _ => Err(Error::Validation(format!(
163                        "Unsupported reference type: &{path_str}"
164                    ))),
165                }
166            } else {
167                Err(Error::Validation(
168                    "Complex reference types not supported".to_string(),
169                ))
170            }
171        }
172        _ => Err(Error::Validation("Complex types not supported".to_string())),
173    }
174}

176fn convert_block(block: &Block) -> Result<Vec<Stmt>> {
177    let mut statements = Vec::new();
178
179    for stmt in &block.stmts {
180        statements.push(convert_stmt(stmt)?);
181    }
182
183    Ok(statements)
184}

186fn convert_stmt(stmt: &SynStmt) -> Result<Stmt> {
187    match stmt {
188        SynStmt::Local(local) => convert_let_stmt(local),
189        SynStmt::Expr(expr, _) => convert_expr_stmt(expr),
190        SynStmt::Macro(macro_stmt) => convert_macro_stmt(macro_stmt),
191        _ => Err(Error::Validation("Unsupported statement type".to_string())),
192    }
193}

195fn convert_let_stmt(local: &syn::Local) -> Result<Stmt> {
196    let Pat::Ident(pat_ident) = &local.pat else {
197        return Err(Error::Validation(
198            "Complex patterns not supported".to_string(),
199        ));
200    };
201
202    let name = pat_ident.ident.to_string();
203
204    let Some(init) = &local.init else {
205        return Err(Error::Validation(
206            "Let bindings must have initializers".to_string(),
207        ));
208    };
209
210    let value = convert_expr(&init.expr)?;
211    Ok(Stmt::Let { name, value })
212}

214fn convert_expr_stmt(expr: &SynExpr) -> Result<Stmt> {
215    match expr {
216        SynExpr::If(expr_if) => convert_if_stmt(expr_if),
217        SynExpr::ForLoop(for_loop) => convert_for_loop(for_loop),
218        SynExpr::While(expr_while) => convert_while_loop(expr_while),
219        SynExpr::Match(expr_match) => convert_match_stmt(expr_match),
220        SynExpr::Break(_) => Ok(Stmt::Break),
221        SynExpr::Continue(_) => Ok(Stmt::Continue),
222        _ => Ok(Stmt::Expr(convert_expr(expr)?)),
223    }
224}

226fn convert_macro_stmt(macro_stmt: &syn::StmtMacro) -> Result<Stmt> {
227    let macro_path = &macro_stmt.mac.path;
228    let macro_name = macro_path
229        .segments
230        .last()
231        .ok_or_else(|| Error::Validation("Empty macro path".to_string()))?
232        .ident
233        .to_string();
234
235    if macro_name == "println" {
236        // Parse macro tokens to extract the format string
237        let tokens = macro_stmt.mac.tokens.clone();
238        let parsed: syn::Expr = syn::parse2(tokens)
239            .map_err(|_| Error::Validation("Invalid println! arguments".to_string()))?;
240
241        // Convert the first argument as the format string
242        let arg = convert_expr(&parsed)?;
243
244        // Generate a function call to rash_println
245        Ok(Stmt::Expr(Expr::FunctionCall {
246            name: "rash_println".to_string(),
247            args: vec![arg],
248        }))
249    } else {
250        Err(Error::Validation(format!(
251            "Unsupported macro: {macro_name}!"
252        )))
253    }
254}

256fn convert_if_stmt(expr_if: &syn::ExprIf) -> Result<Stmt> {
257    let condition = convert_expr(&expr_if.cond)?;
258    let then_block = convert_block(&expr_if.then_branch)?;
259    let else_block = convert_else_block(&expr_if.else_branch)?;
260
261    Ok(Stmt::If {
262        condition,
263        then_block,
264        else_block,
265    })
266}

268fn convert_else_block(else_branch: &Option<(syn::token::Else, Box<SynExpr>)>) -> Result<Option<Vec<Stmt>>> {
269    let Some((_, else_expr)) = else_branch else {
270        return Ok(None);
271    };
272
273    match &**else_expr {
274        SynExpr::Block(block) => Ok(Some(convert_block(&block.block)?)),
275        SynExpr::If(nested_if) => convert_else_if(nested_if),
276        _ => Ok(None),
277    }
278}

280fn convert_else_if(nested_if: &syn::ExprIf) -> Result<Option<Vec<Stmt>>> {
281    let nested_condition = convert_expr(&nested_if.cond)?;
282    let nested_then = convert_block(&nested_if.then_branch)?;
283    let nested_else = convert_nested_else(&nested_if.else_branch)?;
284
285    Ok(Some(vec![Stmt::If {
286        condition: nested_condition,
287        then_block: nested_then,
288        else_block: nested_else,
289    }]))
290}

292fn convert_nested_else(else_branch: &Option<(syn::token::Else, Box<SynExpr>)>) -> Result<Option<Vec<Stmt>>> {
293    let Some((_, nested_else_expr)) = else_branch else {
294        return Ok(None);
295    };
296
297    match &**nested_else_expr {
298        SynExpr::Block(block) => Ok(Some(convert_block(&block.block)?)),
299        SynExpr::If(_) => {
300            // Recursively handle else-if-else-if chains
301            let stmt = SynStmt::Expr((**nested_else_expr).clone(), None);
302            Ok(Some(vec![convert_stmt(&stmt)?]))
303        }
304        _ => Ok(None),
305    }
306}

308fn convert_expr(expr: &SynExpr) -> Result<Expr> {
309    match expr {
310        SynExpr::Lit(expr_lit) => convert_literal_expr(expr_lit),
311        SynExpr::Path(expr_path) => convert_path_expr(expr_path),
312        SynExpr::Call(expr_call) => convert_call_expr(expr_call),
313        SynExpr::Binary(expr_binary) => convert_binary_expr(expr_binary),
314        SynExpr::Unary(expr_unary) => convert_unary_expr(expr_unary),
315        SynExpr::MethodCall(method_call) => convert_method_call_expr(method_call),
316        SynExpr::Return(ret_expr) => convert_return_expr(ret_expr),
317        SynExpr::Paren(expr_paren) => convert_expr(&expr_paren.expr),
318        SynExpr::If(_) => Err(Error::Validation(
319            "If expressions not supported in expression position".to_string(),
320        )),
321        SynExpr::Range(range_expr) => convert_range_expr(range_expr),
322        _ => Err(Error::Validation("Unsupported expression type".to_string())),
323    }
324}

342fn convert_call_expr(expr_call: &syn::ExprCall) -> Result<Expr> {
343    let SynExpr::Path(path) = &*expr_call.func else {
344        return Err(Error::Validation(
345            "Complex function calls not supported".to_string(),
346        ));
347    };
348
349    let name = path
350        .path
351        .segments
352        .iter()
353        .map(|seg| seg.ident.to_string())
354        .collect::<Vec<_>>()
355        .join("::");
356
357    let mut args = Vec::new();
358    for arg in &expr_call.args {
359        args.push(convert_expr(arg)?);

365fn convert_binary_expr(expr_binary: &syn::ExprBinary) -> Result<Expr> {
366    let left = Box::new(convert_expr(&expr_binary.left)?);
367    let right = Box::new(convert_expr(&expr_binary.right)?);
368    let op = convert_binary_op(&expr_binary.op)?;
369    Ok(Expr::Binary { op, left, right })

372fn convert_unary_expr(expr_unary: &syn::ExprUnary) -> Result<Expr> {
373    // Special case: negative integer literals (-1, -42, etc.)
374    // Simplify UnaryOp(Neg, Literal(n)) to Literal(-n)
375    if let UnOp::Neg(_) = &expr_unary.op {
376        if let SynExpr::Lit(lit_expr) = &*expr_unary.expr {
377            if let Lit::Int(lit_int) = &lit_expr.lit {
378                // Special case: i32::MIN (-2147483648)
379                // Can't parse 2147483648 as i32 since i32::MAX = 2147483647
380                let lit_str = lit_int.to_string();
381                if lit_str == "2147483648" {
382                    return Ok(Expr::Literal(Literal::I32(i32::MIN)));
383                }
384
385                // Parse as i32 for other negative numbers
386                let value: i32 = lit_int
387                    .base10_parse()
388                    .map_err(|_| Error::Validation("Invalid integer literal".to_string()))?;
389                // Return as negative literal
390                return Ok(Expr::Literal(Literal::I32(-value)));
391            }
392        }
393    }
394
395    // General case: convert normally
396    let operand = Box::new(convert_expr(&expr_unary.expr)?);
397    let op = convert_unary_op(&expr_unary.op)?;
398    Ok(Expr::Unary { op, operand })
399}

427fn convert_literal(lit: &Lit) -> Result<Literal> {
428    match lit {
429        Lit::Bool(lit_bool) => Ok(Literal::Bool(lit_bool.value)),
430        Lit::Int(lit_int) => {
431            let value: u32 = lit_int
432                .base10_parse()
433                .map_err(|_| Error::Validation("Invalid integer literal".to_string()))?;
434            Ok(Literal::U32(value))
435        }
436        Lit::Str(lit_str) => Ok(Literal::Str(lit_str.value())),
437        _ => Err(Error::Validation("Unsupported literal type".to_string())),
438    }
439}

442    match op {

444        BinOp::Sub(_) => Ok(BinaryOp::Sub),
445        BinOp::Mul(_) => Ok(BinaryOp::Mul),

453        BinOp::And(_) => Ok(BinaryOp::And),

467fn convert_range_expr(range_expr: &syn::ExprRange) -> Result<Expr> {
468    let start = range_expr
469        .start
470        .as_ref()
471        .ok_or_else(|| Error::Validation("Range must have start".to_string()))?;
472    let end = range_expr
473        .end
474        .as_ref()
475        .ok_or_else(|| Error::Validation("Range must have end".to_string()))?;
476
477    let inclusive = matches!(range_expr.limits, syn::RangeLimits::Closed(_));
478
479    Ok(Expr::Range {
480        start: Box::new(convert_expr(start)?),
481        end: Box::new(convert_expr(end)?),
482        inclusive,
483    })
484}

486fn convert_for_loop(for_loop: &syn::ExprForLoop) -> Result<Stmt> {
487    // Extract pattern (e.g., "i" in "for i in...")
488    let Pat::Ident(pat_ident) = &*for_loop.pat else {
489        return Err(Error::Validation(
490            "Complex patterns in for loops not supported".to_string(),
491        ));
492    };
493    let pattern = Pattern::Variable(pat_ident.ident.to_string());
494
495    // Convert iterator expression (e.g., 0..3)
496    let iter = convert_expr(&for_loop.expr)?;
497
498    // Convert body
499    let body = convert_block(&for_loop.body)?;
500
501    Ok(Stmt::For {
502        pattern,
503        iter,
504        body,
505        max_iterations: Some(1000), // Default safety limit
506    })
507}

509fn convert_while_loop(expr_while: &syn::ExprWhile) -> Result<Stmt> {
510    // Convert condition expression
511    let condition = convert_expr(&expr_while.cond)?;
512
513    // Convert body
514    let body = convert_block(&expr_while.body)?;
515
516    Ok(Stmt::While {
517        condition,
518        body,
519        max_iterations: Some(10000), // Safety limit for infinite loops
520    })
521}

523fn convert_match_stmt(expr_match: &syn::ExprMatch) -> Result<Stmt> {
524    // Convert the scrutinee (the expression being matched)
525    let scrutinee = convert_expr(&expr_match.expr)?;
526
527    // Convert each match arm
528    let mut arms = Vec::new();
529    for arm in &expr_match.arms {
530        let pattern = convert_pattern(&arm.pat)?;
531        let guard = if let Some((_, guard_expr)) = &arm.guard {
532            Some(convert_expr(guard_expr)?)
533        } else {
534            None
535        };
536
537        // Convert the arm body
538        let body = match &*arm.body {
539            SynExpr::Block(block) => convert_block(&block.block)?,
540            expr => vec![Stmt::Expr(convert_expr(expr)?)],
541        };
542
543        arms.push(MatchArm {
544            pattern,
545            guard,
546            body,
547        });
548    }
549
550    Ok(Stmt::Match { scrutinee, arms })
551}
552
553fn convert_pattern(pat: &Pat) -> Result<Pattern> {
554    match pat {
555        Pat::Lit(lit_pat) => {
556            // Convert literal patterns like 1, 2, "hello"
557            let literal = convert_literal(&lit_pat.lit)?;
558            Ok(Pattern::Literal(literal))
559        }
560        Pat::Ident(ident_pat) => {
561            // Check if this is a wildcard (_)
562            let name = ident_pat.ident.to_string();
563            if name == "_" {
564                Ok(Pattern::Wildcard)
565            } else {
566                Ok(Pattern::Variable(name))
567            }
568        }
569        Pat::Wild(_) => Ok(Pattern::Wildcard),
570        _ => Err(Error::Validation(format!(
571            "Unsupported pattern type: {:?}",
572            pat
573        ))),
574    }
575}
