11/// Parse Rust source code into a RestrictedAst
12pub fn parse(input: &str) -> Result<RestrictedAst> {
13    let file: File = syn::parse_str(input)?;
14
15    let mut functions = Vec::new();
16    let mut entry_point = None;
17
18    for item in file.items {
19        process_item(item, &mut functions, &mut entry_point)?;
20    }
21
22    let entry_point = entry_point
23        .ok_or_else(|| Error::Validation("No #[bashrs::main] function found".to_string()))?;
24
25    Ok(RestrictedAst {
26        functions,
27        entry_point,
28    })
29}

31fn process_item(
32    item: Item,
33    functions: &mut Vec<Function>,
34    entry_point: &mut Option<String>,
35) -> Result<()> {
36    let Item::Fn(item_fn) = item else {
37        return Err(Error::Validation(
38            "Only functions are allowed in Rash code".to_string(),
39        ));
40    };
41
42    let is_main = has_main_attribute(&item_fn) || item_fn.sig.ident == "main";
43    let function = convert_function(item_fn)?;
44
45    if is_main {
46        check_single_entry_point(entry_point, &function.name)?;
47        *entry_point = Some(function.name.clone());
48    }
49
50    functions.push(function);
51    Ok(())
52}

54fn has_main_attribute(item_fn: &ItemFn) -> bool {
55    item_fn.attrs.iter().any(is_main_attribute)
56}
57
58fn is_main_attribute(attr: &syn::Attribute) -> bool {
59    let path = attr.path();
60    path.segments.len() == 2
61        && path
62            .segments
63            .get(0)
64            .is_some_and(|seg| seg.ident == "bashrs" || seg.ident == "rash")
65        && path.segments.get(1).is_some_and(|seg| seg.ident == "main")
66}

68fn check_single_entry_point(current: &Option<String>, _new_name: &str) -> Result<()> {
69    if current.is_some() {
70        return Err(Error::Validation(
71            "Multiple #[bashrs::main] functions found".to_string(),
72        ));
73    }
74    Ok(())
75}

77fn convert_function(item_fn: ItemFn) -> Result<Function> {
78    let name = item_fn.sig.ident.to_string();
79
80    // Convert parameters
81    let mut params = Vec::new();
82    for input in item_fn.sig.inputs {
83        match input {
84            FnArg::Typed(pat_type) => {
85                if let Pat::Ident(pat_ident) = &*pat_type.pat {
86                    let param_name = pat_ident.ident.to_string();
87                    let param_type = convert_type(&pat_type.ty)?;
88                    params.push(Parameter {
89                        name: param_name,
90                        param_type,
91                    });
92                } else {
93                    return Err(Error::Validation(
94                        "Complex parameter patterns not supported".to_string(),
95                    ));
96                }
97            }
98            FnArg::Receiver(_) => {
99                return Err(Error::Validation("Self parameters not allowed".to_string()));
100            }
101        }
102    }
103
104    // Convert return type
105    let return_type = match &item_fn.sig.output {
106        ReturnType::Default => Type::Void, // Default to void/unit type
107        ReturnType::Type(_, ty) => convert_type(ty)?,
108    };
109
110    // Convert function body
111    let body = convert_block(&item_fn.block)?;
112
113    Ok(Function {
114        name,
115        params,
116        return_type,
117        body,
118    })
119}

121fn convert_type(ty: &SynType) -> Result<Type> {
122    match ty {
123        SynType::Path(type_path) => {
124            let path_str = type_path
125                .path
126                .segments
127                .iter()
128                .map(|seg| seg.ident.to_string())
129                .collect::<Vec<_>>()
130                .join("::");
131
132            match path_str.as_str() {
133                "bool" => Ok(Type::Bool),
134                "u32" | "i32" => Ok(Type::U32), // Treat i32 as u32 for now
135                "str" | "String" => Ok(Type::Str),
136                path if path.starts_with("Result") => {
137                    // Simple Result type parsing - assumes Result<T, E>
138                    Ok(Type::Result {
139                        ok_type: Box::new(Type::Str), // Simplified
140                        err_type: Box::new(Type::Str),
141                    })
142                }
143                path if path.starts_with("Option") => {
144                    // Simple Option type parsing - assumes Option<T>
145                    Ok(Type::Option {
146                        inner_type: Box::new(Type::Str), // Simplified
147                    })
148                }
149                _ => Err(Error::Validation(format!("Unsupported type: {path_str}"))),
150            }
151        }
152        SynType::Reference(type_ref) => {
153            // Handle &str and other reference types
154            if let SynType::Path(path) = &*type_ref.elem {
155                let path_str = path
156                    .path
157                    .segments
158                    .iter()
159                    .map(|seg| seg.ident.to_string())
160                    .collect::<Vec<_>>()
161                    .join("::");
162
163                match path_str.as_str() {
164                    "str" => Ok(Type::Str),
165                    _ => Err(Error::Validation(format!(
166                        "Unsupported reference type: &{path_str}"
167                    ))),
168                }
169            } else {
170                Err(Error::Validation(
171                    "Complex reference types not supported".to_string(),
172                ))
173            }
174        }
175        _ => Err(Error::Validation("Complex types not supported".to_string())),
176    }
177}

179fn convert_block(block: &Block) -> Result<Vec<Stmt>> {
180    let mut statements = Vec::new();
181
182    for stmt in &block.stmts {
183        statements.push(convert_stmt(stmt)?);
184    }
185
186    Ok(statements)
187}

189fn convert_stmt(stmt: &SynStmt) -> Result<Stmt> {
190    match stmt {
191        SynStmt::Local(local) => convert_let_stmt(local),
192        SynStmt::Expr(expr, _) => convert_expr_stmt(expr),
193        SynStmt::Macro(macro_stmt) => convert_macro_stmt(macro_stmt),
194        _ => Err(Error::Validation("Unsupported statement type".to_string())),
195    }
196}

198fn convert_let_stmt(local: &syn::Local) -> Result<Stmt> {
199    // Extract identifier from pattern (handle both Pat::Ident and Pat::Type)
200    // This allows: `let x = 5` and `let x: i32 = 5`
201    let pat_ident = match &local.pat {
202        Pat::Ident(ident) => ident,
203        Pat::Type(pat_type) => {
204            // For type-annotated patterns like `let args: Vec<String> = ...`
205            match &*pat_type.pat {
206                Pat::Ident(ident) => ident,
207                _ => {
208                    return Err(Error::Validation(
209                        "Complex patterns not supported in type annotations".to_string(),
210                    ));
211                }
212            }
213        }
214        _ => {
215            return Err(Error::Validation(
216                "Complex patterns not supported".to_string(),
217            ));
218        }
219    };
220
221    let name = pat_ident.ident.to_string();
222
223    let Some(init) = &local.init else {
224        return Err(Error::Validation(
225            "Let bindings must have initializers".to_string(),
226        ));
227    };
228
229    let value = convert_expr(&init.expr)?;
230    Ok(Stmt::Let { name, value })
231}

233fn convert_expr_stmt(expr: &SynExpr) -> Result<Stmt> {
234    match expr {
235        SynExpr::If(expr_if) => convert_if_stmt(expr_if),
236        SynExpr::ForLoop(for_loop) => convert_for_loop(for_loop),
237        SynExpr::While(expr_while) => convert_while_loop(expr_while),
238        SynExpr::Match(expr_match) => convert_match_stmt(expr_match),
239        SynExpr::Break(_) => Ok(Stmt::Break),
240        SynExpr::Continue(_) => Ok(Stmt::Continue),
241        _ => Ok(Stmt::Expr(convert_expr(expr)?)),
242    }
243}

245fn convert_macro_stmt(macro_stmt: &syn::StmtMacro) -> Result<Stmt> {
246    let macro_path = &macro_stmt.mac.path;
247    let macro_name = macro_path
248        .segments
249        .last()
250        .ok_or_else(|| Error::Validation("Empty macro path".to_string()))?
251        .ident
252        .to_string();
253
254    if macro_name == "println" {
255        // Parse macro tokens to extract the format string
256        let tokens = macro_stmt.mac.tokens.clone();
257        let parsed: syn::Expr = syn::parse2(tokens)
258            .map_err(|_| Error::Validation("Invalid println! arguments".to_string()))?;
259
260        // Convert the first argument as the format string
261        let arg = convert_expr(&parsed)?;
262
263        // Generate a function call to rash_println
264        Ok(Stmt::Expr(Expr::FunctionCall {
265            name: "rash_println".to_string(),
266            args: vec![arg],
267        }))
268    } else {
269        Err(Error::Validation(format!(
270            "Unsupported macro: {macro_name}!"
271        )))
272    }
273}

275fn convert_if_stmt(expr_if: &syn::ExprIf) -> Result<Stmt> {
276    let condition = convert_expr(&expr_if.cond)?;
277    let then_block = convert_block(&expr_if.then_branch)?;
278    let else_block = convert_else_block(&expr_if.else_branch)?;
279
280    Ok(Stmt::If {
281        condition,
282        then_block,
283        else_block,
284    })
285}

287fn convert_else_block(
288    else_branch: &Option<(syn::token::Else, Box<SynExpr>)>,
289) -> Result<Option<Vec<Stmt>>> {
290    let Some((_, else_expr)) = else_branch else {
291        return Ok(None);
292    };
293
294    match &**else_expr {
295        SynExpr::Block(block) => Ok(Some(convert_block(&block.block)?)),
296        SynExpr::If(nested_if) => convert_else_if(nested_if),
297        _ => Ok(None),
298    }
299}

301fn convert_else_if(nested_if: &syn::ExprIf) -> Result<Option<Vec<Stmt>>> {
302    let nested_condition = convert_expr(&nested_if.cond)?;
303    let nested_then = convert_block(&nested_if.then_branch)?;
304    let nested_else = convert_nested_else(&nested_if.else_branch)?;
305
306    Ok(Some(vec![Stmt::If {
307        condition: nested_condition,
308        then_block: nested_then,
309        else_block: nested_else,
310    }]))
311}
312
313fn convert_nested_else(
314    else_branch: &Option<(syn::token::Else, Box<SynExpr>)>,
315) -> Result<Option<Vec<Stmt>>> {
316    let Some((_, nested_else_expr)) = else_branch else {
317        return Ok(None);
318    };
319
320    match &**nested_else_expr {
321        SynExpr::Block(block) => Ok(Some(convert_block(&block.block)?)),
322        SynExpr::If(_) => {
323            // Recursively handle else-if-else-if chains
324            let stmt = SynStmt::Expr((**nested_else_expr).clone(), None);
325            Ok(Some(vec![convert_stmt(&stmt)?]))
326        }
327        _ => Ok(None),
328    }
329}

331fn convert_expr(expr: &SynExpr) -> Result<Expr> {
332    match expr {
333        SynExpr::Lit(expr_lit) => convert_literal_expr(expr_lit),
334        SynExpr::Path(expr_path) => convert_path_expr(expr_path),
335        SynExpr::Call(expr_call) => convert_call_expr(expr_call),
336        SynExpr::Binary(expr_binary) => convert_binary_expr(expr_binary),
337        SynExpr::Unary(expr_unary) => convert_unary_expr(expr_unary),
338        SynExpr::MethodCall(method_call) => convert_method_call_expr(method_call),
339        SynExpr::Return(ret_expr) => convert_return_expr(ret_expr),
340        SynExpr::Paren(expr_paren) => convert_expr(&expr_paren.expr),
341        SynExpr::If(_) => Err(Error::Validation(
342            "If expressions not supported in expression position".to_string(),
343        )),
344        SynExpr::Range(range_expr) => convert_range_expr(range_expr),
345        _ => Err(Error::Validation("Unsupported expression type".to_string())),
346    }
347}

365fn convert_call_expr(expr_call: &syn::ExprCall) -> Result<Expr> {
366    let SynExpr::Path(path) = &*expr_call.func else {
367        return Err(Error::Validation(
368            "Complex function calls not supported".to_string(),
369        ));
370    };
371
372    let name = path
373        .path
374        .segments
375        .iter()
376        .map(|seg| seg.ident.to_string())
377        .collect::<Vec<_>>()
378        .join("::");
379
380    let mut args = Vec::new();
381    for arg in &expr_call.args {

388fn convert_binary_expr(expr_binary: &syn::ExprBinary) -> Result<Expr> {
389    let left = Box::new(convert_expr(&expr_binary.left)?);
390    let right = Box::new(convert_expr(&expr_binary.right)?);

395fn convert_unary_expr(expr_unary: &syn::ExprUnary) -> Result<Expr> {
396    // Special case: negative integer literals (-1, -42, etc.)
397    // Simplify UnaryOp(Neg, Literal(n)) to Literal(-n)
398    if let UnOp::Neg(_) = &expr_unary.op {
399        if let SynExpr::Lit(lit_expr) = &*expr_unary.expr {
400            if let Lit::Int(lit_int) = &lit_expr.lit {
401                // Special case: i32::MIN (-2147483648)
402                // Can't parse 2147483648 as i32 since i32::MAX = 2147483647
403                let lit_str = lit_int.to_string();
404                if lit_str == "2147483648" {
405                    return Ok(Expr::Literal(Literal::I32(i32::MIN)));
406                }
407
408                // Parse as i32 for other negative numbers
409                let value: i32 = lit_int
410                    .base10_parse()
411                    .map_err(|_| Error::Validation("Invalid integer literal".to_string()))?;
412                // Return as negative literal
413                return Ok(Expr::Literal(Literal::I32(-value)));
414            }
415        }
416    }
417
418    // General case: convert normally
419    let operand = Box::new(convert_expr(&expr_unary.expr)?);
420    let op = convert_unary_op(&expr_unary.op)?;
421    Ok(Expr::Unary { op, operand })
422}

424fn convert_method_call_expr(method_call: &syn::ExprMethodCall) -> Result<Expr> {
425    let receiver = Box::new(convert_expr(&method_call.receiver)?);
426    let method = method_call.method.to_string();
427
428    // Special case: std::env::args().collect() â†’ PositionalArgs
429    if method == "collect" && method_call.args.is_empty() {
430        if let Expr::FunctionCall { name, args } = &*receiver {
431            if name == "std::env::args" && args.is_empty() {
432                return Ok(Expr::PositionalArgs);
433            }
434        }
435    }
436
437    // General case: regular method call
438    let mut args = Vec::new();
439    for arg in &method_call.args {
440        args.push(convert_expr(arg)?);
441    }
442
443    Ok(Expr::MethodCall {
444        receiver,
445        method,
446        args,
447    })
448}

460fn convert_literal(lit: &Lit) -> Result<Literal> {
461    match lit {

463        Lit::Int(lit_int) => {

500fn convert_range_expr(range_expr: &syn::ExprRange) -> Result<Expr> {
501    let start = range_expr
502        .start
503        .as_ref()
504        .ok_or_else(|| Error::Validation("Range must have start".to_string()))?;
505    let end = range_expr
506        .end
507        .as_ref()
508        .ok_or_else(|| Error::Validation("Range must have end".to_string()))?;
509
510    let inclusive = matches!(range_expr.limits, syn::RangeLimits::Closed(_));
511
512    Ok(Expr::Range {
513        start: Box::new(convert_expr(start)?),
514        end: Box::new(convert_expr(end)?),
515        inclusive,
516    })
517}

519fn convert_for_loop(for_loop: &syn::ExprForLoop) -> Result<Stmt> {
520    // Extract pattern (e.g., "i" in "for i in...")
521    let Pat::Ident(pat_ident) = &*for_loop.pat else {
522        return Err(Error::Validation(
523            "Complex patterns in for loops not supported".to_string(),
524        ));
525    };
526    let pattern = Pattern::Variable(pat_ident.ident.to_string());
527
528    // Convert iterator expression (e.g., 0..3)
529    let iter = convert_expr(&for_loop.expr)?;
530
531    // Convert body
532    let body = convert_block(&for_loop.body)?;
533
534    Ok(Stmt::For {
535        pattern,
536        iter,
537        body,
538        max_iterations: Some(1000), // Default safety limit
539    })
540}

556fn convert_match_stmt(expr_match: &syn::ExprMatch) -> Result<Stmt> {
557    // Convert the scrutinee (the expression being matched)
558    let scrutinee = convert_expr(&expr_match.expr)?;
559
560    // Convert each match arm
561    let mut arms = Vec::new();
562    for arm in &expr_match.arms {
563        let pattern = convert_pattern(&arm.pat)?;
564        let guard = if let Some((_, guard_expr)) = &arm.guard {
565            Some(convert_expr(guard_expr)?)
566        } else {
567            None
568        };
569
570        // Convert the arm body
571        let body = match &*arm.body {
572            SynExpr::Block(block) => convert_block(&block.block)?,
573            expr => vec![Stmt::Expr(convert_expr(expr)?)],
574        };
575
576        arms.push(MatchArm {
577            pattern,
578            guard,
579            body,
580        });
581    }
582
583    Ok(Stmt::Match { scrutinee, arms })
584}

586fn convert_pattern(pat: &Pat) -> Result<Pattern> {
587    match pat {
588        Pat::Lit(lit_pat) => {
589            // Convert literal patterns like 1, 2, "hello"
590            let literal = convert_literal(&lit_pat.lit)?;
591            Ok(Pattern::Literal(literal))
592        }
593        Pat::Ident(ident_pat) => {
594            // Check if this is a wildcard (_)
595            let name = ident_pat.ident.to_string();
596            if name == "_" {
597                Ok(Pattern::Wildcard)
598            } else {
599                Ok(Pattern::Variable(name))
600            }
601        }
602        Pat::Wild(_) => Ok(Pattern::Wildcard),
603        _ => Err(Error::Validation(format!(
604            "Unsupported pattern type: {:?}",
605            pat
606        ))),
607    }
608}
