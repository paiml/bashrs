Shell type detection for correct linting.

Automatically detects which shell (bash, zsh, sh, ksh) a script is written for, allowing bashrs to apply appropriate linting rules based on shell compatibility.

use std::path::Path;

let path = Path::new("script.sh");
let shell = detect_shell_type(path, content);
assert_eq!(shell, ShellType::Zsh);
